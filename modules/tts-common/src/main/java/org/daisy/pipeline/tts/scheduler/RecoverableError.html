3   /**
4    * Exception to be raised by Schedulable object when an error occurred but that object can be
5    * rescheduled.
6    *
7    * @author Nicolas Pavie @ braillenet.org
8    */
9   public class RecoverableError extends Exception {
10  
11  	private static final long serialVersionUID = 1L;
12  
13  	public RecoverableError(String message) {
14  		super(message);
15  	}
16  
17  	public RecoverableError(String message, Throwable cause) {
18  		super(message, cause);
19  	}
20  	
21  	public RecoverableError(Throwable cause) {
22  		super(cause);
23  	}
24  }
