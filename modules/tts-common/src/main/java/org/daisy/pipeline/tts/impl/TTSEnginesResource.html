40  public class TTSEnginesResource extends AuthenticatedResource {
41  
42  	static final String TTS_REGISTRY_KEY = "tts-registry";
43  
44  	private static final Logger logger = LoggerFactory.getLogger(TTSEnginesResource.class.getName());
45  	private static final Processor saxonProcessor = new Processor(false);
46  
47  	private TTSRegistry ttsRegistry;
48  
49  	@Override
50  	public void doInit() {
51  		super.doInit();
52  		if (!isAuthenticated()) {
53  			return;
54  		}
55  		ttsRegistry = (TTSRegistry)getContext().getAttributes().get(TTS_REGISTRY_KEY);
56  	}
57  
58  	/**
59  	 * Gets the resource.
60  	 *
61  	 * @return the resource
62  	 */
63  	@Get("xml")
64  	public Representation getResource() {
65  		return getResource(null);
66  	}
67  
68  	/**
69  	 * @param ttsConfig
70  	 */
71  	@Post("xml")
72  	public Representation getResource(Representation ttsConfig) {
73  		logRequest();
74  		maybeEnableCORS();
75  		if (!isAuthenticated()) {
76  			setStatus(Status.CLIENT_ERROR_UNAUTHORIZED);
77  			return null;
78  		}
79  		Collection<TTSService> ttsServices = ttsRegistry.getServices();
80  		Map<String,String> properties; {
81  			DynamicPropertiesExtension propsExt = new DynamicPropertiesExtension();
82  			XdmNode configXML = null; {
83  				if (ttsConfig != null) {
84  					try {
85  						configXML = saxonProcessor.newDocumentBuilder().build(
86  							new SAXSource(new InputSource(new StringReader(ttsConfig.getText()))));
87  					} catch (IOException|SaxonApiException e) {
88  						logger.error("bad request:", e);
89  						setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
90  						return getErrorRepresentation(e.getMessage());
91  					}
92  				}
93  			}
94  			new ConfigReader(saxonProcessor, configXML, propsExt);
95  			if (configXML != null)
96  				logger.debug("TTS configuration XML:\n" + configXML);
97  			properties = Properties.getSnapshot();
98  			Map<String,String> dynProperties = propsExt.getDynamicProperties();
99  			if (dynProperties != null && !dynProperties.isEmpty()) {
100 				properties = new HashMap<>(properties);
101 				properties.putAll(dynProperties);
102 			}
103 		}
104 		Document enginesDoc; {
105 			enginesDoc = XmlUtils.createDom("tts-engines");
106 			Element enginesElem = enginesDoc.getDocumentElement();
107 			String baseURL = getRequest().getRootRef().toString();
108 			enginesElem.setAttribute("href", baseURL + TTSEnginesWebServiceExtension.TTS_ENGINES_ROUTE);
109 			for (TTSService s : ttsServices) {
110 				Element engineElem = enginesDoc.createElementNS(XmlUtils.NS_PIPELINE_DATA, "tts-engine");
111 				String name = s.getName();
112 				engineElem.setAttribute("name", name);
113 				engineElem.setAttribute("nicename", s.getDisplayName());
114 				TTSEngine e = null;
115 				Throwable error = null;
116 				try {
117 				    e = s.newEngine(properties);
118 					engineElem.setAttribute("status", "available");
119 				} catch (ServiceDisabledException ex) {
120 					logger.debug(name + " is disabled", ex);
121 					error = ex;
122 					engineElem.setAttribute("status", "disabled");
123 				} catch (Throwable ex) {
124 					logger.debug(name + " could not be activated", ex);
125 					error = ex;
126 					engineElem.setAttribute("status", "error");
127 				}
128 				if (e != null) {
129 					engineElem.setAttribute("voices", baseURL + VoicesWebServiceExtension.VOICES_ROUTE + "?engine=" + name);
130 					if (e.handlesSpeakingRate())
131 						engineElem.setAttribute("features", "speech-rate");
132 				} else if (error != null) {
133 					// Clients should use first line as the short message. The short message is
134 					// followed by the full message after a blank line.
135 					String shortMessage = error.getMessage();
136 					error = error.getCause();
137 					String detailedMessage = error != null ? error.getMessage() : null;
138 					if (shortMessage.length() > 80) {
139 						// Use the heuristic that if a message is longer than 80 characters, it is possible
140 						// that it is too technical for the average user. It also becomes difficult to fit it
141 						// in the UI. So provide this backup:
142 						if (detailedMessage == null)
143 							detailedMessage = shortMessage;
144 						else {
145 							detailedMessage = detailedMessage.trim();
146 							shortMessage = shortMessage.trim();
147 							if (detailedMessage.startsWith(shortMessage))
148 								detailedMessage = detailedMessage.substring(shortMessage.length());
149 							if (detailedMessage.startsWith(":")) {
150 								detailedMessage = detailedMessage.substring(1);
151 								detailedMessage = detailedMessage.trim();
152 							}
153 							if (detailedMessage.length() > 0) {
154 								if (!shortMessage.matches(".*\\p{Punct}$"))
155 									shortMessage += ":";
156 								detailedMessage = shortMessage + " " + detailedMessage;
157 							} else
158 								detailedMessage = shortMessage;
159 						}
160 						shortMessage = "Could not connect to the service";
161 					}
162 					String message = shortMessage;
163 					if (detailedMessage != null)
164 						message += ("\n\n" + detailedMessage);
165 					engineElem.setAttribute("message", message);
166 				}
167 				enginesElem.appendChild(engineElem);
168 			}
169 		}
170 		DomRepresentation dom; {
171 			try {
172 				dom = new DomRepresentation(MediaType.APPLICATION_XML, enginesDoc);
173 				setStatus(Status.SUCCESS_OK);
174 			} catch (Exception e) {
175 				setStatus(Status.SERVER_ERROR_INTERNAL);
176 				return getErrorRepresentation(e.getMessage());
177 			}
178 		}
179 		logResponse(dom);
180 		return dom;
181 	}
182 }
