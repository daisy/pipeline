1   package org.daisy.pipeline.tts.calabash.impl;
2   
3   import net.sf.saxon.s9api.Processor;
4   import net.sf.saxon.s9api.QName;
5   import net.sf.saxon.s9api.SaxonApiException;
6   import net.sf.saxon.s9api.XdmNode;
7   
8   import org.daisy.common.xproc.calabash.XProcStep;
9   import org.daisy.common.xproc.calabash.XProcStepProvider;
10  import org.daisy.pipeline.tts.config.AnnotationsConfigExtension;
11  import org.daisy.pipeline.tts.config.ConfigReader;
12  
13  import com.xmlcalabash.core.XProcRuntime;
14  import com.xmlcalabash.io.ReadablePipe;
15  import com.xmlcalabash.io.WritablePipe;
16  import com.xmlcalabash.model.RuntimeValue;
17  import com.xmlcalabash.runtime.XAtomicStep;
18  
19  import org.osgi.service.component.annotations.Component;
20  
21  @Component(
22  	name = "get-tts-annotations",
23  	service = { XProcStepProvider.class },
24  	property = { "type:String={http://www.daisy.org/ns/pipeline/xproc}get-tts-annotations" }
25  )
26  public class GetAnnotationsProvider implements XProcStepProvider {
27  
28  	@Override
29  	public XProcStep newStep(final XProcRuntime runtime, XAtomicStep step) {
30  		return new XProcStep() {
31  
32  			private String mContentType;
33  			private ReadablePipe mConfig;
34  			private WritablePipe mResult;
35  
36  			public void setInput(String port, ReadablePipe pipe) {
37  				if ("config".equalsIgnoreCase(port))
38  					mConfig = pipe;
39  				else {
40  					runtime.error(new Throwable("unknown port " + port));
41  				}
42  			}
43  
44  			@Override
45  			public void setOption(QName name, RuntimeValue value) {
46  				String optName = name.getLocalName();
47  				if ("content-type".equalsIgnoreCase(optName)) {
48  					mContentType = value.getString();
49  				} else {
50  					runtime.error(new Throwable("unknown option " + optName));
51  					return;
52  				}
53  			}
54  
55  			public void setOutput(String port, WritablePipe pipe) {
56  				mResult = pipe;
57  			}
58  
59  			public void reset() {
60  				mConfig.resetReader();
61  				mResult.resetWriter();
62  			}
63  
64  			public void run() throws SaxonApiException {
65  
66  				Processor proc = runtime.getProcessor();
67  				AnnotationsConfigExtension annoExt = new AnnotationsConfigExtension(proc);
68  				new ConfigReader(proc, mConfig.read(), annoExt);
69  
70  				for (XdmNode annotations : annoExt.getAnnotations(mContentType)) {
71  					mResult.write(annotations);
72  				}
73  			}
74  
75  			@Override
76  			public void setParameter(QName arg0, RuntimeValue arg1) {
77  			}
78  
79  			@Override
80  			public void setParameter(String arg0, QName arg1, RuntimeValue arg2) {
81  			}
82  		};
83  	}
84  }
