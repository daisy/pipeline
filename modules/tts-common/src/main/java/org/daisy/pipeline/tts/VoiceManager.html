31  public class VoiceManager {
32  
33  	private Logger ServerLogger = LoggerFactory.getLogger(VoiceManager.class);
34  
35  	/**
36  	 * Map of the best services for each available voice, given that two different services can serve the same voice
37  	 */
38  	private final Map<Voice,TTSEngine> bestEngines;
39  	/**
40  	 * Map from voice name/engine to voice
41  	 */
42  	private final Map<VoiceKey,Voice> primaryVoices;
43  	/**
44  	 * Map from voice (name/engine) to a collection of fallback voices (sorted by matching engine,
45  	 * matching gender and priority).
46  	 */
47  	private final Map<VoiceKey,Collection<Voice>> secondaryVoices;
48  	/**
49  	 * Map from voice properties (language/gender/engine) to voices. Each combination of primary
50  	 * language, gender and engine is mapped to a collection of voices with corresponding language
51  	 * range, sorted by priority.
52  	 */
53  	private final Map<VoiceKey,Collection<Entry<Voice,Collection<LanguageRange>>>> voiceIndex;
54  
55  	public VoiceManager(Collection<TTSEngine> engines, Collection<VoiceInfo> voiceInfoFromConfig) {
56  
57  		// create a map of the best services for each available voice
58  		bestEngines = new LinkedHashMap<Voice,TTSEngine>(); { // LinkedHashMap: iteration order = insertion order
59  			TTSTimeout timeout = new TTSTimeout();
60  			int timeoutSecs = 30;
61  			// sort engines by engine priority, so that voice info from engines (see below) is sorted by engine priority
62  			// as well
63  			List<TTSEngine> sortedEngines = new ArrayList<>(engines);
64  			Collections.sort(sortedEngines, Comparator.comparingInt(TTSEngine::getOverallPriority).reversed());
65  			for (TTSEngine tts : sortedEngines) {
66  				timeout.enableForCurrentThread(timeoutSecs);
67  				try {
68  					Collection<Voice> voices = tts.getAvailableVoices();
69  					if (voices != null)
70  						for (Voice v : voices)
71  							if (!bestEngines.containsKey(v))
72  								bestEngines.put(v, tts);
73  				} catch (SynthesisException e) {
74  					ServerLogger.error("error while retrieving the voices of "
75  					                   + tts.getProvider().getName());
76  					ServerLogger.debug(tts.getProvider().getName()
77  					                   + " getAvailableVoices error: " + getStack(e));
78  				} catch (InterruptedException e) {
79  					ServerLogger.error("timeout while retrieving the voices of "
80  					                   + tts.getProvider().getName()
81  					                   + " (exceeded " + timeoutSecs + " seconds)");
82  				} finally {
83  					timeout.disable();
84  				}
85  			}
86  			timeout.close();
87  		}
88  
89  		// get mappings from voice properties (language/gender) to voice (name/engine)
90  		List<VoiceInfo> voiceInfo = new ArrayList<>();
91  
92  		// get info from engines (lowest priority)
93  		primaryVoices = new HashMap<>();
94  		for (Voice v : bestEngines.keySet()) {
95  			primaryVoices.put(new VoiceKey(v.getEngine(), v.getName()), v);
96  			if (!v.getLocale().isEmpty() && v.getGender().isPresent()) {
97  				Gender g = v.getGender().get();
98  				for (LanguageRange l : v.getLocale())
99  					voiceInfo.add(new VoiceInfo(v.getEngine(), v.getName(), l, g, 0));
100 			}
101 		}
102 
103 		// voices also match less specific locales (without region tag)
104 		final float priorityVariantPenalty = 0.1f;
105 		List<VoiceInfo> derivedVoiceInfo = new ArrayList<>(); {
106 			for (VoiceInfo vi : voiceInfo) {
107 				Locale shortLang = vi.language.getPrimaryLanguageSubTag();
108 				if (shortLang != null && !shortLang.toString().equals(vi.language.toString()))
109 					derivedVoiceInfo.add(
110 						new VoiceInfo(vi.voiceEngine, vi.voiceName, shortLang, vi.gender,
111 						              vi.priority - priorityVariantPenalty));
112 			}
113 		}
114 		voiceInfo.addAll(derivedVoiceInfo);
115 
116 		// get info from configuration
117 		// the configuration is interpreted as-is. region and other subtags are significant
118 		// due to fuzzy voice selection, a voice can be applied to less specific locales, but it is
119 		// not an exact match
120 		voiceInfo.addAll(voiceInfoFromConfig);
121 
122 		// sort by priority and language range specificity (descending)
123 		// voices with equal priority and specificity will not be reordered as a result of the sort
124 		Collections.sort(voiceInfo, sortByPriority.thenComparing(sortBySpecificity));
125 
126 		// filter available voices
127 		List<VoiceInfo> availableVoiceInfo = new ArrayList<>(); {
128 			for (VoiceInfo i : voiceInfo)
129 				if (primaryVoices.containsKey(new VoiceKey(i.voiceEngine, i.voiceName)))
130 					availableVoiceInfo.add(i);
131 		}
132 
133 		// create map of the best fallback for each voice
134 		// engine is more important than gender and priority, gender is more important than priority
135 		secondaryVoices = new HashMap<>(); {
136 			for (boolean sameEngine : new boolean[]{true, false})
137 				for (boolean sameGender : new boolean[]{true, false})
138 					for (VoiceInfo best : voiceInfo) {
139 						VoiceKey bestKey = new VoiceKey(best.voiceEngine, best.voiceName);
140 						for (VoiceInfo fallback : availableVoiceInfo) {
141 							VoiceKey fallbackKey = new VoiceKey(fallback.voiceEngine, fallback.voiceName);
142 							if (!fallbackKey.equals(bestKey))
143 								if (best.language.getPrimaryLanguageSubTag() == null) {
144 									if (!sameGender && !sameEngine) {
145 										// multilang fallback voices are only considered when gender
146 										// and engine are not criteria so as to prevent the algo
147 										// from choosing a multilang voice with the same engine over
148 										// a regular voice with a different engine.
149 										Collection<Voice> vv = secondaryVoices.get(bestKey);
150 										if (vv == null) {
151 											vv = new LinkedHashSet<>();
152 											secondaryVoices.put(bestKey, vv);
153 										}
154 										vv.add(primaryVoices.get(fallbackKey));
155 									}
156 								} else if (fallback.language.equals(best.language)
157 								           && (!sameGender
158 								               // fallback voices with unknown gender are only considered
159 								               // when gender is not a criteria
160 								               || (fallback.gender != NO_DEFINITE_GENDER
161 								                   && fallback.gender == best.gender))
162 								           && (!sameEngine
163 								               || fallback.voiceEngine.equals(best.voiceEngine))) {
164 									Collection<Voice> vv = secondaryVoices.get(bestKey);
165 									if (vv == null) {
166 										vv = new LinkedHashSet<>();
167 										secondaryVoices.put(bestKey, vv);
168 									}
169 									vv.add(primaryVoices.get(fallbackKey));
170 								}
171 						}
172 					}
173 		}
174 
175 		// create index of voices with language, engine and gender as keys
176 		Set<Locale> allLangs = new HashSet<Locale>(); {
177 			for (VoiceInfo vi : availableVoiceInfo) {
178 				Locale shortLang = vi.language.getPrimaryLanguageSubTag();
179 				if (shortLang != null)
180 					allLangs.add(shortLang); }}
181 		Set<Gender> allGenders = new HashSet<Gender>(); {
182 			for (VoiceInfo vi : availableVoiceInfo) allGenders.add(vi.gender);
183 			allGenders.remove(NO_DEFINITE_GENDER); }
184 		Map<VoiceKey,LinkedHashMap<Voice,Collection<LanguageRange>>> voiceIndex = new HashMap<>(); {
185 			for (VoiceInfo vi : availableVoiceInfo) {
186 				if (vi.language.getPrimaryLanguageSubTag() == null)
187 					// this is to make sure that a multi-lang voice wins from a regular voice if it has
188 					// a higher priority
189 					for (Locale shortLang : allLangs)
190 						for (boolean sameEngine : new boolean[]{true, false}) {
191 							if (vi.gender == NO_DEFINITE_GENDER)
192 								// this is to make sure that a voice with unknown gender wins from a regular voice
193 								// if it has a higher priority
194 								for (Gender g : allGenders) {
195 									VoiceKey k = new VoiceKey(shortLang, g, sameEngine ? vi.voiceEngine : null);
196 									LinkedHashMap<Voice,Collection<LanguageRange>> vv = voiceIndex.get(k);
197 									if (vv == null) {
198 										vv = new LinkedHashMap<>();
199 										voiceIndex.put(k, vv);
200 									}
201 									Voice v = primaryVoices.get(new VoiceKey(vi.voiceEngine, vi.voiceName));
202 									Collection<LanguageRange> lang = vv.get(v);
203 									if (lang == null) {
204 										lang = new HashSet<>();
205 										vv.put(v, lang);
206 									}
207 									lang.add(vi.language);
208 								}
209 							for (boolean sameGender : new boolean[]{true, false}) {
210 								VoiceKey k = new VoiceKey(shortLang,
211 								                          sameGender ? vi.gender : null,
212 								                          sameEngine ? vi.voiceEngine : null);
213 								LinkedHashMap<Voice,Collection<LanguageRange>> vv = voiceIndex.get(k);
214 								if (vv == null) {
215 									vv = new LinkedHashMap<>();
216 									voiceIndex.put(k, vv);
217 								}
218 								Voice v = primaryVoices.get(new VoiceKey(vi.voiceEngine, vi.voiceName));
219 								Collection<LanguageRange> lang = vv.get(v);
220 								if (lang == null) {
221 									lang = new HashSet<>();
222 									vv.put(v, lang);
223 								}
224 								lang.add(vi.language);
225 							}
226 						}
227 				for (boolean sameLanguage : new boolean[]{true, false}) {
228 					Locale shortLang = vi.language.getPrimaryLanguageSubTag();
229 					if (shortLang == null) shortLang = VoiceKey.MUL;
230 					for (boolean sameEngine : new boolean[]{true, false}) {
231 						if (vi.gender == NO_DEFINITE_GENDER)
232 							for (Gender g : allGenders) {
233 								VoiceKey k = new VoiceKey(sameLanguage ? shortLang : null,
234 								                          g,
235 								                          sameEngine ? vi.voiceEngine : null);
236 								LinkedHashMap<Voice,Collection<LanguageRange>> vv = voiceIndex.get(k);
237 								if (vv == null) {
238 									vv = new LinkedHashMap<>();
239 									voiceIndex.put(k, vv);
240 								}
241 								Voice v = primaryVoices.get(new VoiceKey(vi.voiceEngine, vi.voiceName));
242 								Collection<LanguageRange> lang = vv.get(v);
243 								if (lang == null) {
244 									lang = new HashSet<>();
245 									vv.put(v, lang);
246 								}
247 								lang.add(vi.language);
248 							}
249 						for (boolean sameGender : new boolean[]{true, false}) {
250 							VoiceKey k = new VoiceKey(sameLanguage ? shortLang : null,
251 							                          sameGender ? vi.gender : null,
252 							                          sameEngine ? vi.voiceEngine : null);
253 							LinkedHashMap<Voice,Collection<LanguageRange>> vv = voiceIndex.get(k);
254 							if (vv == null) {
255 								vv = new LinkedHashMap<>();
256 								voiceIndex.put(k, vv);
257 							}
258 							Voice v = primaryVoices.get(new VoiceKey(vi.voiceEngine, vi.voiceName));
259 							Collection<LanguageRange> lang = vv.get(v);
260 							if (lang == null) {
261 								lang = new HashSet<>();
262 								vv.put(v, lang);
263 							}
264 							lang.add(vi.language);
265 						}
266 					}
267 				}
268 			}
269 		}
270 		this.voiceIndex = Maps.transformValues(voiceIndex, Map::entrySet);
271 
272 		// log
273 		StringBuilder sb = new StringBuilder("Available voices:");
274 		for (Entry<Voice,TTSEngine> e : bestEngines.entrySet())
275 			sb.append("\n * ")
276 			  .append(e.getKey());
277 		ServerLogger.debug(sb.toString());
278 		sb = new StringBuilder("Voice selection data:");
279 		Collection<VoiceInfo> sortedAvailableVoiceInfo = new TreeSet<VoiceInfo>(
280 			new Comparator<VoiceInfo>() {
281 				public int compare(VoiceInfo vi1, VoiceInfo vi2) {
282 					// first group by locale
283 					// multi-lang voices last
284 					if (vi1.language.getPrimaryLanguageSubTag() == null && vi2.language.getPrimaryLanguageSubTag() != null)
285 						return 1;
286 					else if (vi1.language.getPrimaryLanguageSubTag() != null && vi2.language.getPrimaryLanguageSubTag() == null)
287 						return -1;
288 					int compare = vi1.language.toString().compareTo(vi2.language.toString());
289 					if (compare != 0) return compare;
290 					// then group by gender
291 					compare = vi1.gender.compareTo(vi2.gender);
292 					if (compare != 0) return compare;
293 					// remove duplicates
294 					// (Note that the duplicate with the highest priority is kept because availableVoiceInfo
295 					// is sorted by descending priority and because of the way TreeSet works)
296 					if (vi1.voiceEngine.equals(vi2.voiceEngine) && vi1.voiceName.equals(vi2.voiceName))
297 						return 0;
298 					// highest priority first
299 					compare = Float.valueOf(vi2.priority).compareTo(Float.valueOf(vi1.priority));
300 					if (compare != 0)
301 						return compare;
302 					compare = vi2.voiceEngine.compareTo(vi1.voiceEngine);
303 					if (compare != 0)
304 						return compare;
305 					return vi2.voiceName.compareTo(vi1.voiceName);
306 				}
307 			}
308 		);
309 		sortedAvailableVoiceInfo.addAll(availableVoiceInfo);
310 		for (VoiceInfo vi : sortedAvailableVoiceInfo)
311 			sb.append("\n * {")
312 			  .append("locale:").append(vi.language)
313 			  .append(", gender:").append(vi.gender)
314 			  .append("}")
315 			  .append(" -> ")
316 			  .append(primaryVoices.get(new VoiceKey(vi.voiceEngine, vi.voiceName)));
317 		ServerLogger.debug(sb.toString());
318 	}

348 	public Iterable<Voice> findAvailableVoices(String voiceEngine, String voiceName, Locale lang, Gender gender) {
349 		Set<Voice> voices = new LinkedHashSet<>();
350 		if (lang == null && gender == null &&
351 		    voiceEngine != null && !voiceEngine.isEmpty() && voiceName != null && !voiceName.isEmpty()) {
352 			VoiceKey preferred = new VoiceKey(voiceEngine, voiceName);
353 			Voice primary = primaryVoices.get(preferred);
354 			if (primary != null)
355 				voices.add(primary);
356 			Collection<Voice> fallback = secondaryVoices.get(preferred);
357 			if (fallback != null)
358 				voices.addAll(fallback);
359 		}
360 		if (lang != null || gender != null || voiceName == null) {
361 			Locale shortLang = lang != null ? new Locale(lang.getLanguage()) : lang;
362 
363 			// engine is more important than gender, region and priority, gender is more important than region
364 			// and priority, region is more important than priority
365 			addExactMatches(voices, lang, shortLang, gender, voiceEngine);
366 			if (gender != null && gender != NO_DEFINITE_GENDER)
367 				addExactMatches(voices, lang, shortLang, NO_DEFINITE_GENDER, voiceEngine);
368 			if (lang != null) {
369 				if (!lang.equals(shortLang)) {
370 					addExactMatches(voices, shortLang, shortLang, gender, voiceEngine);
371 					if (gender != null && gender != NO_DEFINITE_GENDER)
372 						addExactMatches(voices, shortLang, shortLang, NO_DEFINITE_GENDER, voiceEngine);
373 				}
374 				if (!lang.equals(VoiceKey.MUL)) {
375 					addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, gender, voiceEngine);
376 					if (gender != null && gender != NO_DEFINITE_GENDER)
377 						addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, NO_DEFINITE_GENDER, voiceEngine);
378 				}
379 			}
380 			if (gender != null) {
381 				addExactMatches(voices, lang, shortLang, null, voiceEngine);
382 				if (lang != null) {
383 					if (!lang.equals(shortLang))
384 						addExactMatches(voices, shortLang, shortLang, null, voiceEngine);
385 					if (!lang.equals(VoiceKey.MUL))
386 						addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, null, voiceEngine);
387 				}
388 			}
389 			if (voiceEngine != null) {
390 				addExactMatches(voices, lang, shortLang, gender, null);
391 				if (gender != null && gender != NO_DEFINITE_GENDER)
392 					addExactMatches(voices, lang, shortLang, NO_DEFINITE_GENDER, null);
393 				if (lang != null) {
394 					if (!lang.equals(shortLang)) {
395 						addExactMatches(voices, shortLang, shortLang, gender, null);
396 						if (gender != null && gender != NO_DEFINITE_GENDER)
397 							addExactMatches(voices, shortLang, shortLang, NO_DEFINITE_GENDER, null);
398 					}
399 					if (!lang.equals(VoiceKey.MUL)) {
400 						addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, gender, null);
401 						if (gender != null && gender != NO_DEFINITE_GENDER)
402 							addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, NO_DEFINITE_GENDER, null);
403 					}
404 				}
405 			}
406 			if (gender != null && voiceEngine != null) {
407 				addExactMatches(voices, lang, shortLang, null, null);
408 				if (lang != null) {
409 					if (!lang.equals(shortLang))
410 						addExactMatches(voices, shortLang, shortLang, null, null);
411 					if (!lang.equals(VoiceKey.MUL))
412 						addExactMatches(voices, VoiceKey.MUL, VoiceKey.MUL, null, null);
413 				}
414 			}
415 		}
416 		return voices;
417 	}

419 	private void addExactMatches(Collection<Voice> collect, Locale lang, Locale shortLang, Gender gender, String voiceEngine) {
420 		Collection<Entry<Voice,Collection<LanguageRange>>> vv = voiceIndex.get(new VoiceKey(shortLang, gender, voiceEngine));
421 		if (vv != null)
422 			for (Entry<Voice,Collection<LanguageRange>> v : vv)
423 				if (lang == null || LanguageRange.matches(v.getValue(), lang))
424 					collect.add(v.getKey());
425 	}

427 	/**
428 	 * Whether the given voice matches exactly the given voice engine, voice name, language and/or gender.
429 	 */
430 	public boolean matches(Voice voice, String voiceEngine, String voiceName, Locale lang, Gender gender) {
431 		if (voiceEngine != null && !voiceEngine.equalsIgnoreCase(voice.getEngine()))
432 			return false;
433 		if (voiceName != null && !voiceName.equalsIgnoreCase(voice.getName()))
434 			return false;
435 		if (lang == null && gender == null)
436 			return true;
437 		for (VoiceKey k : voiceIndex.keySet()) {
438 			if (k.primaryLang != null && k.gender != null && k.engine != null)
439 				for (Entry<Voice,Collection<LanguageRange>> v : voiceIndex.get(k))
440 					if (v.getKey().equals(voice)) {
441 						if (lang != null && !LanguageRange.matches(v.getValue(), lang))
442 							break;
443 						else if (gender != null
444 						    && k.gender != NO_DEFINITE_GENDER
445 						    && !k.gender.equals(gender))
446 							break;
447 						else
448 							return true;
449 					}
450 		}
451 		return false;
452 	}

454 	private static String getStack(Throwable t) {
455 		StringWriter writer = new StringWriter();
456 		PrintWriter printWriter = new PrintWriter(writer);
457 		t.printStackTrace(printWriter);
458 		printWriter.flush();
459 		return writer.toString();

463 	private static final Comparator<VoiceInfo> sortByPriority = new Comparator<VoiceInfo>() {
464 			public int compare(VoiceInfo vi1, VoiceInfo vi2) {
465 				return Float.valueOf(vi2.priority).compareTo(Float.valueOf(vi1.priority));
466 			}

469 	// sort VoiceInfo by specificity of language range (descending)
470 	private static final Comparator<VoiceInfo> sortBySpecificity
471 		= Comparator.<VoiceInfo>comparingInt(vi -> vi.language.getSpecificity()).reversed();

473 	private static class VoiceKey {
474 
475 		private static final Locale MUL = new Locale("mul");
476 
477 		private final String engine;
478 		private final String name;
479 		private final Locale primaryLang; // primary language subtag or "mul"
480 		private final Gender gender;
481 
482 		public VoiceKey(String engine, String name) {
483 			this.engine = engine == null ? null : engine.toLowerCase();
484 			this.name = name == null ? null : name.toLowerCase();
485 			this.primaryLang = null;
486 			this.gender = null;
487 		}
488 
489 		public VoiceKey(Locale primaryLang, Gender gender, String engine) {
490 			this.primaryLang = primaryLang;
491 			this.gender = gender;
492 			this.engine = engine == null ? null : engine.toLowerCase();
493 			this.name = null;
494 		}
495 
496 		@Override
497 		public int hashCode() {
498 			int res = 0;
499 			if (this.primaryLang != null)
500 				res ^=  this.primaryLang.hashCode();
501 			if (this.gender != null)
502 				res ^= this.gender.hashCode();
503 			if (this.engine != null)
504 				res ^= this.engine.hashCode();
505 			if (this.name != null)
506 				res ^= this.name.hashCode();
507 			return res;
508 		}
509 
510 		@Override
511 		public boolean equals(Object other) {
512 			VoiceKey o = (VoiceKey) other;
513 			return (primaryLang == o.primaryLang || (primaryLang != null && primaryLang.equals(o.primaryLang)))
514 			        && (gender == o.gender || (gender != null && gender.equals(o.gender)))
515 			        && (engine == o.engine || (engine != null && engine.equals(o.engine)))
516 			        && (name == o.name || (name != null && name.equals(o.name)));
517 		}
518 
519 		@Override
520 		public String toString() {
521 			StringBuilder s = new StringBuilder();
522 			s.append("{");
523 			if (engine != null)
524 				s.append("engine: " + engine);
525 			if (name != null)
526 				s.append(", name: " + name);
527 			if (primaryLang != null)
528 				s.append(", primaryLang: " + primaryLang);
529 			if (gender != null)
530 				s.append(", gender: " + gender);
