1   package org.daisy.pipeline.tts;
2   
3   import java.util.Collection;
4   import java.util.List;
5   
6   import javax.sound.sampled.AudioFormat;
7   
8   import net.sf.saxon.s9api.XdmNode;
9   
10  import org.daisy.pipeline.audio.AudioBuffer;
11  import org.daisy.pipeline.tts.AudioBufferAllocator.MemoryException;
12  import org.daisy.pipeline.tts.TTSRegistry.TTSResource;
13  import org.daisy.pipeline.tts.TTSService.Mark;
14  import org.daisy.pipeline.tts.TTSService.SynthesisException;
15  
16  /**
17   * Classes that inherit from TTSEngine are the ones that deal with adapting
18   * external TTS processors (e.g. eSpeak and SAPI) to the Pipeline interface,
19   * which is used notably by the TextToPcm threads in the conversion from SSML to
20   * mp3. TTSEngines are meant to be allocated for every new session so that the
21   * end user can change the TTSEngine's parameters between two Pipeline jobs.
22   * Most of the methods of TTSEngine must be thread-safe.
23   */
24  public abstract class TTSEngine {
25  
26  	/**
27  	 * @param provider is the service from which the TTSEngine has been
28  	 *            allocated.
29  	 */
30  	protected TTSEngine(TTSService provider) {
31  		mProvider = provider;
32  	}
33  
34  	/**
35  	 * @return the service from which the TTSEngine has been allocated.
36  	 */
37  	public TTSService getProvider() {
38  		return mProvider;
39  	}
40  
41  	protected TTSService mProvider;
42  
43  	/**
44  	 * This method must be thread-safe. But @param threadResources is here to
45  	 * prevent the service from locking internal resources.
46  	 * 
47  	 * @param sentence is the sentence to synthesize. It is the serialized
48  	 *            result of the SSML conversion performed by the XSLT whose URL
49  	 *            is returned by TTSService.getSSMLxslTransformerURL().
50  	 * @param xmlSentence is the XML version of @param sentence. It should not
51  	 *            be used unless one needs information that has been lost during
52  	 *            the serialization process (e.g. marks or CSS)
53  	 * @param voice is the voice the synthesizer must use. It is guaranteed to
54  	 *            be one of those returned by getAvailableVoices(). This
55  	 *            parameter can't be null.
56  	 * @param threadResources is the object returned by
57  	 *            allocateThreadResource(). It may contain small persistent
58  	 *            buffers, opened file streams, TCP connections and so on. The
59  	 *            boolean field 'released' is guaranteed to be false, i.e. the
60  	 *            resource provided is always valid and will remain so during
61  	 *            the call.
62  	 * @param marks are the returned pairs (markName, offset-in-output)
63  	 *            corresponding to the ssml:marks of @param sentence. The order
64  	 *            must be kept. The provided list is always empty. The
65  	 *            offset-in-outputs are relative to the new buffers returned by
66  	 *            synthesize(). That is, they start at 0. If the service doesn't
67  	 *            handle SSML marks (i.e. endingmark() returns null). This
68  	 *            parameter may be set to null.
69  	 * @param expectedMarks are the mark names extracted from @param xmlSentence. The
70  	 * 	          TTS engine can use them if it doesn't play well with complicated/long
71  	 *            mark names, but it will harder to detect errors. It does include
72  	 *            the TTS provider's ending mark. The ending-mark can be set to anything if
73  	 *            the TTS processor or @param voice cannot handle marks. 
74  	 * @param bufferAllocator is the object that the TTS Service must use to
75  	 *            allocate new audio buffers.
76  	 * @param retry is true when this is not the first time the thread attempts
77  	 *            to synthesize @param sentence. In such cases, you may
78  	 *            reinitialize the connection to remote TTS servers if there are
79  	 *            any.
80  	 * 
81  	 * 
82  	 * @return a list of adjacent PCM chunks produced by the TTS processor.
83  	 */
84  	abstract public Collection<AudioBuffer> synthesize(String sentence, XdmNode xmlSentence,
85  	        Voice voice, TTSResource threadResources, List<Mark> marks, List<String> expectedMarks,
86  	        AudioBufferAllocator bufferAllocator, boolean retry) throws SynthesisException,
87  	        InterruptedException, MemoryException;
88  
89  	/**
90  	 * @return the audio format (sample rate etc.) of the data produced by
91  	 *         synthesize(). The engine is assumed to use the same audio format
92  	 *         every time. It is okay to return null before the first call to
93  	 *         synthesize(), though it is better to return non-null values for
94  	 *         optimization purposes. It must, however, return a non-null value
95  	 *         once synthesize() has been called. Must be thread-safe.
96  	 */
97  	abstract public AudioFormat getAudioOutputFormat();
98  
99  	/**
100 	 * Need not be thread-safe. This method is called from the main thread.
101 	 */
102 	abstract public Collection<Voice> getAvailableVoices() throws SynthesisException,
103 	        InterruptedException;
104 
105 	/**
106 	 * This method must be thread-safe. It allocates new resources (such as TCP
107 	 * connections) unique for each thread. Allocations can be made on-the-fly
108 	 * from different threads. It must not catch InterruptuedExceptions.
109 	 * 
110 	 * @return the resources. Must not be null.
111 	 * @throws SynthesisException
112 	 */
113 	public TTSResource allocateThreadResources() throws SynthesisException,
114 	        InterruptedException {
115 		return new TTSResource();
116 	}
117 
118 	/**
119 	 * This method must be thread-safe. Deallocations may be performed from
120 	 * different threads but are always performed in the same thread as the one
121 	 * exploiting @param resources.
122 	 * 
123 	 * @param resources is the object returned by allocateThreadResource()
124 	 */
125 	public void releaseThreadResources(TTSResource resources) throws SynthesisException,
126 	        InterruptedException {
127 	}
128 
129 	/**
130 	 * Force interruption of the execution of synthesize() when the thread-level
131 	 * interruption is not enough to make synthesize() finish. Must be
132 	 * thread-safe, although the method must not wait for locks.
133 	 * 
134 	 * @param resource is the same as the one provided to synthesize()
135 	 */
136 	public void interruptCurrentWork(TTSResource resource) {
137 	}
138 
139 	/**
140 	 * Need not be thread-safe. This method is called from the main thread.
141 	 */
142 	public int getOverallPriority() {
143 		return 1;
144 	}
145 
146 	/**
147 	 * @return the number of text-to-pcm threads reserved for this TTS
148 	 *         processor. Different values than zero make only sense if the TTS
149 	 *         speed is limited by the number of opened resources rather than by
150 	 *         the number of cores. In such cases, we want to maximize the time
151 	 *         spent on using the TTS resources by avoiding using threads for
152 	 *         something else (e.g. audio encoding or calling other TTS
153 	 *         processors).
154 	 */
155 	public int reservedThreadNum() {
156 		return 0;
157 	}
158 
159 	/**
160 	 * @return the average number of milliseconds the TTS processors is expected
161 	 *         to spend to process a single word on a single CPU core. It
162 	 *         doesn't need to be accurate. It is used for tuning the timeouts.
163 	 */
164 	public int expectedMillisecPerWord() {
165 		return 100;
166 	}
167 
168 	/**
169 	 * @return the name of the mark that will be added to check whether all the
170 	 *         SSML have been successfully synthesized. TTS processors that
171 	 *         cannot handle marks must return null. Must be thread-safe.
172 	 */
173 	public String endingMark() {
174 		return null; //marks not handled
175 	}
176 }
