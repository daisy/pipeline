17  public class VoiceInfo {
18  	private static Pattern localePattern = Pattern
19  	        .compile("(\\p{Alpha}{2})(?:[-_](\\p{Alpha}{2}))?(?:[-_](\\p{Alnum}{1,8}))*");
20  
21  	public enum Gender {
22  		MALE_ADULT("male", "man", "male-adult"),
23  		MALE_CHILD("boy", "male-young", "male-child"),
24  		MALE_ELDERY("man-old", "male-old", "male-elder", "man-elder"),
25  		FEMALE_CHILD("girl", "female-young", "female-child"),
26  		FEMALE_ADULT("woman", "female", "female-adult"),
27  		FEMALE_ELDERY("woman-old", "female-old", "woman-elder", "female-elder");
28  
29  		private final List<String> variants;
30  
31  		private Gender(String... variants) {
32  			this.variants = Lists.newArrayList(variants);
33  		}
34  
35  		private static final Map<String, Gender> lookup = new HashMap<String, Gender>();
36  
37  		static {
38  			Splitter splitter = Splitter.on('-').omitEmptyStrings();
39  			for (Gender gender : EnumSet.allOf(Gender.class)) {
40  				for (String variant : gender.variants) {
41  					lookup.put(variant, gender);
42  					List<String> parts = splitter.splitToList(variant);
43  					if (parts.size() > 1) {
44  						lookup.put(parts.get(0) + parts.get(1), gender);
45  						lookup.put(parts.get(1) + parts.get(0), gender);
46  						lookup.put(parts.get(0) + '-' + parts.get(1), gender);
47  						lookup.put(parts.get(1) + '-' + parts.get(0), gender);
48  						lookup.put(parts.get(0) + '_' + parts.get(1), gender);
49  						lookup.put(parts.get(1) + '_' + parts.get(0), gender);
50  					}
51  				}
52  			}
53  		}
54  
55  		public static Gender of(String gender) {
56  			return lookup.get(gender);
57  		}
58  
59  	}
60  	
61  	public static class UnknownLanguage extends Exception{
62  		public UnknownLanguage(String message) {
63  	        super(message);
64  	    }
65  	}
66  
67  	public static Locale tagToLocale(String langtag) throws UnknownLanguage {
68  		if ("*".equals(langtag) || langtag == null)
69  			return NO_DEFINITE_LANG;
70  		
71  		//TODO: in Java7 we would use:
72  		//return Locale.forLanguageTag(lang)
73  		//=> this works with BCP47 tags, and should work with old tags from RFC 3066
74  		//TODO: use a common function for pipeline-mod-nlp and pipeline-mod-tts
75  		Locale locale = null;
76  		if (langtag != null) {
77  			Matcher m = localePattern.matcher(langtag.toLowerCase());
78  			if (m.matches()) {
79  				locale = new Locale(m.group(1), m.group(2) != null ? m.group(2) : "");
80  			}
81  		}
82  		
83  		if (locale == null)
84  			throw new UnknownLanguage(langtag);
85  		
86  		return locale;
87  	}

89  	public VoiceInfo(String voiceEngine, String voiceName, String language, Gender gender,
90  	        float priority) throws UnknownLanguage {
91  		this(voiceEngine, voiceName, MarkSupport.DEFAULT, language, gender, priority);
92  	}
93  
94  	public VoiceInfo(String voiceEngine, String voiceName, MarkSupport markSupport,
95  	        String language, Gender gender, float priority) throws UnknownLanguage {
96  		this(new Voice(voiceEngine, voiceName, markSupport), language, gender, priority);
97  	}
98  
99  	public VoiceInfo(Voice v, String language, Gender gender, float priority) throws UnknownLanguage {
100 		this(v, tagToLocale(language), gender, priority);
101 	}
102 
103 	public VoiceInfo(Voice v, Locale language, Gender gender) {
104 		this(v, language, gender, -1);
105 	}
106 
107 	private VoiceInfo(Voice v, Locale locale, Gender gender, float priority) {
108 		Preconditions.checkNotNull(v);
109 		Preconditions.checkNotNull(gender);
110 		this.voice = v;
111 		this.language = locale;
112 		this.priority = priority;
113 		this.gender = gender;
114 	}
115 
116 	@Override
117 	public int hashCode() {
118 		return this.voice.hashCode() ^ (this.language == null ? 0 : this.language.hashCode());
119 	}
120 
121 	public boolean equals(Object other) {
122 		VoiceInfo o = (VoiceInfo) other;
123 		return voice.equals(o.voice) && ((language == null && o.language == null) ||
124 				(language != null && language.equals(o.language)));
125 	}
126 	
127 	public boolean isMultiLang(){
128 		return this.language == NO_DEFINITE_LANG;
129 	}
130 
131 	public Gender gender;
132 	public Voice voice;
133 	public Locale language;
134 	public float priority;
135 	
136 	public static Locale NO_DEFINITE_LANG = null;
137 }
