51  /**
52   * SSMLtoAudio splits the input SSML sentences into sections called
53   * ContiguousText. Every section contains a list of contiguous sentences that
54   * may be eventually stored into different audio files, but it is guaranteed
55   * that such audio files won't contain sentences of other sections in such a
56   * manner that the list of audio files will mirror the document order. Within
57   * a section, every sentence has the same single sample rate to prevent us from
58   * resampling the audio data before sending them to encoders.
59   * 
60   * Once all the sentences have been assigned to TTS voices, they are sorted by
61   * size and stored in a shared queue of ContiguousText (actually, only the
62   * smallest sentences are sorted that way). SSMLtoAudio creates threads to
63   * consume this queue.
64   * 
65   * The TextToPcmThreads send PCM data to EncodingThreads via a queue of
66   * ContiguousPCM. These PCM packets are then processed from the longest (with
67   * respect to the number of samples) to the shortest in order to make it likely
68   * that the threads will finish at the same time. When all the TextToPcmThreads
69   * are joined, the pipeline pushes an EndOfQueue marker to every EncodingThreads
70   * to notify that they must stop waiting for more PCM packets than the ones
71   * already pushed.
72   * 
73   * The queue of PCM chunks, along with the queues of other TTS steps, share a
74   * global max size so as to make sure that they won't grow too much if the
75   * encoding is slower than the synthesizing (see AudioFootprintMonitor).
76   * 
77   * If an EncodingThread fails to encode samples, it won't set the URI attribute
78   * of the audio chunks. In that way, SSMLtoAudio is informed not to include the
79   * corresponding text into the list of audio clips.
80   * 
81   */
82  public class SSMLtoAudio implements FormatSpecifications {

157 		XdmSequenceIterator iter = doc.axisIterator(Axis.CHILD);
158 		if (iter.hasNext()) {

164 	private void traverse(XdmNode node, Locale lang) throws SynthesisException {
165 		if (SentenceTag.equals(node.getNodeName())) {
166 			if (!dispatchSSML(node, lang))
167 				mErrorCounter++;
168 			if (mMaxSentencesPerSection > 0 && ++mSentenceCounter % mMaxSentencesPerSection == 0)
169 				endSection();
170 		} else {
171 			String langAttr = node.getAttributeValue(Sentence_attr_lang);
172 			if (langAttr != null) {
173 				try {
174 					lang = (new Locale.Builder()).setLanguageTag(langAttr).build();
175 				} catch (IllformedLocaleException e) {
176 					lang = null;
177 				}
178 			}
179 			XdmSequenceIterator iter = node.axisIterator(Axis.CHILD);
180 			while (iter.hasNext())
181 				traverse((XdmNode)iter.next(), lang);
182 		}
183 	}

186 	 * The SSML is assumed to be pushed in document order.
187 	 *
188 	 * @param ssml The input SSML
189 	 * @param lang The parent language
190 	 * @return true when the SSML was successfully converted to speech, false when there was an error
191 	 **/
192 	// package private for tests
193 	boolean dispatchSSML(XdmNode ssml, Locale lang) throws SynthesisException {
194 		String voiceEngine = ssml.getAttributeValue(Sentence_attr_select1);
195 		String voiceName = ssml.getAttributeValue(Sentence_attr_select2);
196 		Gender gender; {
197 			String attr = ssml.getAttributeValue(Sentence_attr_gender);
198 			String ageAttr = ssml.getAttributeValue(Sentence_attr_age);
199 			if (attr != null && ageAttr != null) {
200 				try {
201 					int age = Integer.parseInt(ageAttr);
202 					if (age <= 16) {
203 						gender = Gender.of(attr + "-child");
204 					} else if (age >= 70) {
205 						gender = Gender.of(attr + "-eldery");
206 					} else {
207 						gender = Gender.of(attr);
208 					}
209 				} catch (NumberFormatException e) {
210 					gender = Gender.of(attr);
211 				}
212 			} else {
213 				gender = Gender.of(attr);
214 			}
215 		}
216 		{
217 			String langAttr = ssml.getAttributeValue(Sentence_attr_lang);
218 			if (langAttr != null) {
219 				try {
220 					lang = (new Locale.Builder()).setLanguageTag(langAttr).build();
221 				} catch (IllformedLocaleException e) {
222 					lang = null;
223 				}
224 			}
225 		}
226 		String id = ssml.getAttributeValue(Sentence_attr_id);
227 
228 		TTSLog.Entry logEntry = mTTSlog.getOrCreateEntry(id);
229 		logEntry.setSSML(ssml);
230 		Iterable<Voice> voices = mVoiceManager.findAvailableVoices(voiceEngine, voiceName, lang, gender);
231 		Voice preferredVoice = Iterables.getFirst(voices, null);
232 		logEntry.setSelectedVoice(preferredVoice);
233 		if (preferredVoice == null) {
234 			String err = "could not find any installed voice matching with "
235 			                + new Voice(voiceEngine, voiceName)
236 			                + " or providing the language '" + lang + "'";
237 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, err));
238 			return false;
239 		}
240 
241 		TTSEngine newSynth = mVoiceManager.getTTS(preferredVoice);
242 		if (newSynth == null) {
243 			/*
244 			 * Should not happen since findAvailableVoice() returns only a
245 			 * non-null voice if a TTSService can provide it
246 			 */
247 			String err = "could not find any TTS engine for the voice "
248 				+ new Voice(voiceEngine, voiceName);
249 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, err));
250 			return false;
251 		}
252 
253 		if (!mVoiceManager.matches(preferredVoice, voiceEngine, voiceName, lang, gender)) {
254 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.UNEXPECTED_VOICE,
255 			        "no voice matches exactly with the requested characteristics"));
256 		}
257 
258 		/*
259 		 * If a TTS engine has no reserved threads, its sentences are pushed to
260 		 * a global list whose key is null. Otherwise the TTS Service is used as
261 		 * key.
262 		 */
263 		TTSEngine poolkey = null;
264 		if (newSynth.reservedThreadNum() > 0) {
265 			poolkey = newSynth;
266 		}
267 
268 		if (newSynth != mLastTTS) {
269 			if (mLastTTS != null
270 			        && (poolkey != null || mLastTTS.reservedThreadNum() > 0))
271 				endSection(); // necessary because the same thread wouldn't be able to
272 				              // concatenate outputs of different formats
273 			mLastTTS = newSynth;
274 		}
275 
276 		if (mCurrentSection == null) {
277 			//happen the first time and whenever endSection() is called
278 			mCurrentSection = new ContiguousText(mDocumentPosition++, mAudioDir);
279 			List<ContiguousText> listOfSections = mOrganizedText.get(poolkey);
280 			if (listOfSections == null) {
281 				listOfSections = new ArrayList<ContiguousText>();
282 				mOrganizedText.put(poolkey, listOfSections);
283 			}
284 
285 			listOfSections.add(mCurrentSection);
286 		}
287 		mCurrentSection.sentences.add(new Sentence(newSynth, voices, ssml));
288 		return true;
289 	}
290 
291 	private void endSection() {
292 		mCurrentSection = null;
293 	}

295 	public Iterable<SoundFileLink> blockingRun(AudioServices audioServices)
296 	        throws SynthesisException, InterruptedException, EncodingException {
297 
298 		MessageAppender activeBlock = MessageAppender.getActiveBlock(); // px:ssml-to-audio step
299 
300 		//SSML mark splitter shared by the threads:
301 		SSMLMarkSplitter ssmlSplitter = new DefaultSSMLMarkSplitter(mProc); // StructuredSSMLSplitter makes assumptions
302 		                                                                    // that are not always met, e.g. marks may
303 		                                                                    // be contained within a a prosody element
304 
305 		reorganizeSections();
306 
307 		//threading layout
308 		int reservedThreadNum = 0;
309 		for (TTSEngine tts : mOrganizedText.keySet()) {
310 			if (tts != null && tts.reservedThreadNum() > 0)
311 				reservedThreadNum += tts.reservedThreadNum();
312 		}
313 		int ttsThreadNum = numberOfCores;
314 		int encodingThreadNum = getPropertyAsInt(ENCODING_THREADS_NUMBER).orElse(ttsThreadNum);
315 		int regularTTSthreadNum = getPropertyAsInt(SPEAKING_THREADS_NUMBER).orElse(ttsThreadNum);
316 		int totalTTSThreads = regularTTSthreadNum + reservedThreadNum;
317 		int maxMemPerTTSThread = getPropertyAsInt(THREADS_MEMLIMIT).get() * 1048576;
318 		mLogger.info("Number of encoding threads: " + encodingThreadNum);
319 		mLogger.info("Number of regular text-to-speech threads: " + regularTTSthreadNum);
320 		mLogger.info("Number of reserved text-to-speech threads: " + reservedThreadNum);
321 		mLogger.info("Max TTS memory footprint (encoding excluded): "
322 		        + mAudioFootprintMonitor.getSpaceForTTS() / 1000000 + "MB");
323 		mLogger.info("Max encoding memory footprint: "
324 		        + mAudioFootprintMonitor.getSpaceForEncoding() / 1000000 + "MB");
325 
326 		//input queue common to all the threads
327 		BlockingQueue<ContiguousPCM> pcmQueue = new PriorityBlockingQueue<ContiguousPCM>();
328 
329 		//start the TTS threads
330 		TextToPcmThread[] tpt = new TextToPcmThread[totalTTSThreads];
331 		List<ContiguousText> text = mOrganizedText.get(null);
332 		if (text == null) {
333 			text = Collections.EMPTY_LIST;
334 		}
335 		int i = 0;
336 		ConcurrentLinkedQueue<ContiguousText> stext = new ConcurrentLinkedQueue<ContiguousText>(text);
337 		if (regularTTSthreadNum > 0) {
338 			long textSize = stext.stream().mapToLong(ContiguousText::getStringSize).sum();
339 			BigDecimal portion = textSize == 0
340 				? BigDecimal.ZERO
341 				: new BigDecimal(textSize).divide(new BigDecimal(mTotalTextSize), MathContext.DECIMAL128)
342 				                          .divide(new BigDecimal(regularTTSthreadNum), MathContext.DECIMAL128);
343 			for (; i < regularTTSthreadNum; ++i) {
344 				tpt[i] = new TextToPcmThread();
345 				tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter,
346 				             mLogger, mAudioFootprintMonitor, maxMemPerTTSThread, mTTSlog, activeBlock,
347 				             mTotalTextSize, portion);
348 			}
349 		}
350 		for (Map.Entry<TTSEngine, List<ContiguousText>> e : mOrganizedText.entrySet()) {
351 			TTSEngine tts = e.getKey();
352 			if (tts != null) { //tts = null is handled by the previous loop
353 				stext = new ConcurrentLinkedQueue<ContiguousText>(e.getValue());
354 				if (tts.reservedThreadNum() > 0) {
355 					long textSize = stext.stream().mapToLong(ContiguousText::getStringSize).sum();
356 					BigDecimal portion = textSize == 0
357 						? BigDecimal.ZERO
358 						: new BigDecimal(textSize).divide(new BigDecimal(mTotalTextSize), MathContext.DECIMAL128)
359 						                          .divide(new BigDecimal(tts.reservedThreadNum()), MathContext.DECIMAL128);
360 					for (int j = 0; j < tts.reservedThreadNum(); ++i, ++j) {
361 						tpt[i] = new TextToPcmThread();
362 						tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter,
363 						             mLogger, mAudioFootprintMonitor, maxMemPerTTSThread, mTTSlog,
364 						             activeBlock, mTotalTextSize, portion);
365 					}
366 				}
367 			}
368 		}
369 		mLogger.info("Text-to-speech threads started.");
370 
371 		//start the encoding threads
372 		EncodingThread[] encodingTh = new EncodingThread[encodingThreadNum];
373 		for (int j = 0; j < encodingTh.length; ++j) {
374 			encodingTh[j] = new EncodingThread();
375 			encodingTh[j].start(mAudioFileFormat, audioServices, pcmQueue,
376 			                    mAudioFootprintMonitor, mProperties, mTTSlog, activeBlock);
377 		}
378 		mLogger.info("Encoding threads started.");
379 
380 		//collect the sound fragments
381 		Collection<SoundFileLink>[] fragments = new Collection[tpt.length];
382 		for (int j = 0; j < tpt.length; ++j) {
383 			fragments[j] = tpt[j].getSoundFragments();
384 		}
385 		for (int j = 0; j < tpt.length; ++j) {
386 			mErrorCounter += tpt[j].getErrorCount();
387 		}
388 
389 		//send END notifications and wait for the encoding threads to finish
390 		mLogger.info("Text-to-speech finished. Waiting for audio encoding to finish...");
391 		for (int k = 0; k < encodingTh.length; ++k) {
392 			pcmQueue.add(ContiguousPCM.EndOfQueue);
393 		}
394 		for (int j = 0; j < encodingTh.length; ++j) {
395 			try {
396 				encodingTh[j].waitToFinish();
397 			} catch (EncodingException e) {
398 				while (++j < encodingTh.length)
399 					// FIXME: interrupt instead of waiting
400 					try { encodingTh[j].waitToFinish(); }
401 					catch (EncodingException _e) {}
402 				throw e;
403 			}
404 		}
405 
406 		mLogger.info("Audio encoding finished.");
407 
408 		return Iterables.concat(fragments);
409 	}

415 	private void reorganizeSections() {
416 		mTotalTextSize = 0;
417 		int sectionCount = 0;
418 		for (List<ContiguousText> sections : mOrganizedText.values()) {
419 
420 			// compute the sections' size: needed for displaying the progress,
421 			// splitting the sections and sorting them
422 			for (ContiguousText section : sections) {
423 				section.computeSize();
424 				mTotalTextSize += section.getStringSize();
425 			}
426 
427 			// split up the sections that are too big
428 			// FIXME: Explain what is "too big".
429 			// FIXME: Code was disabled for now because it does not make sense to let the maximum
430 			// size depend on the total size (and mTotalTextSize isn't even constant).
431 			/*int maxSize = (int) (mTotalTextSize / 15);
432 			List<ContiguousText> newSections = new ArrayList<ContiguousText>();
433 			List<ContiguousText> toRemove = new ArrayList<ContiguousText>();
434 			for (ContiguousText section : sections) {
435 				if (section.getStringSize() >= maxSize) {
436 					toRemove.add(section);
437 					splitSection(section, maxSize, newSections);
438 				}
439 			}
440 			sections.removeAll(toRemove);
441 			sections.addAll(newSections);*/
442 
443 			// sort the sections according to their size in descending order
444 			// FIXME: Explain why this is done.
445 			Collections.sort(sections);
446 
447 			// keep sorted only the smallest sections (50% of total) so the biggest sections won't
448 			// necessarily be processed at the same time, as that may consume too much memory.
449 			Collections.shuffle(sections.subList(0, sections.size() / 2));
450 
451 			sectionCount += sections.size();
452 		}
453 		mLogger.debug("Number of synthesizable TTS sections: " + sectionCount);
454 	}

456 	//we can dispense with this function as soon as we take into consideration the size of the SSML
457 	//sentences, rather than creating a new section every 10 sentences or so.
458 	private void splitSection(ContiguousText section, int maxSize,
459 	        List<ContiguousText> newSections) {
460 		int left = 0;
461 		int count = 0;
462 		ContiguousText currentSection = new ContiguousText(section.getDocumentPosition(),
463 		        section.getAudioOutputDir());
464 		currentSection.setStringsize(0);
465 		for (int right = 0; right < section.sentences.size(); ++right) {
466 			if (currentSection.getStringSize() > maxSize) {
467 				currentSection.sentences = section.sentences.subList(left, right);
468 				currentSection.setDocumentSplitPosition(count);
469 				newSections.add(currentSection);
470 				currentSection = new ContiguousText(section.getDocumentPosition(), section
471 				        .getAudioOutputDir());
472 				currentSection.setStringsize(0);
473 				left = right;
474 				++count;
475 			}
476 			currentSection.setStringsize(currentSection.getStringSize()
477 			        + section.sentences.get(right).getSize());
478 		}
479 		currentSection.sentences = section.sentences.subList(left, section.sentences.size());
480 		currentSection.setDocumentSplitPosition(count);
481 		newSections.add(currentSection);
482 	}

484 	private Optional<Integer> getPropertyAsInt(Property prop) {
485 		String str = prop.getValue(mProperties);
486 		if (str != null) {
487 			try {
488 				return Optional.of(Integer.valueOf(str));
489 			} catch (NumberFormatException e) {
490 				mTTSlog.addGeneralError(
491 					ErrorCode.WARNING, str + " is not a valid value for property " + prop.getName(), e);
492 			}
493 		}
494 		return Optional.empty();
495 	}
