55  /**
56   * SSMLtoAudio splits the input SSML sentences into sections called
57   * ContiguousText. Every section contains a list of contiguous sentences that
58   * may be eventually stored into different audio files, but it is guaranteed
59   * that such audio files won't contain sentences of other sections in such a
60   * manner that the list of audio files will mirror the document order. Within
61   * a section, every sentence has the same single sample rate to prevent us from
62   * resampling the audio data before sending them to encoders.
63   * 
64   * Once all the sentences have been assigned to TTS voices, they are sorted by
65   * size and stored in a shared queue of ContiguousText (actually, only the
66   * smallest sentences are sorted that way). SSMLtoAudio creates threads to
67   * consume this queue.
68   * 
69   * The TextToPcmThreads send PCM data to EncodingThreads via a queue of
70   * ContiguousPCM. These PCM packets are then processed from the longest (with
71   * respect to the number of samples) to the shortest in order to make it likely
72   * that the threads will finish at the same time. When all the TextToPcmThreads
73   * are joined, the pipeline pushes an EndOfQueue marker to every EncodingThreads
74   * to notify that they must stop waiting for more PCM packets than the ones
75   * already pushed.
76   * 
77   * The queue of PCM chunks, along with the queues of other TTS steps, share a
78   * global max size so as to make sure that they won't grow too much if the
79   * encoding is slower than the synthesizing (see AudioFootprintMonitor).
80   * 
81   * If an EncodingThread fails to encode samples, it won't set the URI attribute
82   * of the audio chunks. In that way, SSMLtoAudio is informed not to include the
83   * corresponding text into the list of audio clips.
84   * 
85   */

195 	 * @throws Throwable if an engine could not be created or when the test failed. The exception is

199 	private TTSEngine createAndTestEngine(final TTSService service, Map<String, String> properties,
200 	                                      XdmNode testingSSMLWithoutMark, XdmNode testingSSMLWithMark,
201 	                                      TTSTimeout timeout)
202 			throws Throwable {
203 
204 		//create the engine
205 		TTSEngine engine = null;
206 		timeout.enableForCurrentThread(2);
207 		try {
208 			engine = service.newEngine(properties);
209 		} finally {
210 			timeout.disable();
211 		}
212 
213 		//get a voice supporting SSML marks (so far as they are supported by the engine)
214 		Voice firstVoice = null;
215 		int timeoutSecs = 30;
216 		timeout.enableForCurrentThread(timeoutSecs);
217 		try {
218 			for (Voice v : engine.getAvailableVoices()) {
219 				if (!engine.handlesMarks() || v.getMarkSupport() != MarkSupport.MARK_NOT_SUPPORTED) {
220 					firstVoice = v;
221 					break;
222 				}
223 			}
224 			if (firstVoice == null) {
225 				throw new Exception("no voices available");
226 			}
227 		} catch (InterruptedException e) {
228 			throw new Exception("timeout while retrieving voices (exceeded "
229 			                    + timeoutSecs + " seconds)");
230 		} catch (Exception e) {
231 			throw new Exception("failed to retreive voices: " + e.getMessage(), e);
232 		} finally {
233 			timeout.disable();
234 		}
235 
236 		//allocate resources for testing purpose
237 		final TTSEngine fengine = engine;
238 		TTSResource resource = null;
239 		timeout.enableForCurrentThread(2);
240 		try {
241 			resource = engine.allocateThreadResources();
242 		} catch (Exception e) {
243 			throw new Exception("could not allocate resources: " + e.getMessage(), e);
244 		} finally {
245 			timeout.disable();
246 		}
247 
248 		//create a custom interrupter in case the engine hangs
249 		final TTSResource res = resource;
250 		TTSTimeout.ThreadFreeInterrupter interrupter = new ThreadFreeInterrupter() {
251 			@Override
252 			public void threadFreeInterrupt() {
253 				String msg = "Timeout while initializing "
254 				        + service.getName()
255 				        + ". Forcing interruption of the current work of "
256 				        + service.getName() + "... ";
257 				mLogger.warn(msg);
258 				mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
259 				fengine.interruptCurrentWork(res);
260 			}
261 		};
262 
263 		//run the text-to-speech on the testing input
264 		SynthesisResult result = null;
265 		try {
266 			XdmNode ssml = engine.handlesMarks() ? testingSSMLWithMark : testingSSMLWithoutMark;
267 			result = mExecutor.synthesizeWithTimeout(
268 				timeout, interrupter, null, ssml, Sentence.computeSize(ssml),
269 				engine, firstVoice, res);
270 		} catch (Exception e) {
271 			throw new Exception("test failed: " + e.getMessage(), e);
272 		} finally {
273 			if (res != null)
274 				timeout.enableForCurrentThread(2);
275 				try {
276 					engine.releaseThreadResources(res);
277 				} catch (Exception e) {
278 					String msg = "Error while releasing resource of "
279 					        + service.getName() + "; " + e.getMessage();
280 					String stack = getStack(e);
281 					mLogger.warn(msg);
282 					mLogger.debug(stack);
283 					mTTSlog.addGeneralError(ErrorCode.WARNING, msg + ": " + stack);
284 				} finally {
285 					timeout.disable();
286 				}
287 		}
288 
289 		//check that the output buffer is big enough
290 		String msg = "";
291 		if (result.audio.getFrameLength() * result.audio.getFormat().getFrameSize() < 2500) {
292 			msg = "Audio output is not big enough. ";
293 		}
294 
295 		if (engine.handlesMarks()) {
296 			// test SSML contains one mark
297 			String details = " voice: "+firstVoice;
298 			if (result.marks.size() != 1) {
299 				msg += "One bookmark events expected, but received " + result.marks.size() + " events instead. "+details;
300 			} else {
301 				int offset = result.marks.get(0);
302 				if (offset < 2500) {
303 					msg += "Expecting mark offset to be bigger, got "
304 					        + offset + " as offset. "+details;
305 				}
306 			}
307 		}
308 		if (!msg.isEmpty()) {
309 			throw new Exception("test failed: " + msg);
310 		}
311 
312 		return engine;
313 	}

315 	/**
316 	 * The SSML is assumed to be pushed in document order.
317 	 *
318 	 * @param ssml The input SSML
319 	 * @return true when the SSML was successfully converted to speech, false when there was an error
320 	 **/
321 	public boolean dispatchSSML(XdmNode ssml) throws SynthesisException {
322 		String voiceEngine = ssml.getAttributeValue(Sentence_attr_select1);
323 		String voiceName = ssml.getAttributeValue(Sentence_attr_select2);
324 		String gender = ssml.getAttributeValue(Sentence_attr_gender);
325 		String age = ssml.getAttributeValue(Sentence_attr_age);
326 		String id = ssml.getAttributeValue(Sentence_attr_id);
327 		String lang = ssml.getAttributeValue(Sentence_attr_lang);
328 
329 		TTSLog.Entry logEntry = mTTSlog.getOrCreateEntry(id);
330 		logEntry.setSSML(ssml);
331 
332 		if (age != null) {
333 			try {
334 				int age_i = Integer.parseInt(age);
335 				if (age_i <= 16) {
336 					gender += "-child";
337 				} else if (age_i >= 70) {
338 					gender += "-eldery";
339 				}
340 			} catch (NumberFormatException e) {
341 				//ignore
342 			}
343 		}
344 
345 		boolean[] exactMatch = new boolean[1];
346 		Voice voice = mVoiceManager.findAvailableVoice(voiceEngine, voiceName, lang, gender,
347 		        exactMatch);
348 		logEntry.setSelectedVoice(voice);
349 		if (voice == null) {
350 			String err = "could not find any installed voice matching with "
351 			                + new Voice(voiceEngine, voiceName)
352 			                + " or providing the language '" + lang + "'";
353 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, err));
354 			endSection();
355 			mLogger.warn(err);
356 			return false;
357 		}
358 
359 		TTSEngine newSynth = mVoiceManager.getTTS(voice);
360 		if (newSynth == null) {
361 			/*
362 			 * Should not happen since findAvailableVoice() returns only a
363 			 * non-null voice if a TTSService can provide it
364 			 */
365 			String err = "could not find any TTS engine for the voice "
366 				+ new Voice(voiceEngine, voiceName);
367 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, err));
368 			endSection();
369 			mLogger.warn(err);
370 			return false;
371 		}
372 
373 		if (!exactMatch[0]) {
374 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.UNEXPECTED_VOICE,
375 			        "no voice matches exactly with the requested characteristics"));
376 		}
377 
378 		/*
379 		 * If a TTS engine has no reserved threads, its sentences are pushed to
380 		 * a global list whose key is null. Otherwise the TTS Service is used as
381 		 * key.
382 		 */
383 		TTSEngine poolkey = null;
384 		if (newSynth.reservedThreadNum() > 0) {
385 			poolkey = newSynth;
386 		}
387 
388 		if (newSynth != mLastTTS) {
389 			if (mLastTTS != null
390 			        && (poolkey != null || mLastTTS.reservedThreadNum() > 0))
391 				endSection(); // necessary because the same thread wouldn't be able to
392 				              // concatenate outputs of different formats
393 			mLastTTS = newSynth;
394 		}
395 
396 		if (mCurrentSection == null) {
397 			//happen the first time and whenever endSection() is called
398 			mCurrentSection = new ContiguousText(mDocumentPosition++, mAudioDir);
399 			List<ContiguousText> listOfSections = mOrganizedText.get(poolkey);
400 			if (listOfSections == null) {
401 				listOfSections = new ArrayList<ContiguousText>();
402 				mOrganizedText.put(poolkey, listOfSections);
403 			}
404 
405 			listOfSections.add(mCurrentSection);
406 		}
407 		mCurrentSection.sentences.add(new Sentence(newSynth, voice, ssml));
408 		return true;
409 	}

417 	public Iterable<SoundFileLink> blockingRun(AudioServices audioServices)
418 	        throws SynthesisException, InterruptedException, EncodingException {
419 
420 		MessageAppender activeBlock = MessageAppender.getActiveBlock();
421 		if (activeBlock != null)
422 			progress = activeBlock.append(new MessageBuilder().withProgress(BigDecimal.ONE));
423 		try {
424 
425 		//SSML mark splitter shared by the threads:
426 		SSMLMarkSplitter ssmlSplitter = new StructuredSSMLSplitter(mProc);
427 
428 		reorganizeSections();
429 		mProgress = 0;
430 		mPrintedProgress = 0;
431 
432 		//threading layout
433 		int reservedThreadNum = 0;
434 		for (TTSEngine tts : mOrganizedText.keySet()) {
435 			if (tts != null && tts.reservedThreadNum() > 0)
436 				reservedThreadNum += tts.reservedThreadNum();
437 		}
438 		int cores = Runtime.getRuntime().availableProcessors();
439 		int ttsThreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.number", cores);
440 		int encodingThreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.encoding.number",
441 		        ttsThreadNum);
442 		int regularTTSthreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.speaking.number",
443 		        ttsThreadNum);
444 		int totalTTSThreads = regularTTSthreadNum + reservedThreadNum;
445 		int maxMemPerTTSThread = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.each.memlimit", 20)*1048576; //20MB
446 		mLogger.info("Number of encoding threads: " + encodingThreadNum);
447 		mLogger.info("Number of regular text-to-speech threads: " + regularTTSthreadNum);
448 		mLogger.info("Number of reserved text-to-speech threads: " + reservedThreadNum);
449 		mLogger.info("Max TTS memory footprint (encoding excluded): "
450 		        + mAudioFootprintMonitor.getSpaceForTTS() / 1000000 + "MB");
451 		mLogger.info("Max encoding memory footprint: "
452 		        + mAudioFootprintMonitor.getSpaceForEncoding() / 1000000 + "MB");
453 
454 		//input queue common to all the threads
455 		BlockingQueue<ContiguousPCM> pcmQueue = new PriorityBlockingQueue<ContiguousPCM>();
456 
457 		//start the TTS threads
458 		TextToPcmThread[] tpt = new TextToPcmThread[totalTTSThreads];
459 		List<ContiguousText> text = mOrganizedText.get(null);
460 		if (text == null) {
461 			text = Collections.EMPTY_LIST;
462 		}
463 		ConcurrentLinkedQueue<ContiguousText> stext = new ConcurrentLinkedQueue<ContiguousText>(
464 		        text);
465 		int i = 0;
466 		for (; i < regularTTSthreadNum; ++i) {
467 			tpt[i] = new TextToPcmThread();
468 			tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter, this,
469 			             mLogger, mAudioFootprintMonitor, maxMemPerTTSThread, mTTSlog);
470 		}
471 		for (Map.Entry<TTSEngine, List<ContiguousText>> e : mOrganizedText.entrySet()) {
472 			TTSEngine tts = e.getKey();
473 			if (tts != null) { //tts = null is handled by the previous loop
474 				stext = new ConcurrentLinkedQueue<ContiguousText>(e.getValue());
475 				for (int j = 0; j < tts.reservedThreadNum(); ++i, ++j) {
476 					tpt[i] = new TextToPcmThread();
477 					tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter,
478 					        this, mLogger, mAudioFootprintMonitor, maxMemPerTTSThread, mTTSlog);
479 				}
480 			}
481 		}
482 		mLogger.info("Text-to-speech threads started.");
483 
484 		//start the encoding threads
485 		EncodingThread[] encodingTh = new EncodingThread[encodingThreadNum];
486 		for (int j = 0; j < encodingTh.length; ++j) {
487 			encodingTh[j] = new EncodingThread();
488 			encodingTh[j].start(mAudioFileFormat, audioServices, pcmQueue, mLogger,
489 					 mAudioFootprintMonitor, mProperties, mTTSlog);
490 		}
491 		mLogger.info("Encoding threads started.");
492 
493 		//collect the sound fragments
494 		Collection<SoundFileLink>[] fragments = new Collection[tpt.length];
495 		for (int j = 0; j < tpt.length; ++j) {
496 			fragments[j] = tpt[j].getSoundFragments();
497 		}
498 		for (int j = 0; j < tpt.length; ++j) {
499 			mErrorCounter += tpt[j].getErrorCount();
500 		}
501 
502 		//send END notifications and wait for the encoding threads to finish
503 		mLogger.info("Text-to-speech finished. Waiting for audio encoding to finish...");
504 		for (int k = 0; k < encodingTh.length; ++k) {
505 			pcmQueue.add(ContiguousPCM.EndOfQueue);
506 		}
507 		for (int j = 0; j < encodingTh.length; ++j) {
508 			try {
509 				encodingTh[j].waitToFinish();
510 			} catch (EncodingException e) {
511 				while (++j < encodingTh.length)
512 					// FIXME: interrupt instead of waiting
513 					try { encodingTh[j].waitToFinish(); }
514 					catch (EncodingException _e) {}
515 				throw e;
516 			}
517 		}
518 
519 		mLogger.info("Audio encoding finished.");
520 
521 		return Iterables.concat(fragments);
522 
523 		} finally {
524 			if (progress != null)
525 				progress.close();
526 			progress = null;
527 		}
528 	}

534 	@Override
535 	synchronized public void notifyFinished(ContiguousText section) {
536 		MessageBuilder m = new MessageBuilder()
537 			.withProgress(
538 				new BigDecimal(section.getStringSize()).divide(new BigDecimal(mTotalTextSize), MathContext.DECIMAL128));
539 		mProgress += section.getStringSize();
540 		if (mProgress - mPrintedProgress > mTotalTextSize / 15) {
541 			int TTSMem = mAudioFootprintMonitor.getUnreleasedTTSMem() / 1000000;
542 			int EncodeMem = mAudioFootprintMonitor.getUnreleasedEncondingMem() / 1000000;
543 			m = m.withLevel(Level.INFO)
544 			     .withText("progress: " + 100 * mProgress / mTotalTextSize + "%  [TTS: "
545 			               + TTSMem + "MB encoding: " + EncodeMem + "MB]");
546 			mPrintedProgress = mProgress;
547 		}
548 		if (progress != null)
549 			progress.append(m).close();
550 	}

552 	private void reorganizeSections() {
553 		mTotalTextSize = 0;
554 		int sectionCount = 0;
555 		for (List<ContiguousText> sections : mOrganizedText.values()) {
556 			//compute the sections' size: needed for displaying the progress,
557 			//splitting the sections and sorting them
558 			for (ContiguousText section : sections)
559 				section.computeSize();
560 
561 			for (ContiguousText section : sections) {
562 				mTotalTextSize += section.getStringSize();
563 			}
564 			//split up the sections that are too big
565 			int maxSize = (int) (mTotalTextSize / 15); //should it depend on the total size or be an absolute max?
566 			List<ContiguousText> newSections = new ArrayList<ContiguousText>();
567 			List<ContiguousText> toRemove = new ArrayList<ContiguousText>();
568 			for (ContiguousText section : sections) {
569 				if (section.getStringSize() >= maxSize) {
570 					toRemove.add(section);
571 					splitSection(section, maxSize, newSections);
572 				}
573 			}
574 
575 			sections.removeAll(toRemove);
576 			sections.addAll(newSections);
577 
578 			//sort the sections according to their size in descending-order
579 			Collections.sort(sections);
580 
581 			//keep sorted only the smallest sections (50% of total) so the biggest sections won't
582 			//necessarily be processed at the same time, as that may consume too much memory.
583 			Collections.shuffle(sections.subList(0, sections.size() / 2));
584 
585 			sectionCount += sections.size();
586 		}
587 		mLogger.info("Number of synthesizable TTS sections: " + sectionCount);
588 	}

590 	//we can dispense with this function as soon as we take into consideration the size of the SSML
591 	//sentences, rather than creating a new section every 10 sentences or so.
592 	private void splitSection(ContiguousText section, int maxSize,
593 	        List<ContiguousText> newSections) {
594 		int left = 0;
595 		int count = 0;
596 		ContiguousText currentSection = new ContiguousText(section.getDocumentPosition(),
597 		        section.getAudioOutputDir());
598 		currentSection.setStringsize(0);
599 		for (int right = 0; right < section.sentences.size(); ++right) {
600 			if (currentSection.getStringSize() > maxSize) {
601 				currentSection.sentences = section.sentences.subList(left, right);
602 				currentSection.setDocumentSplitPosition(count);
603 				newSections.add(currentSection);
604 				currentSection = new ContiguousText(section.getDocumentPosition(), section
605 				        .getAudioOutputDir());
606 				currentSection.setStringsize(0);
607 				left = right;
608 				++count;
609 			}
610 			currentSection.setStringsize(currentSection.getStringSize()
611 			        + section.sentences.get(right).getSize());
612 		}
613 		currentSection.sentences = section.sentences.subList(left, section.sentences.size());
614 		currentSection.setDocumentSplitPosition(count);
615 		newSections.add(currentSection);
616 	}

618 	private static String getStack(Throwable t) {
619 		StringWriter writer = new StringWriter();
620 		PrintWriter printWriter = new PrintWriter(writer);
621 		t.printStackTrace(printWriter);
622 		printWriter.flush();
623 		return writer.toString();

626 	private int convertToInt(Map<String, String> params, String prop, int defaultVal) {
627 		String str = params.get(prop);
628 		if (str != null) {
629 			try {
630 				defaultVal = Integer.valueOf(str);
631 			} catch (NumberFormatException e) {
632 				String msg = str + " is not a valid value for property " + prop;
633 				mLogger.warn(msg);
634 				mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
635 			}
636 		}
637 		return defaultVal;
638 	}
