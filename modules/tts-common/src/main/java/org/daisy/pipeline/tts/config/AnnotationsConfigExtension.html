1   package org.daisy.pipeline.tts.config;
2   
3   import java.net.URI;
4   import java.util.ArrayList;
5   import java.util.Collection;
6   import java.util.Collections;
7   import java.util.HashMap;
8   import java.util.List;
9   import java.util.Map;
10  
11  import net.sf.saxon.s9api.Processor;
12  import net.sf.saxon.s9api.QName;
13  import net.sf.saxon.s9api.XdmNode;
14  
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  public class AnnotationsConfigExtension implements ConfigReader.Extension {
19  
20  	private Logger Logger = LoggerFactory.getLogger(AnnotationsConfigExtension.class);
21  	private final Processor saxonproc;
22  	
23  	public AnnotationsConfigExtension(Processor saxonproc) {
24  		this.saxonproc = saxonproc;
25  	}
26  	
27  	@Override
28  	public boolean parseNode(XdmNode node, URI documentURI) {
29  		String name = node.getNodeName().getLocalName();
30  		if ("annotations".equalsIgnoreCase(name)) {
31  			String href = node.getAttributeValue(new QName(null, "href"));
32  			String type = node.getAttributeValue(new QName(null, "type"));
33  			if (type == null) {
34  				Logger.debug("missing attribute @type in node <annotations> of the config file");
35  				return false;
36  			}
37  			if (!mAnnotations.containsKey(type))
38  				mAnnotations.put(type, new ArrayList<XdmNode>());
39  			if (href != null) {
40  				XdmNode external = ConfigReader.readFromURIinsideConfig(href, saxonproc, documentURI);
41  				if (external != null) {
42  					Logger.info("custom annotations read from " + external.getBaseURI());
43  					mAnnotations.get(type).add(external);
44  				}
45  			} else {
46  				Logger.info("custom embedded annotations read from " + documentURI);
47  				mAnnotations.get(type).add(node);
48  			}
49  			return true;
50  		}
51  		return false;
52  	}
53  
54  	public Collection<XdmNode> getAnnotations(String mediatype) {
55  		Collection<XdmNode> res = mAnnotations.get(mediatype);
56  		if (res != null)
57  			return res;
58  		return Collections.EMPTY_LIST;
59  	}
60  
61  	@Override
62  	public void setParentReader(ConfigReader cr) {
63  	}
64  
65  	private Map<String, List<XdmNode>> mAnnotations = new HashMap<String, List<XdmNode>>();
66  }
