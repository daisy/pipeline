1   package org.daisy.pipeline.tts;
2   
3   import java.util.Map;
4   
5   /**
6    * Classes that implements the TTSService interface are meant to be allocated
7    * only once, as opposed to TTSEngines. The main role of TTSService is to
8    * allocate TTSEngines depending on given parameters.
9    * 
10   */
11  public interface TTSService {
12  
13  	class SynthesisException extends Exception {
14  
15  		public SynthesisException(String message, Throwable cause) {
16  			super(message, cause);
17  		}
18  
19  		public SynthesisException(Throwable t) {
20  			super(t);
21  		}
22  
23  		public SynthesisException(String message) {
24  			super(message);
25  		}
26  	}
27  
28  	/**
29  	 * Allocate a new TTSEngine (e.g. eSpeak or SAPI).
30  	 * 
31  	 * @param params contains various key-value pairs. Some of them might be options
32  	 *         for the TTS processor under consideration, e.g. server IPs,
33  	 *         priorities or sound quality. It can also contain options which
34  	 *         have nothing to do with the returned TTSEngine. Such options must
35  	 *         be ignored. This method is allowed to perform heavy
36  	 *         initializations as long as they can be interrupted with the
37  	 *         regular Java thread interruption mechanism. In particular, it is
38  	 *         not recommended to test a full text-to-speech step in this
39  	 *         method.
40  	 */
41  	public TTSEngine newEngine(Map<String, String> params) throws Throwable;
42  
43  	/**
44  	 * @return the same name as in the CSS voice-family property. If several TTS
45  	 *         services share the same name, then the one with the highest
46  	 *         priority will be chosen. Must be thread-safe.
47  	 */
48  	public String getName();
49  
50  }
