1   package org.daisy.pipeline.tts.calabash.impl;
2   
3   import java.io.File;
4   
5   import javax.sound.sampled.AudioInputStream;
6   
7   import org.daisy.pipeline.tts.AudioFootprintMonitor;
8   
9   /**
10   * ContiguousPCMs are the message objects sent from TTS processors to the
11   * encoders. Every ContiguousPCM contains a AudioInputStream and attributes
12   * related to the URI of the sound file to produce.
13   */
14  class ContiguousPCM implements Comparable<ContiguousPCM> {
15  
16  	static ContiguousPCM EndOfQueue = new ContiguousPCM();
17  
18  	ContiguousPCM() {
19  		mEncodingTimeApprox = -1;
20  		mSizeInBytes = minByteSize();
21  	}
22  
23  	boolean isEndOfQueue() {
24  		return (mEncodingTimeApprox == -1);
25  	}
26  
27  	ContiguousPCM(AudioInputStream audio, File destdir, String destFilePrefix) {
28  		mDestURI = new StringBuilder();
29  		mAudio = audio;
30  		mDestDir = destdir;
31  		mDestFilePrefix = destFilePrefix;
32  		int size = AudioFootprintMonitor.getFootprint(audio);
33  		mSizeInBytes = minByteSize() + size;
34  		mEncodingTimeApprox = size / audio.getFormat().getSampleSizeInBits();
35  	}
36  
37  	@Override
38  	public int compareTo(ContiguousPCM o) {
39  		return (o.mEncodingTimeApprox - mEncodingTimeApprox); //descending order => biggest chunks will be polled first
40  	}
41  
42  	String getDestinationFilePrefix() {
43  		return mDestFilePrefix;
44  	}
45  
46  	File getDestinationDirectory() {
47  		return mDestDir;
48  	}
49  
50  	AudioInputStream getAudio() {
51  		return mAudio;
52  	}
53  
54  	StringBuilder getURIholder() {
55  		return mDestURI;
56  	}
57  
58  	public int sizeInBytes() {
59  		return mSizeInBytes;
60  	}
61  
62  	private int minByteSize() {
63  		return 500;//rough approximation of an empty ContiguousPCM's memory footprint
64  	}
65  
66  	private AudioInputStream mAudio;
67  	private int mEncodingTimeApprox; //used for sorting
68  	private int mSizeInBytes; //used for monitoring the memory footprint
69  	private File mDestDir;
70  	private String mDestFilePrefix;
71  	private StringBuilder mDestURI; //simple way to hold a string
72  
73  }
