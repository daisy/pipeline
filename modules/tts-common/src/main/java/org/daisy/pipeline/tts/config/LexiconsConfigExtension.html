1   package org.daisy.pipeline.tts.config;
2   
3   import java.net.URI;
4   import java.util.ArrayList;
5   import java.util.Collection;
6   import java.util.List;
7   
8   import net.sf.saxon.s9api.Processor;
9   import net.sf.saxon.s9api.QName;
10  import net.sf.saxon.s9api.XdmNode;
11  
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  public class LexiconsConfigExtension implements ConfigReader.Extension {
16  
17  	private Logger Logger = LoggerFactory.getLogger(LexiconsConfigExtension.class);
18  	private final Processor saxonproc;
19  	
20  	public LexiconsConfigExtension(Processor saxonproc) {
21  		this.saxonproc = saxonproc;
22  	}
23  
24  	@Override
25  	public boolean parseNode(XdmNode node, URI documentURI) {
26  		String name = node.getNodeName().getLocalName();
27  		if ("lexicon".equalsIgnoreCase(name)) {
28  			String href = node.getAttributeValue(new QName(null, "href"));
29  			if (href != null) {
30  				XdmNode external = ConfigReader.readFromURIinsideConfig(href, saxonproc, documentURI);
31  				if (external != null) {
32  					Logger.info("custom annotations read from " + external.getBaseURI());
33  					mLexicons.add(external);
34  				}
35  			} else {
36  				Logger.info("custom embedded annotations read from " + documentURI);
37  				mLexicons.add(node);
38  			}
39  			return true;
40  		}
41  		return false;
42  	}
43  
44  	public Collection<XdmNode> getLexicons() {
45  		return mLexicons;
46  	}
47  
48  	@Override
49  	public void setParentReader(ConfigReader cr) {
50  	}
51  
52  	private List<XdmNode> mLexicons = new ArrayList<XdmNode>();
53  }
