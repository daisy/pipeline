1   package org.daisy.pipeline.tts.config;
2   
3   import java.net.URI;
4   import java.util.ArrayList;
5   import java.util.Collection;
6   import java.util.List;
7   
8   import net.sf.saxon.s9api.QName;
9   import net.sf.saxon.s9api.XdmNode;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  public class LexiconsConfigExtension implements ConfigReader.Extension {
15  
16  	private final static Logger logger = LoggerFactory.getLogger(LexiconsConfigExtension.class);
17  
18  	private List<XdmNode> lexicons = new ArrayList<>();
19  
20  	@Override
21  	public boolean parseNode(XdmNode node, URI documentURI, ConfigReader parent) {
22  		String name = node.getNodeName().getLocalName();
23  		if ("lexicon".equalsIgnoreCase(name)) {
24  			String href = node.getAttributeValue(new QName(null, "href"));
25  			if (href != null) {
26  				XdmNode external = parent.parseXML(href, documentURI);
27  				if (external != null) {
28  					logger.info("custom annotations read from " + external.getBaseURI());
29  					lexicons.add(external);
30  				}
31  			} else {
32  				logger.info("custom embedded annotations read from " + documentURI);
33  				lexicons.add(node);
34  			}
35  			return true;
36  		}
37  		return false;
38  	}
39  
40  	public Collection<XdmNode> getLexicons() {
41  		return lexicons;
42  	}
43  }
