8   import javax.xml.namespace.QName;

10  import org.daisy.common.transform.InputValue;

12  import org.daisy.common.transform.XMLInputValue;

17  /**
18   * Prepares an input XML stream for TTS processing. The output is an SSML document that can be
19   * passed to a {@link TTSEngine}.
20   *
21   * Objects that implement this interface are expected to also implement {@link
22   * SingleInSingleOutXMLTransformer}, or a {@link XProcStepProvider} that provides {@link
23   * SingleInSingleOutXMLTransformer}.
24   *
25   * The language of the input can be specified through a "language" parameter. It should not be
26   * assumed that a {@code xml:lang} attribute is present when the input is an element.
27   */
28  public interface TTSInputProcessor {
29  
30  	public boolean supportsInputMediaType(String mediaType);
31  
32  	public TTSInputProcessor forInputMediaType(String mediaType) throws UnsupportedOperationException;
33  
34  	/**
35  	 * Priority of this processor compared to other processors (that supports a given media type).
36  	 */
37  	public int getPriority();
38  
39  	public static Optional<Locale> getLanguage(Map<QName,InputValue<?>> params) throws IllegalArgumentException {
40  		InputValue<?> v = params.get(new QName("language"));
41  		if (v != null)
42  			try {
43  				Object o = v.asObject();
44  				if (o instanceof Locale)
45  					return Optional.of((Locale)o);
46  				else if (o instanceof String)
47  					try {
48  						return Optional.of(new Locale.Builder().setLanguageTag((String)o).build());
49  					} catch (IllformedLocaleException e) {
50  						throw new IllegalArgumentException("language can not be parsed: " + v, e);
51  					}
52  				else
53  					throw new IllegalArgumentException();
54  			} catch (UnsupportedOperationException e) {
55  				throw new IllegalArgumentException();
56  			}
57  		return Optional.empty();
58  	}
59  }
