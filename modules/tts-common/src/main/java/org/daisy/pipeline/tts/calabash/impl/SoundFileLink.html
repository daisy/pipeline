1   package org.daisy.pipeline.tts.calabash.impl;
2   
3   import java.time.Duration;
4   import java.util.concurrent.atomic.AtomicReference;
5   
6   import org.daisy.pipeline.audio.AudioClip;
7   
8   /**
9    * Mapping between a fragment in the XML document and a fragment in an audio file.
10   */
11  class SoundFileLink {
12  
13  	/** ID of fragment in XML document */
14  	private final String xmlID;
15  	/** Time position where the audio fragment starts within <code>clipBase</code> */
16  	private final Duration clipBegin;
17  	/** Time position where the audio fragment ends within <code>clipBase</code> */
18  	private final Duration clipEnd;
19  
20  	/**
21  	 * Audio clip that is used as the base for <code>clipBegin</code> and <code>clipEnd</code>. The
22  	 * purpose of the reference is so that it can be easily shared between {@link TextToPcmThread}
23  	 * and {@link EncodingThread} (via {@link ContiguousPCM}), because {@link AudioClip} is
24  	 * immutable.
25  	 */
26  	public AtomicReference<AudioClip> clipBase;
27  
28  	public SoundFileLink(String xmlID, Duration clipBegin, Duration clipEnd) {
29  		if (clipBegin == null || clipEnd == null || xmlID == null)
30  			throw new NullPointerException();
31  		if (clipBegin.compareTo(Duration.ZERO) < 0)
32  			throw new IllegalArgumentException("Clip begin can not be negative: " + clipBegin);
33  		if (clipEnd.compareTo(clipBegin) < 0)
34  			throw new IllegalArgumentException("Clip end can not come before clip begin: " + clipEnd + " < " + clipBegin);
35  		this.xmlID = xmlID;
36  		this.clipBegin = clipBegin;
37  		this.clipEnd = clipEnd;
38  	}
39  
40  	public String getTextFragment() {
41  		return xmlID;
42  	}
43  
44  	private AudioClip clip;
45  
46  	public AudioClip getAudioFragment(TTSLog log) {
47  		if (clip == null) {
48  			if (clipBase == null)
49  				throw new IllegalStateException(); // should not happen
50  			AudioClip base = clipBase.get();
51  			if (base == null)
52  				return null; // was not set in EncodingThread due to an error
53  			Duration begin = base.clipBegin.plus(clipBegin);
54  			Duration end = base.clipBegin.plus(clipEnd);
55  			if (begin.compareTo(base.clipEnd) > 0) {
56  				if (begin.minus(base.clipEnd).toMillis() > 1)
57  					// this can not be a rounding error due to sampling
58  					log.getOrCreateEntry(getTextFragment()).addError(
59  						new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, "clip not encoded"));
60  				begin = base.clipEnd;
61  				end = base.clipEnd;
62  			} else if (end.compareTo(base.clipEnd) > 0) {
63  				if (end.minus(base.clipEnd).toMillis() > 1)
64  					// this can not be a rounding error due to sampling
65  					log.getOrCreateEntry(getTextFragment()).addError(
66  						new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING, "part of clip not encoded"));
67  				end = base.clipEnd;
68  			}
69  			clip = new AudioClip(base.src, begin, end);
70  		}
71  		return clip;
72  	}
73  }
