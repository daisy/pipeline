4   import java.util.Collection;

8   import java.util.concurrent.CopyOnWriteArrayList;

10  import javax.xml.transform.URIResolver;

12  import net.sf.saxon.s9api.Processor;
13  
14  import org.daisy.pipeline.tts.TTSService.SynthesisException;
15  import org.daisy.pipeline.tts.config.ConfigReader;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  import org.osgi.service.component.annotations.Component;
20  import org.osgi.service.component.annotations.Reference;
21  import org.osgi.service.component.annotations.ReferenceCardinality;
22  import org.osgi.service.component.annotations.ReferencePolicy;
23  
24  @Component(
25  	name = "tts-registry",
26  	service = { TTSRegistry.class }
27  )
28  public class TTSRegistry {
29  
30  	public static class TTSResource {
31  		public boolean invalid = false;
32  	}
33  
34  	private Logger ServerLogger = LoggerFactory.getLogger(TTSRegistry.class);
35  	private URIResolver mURIResolver; //not used so far
36  	private List<TTSService> mServices = new CopyOnWriteArrayList<TTSService>(); //List of active services
37  	//Services and resources used by the current running steps (some of them may not be active anymore):
38  	private Map<TTSService, List<TTSResource>> mTTSResources = new HashMap<TTSService, List<TTSResource>>();
39  
40  	/**
41  	 * Service component callback
42  	 */
43  	@Reference(
44  		name = "uri-resolver",
45  		unbind = "-",
46  		service = URIResolver.class,
47  		cardinality = ReferenceCardinality.MANDATORY,
48  		policy = ReferencePolicy.STATIC
49  	)
50  	public void setURIResolver(URIResolver uriResolver) {
51  		mURIResolver = uriResolver;
52  	}
53  
54  	/**
55  	 * Service component callback
56  	 */
57  	public void unsetURIResolver(URIResolver uriResolver) {
58  		mURIResolver = null;
59  	}
60  
61  	/**
62  	 * Service component callback
63  	 */
64  	@Reference(
65  		name = "TTSService",
66  		unbind = "removeTTS",
67  		service = TTSService.class,
68  		cardinality = ReferenceCardinality.MULTIPLE,
69  		policy = ReferencePolicy.DYNAMIC
70  	)
71  	public void addTTS(TTSService tts) {
72  		ServerLogger.info("Adding TTSService " + tts.getName());
73  		mServices.add(tts);
74  		synchronized (mTTSResources) {
75  			mTTSResources.put(tts, new ArrayList<TTSResource>());
76  		}
77  	}
78  
79  	/**
80  	 * Service component callback
81  	 */
82  	public void removeTTS(TTSService tts) {
83  		ServerLogger.info("Removing TTSService " + tts.getName());
84  
85  		List<TTSResource> resources = null;
86  		synchronized (mTTSResources) {
87  			resources = mTTSResources.get(tts);
88  		}
89  		if (resources != null) {
90  			if (resources.size() > 0)
91  				ServerLogger.warn("Stopping bundle of " + tts.getName()
92  				        + " while a TTS job is running");
93  			for (TTSResource resource : resources) {
94  				synchronized (resource) {
95  					resource.invalid = true;
96  				}
97  			}
98  		}
99  
100 		mServices.remove(tts);
101 	}
102 
103 	public Collection<TTSService> getServices() {
104 		return mServices;
105 	}
106 
107 	/**
108 	 * Return the list of available voices. The SSML adapters and the engines
109 	 * are not checked, so it might be the case that some of the returned voices
110 	 * are actually not usable.
111 	 */
112 	public Collection<Voice> getAllAvaibleVoices(Processor saxonProcessor) {
113 		ConfigReader cr = new ConfigReader(saxonProcessor);
114 		TTSTimeout timeout = new TTSTimeout();
115 		List<Voice> result = new ArrayList<Voice>();
116 		for (TTSService service : mServices) {
117 			timeout.enableForCurrentThread(30);
118 			try {
119 				TTSEngine engine = service.newEngine(cr.getStaticProperties());
120 				result.addAll(engine.getAvailableVoices());
121 			} catch (Throwable e) {
122 				//ignore
123 			} finally {
124 				timeout.disable();
125 			}
126 		}
127 		timeout.close();
128 		return result;
129 	}
130 
131 	public TTSResource allocateResourceFor(TTSEngine tts) throws SynthesisException,
132 	        InterruptedException {
133 		List<TTSResource> resources = null;
134 		synchronized (mTTSResources) {
135 			resources = mTTSResources.get(tts.getProvider());
136 		}
137 
138 		if (resources == null)
139 			return null; //mTTSResources has been clear because the OSGi component has been stopped
140 
141 		TTSResource r = tts.allocateThreadResources();
142 		if (r == null)
143 			r = new TTSResource();
144 		resources.add(r);
145 
146 		return r;
147 	}
148 }
