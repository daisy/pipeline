1   package org.daisy.pipeline.tts;
2   
3   import java.util.Locale;
4   import java.util.Optional;
5   
6   import org.daisy.pipeline.tts.VoiceInfo.Gender;
7   
8   /** Voice instances are expected to be created only by TTSEngine/TTSService implementations */
9   public class Voice {
10  
11  	public enum MarkSupport {
12  		DEFAULT,
13  		MARK_SUPPORTED,
14  		MARK_NOT_SUPPORTED
15  	}
16  
17  	public Voice(String engine, String name) {
18  		this(engine, name, MarkSupport.DEFAULT);
19  	}
20  
21  	public Voice(String engine, String name, MarkSupport markSupport) {
22  		this(engine, name, null, null, markSupport);
23  	}
24  
25  	public Voice(String engine, String name, Locale locale, Gender gender) {
26  		this(engine, name, locale, gender, MarkSupport.DEFAULT);
27  	}
28  
29  	public Voice(String engine, String name, Locale locale, Gender gender, MarkSupport markSupport) {
30  		//we keep the strings in their full case form because some engines might be case sensitive
31  		this.engine = engine == null ? "" : engine;
32  		this.name = name == null ? "" : name;
33  		this.locale = Optional.ofNullable(locale);
34  		this.gender = Optional.ofNullable(gender);
35  		this.mMarkSupport = markSupport;
36  		this.mEngine_lo = this.engine.toLowerCase();
37  		this.mName_lo = this.name.toLowerCase();
38  	}
39  
40  	public int hashCode() {
41  		return mEngine_lo.hashCode() ^ mName_lo.hashCode();
42  	}
43  
44  	public boolean equals(Object other) {
45  		if (other == null)
46  			return false;
47  		Voice v2 = (Voice) other;
48  		return mEngine_lo.equals(v2.mEngine_lo) && mName_lo.equals(v2.mName_lo);
49  	}
50  
51  	public String toString() {
52  		StringBuilder s = new StringBuilder();
53  		boolean first = true;
54  		s.append("{");
55  		if (!engine.isEmpty()) {
56  			s.append("engine:").append(engine);
57  			first = false;
58  		}
59  		if (!name.isEmpty()) {
60  			if (!first) s.append(", ");
61  			s.append("name:").append(name);
62  			first = false;
63  		}
64  		if (locale.isPresent()) {
65  			if (!first) s.append(", ");
66  			s.append("locale:").append(locale.get());
67  			first = false;
68  		}
69  		if (gender.isPresent()) {
70  			if (!first) s.append(", ");
71  			s.append("gender:").append(gender.get());
72  		}
73  		s.append("}");
74  		return s.toString();
75  	}
76  
77  	public MarkSupport getMarkSupport() {
78  		return mMarkSupport;
79  	}
80  
81  	/**
82  	 * The locale of this voice, or absent if it is not known.
83  	 */
84  	public Optional<Locale> getLocale() {
85  		return locale;
86  	}
87  
88  	/**
89  	 * The gender of this voice, or absent if it is not known.
90  	 */
91  	public Optional<Gender> getGender() {
92  		return gender;
93  	}
94  
95  	//the upper-case versions need to be kept because some TTS Processors like SAPI
96  	//are case-sensitive. Lower-case versions are only used for comparison.
97  	public final String engine;
98  	public final String name;
99  	private String mEngine_lo;
100 	private String mName_lo;
101 	private final MarkSupport mMarkSupport;
102 	private final Optional<Locale> locale;
103 	private final Optional<Gender> gender;
104 }
