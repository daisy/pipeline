1   package org.daisy.pipeline.tts.calabash.impl;
2   
3   import java.io.File;
4   import java.util.Map;
5   import java.util.concurrent.BlockingQueue;
6   
7   import javax.sound.sampled.AudioFileFormat;
8   import javax.sound.sampled.AudioInputStream;
9   
10  import org.daisy.common.messaging.MessageAppender;
11  import org.daisy.common.messaging.MessageBuilder;
12  import org.daisy.pipeline.audio.AudioClip;
13  import org.daisy.pipeline.audio.AudioEncoder;
14  import org.daisy.pipeline.audio.AudioServices;
15  import org.daisy.pipeline.tts.AudioFootprintMonitor;
16  import org.daisy.pipeline.tts.TTSTimeout;
17  import org.daisy.pipeline.tts.calabash.impl.TTSLog.ErrorCode;
18  
19  /**
20   * Consumes a shared queue of PCM packets. PCM packets are then provided to
21   * audio encoders. The thread stops when it receives an 'EndOfQueue' marker.
22   */
23  public class EncodingThread {
24  
25  	class EncodingException extends RuntimeException {
26  		public EncodingException(String message, Throwable cause) {
27  			super(message, cause);
28  		}
29  		public EncodingException(Throwable t) {
30  			super(t);
31  		}
32  		public EncodingException(String message) {
33  			super(message);
34  		}
35  	}
36  	
37  	private Thread mThread;
38  	private Throwable criticalError;
39  
40  	void start(final AudioFileFormat.Type fileType, final AudioServices encoderRegistry,
41  	        final BlockingQueue<ContiguousPCM> inputPCM, final AudioFootprintMonitor audioFootprintMonitor,
42  	        Map<String, String> TTSproperties, final TTSLog ttslog, MessageAppender messageAppender) {
43  
44  		//max seconds of encoded audio per seconds of encoding
45  		//it would be more accurate with a byte rate instead, but less intuitive
46  		float encodingSpeed = 2.0f;
47  		String speedProp = "org.daisy.pipeline.tts.encoding.speed";
48  		String speedParam = TTSproperties.get(speedProp);
49  		if (speedParam != null) {
50  			try {
51  				encodingSpeed = Float.valueOf(speedParam);
52  			} catch (NumberFormatException e) {
53  				ttslog.addGeneralError(
54  					ErrorCode.WARNING,
55  					"wrong format for property " + speedProp + ". A float is expected, not " + speedParam,
56  					e);
57  			}
58  		}
59  
60  		//Eventually, we should select the encoder using the audio format as criterion, but for now
61  		//we always employ the same encoder for every chunk of PCM
62  		AudioEncoder encoder = encoderRegistry.newEncoder(fileType, TTSproperties).orElse(null);
63  		if (encoder == null) {
64  			ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, "No audio encoder found");
65  		}
66  
67  		final AudioEncoder fencoder = encoder;
68  		final float fEncodingSpeed = encodingSpeed;
69  		final TTSTimeout timeout = new TTSTimeout();
70  
71  		mThread = new Thread() {
72  			@Override
73  			public void run() {
74  				// wrap the messages from this thread in a (empty) block so that there is always an
75  				// active block for this thread, so that SLF4J log messages always have a destination
76  				MessageAppender messageThread = messageAppender != null
77  					? messageAppender.append(new MessageBuilder())
78  					: null;
79  				try {
80  					while (!interrupted()) {
81  						ContiguousPCM job;
82  						try {
83  							job = inputPCM.take();
84  						} catch (InterruptedException e) {
85  							ttslog.addGeneralError(
86  								ErrorCode.CRITICAL_ERROR, "encoding thread has been interrupted");
87  							break; //warning: encoding bytes are not freed
88  						}
89  						if (job.isEndOfQueue()) {
90  							//nothing to release
91  							break;
92  						}
93  						int jobSize = job.sizeInBytes();
94  						// FIXME: why do we end up in endless loop when encoder is null??
95  						if (fencoder != null) {
96  							AudioInputStream audio = job.getAudio();
97  							float secs = jobSize / audio.getFormat().getFrameRate();
98  							int maxTime = (int) (1.0 + secs / fEncodingSpeed);
99  							timeout.enableForCurrentThread(maxTime);
100 							try {
101 								File encodedFile = new File(
102 									job.getDestinationDirectory(),
103 									job.getDestinationFilePrefix() + "." + fileType.getExtension());
104 								AudioClip clip = fencoder.encode(
105 									audio,
106 									fileType,
107 									encodedFile);
108 								audio.close();
109 								job.getDestinationFile().set(clip);
110 							} catch (InterruptedException e) {
111 								ttslog.addGeneralError(
112 									ErrorCode.CRITICAL_ERROR,
113 									"timeout while encoding audio to " + job.getDestinationFilePrefix(),
114 									e);
115 								audioFootprintMonitor.releaseEncodersMemory(jobSize);
116 								throw new EncodingException(e);
117 							} catch (Throwable t) {
118 								ttslog.addGeneralError(
119 									ErrorCode.CRITICAL_ERROR,
120 									"error while encoding audio to " + job.getDestinationFilePrefix(),
121 									t);
122 								audioFootprintMonitor.releaseEncodersMemory(jobSize);
123 								throw new EncodingException(t);
124 							} finally {
125 								timeout.disable();
126 							}
127 						}
128 						audioFootprintMonitor.releaseEncodersMemory(jobSize);
129 					}
130 				} finally {
131 					timeout.close();
132 					if (messageThread != null)
133 						messageThread.close();
134 				}
135 			}
136 		};
137 		mThread.setUncaughtExceptionHandler(
138 			(thread, throwable) -> { criticalError = throwable; }
139 		);
140 		mThread.start();
141 	}
142 
143 	void waitToFinish() throws EncodingException {
144 		try {
145 			mThread.join();
146 		} catch (InterruptedException e) {
147 			throw new RuntimeException(); // should not happen
148 		}
149 		if (criticalError != null) {
150 			if (criticalError instanceof EncodingException)
151 				throw (EncodingException)criticalError;
152 			else
153 				throw new RuntimeException("coding error", criticalError); // should not happen
154 		}
155 	}
156 }
