1   package org.daisy.pipeline.tts.calabash.impl;
2   
3   import java.util.concurrent.Semaphore;
4   
5   import org.daisy.common.xproc.calabash.XProcStep;
6   import org.daisy.common.xproc.calabash.XProcStepProvider;
7   import org.daisy.pipeline.audio.AudioServices;
8   import org.daisy.pipeline.tts.AudioFootprintMonitor;
9   import org.daisy.pipeline.tts.TTSRegistry;
10  
11  import com.xmlcalabash.core.XProcRuntime;
12  import com.xmlcalabash.runtime.XAtomicStep;
13  
14  import org.osgi.service.component.annotations.Component;
15  import org.osgi.service.component.annotations.Reference;
16  import org.osgi.service.component.annotations.ReferenceCardinality;
17  import org.osgi.service.component.annotations.ReferencePolicy;
18  
19  @Component(
20  	name = "ssml-to-audio",
21  	service = { XProcStepProvider.class },
22  	property = { "type:String={http://www.daisy.org/ns/pipeline/xproc}ssml-to-audio" }
23  )
24  public class SynthesizeProvider implements XProcStepProvider {
25  	private TTSRegistry mRegistry;
26  	private AudioServices mAudioServices;
27  	private Semaphore mStartSemaphore; //counter to limit the number of simultaneous text-to-speech steps
28  	private AudioFootprintMonitor mAudioFootprintMonitor;
29  
30  	@Override
31  	public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
32  		if (mStartSemaphore == null) {
33  			mStartSemaphore = new Semaphore(3, true);
34  		}
35  
36  		if (mAudioFootprintMonitor == null) {
37  			mAudioFootprintMonitor = new AudioFootprintMonitor();
38  		}
39  
40  		boolean error = false;
41  		if (mRegistry == null) {
42  			runtime.error(new RuntimeException("Registry of TTS engines is missing."));
43  			error = true;
44  		}
45  
46  		if (mAudioServices == null) {
47  			runtime.error(new RuntimeException("Registry of audio encoders is missing."));
48  			error = true;
49  		}
50  
51  		if (error)
52  			return null;
53  
54  		//warning: a reference is kept on the audio encoder during all the synthesizing process,
55  		//even if it is unregistered.
56  
57  		return new SynthesizeStep(runtime, step, mRegistry, mAudioServices, mStartSemaphore,
58  		        mAudioFootprintMonitor);
59  	}
60  
61  	@Reference(
62  		name = "TTSRegistry",
63  		unbind = "unsetTTSRegistry",
64  		service = TTSRegistry.class,
65  		cardinality = ReferenceCardinality.MANDATORY,
66  		policy = ReferencePolicy.STATIC
67  	)
68  	protected void setTTSRegistry(TTSRegistry registry) {
69  		mRegistry = registry;
70  	}
71  
72  	protected void unsetTTSRegistry(TTSRegistry registry) {
73  		mRegistry = null;
74  	}
75  
76  	@Reference(
77  		name = "AudioServices",
78  		unbind = "unsetAudioServices",
79  		service = AudioServices.class,
80  		cardinality = ReferenceCardinality.OPTIONAL,
81  		policy = ReferencePolicy.DYNAMIC
82  	)
83  	protected void setAudioServices(AudioServices audioServices) {
84  		mAudioServices = audioServices;
85  	}
86  
87  	protected void unsetAudioServices(AudioServices audioServices) {
88  		mAudioServices = null;
89  	}
90  }
