27  	/**
28  	 * Create an {@link AudioInputStream} from an {@link AudioFormat} and the audio data.
29  	 *
30  	 * @param format Expected to be PCM.
31  	 */
32  	public static AudioInputStream createAudioStream(AudioFormat format, byte[] data) {
33  		return createAudioStream(format, new ByteArrayInputStream(data));
34  	}

36  	public static AudioInputStream createAudioStream(AudioFormat format, ByteArrayInputStream data) {
37  		if (format.getFrameSize() == AudioSystem.NOT_SPECIFIED)
38  			throw new IllegalArgumentException("unknown frame size");
39  		return new AudioInputStream(data,
40  		                            format,
41  		                            // ByteArrayInputStream.available() returns
42  		                            // the total number of bytes
43  		                            data.available() / format.getFrameSize());
44  	}

46  	/**
47  	 * Create a {@link AudioInputStream} from a {@link InputStream}.
48  	 *
49  	 * This is to work around a bug in {@link javax.sound.sampled.AudioSystem}
50  	 * which may return {@link AudioInputStream} with a wrong {@link
51  	 * AudioInputStream#getFrameLength()}.
52  	 */
53  	public static AudioInputStream createAudioStream(InputStream stream)
54  			throws UnsupportedAudioFileException, IOException {
55  		AudioInputStream audio = AudioSystem.getAudioInputStream(new BufferedInputStream(stream));
56  		AudioFormat format = audio.getFormat();
57  		if (format.getFrameSize() != AudioSystem.NOT_SPECIFIED) {
58  			byte[] data = ByteStreams.toByteArray(audio);
59  			audio.close();
60  			audio = createAudioStream(format, data);
61  		}
62  		return audio;
63  	}

65  	/**
66  	 * Get the duration of a PCM encoded audio stream.
67  	 *
68  	 * @throws IllegalArgumentException if format is not PCM encoded with specified frame rate.
69  	 */
70  	public static Duration getDuration(AudioInputStream stream) {
71  		return getDurationFromFrames(PCMAudioFormat.of(stream.getFormat()), stream.getFrameLength());
72  	}

74  	/**
75  	 * @throws IllegalArgumentException if format is not PCM encoded with specified frame rate
76  	 *                                  and frame size.
77  	 */
78  	public static Duration getDuration(AudioFormat format, int bytes) {
79  		format = PCMAudioFormat.of(format);
80  		return getDurationFromFrames((PCMAudioFormat)format, (long)(bytes / format.getFrameSize()));
81  	}

83  	private static Duration getDurationFromFrames(PCMAudioFormat format, Long frames) {
84  		return Duration.ofMillis((long)(frames.floatValue() / (format.getFrameRate() / 1000)));
85  	}

87  	/**
88  	 * @return the number of frames needed to contain the given duration of audio.
89  	 */
90  	public static long getLengthInFrames(PCMAudioFormat format, Duration duration) {
91  		// rounding to the next natural number if needed
92  		return (long)Math.ceil((double)(format.getFrameRate() / 1000) * duration.toMillis());
93  	}

95  	/**
96  	 * Concatenate a sequence of {@link AudioInputStream} into a single {@link AudioInputStream}.
97  	 *
98  	 * @param streams Must be streams with the same (PCM) audio format.
99  	 */
100 	public static AudioInputStream concat(Iterable<AudioInputStream> streams) {
101 		int count = 0;
102 		long _totalLength = 0;
103 		PCMAudioFormat format = null;
104 		for (AudioInputStream s : streams) {
105 			count++;
106 			_totalLength += s.getFrameLength();
107 			if (format == null) {
108 				try {
109 					format = PCMAudioFormat.of(s.getFormat());
110 				} catch (IllegalArgumentException e) {
111 					throw new IllegalArgumentException("AudioInputStream must be PCM encoded, but got: "+ s.getFormat());
112 				}
113 			} else if (!format.matches(s.getFormat()))
114 				throw new IllegalArgumentException("Can not concatenate AudioInputStream with different audio formats");
115 		}
116 		if (count == 0)
117 			throw new IllegalArgumentException("At least one AudioInputStream expected");
118 		if (count == 1)
119 			return streams.iterator().next();
120 		long totalLength = _totalLength;
121 		int frameSize = format.getFrameSize();
122 		return new AudioInputStream(
123 			new InputStream() {
124 				Iterator<AudioInputStream> nextStreams = streams.iterator();
125 				AudioInputStream stream = null;
126 				byte[] frame = new byte[frameSize];
127 				long availableFrames = totalLength;
128 				int availableInFrame = 0;
129 				public int read() throws IOException {
130 					if (availableFrames == 0 && availableInFrame == 0)
131 						return -1;
132 					if (availableInFrame > 0)
133 						return frame[frameSize - (availableInFrame--)] & 0xFF;
134 					if (stream != null) {
135 						availableInFrame = stream.read(frame);
136 						if (availableInFrame > 0) {
137 							availableFrames--;
138 							return frame[frameSize - (availableInFrame--)] & 0xFF;
139 						}
140 					}
141 					try {
142 						if (stream != null)
143 							stream.close();
144 						stream = nextStreams.next();
145 					} catch (NoSuchElementException e) {
146 						return -1;
147 					}
148 					return read();
149 				}
150 				public int available() {
151 					try {
152 						return Math.toIntExact(Math.multiplyExact(availableFrames, frameSize)) + availableInFrame;
153 					} catch (ArithmeticException e) {
154 						return Integer.MAX_VALUE;
155 					}
156 				}
157 				public void close() throws IOException {
158 					if (stream != null)
159 						stream.close();
160 					while (nextStreams.hasNext())
161 						nextStreams.next().close();
162 				}
163 			},
164 			format,
165 			totalLength);
166 	}

168 	/**
169 	 * Split a {@link AudioInputStream} into a sequence of {@link AudioInputStream}.
170 	 *
171 	 * @param stream Must be PCM encoded.
172 	 * @param splitPoints Are relative to the beginning of the stream.
173 	 * @return A sequence of streams. The number of streams is one more than the number of split points.
174 	 *         The durations are approximately (apart from rounding errors due to sampling) the time
175 	 *         between the split points.
176 	 * @throws IllegalArgumentException if a negative split point is provided, a split point exceeds
177 	 *                                  the total length of the input stream, or a split point is not
178 	 *                                  greater than or equal to the preceding split point ,and also
179 	 *                                  if the audio format is not PCM encoded with specified frame
180 	 *                                  rate and frame size.
181 	 */
182 	public static Iterable<AudioInputStream> split(AudioInputStream stream, Duration... splitPoints) {
183 		long[] splitPointsInFrames = new long[splitPoints.length];
184 		PCMAudioFormat format = PCMAudioFormat.of(stream.getFormat());
185 		for (int i = 0; i < splitPoints.length; i++)
186 			splitPointsInFrames[i] = getLengthInFrames(format, splitPoints[i]);
187 		return split(stream, splitPointsInFrames);
188 	}

190 	public static Iterable<AudioInputStream> split(AudioInputStream stream, long... splitPoints) {
191 		if (splitPoints.length == 0)
192 			return Collections.singleton(stream);
193 		PCMAudioFormat format = PCMAudioFormat.of(stream.getFormat());
194 		long totalFrames = stream.getFrameLength();
195 		{ // validate splitPoints
196 			long begin = 0;
197 			for (long end : splitPoints) {
198 				if (end < begin)
199 					throw new IllegalArgumentException();
200 				begin = end; }
201 			if (totalFrames < begin)
202 				// could be a rounding error due to sampling: provide margin of one frame
203 				if (totalFrames + 1 < begin)
204 					throw new IllegalArgumentException();
205 		}
206 		int frameSize = format.getFrameSize();
207 		return new Iterable<AudioInputStream>() {
208 			public Iterator<AudioInputStream> iterator() {
209 				return new AbstractIterator<AudioInputStream>() {
210 					long framesAvailable = totalFrames;
211 					long framesConsumed = 0;
212 					int i = 0;
213 					protected AudioInputStream computeNext() {
214 						if (i > splitPoints.length)
215 							return endOfData();
216 						else if (i == splitPoints.length)
217 							// we're at the last chunk, so we can simply return remaining stream
218 							return stream;
219 						long splitPoint = splitPoints[i];
220 						long frames = splitPoint - framesConsumed;
221 						if (frames > framesAvailable)
222 							throw new IllegalStateException("coding error");
223 						// In theory a byte array can hold 2 Gb of memory, which should be more than
224 						// enough for audio. Note that when a lot of AudioInputStreams are open,
225 						// there is still a risk for out-of-memory errors, but normally the streams
226 						// are consumed in order so that should be okay as well. An alternative
227 						// solution would be to not use byte arrays and enforce that streams are
228 						// fully consumed before the next is computed.
229 						byte[] bytes = new byte[Math.toIntExact(frames * frameSize)];
230 						if (frames > 0) {
231 							try {
232 								if (stream.read(bytes) < bytes.length)
233 									throw new IllegalStateException("coding error");
234 							} catch (IOException e) {
235 								throw new RuntimeException(e);
236 							}
237 							framesConsumed += frames;
238 							framesAvailable -= frames;
239 						}
240 						i++;
241 						return createAudioStream(format, bytes);
242 					}
243 				};
244 			}
245 		};
246 	}

249 	 * @param duration must not be negative
250 	 * @return <code>null</code> if <code>duration</code> is too short
251 	 */
252 	public static AudioInputStream createSilence(PCMAudioFormat format, Duration duration) {
253 		if (duration.compareTo(Duration.ZERO) < 0)
254 			throw new IllegalArgumentException();
255 		long frames = getLengthInFrames(format, duration);
256 		if (frames == 0)
257 			return null;
258 		else
259 			return createSilence(format, frames);
260 	}

262 	/**
263 	 * @param format must be PCM encoded
264 	 * @param frames must not be negative
265 	 * @return <code>null</code> if <code>frames</code> is <code>0</code>
266 	 */
267 	public static AudioInputStream createSilence(AudioFormat format, long frames) {
268 		format = PCMAudioFormat.of(format);
269 		if (frames < 0)
270 			throw new IllegalArgumentException();
271 		if (frames == 0)
272 			return null;
273 		byte[] data = new byte[Math.multiplyExact(Math.toIntExact(frames), format.getFrameSize())]; // initialized to all zeros
274 		Encoding encoding = format.getEncoding();
275 		if (encoding == Encoding.PCM_SIGNED)
276 			return createAudioStream(format, data);
277 		else if (encoding == Encoding.PCM_UNSIGNED)
278 			return convertAudioStream(
279 				format,
280 				createAudioStream(new AudioFormat(Encoding.PCM_SIGNED,
281 				                                  format.getSampleRate(),
282 				                                  format.getSampleSizeInBits(),
283 				                                  format.getChannels(),
284 				                                  format.getFrameSize(),
285 				                                  format.getFrameRate(),
286 				                                  format.isBigEndian()),
287 				                  data));
288 		else // encoding == Encoding.PCM_FLOAT
289 			throw new UnsupportedOperationException("Not implemented"); // FIXME
290 	}

292 	/**
293 	 * Convert audio from one format to another.
294 	 */
295 	public static AudioInputStream convertAudioStream(AudioFormat newFormat, AudioInputStream stream) {
296 		AudioFormat format = stream.getFormat();
297 		if (format.getSampleSizeInBits() > 8
298 		    && newFormat.getSampleSizeInBits() > 8
299 		    && !format.isBigEndian()
300 		    && !newFormat.isBigEndian()
301 		    && ((format.getEncoding() == Encoding.PCM_UNSIGNED
302 		         && newFormat.getEncoding() == Encoding.PCM_SIGNED)
303 		        || (format.getEncoding() == Encoding.PCM_SIGNED
304 		            && newFormat.getEncoding() == Encoding.PCM_UNSIGNED))) {
305 
306 			// com.sun.media.sound.PCMtoPCMCodec does not correctly convert unsigned little-endian
307 			// to signed little-endian or visa-versa. Work around this by first converting the
308 			// samples to big-endian.
309 			return convertAudioStream(
310 				newFormat,
311 				convertAudioStream(
312 					new AudioFormat(format.getSampleRate(),
313 					                format.getSampleSizeInBits(),
314 					                format.getChannels(),
315 					                format.getEncoding() == Encoding.PCM_SIGNED,
316 					                true), // big endian
317 					stream));
318 		} else if (format.getEncoding() == Encoding.PCM_FLOAT
319 		           && newFormat.getEncoding() == Encoding.PCM_FLOAT
320 		           && format.getSampleSizeInBits() == 64
321 		           && newFormat.getSampleSizeInBits() == 64
322 		           && format.isBigEndian() != newFormat.isBigEndian()) {
323 
324 			// AudioSystem.getAudioInputStream() introduces a rounding error, so we do the byte
325 			// order conversion ourselves.
326 			AudioFormat otherEndian = new AudioFormat(format.getEncoding(),
327 			                                          format.getSampleRate(),
328 			                                          format.getSampleSizeInBits(),
329 			                                          format.getChannels(),
330 			                                          format.getFrameSize(),
331 			                                          format.getFrameRate(),
332 			                                          !format.isBigEndian());
333 			ByteBuffer bytes; {
334 				try {
335 					bytes = ByteBuffer.wrap(ByteStreams.toByteArray(stream));
336 				} catch (IOException e) {
337 					throw new RuntimeException(e); // should not happen
338 				}
339 			}
340 			ByteBuffer reorderedBytes  = ByteBuffer.wrap(new byte[bytes.array().length]);
341 			reorderedBytes = reorderedBytes.order(ByteOrder.LITTLE_ENDIAN);
342 			while (bytes.hasRemaining())
343 				reorderedBytes.putLong(bytes.getLong());
344 			stream = createAudioStream(otherEndian, reorderedBytes.array());
345 			if (otherEndian.matches(newFormat))
346 				return stream;
347 			else
348 				return convertAudioStream(newFormat, stream);
349 		}
350 		return AudioSystem.getAudioInputStream(newFormat, stream);
351 	}

353 	public static boolean isPCM(AudioFormat format) {
354 		Encoding encoding = format.getEncoding();
355 		return encoding == Encoding.PCM_SIGNED
356 			|| encoding == Encoding.PCM_UNSIGNED
357 			|| encoding == Encoding.PCM_FLOAT;
358 	}
