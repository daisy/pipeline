125 	private static class DTBookInsertSyncPoints extends SingleInSingleOutXMLTransformer {
126 
127 		public Runnable transform(XMLInputValue<?> source, XMLOutputValue<?> result, InputValue<?> params) throws IllegalArgumentException {
128 			if (source == null || result == null)
129 				throw new IllegalArgumentException();
130 			return () -> transform(source.asXMLStreamReader(), result.asXMLStreamWriter());
131 		}
132 
133 		public void transform(BaseURIAwareXMLStreamReader reader, BaseURIAwareXMLStreamWriter writer) throws TransformerException {
134 			boolean newBlock = true;
135 			LinkedList<QName> parentElement = new LinkedList<>();
136 			try {
137 				int event = reader.getEventType();
138 				while (true)
139 					try {
140 						switch (event) {
141 						case START_DOCUMENT:
142 							writer.setBaseURI(reader.getBaseURI());
143 							XMLStreamWriterHelper.writeEvent(writer, reader);
144 							break;
145 						case START_ELEMENT: {
146 							QName name = reader.getName();
147 							parentElement.push(name);
148 							XMLStreamWriterHelper.writeStartElement(writer, reader);
149 							XMLStreamWriterHelper.writeAttributes(writer, reader);
150 							if (BLOCK_ELEMENTS.contains(name.getLocalPart()))
151 								newBlock = true;
152 							break; }
153 						case END_ELEMENT: {
154 							QName name = parentElement.pop();
155 							writer.writeEndElement();
156 							if (BLOCK_ELEMENTS.contains(name.getLocalPart()))
157 								newBlock = true;
158 							break; }
159 						case CHARACTERS:
160 							if (newBlock) {
161 								String text = reader.getText();
162 								if (text.trim().isEmpty()) {
163 									writer.writeCharacters(text);
164 								} else if (text.matches("^\\s.*$")) {
165 									// Place sync point in between any leading white space and actual text.
166 									// (Note that by doing this, we're actually trimming the white space because it is
167 									// lost in the transformation to HTML. This is not really a problem, but if it needs
168 									// fixing, it should be fixed on the side of the HTML transformation, not here.)
169 									String leadingSpace = text.replaceFirst("^(\\s+).*$", "$1");
170 									writer.writeCharacters(leadingSpace);
171 									if (insertSyncPoint(writer, parentElement.get(0)))
172 										newBlock = false;
173 									writer.writeCharacters(text.substring(leadingSpace.length()));
174 								} else {
175 									if (insertSyncPoint(writer, parentElement.get(0)))
176 										newBlock = false;
177 									writer.writeCharacters(text);
178 								}
179 							} else {
180 								XMLStreamWriterHelper.writeEvent(writer, reader);
181 							}
182 							break;
183 						default:
184 							XMLStreamWriterHelper.writeEvent(writer, reader);
185 						}
186 						event = reader.next();
187 					} catch (NoSuchElementException e) {
188 						break;
189 					}
190 				writer.flush();
191 			} catch (XMLStreamException e) {
192 				throw new TransformerException(e);
193 			}
194 		}
195 
196 		private static boolean insertSyncPoint(XMLStreamWriter writer, QName parent) throws XMLStreamException {
197 			if (TEXT_ONLY_ELEMENTS.contains(parent.getLocalPart()))
198 				return false;
199 			XMLStreamWriterHelper.writeStartElement(writer, DTB_A.equals(parent) ? DTB_SPAN : DTB_A);
200 			XMLStreamWriterHelper.writeAttribute(writer, _CLASS, SYNC_CLASS);
201 			// note that an ID attribute is added in a subsequent step
202 			writer.writeEndElement();
203 			return true;
204 		}
