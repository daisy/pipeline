52  /**
53   * TextToPcmThread consumes text from a shared queue. It produces PCM data as
54   * output, which in turn are pushed to another shared queue consumed by the
55   * EncodingThreads. PCM is produced by calling TTS processors.
56   *
57   * TTS processors may fail for some reasons, e.g. after a timeout or because the
58   * ending SSML mark is missing. In such cases, TextToPcmThread will clean up the
59   * resources and attempt to synthesize the current sentence with another TTS
60   * processor chosen by the TTSRegistry, unless the error is a MemoryException,
61   * in which case the thread gives up on the guilty sentence.
62   *
63   * The resources of the TTS processors (e.g. sockets) are allocated on-the-fly
64   * and are all released at the end of the thread execution.
65   *
66   */
67  public class TextToPcmThread implements FormatSpecifications {
68  	private Logger ServerLogger = LoggerFactory.getLogger(TextToPcmThread.class);
69  	private Map<TTSEngine, TTSResource> mResources = new HashMap<TTSEngine, TTSResource>();
70  	private Map<TTSService, ThreadUnsafeXslTransformer> mTransforms = new HashMap<TTSService, ThreadUnsafeXslTransformer>();
71  	private int mFileNrInSection; //usually = 0, but incremented when a flush occurs within a section
72  	private List<SoundFileLink> mSoundFileLinks; //result provided back to the SynthesizeStep caller
73  	private List<SoundFileLink> mLinksOfCurrentFile; //links under construction
74  	private Iterable<AudioBuffer> mBuffersOfCurrentFile; //buffers under construction
75  	private int mOffsetInFile; // reset after every flush
76  	private int mMemFootprint; //reset after every flush
77  	private Thread mThread;
78  	private TimedTTSExecutor mExecutor;
79  	private TTSRegistry mTTSRegistry;
80  	private IPipelineLogger mPipelineLogger;
81  	private AudioFormat mLastFormat; //used for knowing if a flush is necessary
82  	private AudioBufferTracker mAudioBufferTracker;
83  	private SSMLMarkSplitter mSSMLSplitter;
84  	private Map<String, Object> mTransformParams = new TreeMap<String, Object>();
85  	private Map<TTSService, CompiledStylesheet> mSSMLTransformers;
86  	private VoiceManager mVoiceManager;
87  	private TTSLog mTTSLog;
88  	private int mErrorCounter;

90  	void start(final ConcurrentLinkedQueue<ContiguousText> input,
91  	        final BlockingQueue<ContiguousPCM> pcmOutput, TimedTTSExecutor executor,
92  	        TTSRegistry ttsregistry, VoiceManager voiceManager, SSMLMarkSplitter ssmlSplitter,
93  	        final IProgressListener progressListener, IPipelineLogger pLogger,
94  	        AudioBufferTracker AudioBufferTracker, final int maxQueueEltSize,
95  	        Map<TTSService, CompiledStylesheet> ssmlTransformers, TTSLog ttsLog) {
96  		mSSMLTransformers = ssmlTransformers;
97  		mSSMLSplitter = ssmlSplitter;
98  		mSoundFileLinks = new ArrayList<SoundFileLink>();
99  		mExecutor = executor;
100 		mTTSRegistry = ttsregistry;
101 		mPipelineLogger = pLogger;
102 		mAudioBufferTracker = AudioBufferTracker;
103 		mVoiceManager = voiceManager;
104 		mTTSLog = ttsLog;
105 		mErrorCounter = 0;
106 		flush(null, pcmOutput);
107 
108 		mThread = new Thread() {
109 			@Override
110 			public void run() {
111 				TTSTimeout timeout = new TTSTimeout();
112 
113 				/* Main loop */
114 				while (true) {
115 					ContiguousText section = input.poll();
116 					if (section == null) { //queue is empty
117 						break;
118 					}
119 					mFileNrInSection = 0;
120 					boolean breakloop = false;
121 					for (Sentence sentence : section.sentences) {
122 						if (breakloop) {
123 							mErrorCounter++;
124 							continue;
125 						}
126 						try {
127 							if (!speak(section, sentence, pcmOutput, timeout, maxQueueEltSize)) mErrorCounter++;
128 						} catch (Throwable t) {
129 							StringWriter sw = new StringWriter();
130 							t.printStackTrace(new PrintWriter(sw));
131 							mErrorCounter++;
132 							mTTSLog.getWritableEntry(sentence.getID()).addError(
133 							        new TTSLog.Error(TTSLog.ErrorCode.CRITICAL_ERROR,
134 							                "the current thread is stopping because of error: "
135 							                        + sw.toString()));
136 							breakloop = true;
137 						}
138 					}
139 					flush(section, pcmOutput);
140 					progressListener.notifyFinished(section);
141 				}
142 
143 				//release the TTS resources
144 				for (Map.Entry<TTSEngine, TTSResource> e : mResources.entrySet()) {
145 					try {
146 						timeout.enableForCurrentThread(2);
147 						releaseResource(e.getKey(), e.getValue());
148 					} catch (Exception ex) {
149 						String msg = "Error while releasing resource of "
150 						        + TTSServiceUtil.displayName(e.getKey().getProvider()) + "; "
151 						        + ex.getMessage();
152 						ServerLogger.warn(msg);
153 						mTTSLog.addGeneralError(ErrorCode.WARNING, msg);
154 					} finally {
155 						timeout.disable();
156 					}
157 				}
158 
159 				timeout.close();
160 			}
161 		};
162 		mThread.start();
163 	}

183 	private void releaseResource(TTSEngine tts, TTSResource r) {
184 		if (r == null) {
185 			return;
186 		}
187 		synchronized (r) {
188 			try {
189 				tts.releaseThreadResources(r);
190 			} catch (Throwable t) {
191 				StringWriter sw = new StringWriter();
192 				t.printStackTrace(new PrintWriter(sw));
193 				String msg = "error while releasing resources of "
194 				        + TTSServiceUtil.displayName(tts.getProvider()) + ": " + sw.toString();
195 				mTTSLog.addGeneralError(ErrorCode.WARNING, msg);
196 			}
197 		}
198 	}

200 	private void flush(ContiguousText section, BlockingQueue<ContiguousPCM> pcmOutput) {
201 		if (section != null && mLinksOfCurrentFile.size() > 0) {
202 			if (mLastFormat == null) {
203 				throw new RuntimeException("coding error"); // should not happen
204 			} else {
205 				String filePrefix = String.format("part%04d_%02d_%03d", section
206 				        .getDocumentPosition(), section.getDocumentSplitPosition(),
207 				        mFileNrInSection);
208 
209 				ContiguousPCM pcm = new ContiguousPCM(mLastFormat, mBuffersOfCurrentFile,
210 				        section.getAudioOutputDir(), filePrefix);
211 				for (SoundFileLink clip : mLinksOfCurrentFile) {
212 					clip.soundFileURIHolder = pcm.getURIholder();
213 				}
214 				try {
215 					mAudioBufferTracker.transferToEncoding(mMemFootprint, pcm.sizeInBytes());
216 				} catch (InterruptedException e) {
217 					// Should never happen since interruptions only occur during calls to TTS processors.
218 					ServerLogger.warn("interruption of memory transfer");
219 				}
220 				pcmOutput.add(pcm);
221 				pcm = null;
222 				mSoundFileLinks.addAll(mLinksOfCurrentFile);
223 				++mFileNrInSection;
224 			}
225 		}
226 		mLinksOfCurrentFile = new ArrayList<SoundFileLink>();
227 		mBuffersOfCurrentFile = new ArrayList<AudioBuffer>();
228 		mOffsetInFile = 0;
229 		mMemFootprint = 0;
230 		mLastFormat = null;
231 	}

249 	/**
250 	 * Wrapper around synthesizeSSML() to handle marks
251 	 */
252 	private Iterable<AudioBuffer> synthesize(
253 			TTSTimeout timeout, TTSTimeout.ThreadFreeInterrupter interrupter, TTSLog.Entry logEntry,
254 			Sentence sentence, TTSEngine tts, Voice voice, TTSResource threadResources,
255 			List<Mark> marks, List<String> expectedMarks
256 	) throws SaxonApiException, SynthesisException, TimeoutException, MemoryException {
257 		logEntry.resetTTSinput();
258 		if (tts.endingMark() != null
259 		        && voice.getMarkSupport() != MarkSupport.MARK_NOT_SUPPORTED){
260 			//can handle mark
261 			expectedMarks.set(expectedMarks.size()-1, tts.endingMark());
262 			return synthesizeSSML(timeout, interrupter, logEntry, sentence.getText(),
263 			                      sentence.getID(), sentence.getSize(), tts, voice,
264 			                      threadResources, marks, expectedMarks);
265 		}
266 		else {
267 			Collection<Chunk> chunks = mSSMLSplitter.split(sentence.getText());
268 			Iterable<AudioBuffer> result = new ArrayList<AudioBuffer>();
269 			int offset = 0;
270 			for (Chunk chunk : chunks) {
271 				Collection<AudioBuffer> buffers = null;
272 				try {
273 					buffers = synthesizeSSML(timeout, interrupter, logEntry, chunk.ssml(),
274 					                         sentence.getID(), Sentence.computeSize(chunk.ssml()),
275 					                         tts, voice, threadResources, new ArrayList<Mark>(),
276 					                         expectedMarks);
277 				} catch (MemoryException | SaxonApiException | SynthesisException
278 				        | TimeoutException e) {
279 					//TODO: flush the buffers here
280 					SoundUtil.cancelFootPrint(result, mAudioBufferTracker);
281 					throw e;
282 				} catch (Throwable t) {
283 
284 					//TODO: flush the buffers here
285 					SoundUtil.cancelFootPrint(result, mAudioBufferTracker);
286 					throw new SynthesisException(t);
287 				}
288 
289 				if (chunk.leftMark() != null) {
290 					marks.add(new Mark(chunk.leftMark(), offset));
291 				}
292 				for (AudioBuffer b : buffers) {
293 					offset += b.size;
294 				}
295 				result = Iterables.concat(result, buffers);
296 			}
297 			
298 			//add an empty ending-mark
299 			marks.add(new Mark(tts.endingMark(), 0));
300 
301 			return result;
302 		}
303 	}

305 	/**
306 	 * @return null if something went wrong
307 	 */
308 	private Iterable<AudioBuffer> speakWithVoice(final Sentence sentence, Voice v,
309 	        final TTSEngine tts, List<Mark> marks, List<String> expectedMarks, TTSTimeout timeout)
310 	        		throws MemoryException {
311 		//allocate a TTS resource if necessary
312 		TTSResource resource = mResources.get(tts);
313 		if (resource == null) {
314 			try {
315 				timeout.enableForCurrentThread(3); //3 seconds
316 				resource = mTTSRegistry.allocateResourceFor(tts);
317 			} catch (SynthesisException e) {
318 				mTTSLog.getWritableEntry(sentence.getID()).addError(
319 				        new TTSLog.Error(ErrorCode.WARNING,
320 				                "Error while allocating resources for "
321 				                        + TTSServiceUtil.displayName(tts.getProvider()) + ": "
322 				                        + e));
323 
324 				return null;
325 			} catch (InterruptedException e) {
326 				mTTSLog.getWritableEntry(sentence.getID()).addError(
327 				        new TTSLog.Error(ErrorCode.WARNING,
328 				                "Timeout while trying to allocate resources for "
329 				                        + TTSServiceUtil.displayName(tts.getProvider())));
330 				return null;
331 			} finally {
332 				timeout.disable();
333 			}
334 			if (resource == null) {
335 				//TTS not working anymore?
336 				mTTSLog.getWritableEntry(sentence.getID()).addError(
337 				        new TTSLog.Error(ErrorCode.WARNING, "Could not allocate resource for "
338 				                + TTSServiceUtil.displayName(tts.getProvider())
339 				                + " (it has probably been stopped)."));
340 				return null; //it will try with another TTS
341 			}
342 			mResources.put(tts, resource);
343 
344 			TTSService service = tts.getProvider();
345 			if (!mTransforms.containsKey(service)) {
346 				mTransforms.put(service, mSSMLTransformers.get(service).newTransformer());
347 			}
348 		}
349 
350 		//convert the input sentence into PCM using the TTS processor
351 		Iterable<AudioBuffer> pcm = null;
352 		final TTSResource fresource = resource;
353 		TTSTimeout.ThreadFreeInterrupter interrupter = new ThreadFreeInterrupter() {
354 			@Override
355 			public void threadFreeInterrupt() {
356 				String msg = "Forcing interruption of the current work of "
357 				        + TTSServiceUtil.displayName(tts.getProvider()) + "...";
358 				ServerLogger.warn(msg);
359 				mTTSLog.getWritableEntry(sentence.getID()).addError(
360 				        new TTSLog.Error(ErrorCode.WARNING, msg));
361 				tts.interruptCurrentWork(fresource);
362 			}
363 		};
364 		TTSLog.Entry logEntry = mTTSLog.getWritableEntry(sentence.getID());
365 		try {
366 			synchronized (resource) {
367 				if (resource.invalid) {
368 					String msg = "Resource of "
369 					        + TTSServiceUtil.displayName(tts.getProvider())
370 					        + " is no longer valid. The corresponding service has probably been stopped.";
371 					mPipelineLogger.printInfo(msg);
372 					mTTSLog.getWritableEntry(sentence.getID()).addError(
373 					        new TTSLog.Error(ErrorCode.WARNING, msg));
374 					return null;
375 				}
376 				pcm = synthesize(timeout, interrupter, logEntry,
377 				                 sentence, tts, v, resource, marks, expectedMarks);
378 			}
379 		} catch (TimeoutException e) {
380 			logEntry.addError(
381 			        new TTSLog.Error(ErrorCode.WARNING, "timeout (" + e.getSeconds()
382 			                + " seconds) fired while speaking with "
383 			                + TTSServiceUtil.displayName(tts.getProvider())));
384 			return null;
385 		} catch (SynthesisException e) {
386 			StringWriter sw = new StringWriter();
387 			e.printStackTrace(new PrintWriter(sw));
388 			logEntry.addError(
389 			        new TTSLog.Error(ErrorCode.WARNING, "error while speaking with "
390 			                + TTSServiceUtil.displayName(tts.getProvider()) + " : " + e + ":"
391 			                + sw.toString()));
392 
393 			return null;
394 		} catch (MemoryException e) {
395 			throw e;
396 		} catch (SaxonApiException e) {
397 			logEntry.addError(
398 			        new TTSLog.Error(ErrorCode.WARNING,
399 			                "error while transforming SSML with the XSLT of "
400 			                        + TTSServiceUtil.displayName(tts.getProvider()) + " : "
401 			                        + e));
402 			return null;
403 		}
404 
405 		//check validity of the result by using the ending mark
406 		if (marks.size() == 0 || (tts.endingMark() != null && !tts.endingMark().equals(
407 		                marks.get(marks.size() - 1).name))) {
408 			SoundUtil.cancelFootPrint(pcm, mAudioBufferTracker);
409 			logEntry.addError(
410 			        new TTSLog.Error(ErrorCode.WARNING, "missing ending mark with "
411 			                + TTSServiceUtil.displayName(tts.getProvider())
412 			                + ". Number of marks received: " + marks.size()));
413 			return null;
414 		}
415 		
416 		int marksReceived = marks.size();
417 		if (expectedMarks.size() != marksReceived){
418 			SoundUtil.cancelFootPrint(pcm, mAudioBufferTracker);
419 			mTTSLog.getWritableEntry(sentence.getID()).addError(
420 			        new TTSLog.Error(ErrorCode.WARNING, "wrong number of marks with "
421 			                + TTSServiceUtil.displayName(tts.getProvider())
422 			                + ". Number of marks received: " + marksReceived +", expected number: "+expectedMarks.size()));
423 			return null;
424 		}
425 		for (int i = 0; i < expectedMarks.size(); ++i){
426 			String expectedMark = expectedMarks.get(i);
427 			String actualMark = marks.get(i).name;
428 			if ((actualMark == null && expectedMark != null) || (actualMark != null && !actualMark.equals(expectedMark))){
429 				SoundUtil.cancelFootPrint(pcm, mAudioBufferTracker);
430 				mTTSLog.getWritableEntry(sentence.getID()).addError(
431 				        new TTSLog.Error(ErrorCode.WARNING, "mark name mismatch with "
432 				                + TTSServiceUtil.displayName(tts.getProvider())
433 				                + " actual: " + actualMark+", expected: "+expectedMark));
434 				return null;
435 			}
436 		}
437 		
438 		return pcm;
439 	}

441 	private List<String> getMarkNames(XdmNode ssml){
442 		XdmSequenceIterator iter = ssml.axisIterator(Axis.DESCENDANT);
443 		ArrayList<String> markNames = new ArrayList<String>();
444 		while (iter.hasNext()){
445 			XdmNode elt = (XdmNode) iter.next();
446 			if (elt.getNodeKind() == XdmNodeKind.ELEMENT && "mark".equals(elt.getNodeName().getLocalName())){
447 				markNames.add(elt.getAttributeValue(new QName("name")));
448 			}
449 		}
450 		return markNames;
451 	}

453 	/**
454 	 * @return true when the sentence was successfully converted to speech, false when there was an error
455 	 */
456 	private boolean speak(ContiguousText section, Sentence sentence,
457 	        BlockingQueue<ContiguousPCM> pcmOutput, TTSTimeout timeout, int maxQueueEltSize) {
458 		
459 		List<String> expectedMarks = getMarkNames(sentence.getText());
460 		expectedMarks.add(null); //makes room for the ending-mark
461 		
462 		TTSEngine tts = sentence.getTTSproc();
463 		Voice originalVoice = sentence.getVoice();
464 		List<Mark> marks = new ArrayList<Mark>();
465 		Iterable<AudioBuffer> pcm;
466 		try {
467 			pcm = speakWithVoice(sentence, originalVoice, tts, marks, expectedMarks, timeout);
468 		} catch (MemoryException e) {
469 			flush(section, pcmOutput);
470 			printMemError(sentence, e);
471 			return false;
472 		}
473 		if (pcm == null) {
474 			//release the resource to make it more likely for the next try to succeed
475 			releaseResource(tts, mResources.get(tts));
476 			mResources.remove(tts);
477 
478 			//Find another voice for this sentence
479 			Voice newVoice = mVoiceManager.findSecondaryVoice(sentence.getVoice());
480 			if (newVoice == null) {
481 				mTTSLog.getWritableEntry(sentence.getID()).addError(
482 				        new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING,
483 				                "  something went wrong but no fallback voice can be found for "
484 				                        + originalVoice));
485 				return false;
486 			}
487 			tts = mVoiceManager.getTTS(newVoice); //cannot return null in this case
488 
489 			//Try with the new engine
490 			marks.clear();
491 			try {
492 				pcm = speakWithVoice(sentence, newVoice, tts, marks, expectedMarks, timeout);
493 			} catch (MemoryException e) {
494 				flush(section, pcmOutput);
495 				printMemError(sentence, e);
496 				return false;
497 			}
498 			if (pcm == null) {
499 				mTTSLog.getWritableEntry(sentence.getID()).addError(
500 				        new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING,
501 				                " something went wrong with " + originalVoice
502 				                        + " and fallback voice " + newVoice
503 				                        + " didn't work either"));
504 				return false;
505 			}
506 
507 			mPipelineLogger.printInfo(IPipelineLogger.UNEXPECTED_VOICE
508 			        + ": something went wrong with " + originalVoice + ". Voice " + newVoice
509 			        + " used instead to synthesize sentence");
510 
511 			if (mLastFormat != null && !tts.getAudioOutputFormat().matches(mLastFormat))
512 				flush(section, pcmOutput);
513 		}
514 		mLastFormat = tts.getAudioOutputFormat();
515 
516 		int begin = mOffsetInFile;
517 		try {
518 			addBuffers(pcm);
519 		} catch (InterruptedException e) {
520 			// Should never happen since interruptions only occur during calls to TTS processors.
521 			ServerLogger.warn("interruption exception while queuing the PCM buffers");
522 		}
523 
524 		if (marks.size() > 0) {
525 			//remove the ending mark
526 			marks = marks.subList(0, marks.size() - 1);
527 		}
528 
529 		// keep track of where the sound begins and where it ends within the audio buffers
530 		if (marks.size() == 0) {
531 			SoundFileLink sf = new SoundFileLink();
532 			sf.xmlid = sentence.getID();
533 			sf.clipBegin = convertBytesToSecond(mLastFormat, begin);
534 			sf.clipEnd = convertBytesToSecond(mLastFormat, mOffsetInFile);
535 			mLinksOfCurrentFile.add(sf);
536 		} else {
537 			Map<String, Integer> starts = new HashMap<String, Integer>();
538 			Map<String, Integer> ends = new HashMap<String, Integer>();
539 			Set<String> all = new HashSet<String>();
540 
541 			for (Mark m : marks) {
542 				String[] mark = m.name.split(FormatSpecifications.MarkDelimiter, -1);
543 				if (!mark[0].isEmpty()) {
544 					ends.put(mark[0], m.offsetInAudio);
545 					all.add(mark[0]);
546 				}
547 				if (!mark[1].isEmpty()) {
548 					starts.put(mark[1], m.offsetInAudio);
549 					all.add(mark[1]);
550 				}
551 			}
552 			for (String id : all) {
553 				SoundFileLink sf = new SoundFileLink();
554 				sf.xmlid = id;
555 				if (starts.containsKey(id))
556 					sf.clipBegin = convertBytesToSecond(mLastFormat, begin + starts.get(id));
557 				else
558 					sf.clipBegin = convertBytesToSecond(mLastFormat, begin);
559 				if (ends.containsKey(id))
560 					sf.clipEnd = convertBytesToSecond(mLastFormat, begin + ends.get(id));
561 				else
562 					sf.clipEnd = convertBytesToSecond(mLastFormat, mOffsetInFile);
563 				mLinksOfCurrentFile.add(sf);
564 			}
565 			/*
566 			 * note: if marks.size() > 0 but all.size() == 0, it means that no
567 			 * marks refer to no ID. It should imply that the sentence contains
568 			 * skippable elements but no text. In such a case, it is important
569 			 * to let the script NOT add any fragment, not even the sentence's
570 			 * parent.
571 			 */
572 		}
573 
574 		if (mMemFootprint > maxQueueEltSize) {
575 			/*
576 			 * This flush prevents the TTS processors from raising too many
577 			 * out-of-memory errors and smoothes the transfers of PCM data to
578 			 * the encoders.
579 			 */
580 			flush(section, pcmOutput);
581 		}
582 		return true;
583 	}

585 	private void printMemError(Sentence sentence, MemoryException e) {
586 		String msg = "out of memory when processing sentence";
587 		ServerLogger.error(msg + " with @id=" + sentence.getID());
588 		mTTSLog.getWritableEntry(sentence.getID()).addError(
589 		        new TTSLog.Error(ErrorCode.AUDIO_MISSING, msg));
590 	}

592 	private void addBuffers(Iterable<AudioBuffer> toadd) throws InterruptedException {
593 		for (AudioBuffer b : toadd) {
594 			mOffsetInFile += b.size;
595 			mMemFootprint += mAudioBufferTracker.getFootPrint(b);
596 		}
597 		mBuffersOfCurrentFile = Iterables.concat(mBuffersOfCurrentFile, toadd);
598 	}

600 	private static double convertBytesToSecond(AudioFormat format, int bytes) {
601 		return (bytes / (format.getFrameRate() * format.getFrameSize()));
602 	}
