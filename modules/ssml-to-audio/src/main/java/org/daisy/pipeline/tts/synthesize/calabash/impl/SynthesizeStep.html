60  	private static String convertSecondToString(double seconds) {
61  		int iseconds = (int) (Math.floor(seconds));
62  		int milliseconds = (int) (Math.floor(1000 * (seconds - iseconds)));
63  		return String.format("%d:%02d:%02d.%03d", iseconds / 3600, (iseconds / 60) % 60,
64  		        (iseconds % 60), milliseconds);
65  	}

67  	public static XdmNode getFirstChild(XdmNode node) {
68  		XdmSequenceIterator iter = node.axisIterator(Axis.CHILD);
69  		if (iter.hasNext()) {
70  			return (XdmNode) iter.next();
71  		} else {
72  			return null;
73  		}
74  	}

134 	public void traverse(XdmNode node, SSMLtoAudio pool) throws SynthesisException {
135 		if (SentenceTag.equals(node.getNodeName())) {
136 			if (!pool.dispatchSSML(node))
137 				mErrorCounter++;
138 			if (++mSentenceCounter % 10 == 0)
139 				pool.endSection();
140 		} else {
141 			XdmSequenceIterator iter = node.axisIterator(Axis.CHILD);
142 			while (iter.hasNext()) {
143 				traverse((XdmNode) iter.next(), pool);
144 			}
145 		}
146 	}

148 	public void run() throws SaxonApiException {
149 		super.run();
150 
151 		try {
152 			mStartSemaphore.acquire();
153 		} catch (InterruptedException e) {
154 			mRuntime.error(e);
155 			return;
156 		}
157 
158 		VoiceConfigExtension configExt = new VoiceConfigExtension();
159 		ConfigReader cr = new ConfigReader(mRuntime.getProcessor(), config.read(), configExt);
160 
161 		String logEnabledProp = cr.getDynamicProperties().get("org.daisy.pipeline.tts.log");
162 		if (logEnabledProp == null)
163 			logEnabledProp = cr.getStaticProperties().get("org.daisy.pipeline.tts.log");
164 		boolean logEnabled = "true".equalsIgnoreCase(logEnabledProp) && mOutputDirOpt != null
165 		        && !mOutputDirOpt.isEmpty();
166 		TTSLog log;
167 		if (logEnabled) {
168 			log = new TTSLogImpl();
169 		} else
170 			log = new TTSLogEmpty();
171 		File audioOutputDir; {
172 			if (mTempDirOpt != null && !mTempDirOpt.isEmpty()) {
173 				try {
174 					audioOutputDir = new File(new URI(mTempDirOpt));
175 				} catch (URISyntaxException e) {
176 					throw new RuntimeException("temp-dir option invalid: " + mTempDirOpt);
177 				}
178 				if (audioOutputDir.exists())
179 					throw new RuntimeException("temp-dir option must be a non-existing directory: "+mTempDirOpt);
180 			} else {
181 				String tmpDir = cr.getAllProperties().get("org.daisy.pipeline.tts.audio.tmpdir");
182 				if (tmpDir == null)
183 					tmpDir = System.getProperty("java.io.tmpdir");
184 				do {
185 					String audioDir = tmpDir + "/";
186 					for (int k = 0; k < 2; ++k)
187 						audioDir += Long.toString(mRandGenerator.nextLong(), 32);
188 					audioOutputDir = new File(audioDir);
189 				} while (audioOutputDir.exists());
190 			}
191 		}
192 		audioOutputDir.mkdirs();
193 		audioOutputDir.deleteOnExit();
194 
195 		SSMLtoAudio ssmltoaudio = new SSMLtoAudio(audioOutputDir, mTTSRegistry, this,
196 		        mAudioBufferTracker, mRuntime.getProcessor(), mURIresolver, configExt, log);
197 
198 		Iterable<SoundFileLink> soundFragments = Collections.EMPTY_LIST;
199 		mErrorCounter = 0;
200 		mSentenceCounter = 0;
201 		try {
202 			while (source.moreDocuments()) {
203 				traverse(getFirstChild(source.read()), ssmltoaudio);
204 				ssmltoaudio.endSection();
205 			}
206 			Iterable<SoundFileLink> newfrags = ssmltoaudio.blockingRun(mAudioServices);
207 			mErrorCounter += ssmltoaudio.getErrorCount();
208 			soundFragments = Iterables.concat(soundFragments, newfrags);
209 		} catch (SynthesisException e) {
210 			mRuntime.error(e);
211 			return;
212 		} catch (EncodingException e) {
213 			throw new XProcException(ENCODING_ERROR, step.getNode(), e, e.getMessage());
214 		} catch (InterruptedException e) {
215 			mRuntime.error(e);
216 			return;
217 		} finally {
218 			mStartSemaphore.release();
219 		}
220 
221 		TreeWriter tw = new TreeWriter(runtime);
222 		tw.startDocument(runtime.getStaticBaseURI());
223 		tw.addStartElement(OutputRootTag);
224 
225 		int num = 0;
226 		for (SoundFileLink sf : soundFragments) {
227 			String soundFileURI = sf.soundFileURIHolder.toString();
228 			if (!soundFileURI.isEmpty()) {
229 				if (sf.clipBegin < sf.clipEnd){
230 					//sf.clipBegin = sf.clipEnd if the input text is empty. Those clips are not useful
231 					//and they can eventually lead to validation errors
232 					tw.addStartElement(ClipTag);
233 					tw.addAttribute(Audio_attr_id, sf.xmlid);
234 					tw.addAttribute(Audio_attr_clipBegin, convertSecondToString(sf.clipBegin));
235 					tw.addAttribute(Audio_attr_clipEnd, convertSecondToString(sf.clipEnd));
236 					tw.addAttribute(Audio_attr_src, soundFileURI);
237 					tw.addEndElement();
238 				}
239 				++num;
240 				TTSLog.Entry entry = log.getOrCreateEntry(sf.xmlid);
241 				entry.setSoundfile(soundFileURI);
242 				entry.setPositionInFile(sf.clipBegin, sf.clipEnd);
243 			} else {
244 				log.getOrCreateEntry(sf.xmlid).addError(
245 				        new TTSLog.Error(ErrorCode.AUDIO_MISSING,
246 				                "not synthesized or not encoded"));
247 			}
248 		}
249 		tw.addEndElement();
250 		tw.endDocument();
251 		result.write(tw.getResult());
252 
253 		printInfo("number of synthesized sound fragments: " + num);
254 		printInfo("audio encoding unreleased bytes : "
255 		        + mAudioBufferTracker.getUnreleasedEncondingMem());
256 		printInfo("TTS unreleased bytes: " + mAudioBufferTracker.getUnreleasedTTSMem());
257 
258 		/*
259 		 * Write status document
260 		 */
261 
262 		tw = new TreeWriter(runtime);
263 		tw.startDocument(runtime.getStaticBaseURI());
264 		tw.addStartElement(StatusRootTag);
265 		if (mErrorCounter == 0)
266 			tw.addAttribute(Status_attr_result, "ok");
267 		else {
268 			tw.addAttribute(Status_attr_result, "error");
269 			tw.addAttribute(Status_attr_success_rate,
270 			                (int)Math.floor(100 * (1 - (double)mErrorCounter/mSentenceCounter)) + "%");
271 		}
272 		tw.addEndElement();
273 		tw.endDocument();
274 		status.write(tw.getResult());
275 
276 		/*
277 		 * Write the log file
278 		 */
279 		if (logEnabled) {
280 			printInfo("writing TTS logs...");
281 			TreeWriter xmlLog = new TreeWriter(runtime);
282 			xmlLog.startDocument(runtime.getStaticBaseURI());
283 			xmlLog.addStartElement(LogRootTag);
284 			xmlLog.startContent();
285 			for (TTSLog.Error err : log.readonlyGeneralErrors()) {
286 				writeXMLerror(xmlLog, err);
287 			}
288 			for (Map.Entry<String, TTSLog.Entry> entry : log.getEntries()) {
289 				TTSLog.Entry le = entry.getValue();
290 				xmlLog.addStartElement(LogTextTag);
291 
292 				xmlLog.addAttribute(Log_attr_id, entry.getKey());
293 				if (le.getSoundFile() != null) {
294 					String basename = new File(le.getSoundFile()).getName();
295 					xmlLog.addAttribute(Log_attr_file, basename);
296 					xmlLog.addAttribute(Log_attr_begin, String.valueOf(le.getBeginInFile()));
297 					xmlLog.addAttribute(Log_attr_end, String.valueOf(le.getEndInFile()));
298 				}
299 
300 				xmlLog.addAttribute(Log_attr_timeout, "" + le.getTimeout() + "s");
301 
302 				if (le.getSelectedVoice() != null)
303 					xmlLog.addAttribute(Log_attr_selected_voice, le.getSelectedVoice()
304 					        .toString());
305 				if (le.getActualVoice() != null)
306 					xmlLog.addAttribute(Log_attr_actual_voice, le.getActualVoice().toString());
307 
308 				for (TTSLog.Error err : le.getReadOnlyErrors())
309 					writeXMLerror(xmlLog, err);
310 
311 				if (le.getSSML() != null) {
312 					xmlLog.addStartElement(LogSsmlTag);
313 					xmlLog.addSubtree(le.getSSML());
314 					xmlLog.addEndElement();
315 				}else{
316 					xmlLog.addText("No SSML available. This piece of text has probably been extracted from inside another sentence.");
317 				}
318 
319 				if (le.getTTSinput() != null && !le.getTTSinput().isEmpty()) {
320 					for (String inp : le.getTTSinput()) {
321 						xmlLog.addStartElement(LogInpTag);
322 						xmlLog.addText(inp);
323 						xmlLog.addEndElement();
324 					}
325 				}
326 
327 				xmlLog.addEndElement(); //LogTextTag
328 			}
329 
330 			xmlLog.addEndElement(); //root
331 			xmlLog.endDocument();
332 			String content = xmlLog.getResult().toString();
333 
334 			FileWriter fw = null;
335 
336 			File outputdir;
337 			try {
338 				outputdir = new File(new URI(mOutputDirOpt).getPath());
339 				if (!outputdir.exists()) {
340 					outputdir.mkdirs();
341 				}
342 				File output = new File(outputdir, "tts-log.xml");
343 				try {
344 
345 					fw = new FileWriter(output);
346 				} catch (IOException e) {
347 					printInfo("Cannot open log file " + output.getAbsolutePath());
348 				}
349 				if (fw != null) {
350 					try {
351 						fw.write(content);
352 					} catch (IOException e) {
353 						printInfo("Cannot write in log file " + output.getAbsolutePath());
354 					}
355 					try {
356 						fw.close();
357 					} catch (IOException e) {
358 						//ignore
359 					}
360 				}
361 
362 			} catch (URISyntaxException e1) {
363 				// TODO Auto-generated catch block
364 				e1.printStackTrace();
365 			}
366 
367 		}
368 	}

370 	private static void writeXMLerror(TreeWriter tw, TTSLog.Error err) {
371 		tw.addStartElement(LogErrorTag);
372 		tw.addAttribute(Log_attr_code, err.getErrorCode().toString());
373 		tw.addText(err.getMessage());
374 		tw.addEndElement();
375 	}
