59  /**
60   * SSMLtoAudio splits the input SSML sentences into sections called
61   * ContiguousText. Every section contains a list of contiguous sentences that
62   * may be eventually stored into different audio files, but it is guaranteed
63   * that such audio files won't contain sentences of other sections in such a
64   * manner that the list of audio files will mirror the document order. Within
65   * a section, every sentence has the same single sample rate to prevent us from
66   * resampling the audio data before sending them to encoders.
67   * 
68   * Once all the sentences have been assigned to TTS voices, they are sorted by
69   * size and stored in a shared queue of ContiguousText (actually, only the
70   * smallest sentences are sorted that way). SSMLtoAudio creates threads to
71   * consume this queue.
72   * 
73   * The TextToPcmThreads send PCM data to EncodingThreads via a queue of
74   * ContiguousPCM. These PCM packets are then processed from the longest (with
75   * respect to the number of samples) to the shortest in order to make it likely
76   * that the threads will finish at the same time. When all the TextToPcmThreads
77   * are joined, the pipeline pushes an EndOfQueue marker to every EncodingThreads
78   * to notify that they must stop waiting for more PCM packets than the ones
79   * already pushed.
80   * 
81   * The queue of PCM chunks, along with the queues of other TTS steps, share a
82   * global max size so as to make sure that they won't grow too much if the
83   * encoding is slower than the synthesizing (see AudioBuffersTracker).
84   * 
85   * If an EncodingThread fails to encode samples, it won't set the URI attribute
86   * of the audio chunks. In that way, SSMLtoAudio is informed not to include the
87   * corresponding text into the list of audio clips.
88   * 
89   */

191 	private TTSEngine createAndTestEngine(final TTSService service,
192 	        Map<String, String> properties, XdmNode testingXML,
193 	        CompiledStylesheet ssmlTransformer, TTSTimeout timeout) {
194 
195 		//create the engine
196 		TTSEngine engine = null;
197 		try {
198 			timeout.enableForCurrentThread(2);
199 			engine = service.newEngine(properties);
200 		} catch (Throwable e) {
201 			String err = TTSServiceUtil.displayName(service)
202 			        + " could not be initialized, cause: " + e.getMessage() + ": "
203 			        + getStack(e);
204 			mTTSlog.addGeneralError(ErrorCode.WARNING, err);
205 			ServerLogger.error(err);
206 			return null;
207 		} finally {
208 			timeout.disable();
209 		}
210 
211 		//transform the SSML with the custom SSML adapter
212 		String ttsInput = null;
213 		try {
214 			Map<String, Object> params = new TreeMap<String, Object>();
215 			if (engine.endingMark() != null)
216 				params.put("ending-mark", engine.endingMark());
217 			ttsInput = ssmlTransformer.newTransformer().transformToString(testingXML, params);
218 		} catch (SaxonApiException e) {
219 			String err = "error while using the SSML adapter of "
220 			        + TTSServiceUtil.displayName(service) + " on " + testingXML.toString();
221 			ServerLogger.error(err);
222 			mTTSlog.addGeneralError(ErrorCode.WARNING, err);
223 			return null;
224 		}
225 
226 		//get a voice supporting SSML marks (so far as they are supported by the engine)
227 		Voice firstVoice = null;
228 		int timeoutSecs = 5;
229 		try {
230 			timeout.enableForCurrentThread(timeoutSecs);
231 			for (Voice v : engine.getAvailableVoices()) {
232 				if (engine.endingMark() == null
233 				        || v.getMarkSupport() != MarkSupport.MARK_NOT_SUPPORTED) {
234 					firstVoice = v;
235 					break;
236 				}
237 			}
238 			if (firstVoice == null) {
239 				String err = TTSServiceUtil.displayName(service)
240 				        + " cannot be tested because no voice seems available.";
241 				mTTSlog.addGeneralError(ErrorCode.WARNING, err);
242 				ServerLogger.error(err);
243 				return null;
244 			}
245 		} catch (InterruptedException e) {
246 			String err = "Timeout while retrieving the voices of "
247 			        + TTSServiceUtil.displayName(service) + " (exceeded " + timeoutSecs
248 			        + " seconds)\n" + getStack(e);
249 			mTTSlog.addGeneralError(ErrorCode.WARNING, err);
250 			ServerLogger.error(err);
251 			return null;
252 		} catch (Exception e) {
253 			String err = TTSServiceUtil.displayName(service)
254 			        + " failed to return voices, cause: " + e.getMessage() + ": "
255 			        + getStack(e);
256 			mTTSlog.addGeneralError(ErrorCode.WARNING, err);
257 			ServerLogger.error(err);
258 			return null;
259 		} finally {
260 			timeout.disable();
261 		}
262 
263 		//allocate resources for testing purpose
264 		final TTSEngine fengine = engine;
265 		TTSResource resource = null;
266 		try {
267 			timeout.enableForCurrentThread(2);
268 			resource = engine.allocateThreadResources();
269 		} catch (Exception e) {
270 			String err = "Could not initialize resource for "
271 			        + TTSServiceUtil.displayName(service) + ", cause: " + e.getMessage()
272 			        + ": " + getStack(e);
273 			mTTSlog.addGeneralError(ErrorCode.WARNING, err);
274 			ServerLogger.error(err);
275 			return null;
276 		} finally {
277 			timeout.disable();
278 		}
279 
280 		//create a custom interrupter in case the engine hangs
281 		final TTSResource res = resource;
282 		TTSTimeout.ThreadFreeInterrupter interrupter = new ThreadFreeInterrupter() {
283 			@Override
284 			public void threadFreeInterrupt() {
285 				String msg = "Timeout while initializing "
286 				        + TTSServiceUtil.displayName(service)
287 				        + ". Forcing interruption of the current work of "
288 				        + TTSServiceUtil.displayName(service) + "... ";
289 				ServerLogger.warn(msg);
290 				mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
291 				fengine.interruptCurrentWork(res);
292 			}
293 		};
294 
295 		//run the text-to-speech on the testing input
296 		Collection<AudioBuffer> audioBuffers = null;
297 		List<TTSService.Mark> marks = new ArrayList<TTSService.Mark>();
298 		List<String> expectedMarks = new ArrayList<String>();
299 		if (engine.endingMark() != null)
300 			expectedMarks.add(engine.endingMark());
301 		try {
302 			audioBuffers = mExecutor.synthesizeWithTimeout(
303 				timeout, interrupter, null, ttsInput, testingXML, Sentence.computeSize(testingXML),
304 				engine, firstVoice, res, marks, expectedMarks, new StraightBufferAllocator(), false);
305 		} catch (Exception e) {
306 			String msg = "Error while testing " + TTSServiceUtil.displayName(service) + "; "
307 			        + e.getMessage() + ": " + getStack(e);
308 			ServerLogger.warn(msg);
309 			mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
310 			return null;
311 		} finally {
312 			if (res != null)
313 				try {
314 					timeout.enableForCurrentThread(2);
315 					engine.releaseThreadResources(res);
316 				} catch (Exception e) {
317 					String msg = "Error while releasing resource of "
318 					        + TTSServiceUtil.displayName(service) + "; " + e.getMessage()
319 					        + ": " + getStack(e);
320 					ServerLogger.warn(msg);
321 					mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
322 				} finally {
323 					timeout.disable();
324 				}
325 		}
326 
327 		//check that the output buffer is big enough
328 		String msg = "";
329 		int size = 0;
330 		for (AudioBuffer buff : audioBuffers)
331 			size += buff.size;
332 		if (size < 2500) {
333 			msg = "Audio output is not big enough. ";
334 		}
335 
336 		//check the ending mark
337 		if (engine.endingMark() != null) {
338 			String details = " input: "+ttsInput+", voice: "+firstVoice;
339 			if (marks.size() != 1) {
340 				msg += "One bookmark events expected, but received " + marks.size() + " events instead. "+details;
341 			} else {
342 				Mark mark = marks.get(0);
343 				if (!engine.endingMark().equals(mark.name)) {
344 					msg += "Expecting ending mark " + engine.endingMark() + ", got "
345 					        + mark.name + " instead. "+details;
346 				}
347 				if (mark.offsetInAudio < 2500) {
348 					msg += "Expecting ending mark offset to be bigger, got "
349 					        + mark.offsetInAudio + " as offset. "+details;
350 				}
351 			}
352 		}
353 		if (!msg.isEmpty()) {
354 			msg = "Errors found while testing, " + TTSServiceUtil.displayName(service) + ": "
355 			        + msg;
356 			ServerLogger.warn(msg);
357 			mTTSlog.addGeneralError(ErrorCode.WARNING, msg);
358 			return null;
359 		}
360 
361 		return engine;
362 	}

364 	/**
365 	 * The SSML is assumed to be pushed in document order.
366 	 *
367 	 * @param ssml The input SSML
368 	 * @return true when the SSML was successfully converted to speech, false when there was an error
369 	 **/
370 	public boolean dispatchSSML(XdmNode ssml) throws SynthesisException {
371 		String voiceEngine = ssml.getAttributeValue(Sentence_attr_select1);
372 		String voiceName = ssml.getAttributeValue(Sentence_attr_select2);
373 		String gender = ssml.getAttributeValue(Sentence_attr_gender);
374 		String age = ssml.getAttributeValue(Sentence_attr_age);
375 		String id = ssml.getAttributeValue(Sentence_attr_id);
376 		String lang = ssml.getAttributeValue(Sentence_attr_lang);
377 
378 		TTSLog.Entry logEntry = mTTSlog.getOrCreateEntry(id);
379 		logEntry.setSSML(ssml);
380 
381 		if (age != null) {
382 			try {
383 				int age_i = Integer.parseInt(age);
384 				if (age_i <= 16) {
385 					gender += "-child";
386 				} else if (age_i >= 70) {
387 					gender += "-eldery";
388 				}
389 			} catch (NumberFormatException e) {
390 				//ignore
391 			}
392 		}
393 
394 		boolean[] exactMatch = new boolean[1];
395 		Voice voice = mVoiceManager.findAvailableVoice(voiceEngine, voiceName, lang, gender,
396 		        exactMatch);
397 		logEntry.setSelectedVoice(voice);
398 		if (voice == null) {
399 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING,
400 			        "could not find any installed voice matching with "
401 			                + new Voice(voiceEngine, voiceName)
402 			                + " or providing the language '" + lang + "'"));
403 			endSection();
404 			return false;
405 		}
406 
407 		TTSEngine newSynth = mVoiceManager.getTTS(voice);
408 		if (newSynth == null) {
409 			/*
410 			 * Should not happen since findAvailableVoice() returns only a
411 			 * non-null voice if a TTSService can provide it
412 			 */
413 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.AUDIO_MISSING,
414 			        "could not find any TTS engine for the voice "
415 			                + new Voice(voiceEngine, voiceName)));
416 			endSection();
417 			return false;
418 		}
419 
420 		if (!exactMatch[0]) {
421 			logEntry.addError(new TTSLog.Error(TTSLog.ErrorCode.UNEXPECTED_VOICE,
422 			        "no voice matches exactly with the requested characteristics"));
423 		}
424 
425 		/*
426 		 * If a TTS engine has no reserved threads, its sentences are pushed to
427 		 * a global list whose key is null. Otherwise the TTS Service is used as
428 		 * key.
429 		 */
430 		TTSEngine poolkey = null;
431 		if (newSynth.reservedThreadNum() > 0) {
432 			poolkey = newSynth;
433 		}
434 
435 		if (newSynth != mLastTTS) {
436 			if (mLastTTS != null
437 			        && (poolkey != null || mLastTTS.reservedThreadNum() != 0
438 			                || mLastTTS.getAudioOutputFormat() == null
439 			                || newSynth.getAudioOutputFormat() == null || !mLastTTS
440 			                .getAudioOutputFormat().matches(newSynth.getAudioOutputFormat())))
441 				endSection(); // necessary because the same thread wouldn't be able to
442 				              // concatenate outputs of different formats
443 			mLastTTS = newSynth;
444 		}
445 
446 		if (mCurrentSection == null) {
447 			//happen the first time and whenever endSection() is called
448 			mCurrentSection = new ContiguousText(mDocumentPosition++, mAudioDir);
449 			List<ContiguousText> listOfSections = mOrganizedText.get(poolkey);
450 			if (listOfSections == null) {
451 				listOfSections = new ArrayList<ContiguousText>();
452 				mOrganizedText.put(poolkey, listOfSections);
453 			}
454 
455 			listOfSections.add(mCurrentSection);
456 		}
457 		mCurrentSection.sentences.add(new Sentence(newSynth, voice, ssml));
458 		return true;
459 	}

467 	public Iterable<SoundFileLink> blockingRun(AudioServices audioServices)
468 	        throws SynthesisException, InterruptedException, EncodingException {
469 
470 		progress = MessageAppender.getActiveBlock().append(new MessageBuilder().withProgress(BigDecimal.ONE));
471 		try {
472 
473 		//SSML mark splitter shared by the threads:
474 		SSMLMarkSplitter ssmlSplitter = new StructuredSSMLSplitter(mProc);
475 
476 		reorganizeSections();
477 		mProgress = 0;
478 		mPrintedProgress = 0;
479 
480 		//threading layout
481 		int reservedThreadNum = 0;
482 		for (TTSEngine tts : mOrganizedText.keySet()) {
483 			if (tts != null && tts.reservedThreadNum() > 0)
484 				reservedThreadNum += tts.reservedThreadNum();
485 		}
486 		int cores = Runtime.getRuntime().availableProcessors();
487 		int ttsThreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.number", cores);
488 		int encodingThreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.encoding.number",
489 		        ttsThreadNum);
490 		int regularTTSthreadNum = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.speaking.number",
491 		        ttsThreadNum);
492 		int totalTTSThreads = regularTTSthreadNum + reservedThreadNum;
493 		int maxMemPerTTSThread = convertToInt(mProperties, "org.daisy.pipeline.tts.threads.each.memlimit", 20)*1048576; //20MB
494 		mLogger.printInfo("Number of encoding threads: " + encodingThreadNum);
495 		mLogger.printInfo("Number of regular text-to-speech threads: " + regularTTSthreadNum);
496 		mLogger.printInfo("Number of reserved text-to-speech threads: " + reservedThreadNum);
497 		mLogger.printInfo("Max TTS memory footprint (encoding excluded): "
498 		        + mAudioBufferTracker.getSpaceForTTS() / 1000000 + "MB");
499 		mLogger.printInfo("Max encoding memory footprint: "
500 		        + mAudioBufferTracker.getSpaceForEncoding() / 1000000 + "MB");
501 
502 		//input queue common to all the threads
503 		BlockingQueue<ContiguousPCM> pcmQueue = new PriorityBlockingQueue<ContiguousPCM>();
504 
505 		//start the TTS threads
506 		TextToPcmThread[] tpt = new TextToPcmThread[totalTTSThreads];
507 		List<ContiguousText> text = mOrganizedText.get(null);
508 		if (text == null) {
509 			text = Collections.EMPTY_LIST;
510 		}
511 		ConcurrentLinkedQueue<ContiguousText> stext = new ConcurrentLinkedQueue<ContiguousText>(
512 		        text);
513 		int i = 0;
514 		for (; i < regularTTSthreadNum; ++i) {
515 			tpt[i] = new TextToPcmThread();
516 			tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter, this,
517 			             mLogger, mAudioBufferTracker, maxMemPerTTSThread, mSSMLtransformers, mTTSlog);
518 		}
519 		for (Map.Entry<TTSEngine, List<ContiguousText>> e : mOrganizedText.entrySet()) {
520 			TTSEngine tts = e.getKey();
521 			if (tts != null) { //tts = null is handled by the previous loop
522 				stext = new ConcurrentLinkedQueue<ContiguousText>(e.getValue());
523 				for (int j = 0; j < tts.reservedThreadNum(); ++i, ++j) {
524 					tpt[i] = new TextToPcmThread();
525 					tpt[i].start(stext, pcmQueue, mExecutor, mTTSRegistry, mVoiceManager, ssmlSplitter,
526 					        this, mLogger, mAudioBufferTracker, maxMemPerTTSThread, mSSMLtransformers,
527 					        mTTSlog);
528 				}
529 			}
530 		}
531 		mLogger.printInfo("Text-to-speech threads started.");
532 
533 		//start the encoding threads
534 		EncodingThread[] encodingTh = new EncodingThread[encodingThreadNum];
535 		for (int j = 0; j < encodingTh.length; ++j) {
536 			encodingTh[j] = new EncodingThread();
537 			encodingTh[j].start(audioServices, pcmQueue, mLogger, mAudioBufferTracker,
538 			        mProperties, mTTSlog);
539 		}
540 		mLogger.printInfo("Encoding threads started.");
541 
542 		//collect the sound fragments
543 		Collection<SoundFileLink>[] fragments = new Collection[tpt.length];
544 		for (int j = 0; j < tpt.length; ++j) {
545 			fragments[j] = tpt[j].getSoundFragments();
546 		}
547 		for (int j = 0; j < tpt.length; ++j) {
548 			mErrorCounter += tpt[j].getErrorCount();
549 		}
550 
551 		//send END notifications and wait for the encoding threads to finish
552 		mLogger.printInfo("Text-to-speech finished. Waiting for audio encoding to finish...");
553 		for (int k = 0; k < encodingTh.length; ++k) {
554 			pcmQueue.add(ContiguousPCM.EndOfQueue);
555 		}
556 		for (int j = 0; j < encodingTh.length; ++j) {
557 			try {
558 				encodingTh[j].waitToFinish();
559 			} catch (EncodingException e) {
560 				while (++j < encodingTh.length)
561 					// FIXME: interrupt instead of waiting
562 					try { encodingTh[j].waitToFinish(); }
563 					catch (EncodingException _e) {}
564 				throw e;
565 			}
566 		}
567 
568 		mLogger.printInfo("Audio encoding finished.");
569 
570 		return Iterables.concat(fragments);
571 
572 		} finally {
573 			progress.close();
574 			progress = null;
575 		}
576 	}

584 		progress.append(

599 	private void reorganizeSections() {
600 		mTotalTextSize = 0;
601 		int sectionCount = 0;
602 		for (List<ContiguousText> sections : mOrganizedText.values()) {
603 			//compute the sections' size: needed for displaying the progress,
604 			//splitting the sections and sorting them
605 			for (ContiguousText section : sections)
606 				section.computeSize();
607 
608 			for (ContiguousText section : sections) {
609 				mTotalTextSize += section.getStringSize();
610 			}
611 			//split up the sections that are too big
612 			int maxSize = (int) (mTotalTextSize / 15); //should it depend on the total size or be an absolute max?
613 			List<ContiguousText> newSections = new ArrayList<ContiguousText>();
614 			List<ContiguousText> toRemove = new ArrayList<ContiguousText>();
615 			for (ContiguousText section : sections) {
616 				if (section.getStringSize() >= maxSize) {
617 					toRemove.add(section);
618 					splitSection(section, maxSize, newSections);
619 				}
620 			}
621 
622 			sections.removeAll(toRemove);
623 			sections.addAll(newSections);
624 
625 			//sort the sections according to their size in descending-order
626 			Collections.sort(sections);
627 
628 			//keep sorted only the smallest sections (50% of total) so the biggest sections won't
629 			//necessarily be processed at the same time, as that may consume too much memory.
630 			Collections.shuffle(sections.subList(0, sections.size() / 2));
631 
632 			sectionCount += sections.size();
633 		}
634 		mLogger.printInfo("Number of synthesizable TTS sections: " + sectionCount);
635 	}

637 	//we can dispense with this function as soon as we take into consideration the size of the SSML
638 	//sentences, rather than creating a new section every 10 sentences or so.
639 	private void splitSection(ContiguousText section, int maxSize,
640 	        List<ContiguousText> newSections) {
641 		int left = 0;
642 		int count = 0;
643 		ContiguousText currentSection = new ContiguousText(section.getDocumentPosition(),
644 		        section.getAudioOutputDir());
645 		currentSection.setStringsize(0);
646 		for (int right = 0; right < section.sentences.size(); ++right) {
647 			if (currentSection.getStringSize() > maxSize) {
648 				currentSection.sentences = section.sentences.subList(left, right);
649 				currentSection.setDocumentSplitPosition(count);
650 				newSections.add(currentSection);
651 				currentSection = new ContiguousText(section.getDocumentPosition(), section
652 				        .getAudioOutputDir());
653 				currentSection.setStringsize(0);
654 				left = right;
655 				++count;
656 			}
657 			currentSection.setStringsize(currentSection.getStringSize()
658 			        + section.sentences.get(right).getSize());
659 		}
660 		currentSection.sentences = section.sentences.subList(left, section.sentences.size());
661 		currentSection.setDocumentSplitPosition(count);
662 		newSections.add(currentSection);
663 	}
