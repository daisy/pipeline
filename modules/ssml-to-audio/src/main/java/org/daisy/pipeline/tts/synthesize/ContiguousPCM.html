1   package org.daisy.pipeline.tts.synthesize;
2   
3   import java.io.File;
4   
5   import javax.sound.sampled.AudioFormat;
6   
7   import org.daisy.pipeline.audio.AudioBuffer;
8   
9   /**
10   * ContiguousPCMs are the message objects sent from TTS processors to the
11   * encoders. Every ContiguousPCM contains a list of audio buffers and attributes
12   * related to the URI of the sound file to produce.
13   */
14  class ContiguousPCM implements Comparable<ContiguousPCM> {
15  
16  	static ContiguousPCM EndOfQueue = new ContiguousPCM();
17  
18  	ContiguousPCM() {
19  		mEncodingTimeApprox = -1;
20  		mSizeInBytes = minByteSize();
21  	}
22  
23  	boolean isEndOfQueue() {
24  		return (mEncodingTimeApprox == -1);
25  	}
26  
27  	ContiguousPCM(AudioFormat audioformat, Iterable<AudioBuffer> buffers, File destdir,
28  	        String destFilePrefix) {
29  		mDestURI = new StringBuilder();
30  		mAudioFormat = audioformat;
31  		mBuffers = buffers;
32  		mDestDir = destdir;
33  		mDestFilePrefix = destFilePrefix;
34  		mEncodingTimeApprox = 0;
35  		for (AudioBuffer buffer : buffers)
36  			mEncodingTimeApprox += buffer.size;
37  		mEncodingTimeApprox /= mAudioFormat.getSampleSizeInBits();
38  		mSizeInBytes = minByteSize();
39  		for (AudioBuffer buffer : buffers)
40  			mSizeInBytes += buffer.data.length;
41  	}
42  
43  	@Override
44  	public int compareTo(ContiguousPCM o) {
45  		return (o.mEncodingTimeApprox - mEncodingTimeApprox); //descending order => biggest chunks will be polled first
46  	}
47  
48  	String getDestinationFilePrefix() {
49  		return mDestFilePrefix;
50  	}
51  
52  	File getDestinationDirectory() {
53  		return mDestDir;
54  	}
55  
56  	AudioFormat getAudioFormat() {
57  		return mAudioFormat;
58  	}
59  
60  	Iterable<AudioBuffer> getBuffers() {
61  		return mBuffers;
62  	}
63  
64  	StringBuilder getURIholder() {
65  		return mDestURI;
66  	}
67  
68  	public int sizeInBytes() {
69  		return mSizeInBytes;
70  	}
71  
72  	private int minByteSize() {
73  		return 500;//rough approximation of an empty ContiguousPCM's memory footprint
74  	}
75  
76  	private Iterable<AudioBuffer> mBuffers;
77  	private int mEncodingTimeApprox; //used for sorting
78  	private int mSizeInBytes; //used for monitoring the memory footprint
79  	private AudioFormat mAudioFormat;
80  	private File mDestDir;
81  	private String mDestFilePrefix;
82  	private StringBuilder mDestURI; //simple way to hold a string
83  
84  }
