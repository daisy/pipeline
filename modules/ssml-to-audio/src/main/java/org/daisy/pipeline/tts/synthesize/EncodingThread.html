1   package org.daisy.pipeline.tts.synthesize;
2   
3   import java.io.PrintWriter;
4   import java.io.StringWriter;
5   import java.util.Map;
6   import java.util.Optional;
7   import java.util.concurrent.BlockingQueue;
8   
9   import org.daisy.pipeline.audio.AudioEncoder;
10  import org.daisy.pipeline.audio.AudioServices;
11  import org.daisy.pipeline.tts.AudioBufferTracker;
12  import org.daisy.pipeline.tts.TTSTimeout;
13  import org.daisy.pipeline.tts.synthesize.TTSLog.ErrorCode;
14  
15  /**
16   * Consumes a shared queue of PCM packets. PCM packets are then provided to
17   * audio encoders. The thread stops when it receives an 'EndOfQueue' marker.
18   */
19  public class EncodingThread {
20  
21  	class EncodingException extends RuntimeException {
22  		public EncodingException(String message, Throwable cause) {
23  			super(message, cause);
24  		}
25  		public EncodingException(Throwable t) {
26  			super(t);
27  		}
28  		public EncodingException(String message) {
29  			super(message);
30  		}
31  	}
32  	
33  	private Thread mThread;
34  	private Throwable criticalError;
35  
36  	void start(final AudioServices encoderRegistry,
37  	        final BlockingQueue<ContiguousPCM> inputPCM, final IPipelineLogger logger,
38  	        final AudioBufferTracker audioBufferTracker, Map<String, String> TTSproperties,
39  	        final TTSLog ttslog) {
40  
41  		//max seconds of encoded audio per seconds of encoding
42  		//it would be more accurate with a byte rate instead, but less intuitive
43  		float encodingSpeed = 2.0f;
44  		String speedProp = "org.daisy.pipeline.tts.encoding.speed";
45  		String speedParam = TTSproperties.get(speedProp);
46  		if (speedParam != null) {
47  			try {
48  				encodingSpeed = Float.valueOf(speedParam);
49  			} catch (NumberFormatException e) {
50  				String msg = "wrong format for property " + speedProp
51  				        + ". A float is expected, not " + speedParam;
52  				logger.printInfo(msg);
53  				ttslog.addGeneralError(ErrorCode.WARNING, msg);
54  			}
55  		}
56  
57  		//Eventually, we should select the encoder using the audio format as criterion, but for now
58  		//we always employ the same encoder for every chunk of PCM
59  		AudioEncoder encoder = encoderRegistry.getEncoder();
60  		AudioEncoder.EncodingOptions encodingOptions = null;
61  		if (encoder == null) {
62  			String msg = "No audio encoder found";
63  			logger.printInfo(msg);
64  			ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
65  		} else {
66  			encodingOptions = encoder.parseEncodingOptions(TTSproperties);
67  			try {
68  				encoder.test(encodingOptions);
69  			} catch (Exception e) {
70  				String msg = "audio encoder does not work: " + getStack(e);
71  				logger.printInfo(msg);
72  				ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
73  				encoder = null;
74  			}
75  		}
76  
77  		final AudioEncoder.EncodingOptions options = encodingOptions;
78  		final AudioEncoder fencoder = encoder;
79  		final float fEncodingSpeed = encodingSpeed;
80  		final TTSTimeout timeout = new TTSTimeout();
81  
82  		mThread = new Thread() {
83  			@Override
84  			public void run() {
85  				try {
86  					while (!interrupted()) {
87  						ContiguousPCM job;
88  						try {
89  							job = inputPCM.take();
90  						} catch (InterruptedException e) {
91  							String msg = "encoding thread has been interrupted";
92  							logger.printInfo(msg);
93  							ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
94  							break; //warning: encoding bytes are not freed
95  						}
96  						if (job.isEndOfQueue()) {
97  							//nothing to release
98  							break;
99  						}
100 						int jobSize = job.sizeInBytes();
101 						if (fencoder != null) {
102 							float secs = jobSize / (job.getAudioFormat().getFrameRate());
103 							int maxTime = (int) (1.0 + secs / fEncodingSpeed);
104 							try {
105 								timeout.enableForCurrentThread(maxTime);
106 								Optional<String> destURI = fencoder.encode(job.getBuffers(), job
107 								                                           .getAudioFormat(), job.getDestinationDirectory(), job
108 								                                           .getDestinationFilePrefix(), options);
109 								if (destURI.isPresent()) {
110 									job.getURIholder().append(destURI.get());
111 								} else {
112 									String msg = "Audio encoder failed to encode to "
113 										+ job.getDestinationFilePrefix();
114 									ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
115 								}
116 							} catch (InterruptedException e) {
117 								String msg = "timeout while encoding audio to "
118 									+ job.getDestinationFilePrefix() + ": " + getStack(e);
119 								ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
120 								audioBufferTracker.releaseEncodersMemory(jobSize);
121 								throw new EncodingException(e);
122 							} catch (Throwable t) {
123 								String msg = "error while encoding audio to "
124 									+ job.getDestinationFilePrefix() + ": " + getStack(t);
125 								ttslog.addGeneralError(ErrorCode.CRITICAL_ERROR, msg);
126 								audioBufferTracker.releaseEncodersMemory(jobSize);
127 								throw new EncodingException(t);
128 							} finally {
129 								timeout.disable();
130 							}
131 						}
132 						audioBufferTracker.releaseEncodersMemory(jobSize);
133 					}
134 				} finally {
135 					timeout.close();
136 				}
137 			}
138 		};
139 		mThread.setUncaughtExceptionHandler(
140 			(thread, throwable) -> { criticalError = throwable; }
141 		);
142 		mThread.start();
143 	}
144 
145 	void waitToFinish() throws EncodingException {
146 		if (criticalError != null) {
147 			if (criticalError instanceof EncodingException)
148 				throw (EncodingException)criticalError;
149 			else
150 				throw new RuntimeException("coding error");
151 		}
152 		try {
153 			mThread.join();
154 		} catch (InterruptedException e) {
155 			throw new RuntimeException(); // should not happen
156 		}
157 	}
158 
159 	//TODO: move this method in some kind of utils/helpers
160 	private static String getStack(Throwable t) {
161 		StringWriter writer = new StringWriter();
162 		PrintWriter printWriter = new PrintWriter(writer);
163 		t.printStackTrace(printWriter);
164 		printWriter.flush();
165 		return writer.toString();
166 	}
167 
168 }
