1   package org.daisy.pipeline.common.calabash.impl;
2   
3   import net.sf.saxon.s9api.QName;
4   import net.sf.saxon.s9api.SaxonApiException;
5   import org.daisy.common.xproc.calabash.XProcStepProvider;
6   
7   import com.xmlcalabash.core.XProcRuntime;
8   import com.xmlcalabash.core.XProcStep;
9   import com.xmlcalabash.runtime.XAtomicStep;
10  
11  import org.osgi.service.component.annotations.Component;
12  
13  @Component(
14  	name = "px:log-message",
15  	service = { XProcStepProvider.class },
16  	property = { "type:String={http://www.daisy.org/ns/pipeline/xproc}log-message" }
17  )
18  public class Message implements XProcStepProvider {
19  
20  	public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
21  		return new MessageStep(runtime, step);
22  	}
23  
24  	public static class MessageStep extends com.xmlcalabash.library.Identity {
25  
26  		private static final QName _message = new QName("message");
27  		private static final QName _severity = new QName("severity");
28  		
29  		private MessageStep(XProcRuntime runtime, XAtomicStep step) {
30  			super(runtime, step);
31  		}
32  
33  		@Override
34  		public void run() throws SaxonApiException {
35  			String severity = getOption(_severity, "INFO");
36  			String message = getOption(_message, (String)null);
37  			if ("WARN".equals(severity)) {
38  				warning(this.step.getNode(), message);
39  			} else if ("DEBUG".equals(severity)) {
40  				// DefaultStep#fine() removed in ndw/xmlcalabash1@ce9b07d
41  				this.runtime.getMessageListener().fine(this,this.step.getNode(), message);
42  			} else {
43  				info(this.step.getNode(), message);
44  			}
45  			
46  			super.run();
47  		}
48  	}
49  }
