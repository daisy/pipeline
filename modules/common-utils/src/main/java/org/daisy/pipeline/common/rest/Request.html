24  /**
25   * REST request to communicate with remote services (such as TTS services like Google and Amazon).
26   *
27   * @author Nicolas Pavie @ braillenet.org
28   */
29  public class Request {
30  
31  	private static List<String> httpMethods =
32  		Arrays.asList(
33  			"GET","POST","PUT","DELETE","HEAD",
34  			"CONNECT","OPTIONS","TRACE","PATCH");
35  
36  	private Map<String,String> headers = new HashMap<String,String>();
37  	private String contentType = null;
38  	private String content = null;
39  	private ByteArrayOutputStream formData = null;
40  	private String multipartBoundary = null;
41  	private URL requestURL;
42  	private String method = "GET";
43  	private HttpURLConnection connection;

45  	/**
46  	 * Fully initialized request
47  	 *
48  	 * @param httpMethod possible values: "GET", "POST", "PUT", "DELETE", "HEAD", "CONNECT", "OPTIONS", "TRACE", "PATCH"
49  	 * @param url a string of the complete request URL (including URL parameters like "?voice=smtg)"
50  	 * @param headers HTTP headers of the request
51  	 *                <p>Use {@code null} to unset all headers.</p>
52  	 * @param content content to send with the request (often associated with POST requests)
53  	 *                <p>Use {@code null} for requests without content</p>
54  	 * @throws Exception if the http method is not one of the list above
55  	 * @throws MalformedURLException if the url is not valid
56  	 */
57  	public Request(String httpMethod, String url, Map<String,String> headers, String content)
58  			throws IllegalArgumentException, MalformedURLException {
59  		this.setMethod(httpMethod);
60  		this.setRequestUrl(url);
61  		this.headers = headers;
62  		this.content = content;
63  	}
64  
65  	public Request(String url) throws MalformedURLException {
66  		setRequestUrl(url);
67  	}
68  
69  	public Request(URL url) {
70  		setRequestUrl(url);
71  	}

73  	/**
74  	 * Add a new header field with specified value to the request
75  	 *
76  	 * @param name name of the header field
77  	 * @param value value set for the field
78  	 */
79  	public Request addHeader(String name, String value) {
80  		if ("Content-Type".equals(name)) {
81  			if (contentType != null && !contentType.equals(value))
82  				throw new IllegalStateException("content type already set to: " + contentType);
83  			else
84  				contentType = value;
85  		} else
86  			headers.put(name, value);
87  		return this;
88  	}
89  
90  	public Request setContentType(String contentType) {
91  		return addHeader("Content-Type", contentType);
92  	}
93  
94  	/**
95  	 * Set the request body, to be sent through the request connection output stream
96  	 *
97  	 * @param content the content
98  	 */
99  	public Request setContent(String content) {
100 		if (this.content != null || formData != null)
101 			throw new IllegalStateException("content already set");
102 		else
103 			this.content = content;
104 		return this;
105 	}
106 
107 	/**
108 	 * Append form data to the multipart body of the request
109 	 *
110 	 * @param name         the name of the subpart
111 	 * @param fileName     the optional file name associated with the subpart
112 	 * @param contentType  the optional mimetype associated with the subpart
113 	 * @param content      the content
114 	 */
115 	public Request setFormDataContent(String name, String fileName, String contentType, InputStream content)
116 			throws IOException {
117 		if (this.content != null)
118 			throw new IllegalStateException("content already set");
119 		if (formData == null) {
120 			formData = new ByteArrayOutputStream();
121 			multipartBoundary = "Boundary-" + new SecureRandom().nextLong();
122 			setContentType("multipart/form-data; boundary=" + multipartBoundary);
123 		}
124 		writeUTF8(formData, String.format("--%s\r\nContent-Disposition: form-data; name=\"%s\"", multipartBoundary, name));
125 		if (fileName != null)
126 			writeUTF8(formData, String.format("; filename=\"%s\"", fileName));
127 		writeUTF8(formData, "\r\n");
128 		if (contentType != null)
129 			writeUTF8(formData, String.format("Content-Type: %s\r\n", contentType));
130 		writeUTF8(formData, "\r\n");
131 		transferTo(content, formData);
132 		writeUTF8(formData, "\r\n");
133 		return this;
134 	}
135 
136 	public Request setFormDataContent(String name, String content) throws IOException {
137 		return setFormDataContent(name, null, content);
138 	}
139 
140 	public Request setFormDataContent(String name, String contentType, String content) throws IOException {
141 		return setFormDataContent(name, null, contentType, new ByteArrayInputStream(content.getBytes("utf-8")));
142 	}
143 
144 	public Request setFormDataContent(String name, String contentType, File content) throws IOException {
145 		return setFormDataContent(name, content.getName(), contentType, new FileInputStream(content));
146 	}
147 
148 	public Request setFormDataContent(String name, Resource content) throws IOException {
149 		return setFormDataContent(name,
150 		                          content.getPath().toASCIIString(),
151 		                          content.getMediaType().orElse(null),
152 		                          content.read());
153 	}

155 	/**
156 	 * Set the request URL.
157 	 *
158 	 * <p>If a previous connection was opened for the current request, close and destroy it.</p>
159 	 */
160 	public Request setRequestUrl(String url) throws MalformedURLException {
161 		return setRequestUrl(new URL(url));
162 	}
163 
164 	public Request setRequestUrl(URL url) {
165 		requestURL = url;
166 		if (this.connection != null) {
167 			this.connection.disconnect();
168 			this.connection = null;
169 		}
170 		return this;
171 	}
172 
173 	/**
174 	 * Set the HTTP method to use with the request
175 	 *
176 	 * @param httpMethod one of the following values: "GET", "POST", "PUT", "DELETE", "HEAD", "CONNECT", "OPTIONS", "TRACE", "PATCH"
177 	 * @throws IllegalArgumentException if the HTTP method value is not a valid one
178 	 */
179 	public Request setMethod(String httpMethod) {
180 		if (!httpMethods.contains(httpMethod.toUpperCase())) {
181 			throw new IllegalArgumentException(httpMethod + " is not a valid HTTP method (valid methods : " + httpMethods.toString() + ")");
182 		}
183 		method = httpMethod.toUpperCase();
184 		return this;
185 	}
186 
187 	/**
188 	 * Send the request to the requested URL and retrieve the server answer
189 	 *
190 	 * @return the {@link Response} object
191 	 * @throws InterruptedException if the current thread is interrupted while getting the response from the server
192 	 * @throws IOException if the response code could not be retrieved, or the body could not be read
193 	 */
194 	public Response send() throws InterruptedException, IOException {
195 		int status = 0;
196 		InputStream body = null;
197 		InputStream error = null;
198 		IOException ioe = null;
199 		try {
200 			connection = (HttpURLConnection)requestURL.openConnection();
201 			connection.setRequestMethod(method);
202 			if (contentType != null)
203 				connection.setRequestProperty("Content-Type", contentType);
204 			if (headers != null)
205 				headers.forEach((String key, String value) -> {
206 						connection.setRequestProperty(key, value);
207 					});
208 			if (content != null || formData != null) {
209 				connection.setDoOutput(true);
210 				try (OutputStream os = connection.getOutputStream()) {
211 					if (formData != null) {
212 						writeUTF8(formData, String.format("--%s--\r\n", multipartBoundary));
213 						formData.flush();
214 						os.write(formData.toByteArray());
215 					} else
216 						writeUTF8(os, content);
217 				}
218 			} else {
219 				connection.setDoOutput(false);
220 			}
221 			// handle thread interrupts (fired by TTSTimeout e.g.)
222 			Thread currentThread = Thread.currentThread();
223 			Thread handleInterrupt = new Thread() {
224 					public void run() {
225 						while (true) {
226 							try {
227 								sleep(1000);
228 							} catch (InterruptedException e) {
229 								return;
230 							}
231 							if (currentThread.isInterrupted()) {
232 								connection.disconnect(); // unblocks connection.getInputStream() below
233 								connection = null;
234 								return;
235 							}
236 						}
237 					}
238 				};
239 			handleInterrupt.start();
240 			try {
241 				body = connection.getInputStream();
242 			} catch (IOException e) {
243 				handleInterrupt.interrupt();
244 				if (connection == null)
245 					// cancelled by interrupt handler
246 					throw new InterruptedException("request was interrupted");
247 				else
248 					throw e;
249 			} finally {
250 				handleInterrupt.interrupt();
251 			}
252 		} catch (IOException e) {
253 			ioe = e;
254 		}
255 		try {
256 			status = getConnection().getResponseCode();
257 		} catch (IOException e) {
258 			throw new IOException(
259 				"Could not retrieve response code for request. Do you have an internet connection?", e);
260 		}
261 		if (ioe != null || status > 299)
262 			error = getConnection().getErrorStream();
263 		return new Response(status, body, error, ioe);
264 	}
265 
266 	/**
267 	 * Get the current connection object
268 	 */
269 	public HttpURLConnection getConnection() {
270 		return connection;
271 	}
272 
273 	/**
274 	 * Cancel the request, disconnecting it from the server
275 	 */
276 	public void cancel() {
277 		if (this.connection != null) {
278 			this.connection.disconnect();
279 			this.connection = null;
280 		}
281 	}
282 
283 	private static void writeUTF8(OutputStream os, String content) throws IOException {
284 		try {
285 			byte[] input = content.getBytes("utf-8");
286 			os.write(input, 0, input.length);
287 		} catch (UnsupportedEncodingException e) {
288 			throw new IllegalStateException("coding error");
289 		}
290 	}

292 	/**
293 	 * Note that this function is part of Java 9
294 	 */
295 	private static long transferTo(InputStream in, OutputStream out) throws IOException {
296 		return ByteStreams.copy(in, out);
297 	}
