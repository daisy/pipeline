1   package org.daisy.pipeline.common.rest;
2   
3   import java.io.BufferedReader;
4   import java.io.IOException;
5   import java.io.InputStream;
6   import java.io.InputStreamReader;
7   
8   public class Response {
9   
10  	/**
11  	 * The response code
12  	 */
13  	public final int status;
14  
15  	/**
16  	 * The response body
17  	 */
18  	public final String body;
19  
20  	/**
21  	 * The error stream (response body when request failed but server sent useful data nonetheless)
22  	 */
23  	public final String error;
24  
25  	/**
26  	 * Error that occured while making the connection or sending data to the server.
27  	 */
28  	public final IOException exception;
29  
30  	Response(int status, InputStream body, InputStream error, IOException exception) throws IOException {
31  		this.status = status;
32  		try {
33  			this.body = body != null ? readStream(body) : null;
34  			this.error = error != null ? readStream(error) : null;
35  		} catch (IOException e) {
36  			throw new IOException("Could not read body", e);
37  		}
38  		this.exception = exception;
39  	}
40  
41  	/**
42  	 * Read InputStream as a UTF-8 encoded string.
43  	 */
44  	private static String readStream(InputStream is) throws IOException {
45  		BufferedReader br = new BufferedReader(new InputStreamReader(is, "utf-8"));
46  		StringBuilder sb = new StringBuilder();
47  		String line;
48  		while ((line = br.readLine()) != null) {
49  			sb.append(line.trim());
50  		}
51  		br.close();
52  		return sb.toString();
53  	}
54  }
