106 	@Override
107 	public void run() throws SaxonApiException {
108 		super.run();
109 		MutableQuery q = mutableQuery(query(getOption(_file_format, "")));
110 		addOption(_line_breaks, q);
111 		addOption(_page_breaks, q);
112 		addOption(_pad, q);
113 		addOption(_charset, q);
114 		RuntimeValue tableQuery = getOption(_table);
115 		if (tableQuery != null) {
116 			Table table;
117 			try {
118 				logger.debug("Finding table for query: " + tableQuery);
119 				table = tableProvider.get(query(tableQuery.getString())).iterator().next();
120 				logger.debug("Found table: " + table); }
121 			catch (NoSuchElementException e) {
122 				
123 				// this fallback is done because in dtbook-to-pef we use the
124 				// query (locale:...) which does not always match something
125 				// FIXME: https://github.com/daisy/pipeline-mod-braille/issues/75
126 				logger.warn("Table not found, falling back to en-US table.");
127 				table = tableProvider.get(EN_US).iterator().next(); }
128 			q.add("table", table.getIdentifier()); }
129 		logger.debug("Finding file format for query: " + q);
130 		Iterable<FileFormat> fileFormats = fileFormatProvider.get(q);
131 		if (!fileFormats.iterator().hasNext()) {
132 			logger.error("pef:pef2text failed: no file format found for query: " + q);
133 			throw new XProcException(step.getNode(), "pef:pef2text failed: no file format found for query: " + q); }
134 		for (FileFormat fileFormat : fileFormats) {
135 			try {
136 				logger.debug("Storing PEF to file format: " + fileFormat);
137 				
138 				// Initialize output directory
139 				File textDir = new File(new URI(getOption(_dir_href).getString()));
140 				textDir.mkdirs();
141 				
142 				// Read source PEF
143 				ByteArrayOutputStream s = new ByteArrayOutputStream();
144 				Serializer serializer = runtime.getProcessor().newSerializer();
145 				serializer.setOutputStream(s);
146 				serializer.setCloseOnCompletion(true);
147 				serializer.serializeNode(source.read());
148 				serializer.close();
149 				InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
150 				s.close();
151 				
152 				// Parse pattern
153 				String singleVolumeName = getOption(_single_volume_name, "");
154 				String pattern = getOption(_name_pattern, "");
155 				if (pattern.isEmpty())
156 					pattern = "volume-{}";
157 				int match = pattern.indexOf("{}");
158 				if (match < 0 || match != pattern.lastIndexOf("{}")) {
159 					// Output to single file
160 					convertPEF2Text(pefStream,
161 							new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
162 				} else {
163 					// Split PEF
164 					pattern = pattern.replaceAll("'", "''")
165 							.replaceAll("([0#\\.,;%\u2030\u00A4-]+)", "'$1'");
166 					// Recalculate after replacement
167 					match = pattern.indexOf("{}");
168 					File splitDir = new File(textDir, "split");
169 					splitDir.mkdir();
170 					PEFFileSplitter splitter = new PEFFileSplitter(validatorFactory);
171 					String prefix = PEFFileSplitter.PREFIX;
172 					String postfix = PEFFileSplitter.POSTFIX;
173 					splitter.split(pefStream, splitDir, prefix, postfix);
174 					File[] pefFiles = splitDir.listFiles();
175 					String formatPattern = pattern.substring(0, match);
176 					int nWidth; {
177 						try {
178 							nWidth = Integer.parseInt(getOption(_number_width, "")); }
179 						catch (NumberFormatException e) {
180 							nWidth = 0; }}
181 					if (nWidth == 0)
182 						formatPattern += "###"; // Assume max 999 volumes
183 					else
184 						while (nWidth > 0) { formatPattern += "0"; nWidth--; }
185 					formatPattern += pattern.substring(match + 2);
186 					NumberFormat format = new DecimalFormat(formatPattern);
187 					for (File pefFile : pefFiles) {
188 						InputStream is = new FileInputStream(pefFile);
189 						if (pefFiles.length == 1 && !singleVolumeName.isEmpty()) {
190 							// Output to single file
191 							convertPEF2Text(is, new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
192 						} else {
193 							String pefName = pefFile.getName();
194 							if (pefName.length() <= prefix.length() + postfix.length()
195 							    || !pefName.substring(0, prefix.length()).equals(prefix)
196 							    || !pefName.substring(pefName.length() - postfix.length()).equals(postfix)) {
197 								is.close();
198 								throw new RuntimeException("Coding error");
199 							}
200 							String textName = format.format(
201 									Integer.parseInt(pefName.substring(prefix.length(), pefName.length() - postfix.length())));
202 							convertPEF2Text(is,
203 									new File(textDir, textName + fileFormat.getFileExtension()),
204 									fileFormat);
205 						}
206 						is.close();
207 						if (!pefFile.delete()) pefFile.deleteOnExit();
208 					}
209 					pefStream.close();
210 					if (!splitDir.delete()) splitDir.deleteOnExit();
211 				}
212 				return; }
213 			catch (Exception e) {
214 				logger.error("Storing PEF to file format '" + fileFormat + "' failed", e); }}
215 		logger.error("pef:pef2text failed");
216 		throw new XProcException(step.getNode(), "pef:pef2text failed");
217 	}

219 	private void convertPEF2Text(InputStream pefStream, File textFile, FileFormat fileFormat)
220 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
221 		// Create EmbosserWriter
222 		OutputStream textStream = new FileOutputStream(textFile);
223 		EmbosserWriter writer = fileFormat.newEmbosserWriter(textStream);
224 		
225 		// Parse PEF to text
226 		PEFHandler.Builder builder = new PEFHandler.Builder(writer);
227 		builder.range(null).align(Alignment.LEFT).offset(0);
228 		parsePefFile(pefStream, builder.build());
229 		textStream.close();
230 	}

303 	// copied from org.daisy.braille.facade.PEFConverterFacade because it is no longer static
304 	/**
305 	 * Parses the given input stream using the supplied PEFHandler.
306 	 * @param is the input stream
307 	 * @param ph the PEFHandler
308 	 * @throws ParserConfigurationException
309 	 * @throws SAXException
310 	 * @throws IOException
311 	 * @throws UnsupportedWidthException
312 	 */
313 	private static void parsePefFile(InputStream is, PEFHandler ph)
314 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
315 		
316 		SAXParserFactory spf = SAXParserFactory.newInstance();
317 		spf.setNamespaceAware(true);
318 		SAXParser sp = spf.newSAXParser();
319 		try {
320 			sp.parse(is, ph); }
321 		catch (SAXException e) {
322 			if (ph.hasWidthError())
323 				throw new UnsupportedWidthException(e);
324 			else
325 				throw e; }
326 	}
