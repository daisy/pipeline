18  import org.daisy.pipeline.braille.common.CSSStyledText;
19  import org.daisy.pipeline.braille.common.Query;
20  import static org.daisy.pipeline.braille.common.Query.util.query;
21  import org.daisy.pipeline.braille.dotify.DotifyTranslator;
22  
23  import org.osgi.service.component.annotations.Component;
24  import org.osgi.service.component.annotations.Reference;
25  import org.osgi.service.component.annotations.ReferenceCardinality;
26  import org.osgi.service.component.annotations.ReferencePolicy;
27  
28  import org.slf4j.Logger;

31  @Component(
32  	name = "dotify:translate",
33  	service = { ExtensionFunctionDefinition.class }
34  )
35  @SuppressWarnings("serial")
36  public class TranslateDefinition extends ExtensionFunctionDefinition {
37  	
38  	private DotifyTranslator.Provider provider = null;
39  	
40  	@Reference(
41  		name = "DotifyTranslatorProvider",
42  		unbind = "unbindTranslatorProvider",
43  		service = DotifyTranslator.Provider.class,
44  		cardinality = ReferenceCardinality.MANDATORY,
45  		policy = ReferencePolicy.STATIC
46  	)
47  	protected void bindTranslatorProvider(DotifyTranslator.Provider provider) {
48  		this.provider = provider;
49  	}
50  	
51  	protected void unbindTranslatorProvider(DotifyTranslator.Provider provider) {
52  		this.provider = null;
53  	}
54  	
55  	private static final StructuredQName funcname = new StructuredQName("dotify",
56  			"http://code.google.com/p/dotify/", "translate");
57  	
58  	@Override
59  	public StructuredQName getFunctionQName() {
60  		return funcname;
61  	}
62  
63  	@Override
64  	public int getMinimumNumberOfArguments() {
65  		return 2;
66  	}
67  
68  	@Override
69  	public int getMaximumNumberOfArguments() {
70  		return 2;
71  	}
72  
73  	@Override
74  	public SequenceType[] getArgumentTypes() {
75  		return new SequenceType[] {
76  			SequenceType.SINGLE_STRING,
77  			SequenceType.SINGLE_STRING };
78  	}
79  
80  	@Override
81  	public SequenceType getResultType(SequenceType[] suppliedArgumentTypes) {
82  		return SequenceType.OPTIONAL_STRING;
83  	}
84  	
85  	@Override
86  	public ExtensionFunctionCall makeCallExpression() {
87  		return new ExtensionFunctionCall() {
88  			
89  			@Override
90  			public Sequence call(XPathContext context, Sequence[] arguments) throws XPathException {
91  				try {
92  					Query query = query(arguments[0].head().getStringValue());
93  					List<String> text = sequenceToList(arguments[1]);
94  					List<CSSStyledText> styledText = new ArrayList<CSSStyledText>();
95  					if (arguments.length > 2) {
96  						List<String> style = sequenceToList(arguments[2]);
97  						if (style.size() != text.size())
98  							throw new RuntimeException("Lengths of text and style sequences must match");
99  						for (int i = 0; i < text.size(); i++)
100 							styledText.add(new CSSStyledText(text.get(i), style.get(i))); }
101 					else
102 						for (int i = 0; i < text.size(); i++)
103 							styledText.add(new CSSStyledText(text.get(i), ""));
104 					for (DotifyTranslator t : provider.get(query))
105 						try {
106 							return iterableToSequence(t.fromStyledTextToBraille().transform(styledText)); }
107 						catch (UnsupportedOperationException e) {}
108 					throw new RuntimeException("Could not find a DotifyTranslator for query: " + query); }
109 				catch (Exception e) {
110 					logger.error("dotify:translate failed", e);
111 					throw new XPathException("dotify:translate failed"); }
112 			}
113 		};
114 	}
115 	
116 	private static List<String> sequenceToList(Sequence seq) throws XPathException {
117 		List<String> list = new ArrayList<String>();
118 		SequenceIterator iterator = seq.iterate();
119 		for (Item item = iterator.next(); item != null; item = iterator.next())
120 			list.add(item.getStringValue());
121 		return list;
122 	}
123 	
124 	private static Sequence iterableToSequence(Iterable<String> iterable) {
125 		List<StringValue> list = new ArrayList<StringValue>();
126 		for (String s : iterable)
127 			list.add(new StringValue(s));
128 		return new SequenceExtent(list);
129 	}
130 	
131 	private static final Logger logger = LoggerFactory.getLogger(TranslateDefinition.class);
132 	
133 }
