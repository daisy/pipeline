92  	private Map<String,Method> findParsingMethods() {
93  		Map<String,Method> map = new HashMap<String,Method>(css.getTotalProperties(), 1.0f);
94  		for (String property : css.getDefinedPropertyNames()) {
95  			try {
96  				String methodName = camelCase(
97  					"process-" + ((prefix.length() > 0 && property.startsWith(prefix))
98  					                  ? property.substring(prefix.length())
99  					                  : property));
100 				Method m = SupportedOBFLProperties.class.getDeclaredMethod(
101 					methodName,
102 					Declaration.class, Map.class, Map.class);
103 				map.put(property, m);
104 			} catch (Exception e) { // assuming this is a NoSuchMethodException
105 				if (forTransformer)
106 					// we're calling from OBFLStyleTransformer, so in the absence of a parsing method
107 					// we can pass the (already parsed) declarations as-is
108 					;
109 				else
110 					log.debug("Unable to find method for property {}.", property); // should not happen
111 			}
112 		}
113 		if (forExtension) {
114 			// add parsing methods for standard properties from SupportedBrailleCSS
115 			for (String property : new String[]{"display",
116 			                                    "volume-break-inside"}) {
117 				try {
118 					Method m = SupportedOBFLProperties.class.getDeclaredMethod(
119 						camelCase("process-" + property),
120 						Declaration.class, Map.class, Map.class);
121 					map.put(property, m);
122 				} catch (Exception e) {
123 					log.debug("Unable to find method for property {}.", property); // should not happen
124 				}
125 			}
126 		}
127 		log.debug("Totally found {} parsing methods", map.size());
128 		return map;
129 	}

131 	@Override
132 	public boolean parseDeclaration(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
133 		String property = d.getProperty().toLowerCase();
134 		if (!(css.isSupportedCSSProperty(property) || "display".equals(property) || "volume-break-inside".equals(property))) {
135 			if (!property.startsWith(prefix) && css.isSupportedCSSProperty(prefix + property))
136 				property = prefix + property;
137 			else
138 				return false;
139 		}
140 		try {
141 			Method m = methods.get(property);
142 			if (m != null)
143 				try {
144 					return (Boolean)m.invoke(this, d, properties, values);
145 				} catch (IllegalAccessException e) {
146 				} catch (IllegalArgumentException e) {
147 				} catch (InvocationTargetException e) {
148 				}
149 			else
150 				// if there is no parsing method for a property, but the property is supported,
151 				// pass the (already parsed) declaration as-is
152 				if (forTransformer && css.isSupportedCSSProperty(property))
153 					// we're calling from OBFLStyleTransformer, so declaration is a PropertyValue
154 					return copyPropertyValue(d, properties, values);
155 		} catch (Exception e) {
156 		}
157 		return false;
158 	}

164 		if (!(d instanceof PropertyValue))
165 			throw new RuntimeException("coding error");
166 		properties.put(d.getProperty(), ((PropertyValue)d).getCSSProperty());
167 		Term<?> value = ((PropertyValue)d).getValue();
168 		if (value != null)
169 			values.put(d.getProperty(), value);
170 		return true;
171 	}

174 	private boolean processDisplay(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
175 		if (forTransformer)
176 			return copyPropertyValue(d, properties, values);
177 		if (d.size() != 1)
178 			return false;
179 		Term t = d.get(0);
180 		if (t instanceof TermIdent) {
181 			String display = ((TermIdent)t).getValue();
182 			boolean normalize = false;
183 			if (!display.startsWith(prefix)) {
184 				display = prefix + display;
185 				normalize = true;
186 			}
187 			if ((prefix + "toc").equals(display) ||
188 			    (prefix + "list-of-references").equals(display)) {
189 				String prop = d.getProperty();
190 				properties.put(prop, Display.custom);
191 				if (normalize)
192 					t = ((TermIdent)t).setValue(display);
193 				values.put(prop, t);
194 				return true;
195 			}
196 		}
197 		return false;
198 	}

474 	@SuppressWarnings("unused")
475 	private boolean processVolumeBreakInside(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
476 		if (forTransformer) {
477 			if (!(d instanceof PropertyValue))
478 				throw new RuntimeException("coding error");
479 			CSSProperty property = ((PropertyValue)d).getCSSProperty();
480 			if (property == VolumeBreakInside.AVOID)
481 				return false;
482 			return copyPropertyValue(d, properties, values);
483 		}
484 		if (d.size() == 1) {
485 			Term<?> term = d.get(0);
486 			if (term instanceof TermFunction) {
487 				TermFunction f = (TermFunction)term;
488 				String funcName = f.getFunctionName();
489 				boolean normalize = false;
490 				if (!funcName.startsWith(prefix)) {
491 					funcName = prefix + funcName;
492 					normalize = true;
493 				}
494 				if ((prefix + "keep").equals(funcName)
495 					&& f.size() == 1
496 					&& f.get(0) instanceof TermInteger) {
497 					int priority = ((TermInteger)f.get(0)).getIntValue();
498 					if (priority >= 1 && priority <= 9) {
499 						properties.put(d.getProperty(), org.daisy.pipeline.braille.dotify.impl.OBFLProperty.VolumeBreakInside.keep);
500 						if (normalize)
501 							f = f.setFunctionName(funcName);
502 						values.put(d.getProperty(), f);
503 						return true;
504 					}
505 				}
506 			}
507 		}
508 		return false;
509 	}

511 	// not reusing SupportedBrailleCSS#genericOneIdentOrDotPattern because method is not static
512 	private <T extends CSSProperty> boolean genericOneIdentOrDotPattern(
513 		Class<T> type, T dotPatternIdentification, Declaration d,
514 		Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
515 		if (d.size() != 1)
516 			return false;
517 		Term<?> term = d.get(0);
518 		if (genericTermIdent(type, term, ALLOW_INH, d.getProperty(), properties))
519 			return true;
520 		try {
521 			if (TermIdent.class.isInstance(term)) {
522 				String propertyName = d.getProperty();
523 				TermDotPattern value = TermDotPattern.createDotPattern((TermIdent)term);
524 				properties.put(propertyName, dotPatternIdentification);
525 				values.put(propertyName, value);
526 				return true;
527 			}
528 		} catch (Exception e) {
529 		}
530 		return false;
531 	}

533 	// not reusing SupportedBrailleCSS#genericOneIdentOrIdentifier because method is not static
534 	private <T extends CSSProperty> boolean genericOneIdentOrIdentifier(
535 		Class<T> type, T identifierIdentification, boolean sanify,
536 		Declaration d, Map<String,CSSProperty> properties,
537 		Map<String,Term<?>> values) {
538 		if (d.size() != 1)
539 			return false;
540 		return genericTermIdent(type, d.get(0), ALLOW_INH, d.getProperty(),
541 				properties)
542 				|| (genericTerm(TermIdent.class, d.get(0), d.getProperty(),
543 						identifierIdentification, sanify, properties, values)
544 					&& !((TermIdent)d.get(0)).getValue().startsWith("-"));
545 	}

643 	private void setSupportedCSS(boolean forExtension, boolean forTransformer, String prefix) {
644 		supportedCSSproperties = new HashSet<String>(TOTAL_SUPPORTED_DECLARATIONS, 1.0f);
645 		defaultCSSproperties = new HashMap<String,CSSProperty>(TOTAL_SUPPORTED_DECLARATIONS, 1.0f);
646 		defaultCSSvalues = new HashMap<String,Term<?>>(TOTAL_SUPPORTED_DECLARATIONS, 1.0f);
647 		if (prefix == null) prefix = "";
648 
649 		// standard properties supported in OBFL
650 
651 		if (forTransformer) {
652 			setProperty("border-bottom-pattern", BorderPattern.NONE);
653 			setProperty("border-left-pattern",   BorderPattern.NONE);
654 			setProperty("border-right-pattern",  BorderPattern.NONE);
655 			setProperty("border-top-pattern",    BorderPattern.NONE);
656 			setProperty("braille-charset",       BrailleCharset.UNICODE);
657 			setProperty("hyphenate-character",   HyphenateCharacter.AUTO);
658 			setProperty("hyphens",               Hyphens.MANUAL);
659 			setProperty("letter-spacing",        LetterSpacing.length,    DEFAULT_UA_LETTER_SPACING);
660 			setProperty("line-height",           LineHeight.number,       DEFAULT_UA_LINE_HEIGHT);
661 			setProperty("margin-bottom",         Margin.integer,          DEFAULT_UA_MARGIN);
662 			setProperty("margin-left",           Margin.integer,          DEFAULT_UA_MARGIN);
663 			setProperty("margin-right",          Margin.integer,          DEFAULT_UA_MARGIN);
664 			setProperty("margin-top",            Margin.integer,          DEFAULT_UA_MARGIN);
665 			setProperty("orphans",               Orphans.integer,         DEFAULT_UA_ORPHANS);
666 			setProperty("padding-bottom",        Padding.integer,         DEFAULT_UA_PADDING);
667 			setProperty("padding-left",          Padding.integer,         DEFAULT_UA_PADDING);
668 			setProperty("padding-right",         Padding.integer,         DEFAULT_UA_PADDING);
669 			setProperty("padding-top",           Padding.integer,         DEFAULT_UA_PADDING);
670 			setProperty("page-break-after",      PageBreak.AUTO);
671 			setProperty("page-break-before",     PageBreak.AUTO);
672 			setProperty("page-break-inside",     PageBreakInside.AUTO);
673 			setProperty("text-align",            TextAlign.LEFT);
674 			setProperty("text-indent",           TextIndent.integer,      DEFAULT_UA_TEXT_IDENT);
675 			setProperty("text-transform",        TextTransform.AUTO);
676 			setProperty("volume-break-after",    VolumeBreak.AUTO);
677 			setProperty("volume-break-before",   VolumeBreak.AUTO);
678 			setProperty("volume-break-inside",   VolumeBreakInside.AUTO);
679 			setProperty("white-space",           WhiteSpace.NORMAL);
680 			setProperty("widows",                Widows.integer,          DEFAULT_UA_WIDOWS);
681 			setProperty("word-spacing",          WordSpacing.length,      DEFAULT_UA_WORD_SPACING);
682 		}
683 
684 		// OBFL specific properties
685 
686 		setProperty(prefix + "right-text-indent",             TextIndent.integer,     DEFAULT_UA_TEXT_IDENT);
687 		setProperty(prefix + "underline",                     BorderPattern.NONE);
688 		setProperty(prefix + "keep-with-next-sheets",         KeepWithSheets.integer, DEFAULT_UA_KEEP_WITH_NEXT_SHEETS);
689 		setProperty(prefix + "keep-with-previous-sheets",     KeepWithSheets.integer, DEFAULT_UA_KEEP_WITH_PREVIOUS_SHEETS);
690 		setProperty(prefix + "preferred-empty-space",         AbsoluteMargin.integer, DEFAULT_UA_PREFERRED_EMPTY_SPACE);
691 		setProperty(prefix + "scenario-cost",                 ScenarioCost.NONE);
692 		setProperty(prefix + "table-col-spacing",             AbsoluteMargin.AUTO);
693 		setProperty(prefix + "table-row-spacing",             AbsoluteMargin.AUTO);
694 		setProperty(prefix + "toc-range",                     TocRange.DOCUMENT);
695 		setProperty(prefix + "use-when-collection-not-empty", Flow.NORMAL);
696 		setProperty(prefix + "vertical-align",                VerticalAlign.AFTER);
697 		setProperty(prefix + "vertical-position",             AbsoluteMargin.AUTO);
698 
699 		if (forExtension) {
700 			setProperty(prefix + "fallback-collection", Flow.NORMAL, null);
701 			setProperty(prefix + "fallback-flow", Flow.NORMAL, null);
702 			setProperty(prefix + "marker", Marker.NONE, null);
703 		}
704 
705 		// the following properties are not supported in OBFL (generate a warning for
706 		// values other than the default):
707 
708 		// - border-bottom-align
709 		// - border-bottom-style
710 		// - border-bottom-width
711 		// - border-left-align
712 		// - border-left-style
713 		// - border-left-width
714 		// - border-right-align
715 		// - border-right-style
716 		// - border-right-width
717 		// - border-top-align
718 		// - border-top-style
719 		// - border-top-width
720 		// - left
721 		// - right
722 
723 		// - content (already handled in css:parse-content)
724 		// - counter-increment (already handled in css:eval-counter)
725 		// - counter-reset (already handled in css:eval-counter)
726 		// - counter-set (already handled in css:eval-counter)
727 		// - display (already handled in css:make-boxes)
728 		// - flow (already handled in css:flow-into and css:flow-from)
729 		// - list-style-type (already handled in css:make-boxes)
730 		// - page (already handled in px:css-cascade)
731 		// - render-table-by (already handled in css:render-table-by)
732 		// - string-set (already handled in css:eval-string-set and css:shift-string-set)
733 		// - table-header-policy (already handled in css:render-table-by)
734 	}

736 	private void setOridinals() {
737 		Map<String,Integer> ords = new HashMap<String,Integer>(getTotalProperties(), 1.0f);
738 		Map<Integer,String> ordsRev = new HashMap<Integer,String>(getTotalProperties(), 1.0f);
739 		int i = 0;
740 		for (String key : defaultCSSproperties.keySet()) {
741 			ords.put(key, i);
742 			ordsRev.put(i, key);
743 			i++;
744 		}
745 		this.ordinals = ords;
746 		this.ordinalsRev = ordsRev;
747 	}
