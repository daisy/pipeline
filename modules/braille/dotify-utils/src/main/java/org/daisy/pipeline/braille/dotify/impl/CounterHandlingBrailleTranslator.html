4   import java.util.List;
5   import java.util.Map;
6   
7   import cz.vutbr.web.css.CSSProperty;
8   import cz.vutbr.web.css.Term;
9   import cz.vutbr.web.css.TermIdent;
10  import cz.vutbr.web.css.TermList;
11  
12  import org.daisy.braille.css.SimpleInlineStyle;
13  import org.daisy.braille.css.BrailleCSSProperty.TextTransform;
14  import org.daisy.pipeline.braille.common.AbstractBrailleTranslator;
15  import org.daisy.pipeline.braille.common.BrailleTranslator;
16  import org.daisy.pipeline.braille.common.BrailleTranslator.LineBreakingFromStyledText;
17  import org.daisy.pipeline.braille.common.CSSStyledText;
18  
19  import org.daisy.pipeline.braille.css.CounterStyle;
20  
21  /**
22   * {@link BrailleTranslator} that handles the <code>text-transform</code> value
23   * "<code>-dotify-counter</code>", which causes numbers to be formatted according to the
24   * value of the <code>-dotify-counter-style</code>
25   * property. <code>-dotify-counter-style</code> can be the name of a custom counter style,
26   * or a <code>symbols()</code> function.
27   */
28  public class CounterHandlingBrailleTranslator extends AbstractBrailleTranslator implements LineBreakingFromStyledText {
29  
30  	private final LineBreakingFromStyledText backingTranslator;
31  	private final Map<String,CounterStyle> customCounterStyles;
32  
33  	public CounterHandlingBrailleTranslator(BrailleTranslator backingTranslator,
34  	                                        Map<String,CounterStyle> customCounterStyles) {
35  		this.backingTranslator = backingTranslator.lineBreakingFromStyledText();
36  		this.customCounterStyles = customCounterStyles;
37  	}
38  
39  	@Override
40  	public LineBreakingFromStyledText lineBreakingFromStyledText() {
41  		return this;
42  	}
43  
44  	public LineIterator transform(Iterable<CSSStyledText> styledText, int from, int to) {
45  		return backingTranslator.transform(handleCounterStyles(styledText), from, to);
46  	}
47  
48  	private Iterable<CSSStyledText> handleCounterStyles(Iterable<CSSStyledText> styledText) {
49  		List<CSSStyledText> segments = new ArrayList<CSSStyledText>();
50  		String segment = null;
51  		SimpleInlineStyle style = null;
52  		Map<String,String> attrs = null;
53  		for (CSSStyledText st : styledText) {
54  			String t = st.getText();
55  			SimpleInlineStyle s = st.getStyle();
56  			Map<String,String> a = st.getTextAttributes();
57  			if (s != null) {
58  				if (s.getProperty("text-transform") == TextTransform.list_values) {
59  					TermList list = s.getValue(TermList.class, "text-transform");
60  					if (((TermIdent)list.get(0)).getValue().equals("-dotify-counter")) {
61  						if (list.size() == 1)
62  							s.removeProperty("text-transform");
63  						else
64  							list.remove(0);
65  						if ("??".equals(t)) {
66  							// If input text is "??", it will be used for creating a placeholder for content that
67  							// can not be computed yet (see org.daisy.dotify.formatter.impl.row.SegmentProcessor).
68  						} else {
69  							int counterValue = Integer.parseInt(t);
70  							Term<?> symbolsFunction = s.getValue("-dotify-counter-style");
71  							if (symbolsFunction != null) {
72  								try {
73  									t = new CounterStyle(symbolsFunction).format(counterValue);
74  								} catch (IllegalArgumentException e) {
75  									// FIXME: show warning
76  								}
77  							} else {
78  								CSSProperty counterStyleName = s.getProperty("-dotify-counter-style");
79  								if (counterStyleName != null
80  								    && customCounterStyles != null
81  								    && customCounterStyles.containsKey(counterStyleName.toString())) {
82  									t = customCounterStyles.get(counterStyleName.toString()).format(counterValue);
83  								}
84  							}
85  						}
86  					}
87  				}
88  				s.removeProperty("-dotify-counter-style"); }
89  			if (segment != null)
90  				segments.add(new CSSStyledText(segment, style, attrs));
91  			segment = t;
92  			style = s;
93  			attrs = a; }
94  		if (segment != null)
95  			segments.add(new CSSStyledText(segment, style, attrs));
96  		return segments;
97  	}
98  }
