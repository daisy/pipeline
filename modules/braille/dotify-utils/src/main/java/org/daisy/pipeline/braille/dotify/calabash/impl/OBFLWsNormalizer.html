35  	public Runnable transform(XMLEventReader input, XMLEventConsumer output) {
36  		Deque<XMLEvent> buffer = new ArrayDeque<>();
37  		return () -> {
38  			try (NormalizedOBFL i = new NormalizedOBFL(input)) {
39  				while (i.hasNext())
40  					output.add(i.next());
41  			} catch (XMLStreamException e) {
42  				throw new TransformerException(e);
43  			}
44  		};
45  	}
46  
47  	private class NormalizedOBFL implements Iterator<XMLEvent>, AutoCloseable {
48  
49  		private final XMLEventReader input;
50  		private final Deque<XMLEvent> buffer;
51  
52  		NormalizedOBFL(XMLEventReader input) {
53  			this.input = input;
54  			buffer = new ArrayDeque<>();
55  		}
56  
57  		@Override
58  		public boolean hasNext() {
59  			return !buffer.isEmpty() || input.hasNext();
60  		}
61  
62  		@Override
63  		public XMLEvent next() throws NoSuchElementException, TransformerException {
64  			while (buffer.isEmpty() && input.hasNext())
65  				try {
66  					readNextChunk();
67  				} catch (XMLStreamException e) {
68  					throw new TransformerException(e);
69  				}
70  			if (!buffer.isEmpty())
71  				return buffer.removeFirst();
72  			else
73  				throw new NoSuchElementException();
74  		}
75  
76  		@Override
77  		public void close() {
78  			buffer.clear();
79  		}
80  
81  		private void readNextChunk() throws XMLStreamException {
82  			XMLEvent event = input.nextEvent();
83  			if (event.getEventType() == XMLStreamConstants.CHARACTERS) {
84  				String chars = normalizeSpace(event.asCharacters().getData());
85  				if (!"".equals(chars))
86  					buffer.add(createCharacters(chars, event.getLocation()));
87  			} else if (event.getEventType() == XMLStreamConstants.START_DOCUMENT)
88  				buffer.add(event);
89  			else if (beginsMixedContent(event))
90  				parseBlock(event);
91  			else
92  				buffer.add(event);
93  		}
94  
95  		/* Group content within block and before and after nested blocks, and call modifyWhiteSpace() on
96  		 * each group.
97  		 */
98  		private void parseBlock(XMLEvent event) throws XMLStreamException {
99  			QName name = event.asStartElement().getName();
100 			List<XMLEvent> events = new ArrayList<>();
101 			events.add(event);
102 			while (input.hasNext()) {
103 				event = input.nextEvent();
104 				if (event.getEventType() == XMLStreamConstants.COMMENT)
105 					; // ignore comments
106 				else if (beginsMixedContent(event)) {
107 					buffer.addAll(modifyWhitespace(events));
108 					events.clear();
109 					parseBlock(event);
110 				} else if (isEndElement(event, name)) {
111 					events.add(event);
112 					buffer.addAll(modifyWhitespace(events));
113 					break;
114 				} else
115 					events.add(event);
116 			}
117 		}

119 		/* Merge text nodes on both sides of a comment
120 		 */
121 		private List<XMLEvent> coalesceTextNodes(List<XMLEvent> events) throws XMLStreamException {
122 			List<XMLEvent> coalesced = new ArrayList<>();
123 			Characters chars = null; // if previous event (that was not a comment) were characters
124 			for (int i = 0; i < events.size(); i++) {
125 				XMLEvent event = events.get(i);
126 				if (event.getEventType() == XMLStreamConstants.CHARACTERS) {
127 					if (chars != null)
128 						chars = createCharacters(chars.getData() + event.asCharacters().getData(),
129 						                         chars.getLocation());
130 					else
131 						chars = event.asCharacters();
132 				} else {
133 					if (chars != null) {
134 						coalesced.add(chars);
135 						chars = null;
136 					}
137 					coalesced.add(event);
138 				}
139 			}
140 			if (chars != null)
141 				coalesced.add(chars);
142 			return coalesced;
143 		}

145 		// Note that there could in principle be some leading or trailing white space inside a toc-entry
146 		// that is moved outside of the toc-entry, and because toc-entry are conditional, this could
147 		// lead to unwanted white space. Because at the moment it is forbidden by the implementation
148 		// (not by OBFL) that two toc-entry's are contained in the same block, this is not an actual
149 		// issue.
150 		private List<XMLEvent> modifyWhitespace(List<XMLEvent> events) throws XMLStreamException {
151 			// code below assumes no two text nodes follow each other immediately
152 			events = coalesceTextNodes(events);
153 			List<XMLEvent> buffer = new ArrayList<>();
154 			// for changing order between white space and markers, anchors and begin and end tags
155 			String pendingSpace = null;
156 			Deque<XMLEvent> pendingTags = new ArrayDeque<>();
157 			boolean firstContent = true;
158 			for (int i = 0; i < events.size(); i++) {
159 				XMLEvent event = events.get(i);
160 				if (event.getEventType() == XMLStreamConstants.CHARACTERS) {
161 					String data = event.asCharacters().getData();
162 					// move any trailing space at the end of an element outside the element, or strip it when
163 					// at the end of the block
164 					if (isSpace(data)) {
165 						// strip space if it does not follow at least one text node that is not white space,
166 						// or an element that is not a span or marker
167 						if (!firstContent)
168 							pendingSpace = " ";
169 					} else {
170 						String pre = !firstContent && (pendingSpace != null || startsWithSpace(data))
171 							? " "
172 							: "";
173 						pendingSpace = null;
174 						if (!pendingTags.isEmpty()) {
175 							if (!"".equals(pre)) {
176 								// move any leading space at the start of an element outside the element, and
177 								// move any leading space in front of marker elements
178 								buffer.add(createCharacters(pre, event.getLocation()));
179 								pre = "";
180 							}
181 							buffer.addAll(pendingTags);
182 							pendingTags.clear();
183 						}
184 						buffer.add(createCharacters(pre + normalizeSpace(data), event.getLocation()));
185 						if (endsWithSpace(data))
186 							pendingSpace = " ";
187 						firstContent = false;
188 					}
189 				} else if (isStartElement(event, ObflQName.SPAN,
190 				                                 ObflQName.STYLE,
191 				                                 ObflQName.TOC_ENTRY,
192 				                                 ObflQName.MARKER,
193 				                                 ObflQName.ANCHOR)
194 				           || isEndElement(event, ObflQName.MARKER,
195 				                                  ObflQName.ANCHOR))
196 					pendingTags.add(event);
197 				else {
198 					if (event.getEventType() == XMLStreamConstants.START_ELEMENT) {
199 						if (pendingSpace != null) {
200 							buffer.add(createCharacters(pendingSpace, event.getLocation()));
201 							pendingSpace = null;
202 						}
203 					} else if (firstContent
204 					           && event.getEventType() == XMLStreamConstants.END_ELEMENT
205 					           && !isEndElement(event, ObflQName.SPAN,
206 					                                   ObflQName.STYLE,
207 					                                   ObflQName.TOC_ENTRY))
208 						firstContent = false;
209 					if (!pendingTags.isEmpty()) {
210 						buffer.addAll(pendingTags);
211 						pendingTags.clear();
212 					}
213 					buffer.add(event);
214 				}
215 			}
216 			if (!pendingTags.isEmpty()) {
217 				buffer.addAll(pendingTags);
218 				pendingTags.clear();
219 			}
220 			// drop empty style, span and toc-entry elements
221 			buffer = dropEmptyStyles(buffer);
222 			return buffer;
223 		}
224 
225 		private List<XMLEvent> dropEmptyStyles(List<XMLEvent> events) throws XMLStreamException {
226 			List<XMLEvent> cleaned = new ArrayList<>();
227 			cleaned.addAll(events);
228 			for (ListIterator<XMLEvent> i = cleaned.listIterator(); i.hasNext();) {
229 				XMLEvent e = i.next();
230 				if (isEndElement(e, ObflQName.SPAN,
231 				                 ObflQName.STYLE,
232 				                 ObflQName.TOC_ENTRY)) {
233 					i.previous();
234 					XMLEvent p = i.previous();
235 					boolean skip = false; {
236 						if (p.getEventType() == XMLStreamConstants.START_ELEMENT) {
237 							skip = true;
238 							// span elements are kept if they have an id attribute
239 							if (p.asStartElement().getName().equals(ObflQName.SPAN)
240 							    && p.asStartElement().getAttributeByName(ObflQName._ID) != null)
241 								skip = false; }}
242 					if (skip) {
243 						i.remove();
244 						i.next();
245 						i.remove();
246 					} else {
247 						i.next();
248 						i.next();
249 					}
250 				}
251 			}
252 			return cleaned;
253 		}
254 	}

256 	private Characters createCharacters(String chars, Location location) throws XMLStreamException {
257 		eventFactory.setLocation(location);
258 		try {
259 			return eventFactory.createCharacters(chars);
260 		} finally {
261 			eventFactory.setLocation(null);
262 		}
263 	}

265 	private static boolean isSpace(String input) {
266 		for (char c : input.toCharArray())
267 			if (!Character.isWhitespace(c))
268 				return false;
269 		return true;
270 	}
271 
272 	private static boolean startsWithSpace(String input) {
273 		return Character.isWhitespace(input.charAt(0));
274 	}
275 
276 	private static boolean endsWithSpace(String input) {
277 		return Character.isWhitespace(input.charAt(input.length() - 1));
278 	}

280 	private static final Pattern WS = Pattern.compile("\\s+");
281 
282 	private static String normalizeSpace(String input) {
283 		return WS.matcher(input).replaceAll(" ").trim();
284 	}
285 
286 	private static boolean isStartElement(XMLEvent event, QName... element) {
287 		if (event.getEventType() != XMLStreamConstants.START_ELEMENT)
288 			return false;
289 		else {
290 			QName name = event.asStartElement().getName();
291 			for (QName n : element) {
292 				if (name.equals(n)) {
293 					return true;
294 				}
295 			}
296 			return false;
297 		}
298 	}
299 
300 	private static boolean isEndElement(XMLEvent event, QName... element) {
301 		if (event.getEventType() != XMLStreamConstants.END_ELEMENT)
302 			return false;
303 		else {
304 			QName name = event.asEndElement().getName();
305 			for (QName n : element) {
306 				if (name.equals(n)) {
307 					return true;
308 				}
309 			}
310 			return false;
311 		}
312 	}

314 	private static boolean beginsMixedContent(XMLEvent event) {
315 		return isStartElement(event, ObflQName.BLOCK,
316 		                             ObflQName.TOC_BLOCK,
317 		                             ObflQName.ITEM,
318 		                             ObflQName.BEFORE,
319 		                             ObflQName.AFTER,
320 		                             ObflQName.TD
321 		);
322 	}
