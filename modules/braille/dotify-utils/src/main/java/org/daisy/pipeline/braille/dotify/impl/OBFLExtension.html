56  	@Override
57  	public boolean parseContentTerm(Term<?> term, TermList list) {
58  		if (term instanceof TermFunction) {
59  			TermFunction f = (TermFunction)term;
60  			String funcName = f.getFunctionName();
61  			boolean normalize = false;
62  			if (!funcName.startsWith(prefix)) {
63  				funcName = prefix + funcName;
64  				normalize = true;
65  			}
66  			if ((prefix + "evaluate").equals(funcName)) {
67  				if (f.size() != 1)
68  					return false;
69  				if (!(f.get(0) instanceof TermString))
70  					return false;
71  			} else if ((prefix + "marker-indicator").equals(funcName)) {
72  				if (f.size() != 2)
73  					return false;
74  				if (!(f.get(0) instanceof TermIdent))
75  					return false;
76  				if (!(f.get(1) instanceof TermString))
77  					return false;
78  			} else if ((prefix + "collection").equals(funcName)) {
79  				if (f.size() < 1 || f.size() > 2)
80  					return false;
81  				if (!(f.get(0) instanceof TermIdent))
82  					return false;
83  				if (f.size() == 2 && !(f.get(1) instanceof TermIdent))
84  					return false;
85  			} else
86  				return false;
87  			if (normalize)
88  				f = f.setFunctionName(funcName);
89  			list.add(term);
90  			return true;
91  		}
92  		return false;
93  	}

95  	@Override
96  	public TermIdent parseCounterName(Term<?> term) {
97  		TermIdent counterName = null;
98  		if (term instanceof TermIdent) {
99  			TermIdent ident = (TermIdent)term;
100 			String name = ident.getValue();
101 			if (name.startsWith(prefix))
102 				name = name.substring(prefix.length());
103 			if ("page".equals(name) ||
104 			    "volume".equals(name))
105 				counterName = ident;
106 			else if ("volumes".equals(name) ||
107 			         "sheets-in-document".equals(name) ||
108 			         "sheets-in-volume".equals(name) ||
109 			         "started-volume-number".equals(name) ||
110 			         "started-page-number".equals(name) ||
111 			         "started-volume-first-content-page".equals(name)) {
112 				counterName = ident;
113 				name = prefix + name;
114 			}
115 			if (counterName != null && !counterName.getValue().equals(name))
116 				// normalize
117 				counterName.setValue(name);
118 		}
119 		if (counterName != null)
120 			return counterName;
121 		else
122 			throw new IllegalArgumentException("Unknown counter name " + term);
123 	}

130 	@Override
131 	public PseudoClass createPseudoClassFunction(String name, String... args) throws IllegalArgumentException {
132 		if (name.startsWith(":"))
133 			name = name.substring(1); // should not happen
134 		if (name.startsWith(prefix))
135 			name = name.substring(prefix.length()); // should not happen
136 		OBFLPseudoClass.Type type = null; {
137 			try {
138 				type = OBFLPseudoClass.Type.valueOf(name.replaceAll("-", "_").toUpperCase());
139 			} catch (IllegalArgumentException e) {
140 			}
141 		}
142 		if (type != null)
143 			return new OBFLPseudoClass(type, args);
144 		throw new IllegalArgumentException("Unknown pseudo-class :" + name);
145 	}

147 	@Override
148 	public PseudoElement createPseudoElement(String name) throws IllegalArgumentException {
149 		if (name.startsWith(":"))
150 			// maybe a single colon was used for a pseudo element
151 			name = name.substring(1);
152 		if (name.startsWith(prefix))
153 			name = name.substring(prefix.length());
154 		try {
155 			OBFLPseudoElement.Type type = OBFLPseudoElement.Type.valueOf(name.replaceAll("-", "_").toUpperCase());
156 			return new OBFLPseudoElement(type);
157 		} catch (IllegalArgumentException e) {
158 		}
159 		// when pseuo class is prefixed with -obfl-, BrailleCSSTreeParser calls RuleFactory.createPseudoElement()
160 		// without first trying RuleFactory.createPseudoClass()
161 		OBFLPseudoClass.Type type = null; {
162 			try {
163 				type = OBFLPseudoClass.Type.valueOf(name.replaceAll("-", "_").toUpperCase());
164 			} catch (IllegalArgumentException e) {
165 			}
166 		}
167 		if (type != null)
168 			return new OBFLPseudoClass(type);
169 		throw new IllegalArgumentException("Unknown pseudo-element ::" + name);
170 	}

173 	public PseudoElement createPseudoElementFunction(String name, String... args) throws IllegalArgumentException {
174 		if (name.startsWith(":"))
175 			// maybe a single colon was used for a pseudo element, or it is a prefixed pseudo class
176 			name = name.substring(1);
177 		if (name.startsWith(prefix))
178 			name = name.substring(prefix.length());
179 		// when pseuo class is prefixed with -obfl-, BrailleCSSTreeParser calls RuleFactory.createPseudoElementFunction()
180 		// without first trying RuleFactory.createPseudoClassFunction()
181 		OBFLPseudoClass.Type type = null; {
182 			try {
183 				type = OBFLPseudoClass.Type.valueOf(name.replaceAll("-", "_").toUpperCase());
184 			} catch (IllegalArgumentException e) {
185 			}
186 		}
187 		if (type != null)
188 			return new OBFLPseudoClass(type, args);
189 		throw new IllegalArgumentException("Unknown pseudo-element ::" + name);
190 	}

192 	// extends PseudoElementImpl to make it stackable (see SelectorImpl)
193 	private static class OBFLPseudoClass extends PseudoElementImpl implements PseudoClass {
194 
195 		private enum Type {
196 			ALTERNATE_SCENARIO("alternate-scenario");
197 
198 			private final String name;
199 
200 			private Type(String name) {
201 				this.name = name;
202 			}
203 		}
204 
205 		private OBFLPseudoClass(Type type, String... args) {
206 			// pass ":" to indicate to PseudoElementImpl that it is a custom pseudo class
207 			super(":" + prefix + type.name, args);
208 			switch (type) {
209 			case ALTERNATE_SCENARIO:
210 				switch (args.length) {
211 				case 0:
212 					break; // :-obfl-alternate-scenario is equivalent to :-obfl-alternate-scenario(1)
213 				case 1:
214 					try {
215 						Integer.parseInt(args[0]);
216 					} catch (NumberFormatException e) {
217 						throw new IllegalArgumentException(
218 							"Argument of :" + prefix + type.name + " pseudo-class must be an integer", e);
219 					}
220 					break;
221 				default:
222 					throw new IllegalArgumentException(":" + prefix + type.name + " pseudo-class accepts at most 1 argument");
223 				}
224 				break;
225 			default: // can not happen
226 			}
227 		}
228 	}

253 	@Override
254 	public VendorAtRule<? extends Rule<?>> createAtRule(String name, List<Rule<?>> content) throws IllegalArgumentException {
255 		if (name.startsWith(prefix))
256 			name = name.substring(prefix.length());
257 		if ("volume-transition".equals(name)) {
258 			for (Rule<?> r : content)
259 				if (!(r instanceof OBFLRuleVolumeTransition))
260 					throw new IllegalArgumentException("Not allowed inside @" + prefix + "volume-transition rule: " + r);
261 			return new OBFLRuleVolumeTransitions((List<OBFLRuleVolumeTransition>)(List)content);
262 		}
263 		OBFLRuleVolumeTransition.Type type = null; {
264 			try {
265 				type = OBFLRuleVolumeTransition.Type.valueOf(name.replaceAll("-", "_").toUpperCase());
266 			} catch (IllegalArgumentException e) {
267 			}
268 		}
269 		if (type != null) {
270 			for (Rule<?> r : content)
271 				if (!(r instanceof Declaration))
272 					throw new IllegalArgumentException("Not allowed inside @" + type.name + ": " + r);
273 			return new OBFLRuleVolumeTransition(type, (List<Declaration>)(List)content);
274 		}
275 		throw new IllegalArgumentException("Unknown at-rule @" + name);
276 	}
