12  import org.daisy.common.file.URLs;
13  import org.daisy.pipeline.braille.common.AbstractTransform;

18  import org.daisy.pipeline.braille.common.Query.Feature;

23  
24  import org.osgi.service.component.annotations.Activate;
25  import org.osgi.service.component.annotations.Component;
26  
27  public interface DotifyOBFLTransform {
28  	
29  	@Component(
30  		name = "org.daisy.pipeline.braille.dotify.impl.DotifyOBFLTransform.Provider",
31  		service = {
32  			TransformProvider.class
33  		}
34  	)
35  	public class Provider extends AbstractTransformProvider<Transform> {
36  		
37  		private URI href;
38  		
39  		@Activate
40  		protected void activate(final Map<?,?> properties) {
41  			href = asURI(URLs.getResourceFromJAR("xml/transform/obfl-to-pef.xpl", DotifyOBFLTransform.class));
42  		}
43  		
44  		private final static Iterable<Transform> empty = Iterables.<Transform>empty();
45  		
46  		protected Iterable<Transform> _get(Query query) {
47  			try {
48  				MutableQuery q = mutableQuery(query);
49  				if (q.containsKey("formatter"))
50  					if (!"dotify".equals(q.removeOnly("formatter").getValue().get()))
51  						return empty;
52  				Iterator<Feature> input = q.get("input").iterator();
53  				while (input.hasNext())
54  					if ("obfl".equals(input.next().getValue().get())) {
55  						input.remove();
56  						for (Feature f : q.removeAll("output"))
57  							if (!("pef".equals(f.getValue().get()) || "braille".equals(f.getValue().get())))
58  								return empty;
59  						q.add("output", "braille");
60  						return Iterables.<Transform>of(
61  							logCreate((Transform)new TransformImpl(q.toString()))); }}
62  			catch (IllegalStateException e) {}
63  			return empty;
64  		}
65  		
66  		private class TransformImpl extends AbstractTransform {
67  			
68  			private final XProc xproc;
69  			
70  			private TransformImpl(String textTransformQuery) {
71  				xproc = new XProc(href, null, ImmutableMap.of("text-transform", textTransformQuery));
72  			}
73  			
74  			@Override
75  			public XProc asXProc() {
76  				return xproc;
77  			}
78  			
79  			@Override
80  			public ToStringHelper toStringHelper() {
81  				return MoreObjects.toStringHelper("o.d.p.b.dotify.impl.DotifyOBFLTransform$Provider$TransformImpl");
82  			}
83  		}
84  		
85  		@Override
86  		public ToStringHelper toStringHelper() {
87  			return MoreObjects.toStringHelper(DotifyOBFLTransform.Provider.class.getName());
88  		}
89  	}
90  }
