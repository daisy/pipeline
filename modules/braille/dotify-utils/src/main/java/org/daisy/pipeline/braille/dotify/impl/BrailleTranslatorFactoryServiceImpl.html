90  	/*
91  	 * Mode for pre-translated text with support for text-level CSS and line
92  	 * breaking according to CSS. Corresponds with translator query
93  	 * `(input:braille)(input:text-css)(output:braille)`
94  	 */
95  	private class BrailleTranslatorFactoryImpl implements BrailleTranslatorFactory {
96  		public BrailleTranslator newTranslator(String locale, String mode) throws TranslatorConfigurationException {
97  			Matcher m = QUERY.matcher(mode);
98  			if (!m.matches())
99  				throw new TranslatorConfigurationException();
100 			Query query = query(mode);
101 			if (locale != null && !"und".equals(locale))
102 				query = mutableQuery(query).add("document-locale", locale);
103 			for (org.daisy.pipeline.braille.common.BrailleTranslator t : translatorRegistry.getWithHyphenator(query))
104 				try {
105 					return new BrailleTranslatorFromBrailleTranslator(mode, t.lineBreakingFromStyledText()); }
106 				catch (UnsupportedOperationException e) {}
107 			throw new TranslatorConfigurationException("Factory does not support " + locale + "/" + mode);
108 		}
109 	}

113 	/**
114 	 * {@link BrailleTranslator} wrapper for a {@link org.daisy.pipeline.braille.common.BrailleTranslator}
115 	 *
116 	 * <a href="http://braillespecs.github.io/braille-css/#h3_white-space-processing">White space
117 	 * processing</a> and <a href="http://braillespecs.github.io/braille-css/#line-breaking">line
118 	 * breaking</a> are done according to braille CSS.
119 	 *
120 	 * White space is normalised. Preserved spaces must have been converted to
121 	 * no-break spaces and preserved line feeds must have been converted to
122 	 * &lt;obfl:br/&gt;.
123 	 *
124 	 * Through setHyphenating() the translator can be made to perform automatic hyphenation or
125 	 * not. Regardless of this setting, hyphenation characters (SHY and ZWSP) in the input are used
126 	 * in line breaking, except when overridden with a <code>hyphens: none</code> style.
127 	 */
128 	private static class BrailleTranslatorFromBrailleTranslator implements BrailleTranslator {
129 		
130 		final String mode;
131 		final org.daisy.pipeline.braille.common.BrailleTranslator.LineBreakingFromStyledText translator;
132 		
133 		private BrailleTranslatorFromBrailleTranslator(
134 				String mode,
135 				org.daisy.pipeline.braille.common.BrailleTranslator.LineBreakingFromStyledText translator) {
136 			this.mode = mode;
137 			this.translator = translator;
138 		}
139 		
140 		public BrailleTranslatorResult translate(Translatable input) throws TranslationException {
141 			if (input.getAttributes() == null && input.isHyphenating() == false) {
142 				String text = input.getText();
143 				if ("".equals(text))
144 					// see org.daisy.dotify.formatter.impl.row.SegmentProcessor.layoutLeader
145 					return new DefaultLineBreaker.LineIterator("", '\u2800', '\u2824', 1);
146 				if (" ".equals(text))
147 					// If input text is a space, it may be used for calculating the margin character
148 					// (see org.daisy.dotify.formatter.impl.common.FormatterCoreContext)
149 					return new DefaultLineBreaker.LineIterator("\u2800", '\u2800', '\u2824', 1); }
150 			return translate(cssStyledTextFromTranslatable(input), 0, -1);
151 		}
152 		
153 		public BrailleTranslatorResult translate(TranslatableWithContext input) throws TranslationException {
154 			List<CSSStyledText> styledText = Lists.newArrayList(cssStyledTextFromTranslatable(input));
155 			int from = input.getPrecedingText().size();
156 			int to = from + input.getTextToTranslate().size();
157 			for (int i = 0; i < styledText.size(); i++)
158 				if ("??".equals(styledText.get(i).getText()))
159 					// If input text is "??", it will be used for creating a placeholder for content that
160 					// can not be computed yet (see org.daisy.dotify.formatter.impl.row.SegmentProcessor).
161 					// Because normally this will never end up in the resulting PEF, it is okay to replace
162 					// it with "0". We choose a number because in some cases the translator may assume that
163 					// the input is numerical.
164 					styledText.set(i, new CSSStyledText("0",
165 					                                    styledText.get(i).getStyle(),
166 					                                    styledText.get(i).getLanguage(),
167 					                                    styledText.get(i).getTextAttributes()));
168 			return translate(styledText, from, to);
169 		}
170 		
171 		private BrailleTranslatorResult translate(Iterable<CSSStyledText> styledText, int from, int to) throws TranslationException {
172 			return translator.transform(styledText, from, to);
173 		}
174 		
175 		public String getTranslatorMode() {
176 			return mode;
177 		}
178 	}

187 	 * Convert Translatable specification to text + CSS style. Text attributes are assumed to
188 	 * contain only CSS or special variable assignments/tests. Computation of the text-transform
189 	 * property is assumed to have been performed already.
190 	 */
191 	protected static Iterable<CSSStyledText> cssStyledTextFromTranslatable(Translatable specification) {
192 		String locale = specification.getLocale();
193 		return cssStyledTextFromTranslatable(
194 			specification.getText(),
195 			specification.getAttributes(),
196 			locale != null ? parseLocale(locale) : null,
197 			specification.isHyphenating(),
198 			null);
199 	}
200 
201 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(String text,
202 	                                                                     TextAttribute attributes,
203 	                                                                     Locale language,
204 	                                                                     boolean hyphenating,
205 	                                                                     SimpleInlineStyle parentStyle) {
206 		if (attributes != null && attributes.getWidth() != text.length())
207 			throw new RuntimeException("Coding error");
208 		SimpleInlineStyle style; {
209 			String s = attributes != null ? attributes.getDictionaryIdentifier() : null;
210 			if (s == null && parentStyle == null)
211 				style = null;
212 			else
213 				// FIXME: extend caching of parsed CSS to support parentStyle!
214 				style = new SimpleInlineStyle(s != null ? s : "", parentStyle);
215 			if (hyphenating && (style == null || style.getProperty("hyphens") == null))
216 				// FIXME: add hyphens declaration through braille-css model
217 				style = new SimpleInlineStyle("hyphens: auto", style); }
218 		if (attributes != null && attributes.hasChildren())
219 			return cssStyledTextFromTranslatable(text, attributes.iterator(), language, false, style);
220 		else
221 			return Collections.singleton(new CSSStyledText(text, style, language));
222 	}
223 
224 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(String text,
225 	                                                                     Iterator<TextAttribute> attributes,
226 	                                                                     Locale language,
227 	                                                                     boolean hyphenating,
228 	                                                                     SimpleInlineStyle parentStyle) {
229 		if (attributes.hasNext()) {
230 			TextAttribute a = attributes.next();
231 			int w = a.getWidth();
232 			return concat(cssStyledTextFromTranslatable(text.substring(0, w), a, language, hyphenating, parentStyle),
233 			              cssStyledTextFromTranslatable(text.substring(w), attributes, language, hyphenating, parentStyle)); }
234 		else
235 			return empty;
236 	}

238 	/**
239 	 * Convert TranslatableWithContext specification to text + CSS style. Text attributes are
240 	 * assumed to contain only CSS or special variable assignments/tests. Computation of the
241 	 * text-transform property is assumed to have been performed already.
242 	 */
243 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(TranslatableWithContext specification) {
244 		return cssStyledTextFromTranslatable(
245 			specification.getPrecedingText(),
246 			specification.getTextToTranslate(),
247 			specification.getFollowingText(),
248 			specification.getAttributes().orElse(null),
249 				null);
250 	}

252 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<PrecedingText> preceding,
253 	                                                                     List<ResolvableText> text,
254 	                                                                     List<FollowingText> following,
255 	                                                                     AttributeWithContext attributes,
256 	                                                                     SimpleInlineStyle parentStyle) {
257 		int textSize = text == null ? 0 : text.size();
258 		int precedingSize = preceding == null ? 0 : preceding.size();
259 		int followingSize = following == null ? 0 : following.size();
260 		if (attributes != null && attributes.getWidth() != precedingSize + textSize + followingSize)
261 			throw new RuntimeException("Coding error");
262 		SimpleInlineStyle style; {
263 			String s = attributes != null ? attributes.getName().orElse(null) : null;
264 			if (s == null && parentStyle == null)
265 				style = null;
266 			else
267 				// FIXME: extend caching of parsed CSS to support parentStyle!
268 				style = new SimpleInlineStyle(s != null ? s : "", parentStyle);
269 		}
270 		if (attributes != null && attributes.hasChildren())
271 			return cssStyledTextFromTranslatable(preceding, text, following, attributes.iterator(), style);
272 		else
273 			return concat(cssStyledTextFromTranslatable(preceding, style),
274 			              cssStyledTextFromTranslatable(text, style),
275 			              cssStyledTextFromTranslatable(following, style));
276 	}

278 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<PrecedingText> preceding,
279 	                                                                     List<ResolvableText> text,
280 	                                                                     List<FollowingText> following,
281 	                                                                     Iterator<AttributeWithContext> attributes,
282 	                                                                     SimpleInlineStyle parentStyle) {
283 		int textSize = text == null ? 0 : text.size();
284 		int precedingSize = preceding == null ? 0 : preceding.size();
285 		int followingSize = following == null ? 0 : following.size();
286 		if (attributes.hasNext()) {
287 			AttributeWithContext a = attributes.next();
288 			int w = a.getWidth();
289 			if (w == 0)
290 				return cssStyledTextFromTranslatable(preceding, text, following, attributes, parentStyle);
291 			if (w <= precedingSize)
292 				return concat(
293 					cssStyledTextFromTranslatable(preceding.subList(0, w), null, null, a, parentStyle),
294 					cssStyledTextFromTranslatable(preceding.subList(w, precedingSize), text, following, attributes, parentStyle));
295 			else if (w - precedingSize <= textSize)
296 				return concat(
297 					cssStyledTextFromTranslatable(preceding, text.subList(0, w - precedingSize), null, a, parentStyle),
298 					cssStyledTextFromTranslatable(null, text.subList(w - precedingSize, textSize), following, attributes, parentStyle));
299 			else
300 				return concat(
301 					cssStyledTextFromTranslatable(preceding, text, following.subList(0, w - precedingSize - textSize), a, parentStyle),
302 					cssStyledTextFromTranslatable(null, null, following.subList(w - precedingSize - textSize, followingSize), attributes, parentStyle));
303 		} else
304 			return empty;
305 	}

307 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<? extends Object/*PrecedingText|FollowingText*/> precedingOrFollowingText,
308 	                                                                     SimpleInlineStyle parentStyle) {
309 		if (precedingOrFollowingText == null)
310 			return empty;
311 		List<CSSStyledText> styledText = new ArrayList<>();
312 		for (Object t : precedingOrFollowingText) {
313 			String text;
314 			String locale;
315 			boolean hyphenate; {
316 				if (t instanceof PrecedingText) {
317 					text = ((PrecedingText)t).resolve();
318 					locale = ((PrecedingText)t).getLocale().orElse(null);
319 					hyphenate = ((PrecedingText)t).shouldHyphenate(); }
320 				else if (t instanceof FollowingText) {
321 					text = ((FollowingText)t).peek();
322 					locale = ((FollowingText)t).getLocale().orElse(null);
323 					hyphenate = ((FollowingText)t).shouldHyphenate(); }
324 				else
325 					throw new RuntimeException();
326 			}
327 			SimpleInlineStyle style; {
328 				if (hyphenate && (parentStyle == null || parentStyle.getProperty("hyphens") == null))
329 					style = new SimpleInlineStyle("hyphens: auto", parentStyle);
330 				else if (parentStyle == null)
331 					style = null;
332 				else
333 					style = new SimpleInlineStyle("", parentStyle);
334 			}
335 			styledText.add(new CSSStyledText(text, style, locale != null ? parseLocale(locale) : null));
336 		}
337 		return styledText;
338 	}
