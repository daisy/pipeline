98  	/*
99  	 * Mode for pre-translated text with support for text-level CSS and line
100 	 * breaking according to CSS. Corresponds with translator query
101 	 * `(input:braille)(input:text-css)(output:braille)`
102 	 */
103 	private class BrailleTranslatorFactoryImpl implements BrailleTranslatorFactory {
104 		public BrailleTranslator newTranslator(String locale, String mode) throws TranslatorConfigurationException {
105 			Matcher m = QUERY.matcher(mode);
106 			if (!m.matches())
107 				throw new TranslatorConfigurationException();
108 			Query query = query(mode);
109 			if (locale != null && !"und".equals(locale))
110 				query = mutableQuery(query).add("document-locale", locale);
111 			for (org.daisy.pipeline.braille.common.BrailleTranslator t : translatorRegistry.get(query))
112 				try {
113 					return new BrailleTranslatorFromBrailleTranslator(mode, t.lineBreakingFromStyledText()); }
114 				catch (UnsupportedOperationException e) {}
115 			throw new TranslatorConfigurationException("Factory does not support " + locale + "/" + mode);
116 		}
117 	}

121 	/**
122 	 * {@link BrailleTranslator} wrapper for a {@link org.daisy.pipeline.braille.common.BrailleTranslator}
123 	 *
124 	 * <a href="http://braillespecs.github.io/braille-css/#h3_white-space-processing">White space
125 	 * processing</a> and <a href="http://braillespecs.github.io/braille-css/#line-breaking">line
126 	 * breaking</a> are done according to braille CSS.
127 	 *
128 	 * White space is normalised. Preserved spaces must have been converted to
129 	 * no-break spaces and preserved line feeds must have been converted to
130 	 * &lt;obfl:br/&gt;.
131 	 *
132 	 * Through setHyphenating() the translator can be made to perform automatic hyphenation or
133 	 * not. Regardless of this setting, hyphenation characters (SHY and ZWSP) in the input are used
134 	 * in line breaking, except when overridden with a <code>hyphens: none</code> style.
135 	 *
136 	 * Support <code>text-transform</code> value "<code>-dotify-counter</code>" which causes numbers
137 	 * to be formatted according to the value of the <code>-dotify-counter-style</code> property.
138 	 */
139 	private static class BrailleTranslatorFromBrailleTranslator implements BrailleTranslator {
140 		
141 		final String mode;
142 		final org.daisy.pipeline.braille.common.BrailleTranslator.LineBreakingFromStyledText translator;
143 		
144 		private BrailleTranslatorFromBrailleTranslator(
145 				String mode,
146 				org.daisy.pipeline.braille.common.BrailleTranslator.LineBreakingFromStyledText translator) {
147 			this.mode = mode;
148 			this.translator = translator;
149 		}
150 		
151 		public BrailleTranslatorResult translate(Translatable input) throws TranslationException {
152 			if (input.getAttributes() == null && input.isHyphenating() == false) {
153 				String text = input.getText();
154 				if ("".equals(text))
155 					// see org.daisy.dotify.formatter.impl.row.SegmentProcessor.layoutLeader
156 					return new DefaultLineBreaker.LineIterator("", '\u2800', '\u2824', 1);
157 				if (" ".equals(text))
158 					// If input text is a space, it may be used for calculating the margin character
159 					// (see org.daisy.dotify.formatter.impl.common.FormatterCoreContext)
160 					return new DefaultLineBreaker.LineIterator("\u2800", '\u2800', '\u2824', 1); }
161 			return translate(cssStyledTextFromTranslatable(input), 0, -1);
162 		}
163 		
164 		public BrailleTranslatorResult translate(TranslatableWithContext input) throws TranslationException {
165 			List<CSSStyledText> styledText = Lists.newArrayList(cssStyledTextFromTranslatable(input));
166 			int from = input.getPrecedingText().size();
167 			int to = from + input.getTextToTranslate().size();
168 			for (int i = 0; i < styledText.size(); i++)
169 				if ("??".equals(styledText.get(i).getText()))
170 					// If input text is "??", it will be used for creating a placeholder for content that
171 					// can not be computed yet (see org.daisy.dotify.formatter.impl.row.SegmentProcessor).
172 					// Because normally this will never end up in the resulting PEF, it is okay to replace
173 					// it with "0". We choose a number because in some cases the translator may assume that
174 					// the input is numerical.
175 					styledText.set(i, new CSSStyledText("0", styledText.get(i).getStyle()));
176 			return translate(styledText, from, to);
177 		}
178 		
179 		private BrailleTranslatorResult translate(Iterable<CSSStyledText> styledText, int from, int to) throws TranslationException {
180 			return translator.transform(styledText, from, to);
181 		}
182 		
183 		public String getTranslatorMode() {
184 			return mode;
185 		}
186 	}

195 	 * Convert Translatable specification to text + CSS style. Text attributes are assumed to
196 	 * contain only CSS or special variable assignments/tests. Computation of the text-transform
197 	 * property is assumed to have been performed already.
198 	 */
199 	protected static Iterable<CSSStyledText> cssStyledTextFromTranslatable(Translatable specification) {
200 		return handleCounterStyles(
201 			cssStyledTextFromTranslatable(
202 				specification.getText(),
203 				specification.getAttributes(),
204 				specification.isHyphenating(),
205 				null));
206 	}

208 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(String text,
209 	                                                                     TextAttribute attributes,
210 	                                                                     boolean hyphenating,
211 	                                                                     SimpleInlineStyle parentStyle) {
212 		if (attributes != null && attributes.getWidth() != text.length())
213 			throw new RuntimeException("Coding error");
214 		SimpleInlineStyle style; {
215 			String s = attributes != null ? attributes.getDictionaryIdentifier() : null;
216 			if (s == null && parentStyle == null)
217 				style = null;
218 			else
219 				// FIXME: extend caching of parsed CSS to support parentStyle!
220 				style = new SimpleInlineStyle(s != null ? s : "", parentStyle);
221 			if (hyphenating && (style == null || style.getProperty("hyphens") == null))
222 				// FIXME: add hyphens declaration through braille-css model
223 				style = new SimpleInlineStyle("hyphens: auto", style); }
224 		if (attributes != null && attributes.hasChildren())
225 			return cssStyledTextFromTranslatable(text, attributes.iterator(), false, style);
226 		else
227 			return Collections.singleton(new CSSStyledText(text, style));
228 	}

230 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(String text,
231 	                                                                     Iterator<TextAttribute> attributes,
232 	                                                                     boolean hyphenating,
233 	                                                                     SimpleInlineStyle parentStyle) {
234 		if (attributes.hasNext()) {
235 			TextAttribute a = attributes.next();
236 			int w = a.getWidth();
237 			return concat(cssStyledTextFromTranslatable(text.substring(0, w), a, hyphenating, parentStyle),
238 			              cssStyledTextFromTranslatable(text.substring(w), attributes, hyphenating, parentStyle)); }
239 		else
240 			return empty;
241 	}

250 			cssStyledTextFromTranslatable(
251 				specification.getPrecedingText(),
252 				specification.getTextToTranslate(),
253 				specification.getFollowingText(),
254 				specification.getAttributes().orElse(null),

258 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<PrecedingText> preceding,
259 	                                                                     List<ResolvableText> text,
260 	                                                                     List<FollowingText> following,
261 	                                                                     AttributeWithContext attributes,
262 	                                                                     SimpleInlineStyle parentStyle) {
263 		int textSize = text == null ? 0 : text.size();
264 		int precedingSize = preceding == null ? 0 : preceding.size();
265 		int followingSize = following == null ? 0 : following.size();
266 		if (attributes != null && attributes.getWidth() != precedingSize + textSize + followingSize)
267 			throw new RuntimeException("Coding error");
268 		SimpleInlineStyle style; {
269 			String s = attributes != null ? attributes.getName().orElse(null) : null;
270 			if (s == null && parentStyle == null)
271 				style = null;
272 			else
273 				// FIXME: extend caching of parsed CSS to support parentStyle!
274 				style = new SimpleInlineStyle(s != null ? s : "", parentStyle);
275 		}
276 		if (attributes != null && attributes.hasChildren())
277 			return cssStyledTextFromTranslatable(preceding, text, following, attributes.iterator(), style);
278 		else
279 			return concat(cssStyledTextFromTranslatable(preceding, style),
280 			              cssStyledTextFromTranslatable(text, style),
281 			              cssStyledTextFromTranslatable(following, style));
282 	}

284 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<PrecedingText> preceding,
285 	                                                                     List<ResolvableText> text,
286 	                                                                     List<FollowingText> following,
287 	                                                                     Iterator<AttributeWithContext> attributes,
288 	                                                                     SimpleInlineStyle parentStyle) {
289 		int textSize = text == null ? 0 : text.size();
290 		int precedingSize = preceding == null ? 0 : preceding.size();
291 		int followingSize = following == null ? 0 : following.size();
292 		if (attributes.hasNext()) {
293 			AttributeWithContext a = attributes.next();
294 			int w = a.getWidth();
295 			if (w == 0)
296 				return cssStyledTextFromTranslatable(preceding, text, following, attributes, parentStyle);
297 			if (w <= precedingSize)
298 				return concat(
299 					cssStyledTextFromTranslatable(preceding.subList(0, w), null, null, a, parentStyle),
300 					cssStyledTextFromTranslatable(preceding.subList(w, precedingSize), text, following, attributes, parentStyle));
301 			else if (w - precedingSize <= textSize)
302 				return concat(
303 					cssStyledTextFromTranslatable(preceding, text.subList(0, w - precedingSize), null, a, parentStyle),
304 					cssStyledTextFromTranslatable(null, text.subList(w - precedingSize, textSize), following, attributes, parentStyle));
305 			else
306 				return concat(
307 					cssStyledTextFromTranslatable(preceding, text, following.subList(0, w - precedingSize - textSize), a, parentStyle),
308 					cssStyledTextFromTranslatable(null, null, following.subList(w - precedingSize - textSize, followingSize), attributes, parentStyle));
309 		} else
310 			return empty;
311 	}

313 	private static Iterable<CSSStyledText> cssStyledTextFromTranslatable(List<? extends Object/*PrecedingText|FollowingText*/> precedingOrFollowingText,
314 	                                                                     SimpleInlineStyle parentStyle) {
315 		if (precedingOrFollowingText == null)
316 			return empty;
317 		List<CSSStyledText> styledText = new ArrayList<>();
318 		for (Object t : precedingOrFollowingText) {
319 			String text;
320 			boolean hyphenate; {
321 				if (t instanceof PrecedingText) {
322 					text = ((PrecedingText)t).resolve();
323 					hyphenate = ((PrecedingText)t).shouldHyphenate(); }
324 				else if (t instanceof FollowingText) {
325 					text = ((FollowingText)t).peek();
326 					hyphenate = ((FollowingText)t).shouldHyphenate(); }
327 				else
328 					throw new RuntimeException();
329 			}
330 			SimpleInlineStyle style; {
331 				if (hyphenate && (parentStyle == null || parentStyle.getProperty("hyphens") == null))
332 					style = new SimpleInlineStyle("hyphens: auto", parentStyle);
333 				else if (parentStyle == null)
334 					style = null;
335 				else
336 					style = new SimpleInlineStyle("", parentStyle);
337 			}
338 			styledText.add(new CSSStyledText(text, style));
339 		}
340 		return styledText;
341 	}

345 	private static Iterable<CSSStyledText> handleCounterStyles(Iterable<CSSStyledText> styledText) {
346 		List<CSSStyledText> segments = new ArrayList<CSSStyledText>();
347 		String segment = null;
348 		SimpleInlineStyle style = null;
349 		Map<String,String> attrs = null;
350 		for (CSSStyledText st : styledText) {
351 			String t = st.getText();
352 			SimpleInlineStyle s = st.getStyle();
353 			Map<String,String> a = st.getTextAttributes();
354 			if (s != null) {
355 				if (s.getProperty("text-transform") == TextTransform.list_values) {
356 					TermList list = s.getValue(TermList.class, "text-transform");
357 					if (((TermIdent)list.get(0)).getValue().equals("-dotify-counter")) {
358 						if (list.size() == 1)
359 							s.removeProperty("text-transform");
360 						else
361 							list.remove(0);
362 						if ("??".equals(t)) {
363 						} else {
364 							int counterValue = Integer.parseInt(t);
365 							Term<?> counterStyle = s.getValue("-dotify-counter-style");
366 							if (counterStyle != null) {
367 								try {
368 									t = new CounterStyle(counterStyle).format(counterValue); }
369 								catch (IllegalArgumentException e) {}}}}}
370 				s.removeProperty("-dotify-counter-style"); }
371 			if (segment != null)
372 				segments.add(new CSSStyledText(segment, style, attrs));
373 			segment = t;
374 			style = s;
375 			attrs = a; }
376 		if (segment != null)
377 			segments.add(new CSSStyledText(segment, style, attrs));
378 		return segments;
379 	}
