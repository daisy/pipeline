30  @Component(
31  	name = "org.daisy.pipeline.braille.dotify.impl.DotifyCSSExtension",
32  	service = {
33  		BrailleCSSExtension.class
34  	}

36  public class DotifyCSSExtension extends BrailleCSSExtension {
37  
38  	private final static String prefix = "-dotify-";
39  
40  	public DotifyCSSExtension() {
41  		// SupportedDotifyProperties is defined at the bottom of this file
42  		// it is a separate class because we need an argument to pass to the constructor of BrailleCSSExtension
43  		super(new SupportedDotifyProperties(prefix));
44  		this.methods = findParsingMethods();
45  	}
46  
47  	///////////////////////////////////////////////////////////////
48  	// BrailleCSSExtension
49  	///////////////////////////////////////////////////////////////
50  
51  	@Override
52  	public String getPrefix() {
53  		return prefix;
54  	}
55  
56  	public TermIdent parseTextTransform(Term<?> term) {
57  		if (term instanceof TermIdent) {
58  			TermIdent ident = (TermIdent)term;
59  			if ((prefix + "counter").equals(ident.getValue()))
60  				return ident;
61  		}
62  		throw new IllegalArgumentException("Unknown text-transform value " + term);
63  	}
64  
65  	///////////////////////////////////////////////////////////////
66  	// DeclarationTransformer
67  	///////////////////////////////////////////////////////////////
68  
69  	private Map<String,Method> findParsingMethods() {
70  		Map<String,Method> map = new HashMap<String,Method>(css.getTotalProperties(), 1.0f);
71  		for (String property : css.getDefinedPropertyNames()) {
72  			try {
73  				Method m = DotifyCSSExtension.class.getDeclaredMethod(
74  					camelCase("process-" + property.substring(prefix.length())),
75  					Declaration.class, Map.class, Map.class);
76  				map.put(property, m);
77  			} catch (Exception e) {
78  				log.debug("Unable to find method for property {}.", property); // should not happen
79  			}
80  		}
81  		log.debug("Totally found {} parsing methods", map.size());
82  		return map;
83  	}
84  
85  	@Override
86  	public boolean parseDeclaration(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
87  		String property = d.getProperty().toLowerCase();
88  		if (!(css.isSupportedCSSProperty(property)))
89  			return false;
90  		try {
91  			Method m = methods.get(property);
92  			if (m != null)
93  				try {
94  					return (Boolean)m.invoke(this, d, properties, values);
95  				} catch (IllegalAccessException e) {
96  				} catch (IllegalArgumentException e) {
97  				} catch (InvocationTargetException e) {
98  				}
99  		} catch (Exception e) {
100 		}
101 		return false;
102 	}
103 
104 	@SuppressWarnings("unused")
105 	private boolean processCounterStyle(Declaration d, Map<String, CSSProperty> properties, Map<String, Term<?>> values) {
106 		String propertyName = d.getProperty();
107 		if (d.size() != 1)
108 			return false;
109 		Term<?> term = d.get(0);
110 		if (genericTermIdent(ListStyleType.class, term, ALLOW_INH, propertyName, properties))
111 			return true;
112 		else
113 			try {
114 				if (term instanceof TermIdent) {
115 					properties.put(propertyName, ListStyleType.counter_style_name);
116 					values.put(propertyName, term);
117 					return true; }
118 				else if (TermFunction.class.isInstance(term)
119 				         && "symbols".equals(((TermFunction)term).getFunctionName().toLowerCase())) {
120 					properties.put(propertyName, ListStyleType.symbols_fn);
121 					values.put(propertyName, term);
122 					return true; }}
123 			catch (Exception e) {}
124 		return false;
125 	}
126 }

128 class SupportedDotifyProperties implements SupportedCSS {
129 
130 	private static Logger log = LoggerFactory.getLogger(SupportedDotifyProperties.class);
131 
132 	private static final int TOTAL_SUPPORTED_DECLARATIONS = 1;
133 
134 	private static final TermFactory tf = CSSFactory.getTermFactory();
135 
136 	private Set<String> supportedCSSproperties;
137 	private Map<String,CSSProperty> defaultCSSproperties;
138 	private Map<String,Integer> ordinals;
139 	private Map<Integer,String> ordinalsRev;
140 
141 	SupportedDotifyProperties(String prefix) {
142 		this.setSupportedCSS(prefix);
143 		this.setOridinals();
144 	}
145 
146 	@Override
147 	public boolean isSupportedMedia(String media) {
148 		if (media == null)
149 			return false;
150 		return media.toLowerCase().equals("embossed");
151 	}
152 
153 	@Override
154 	public final boolean isSupportedCSSProperty(String property) {
155 		return supportedCSSproperties.contains(property);
156 	}
157 
158 	@Override
159 	public final CSSProperty getDefaultProperty(String property) {
160 		CSSProperty value = defaultCSSproperties.get(property);
161 		return value;
162 	}
163 
164 	@Override
165 	public final Term<?> getDefaultValue(String property) {
166 		return null;
167 	}
168 
169 	@Override
170 	public final int getTotalProperties() {
171 		return defaultCSSproperties.size();
172 	}
173 
174 	@Override
175 	public final Set<String> getDefinedPropertyNames() {
176 		return defaultCSSproperties.keySet();
177 	}
178 
179 	@Override
180 	public String getRandomPropertyName() {
181 		final Random generator = new Random();
182 		int o = generator.nextInt(getTotalProperties());
183 		return getPropertyName(o);
184 	}
185 
186 	@Override
187 	public int getOrdinal(String propertyName) {
188 		Integer i = ordinals.get(propertyName);
189 		return (i == null) ? -1 : i.intValue();
190 	}
191 
192 	@Override
193 	public String getPropertyName(int o) {
194 		return ordinalsRev.get(o);
195 	}
196 
197 	private void setProperty(String name, CSSProperty defaultProperty) {
198 		supportedCSSproperties.add(name);
199 		defaultCSSproperties.put(name, defaultProperty);
200 	}
201 
202 	private void setSupportedCSS(String prefix) {
203 		supportedCSSproperties = new HashSet<String>(TOTAL_SUPPORTED_DECLARATIONS, 1.0f);
204 		defaultCSSproperties = new HashMap<String,CSSProperty>(TOTAL_SUPPORTED_DECLARATIONS, 1.0f);
205 		setProperty(prefix + "counter-style", ListStyleType.NONE);
206 	}
207 
208 	private void setOridinals() {
209 		Map<String,Integer> ords = new HashMap<String,Integer>(getTotalProperties(), 1.0f);
210 		Map<Integer,String> ordsRev = new HashMap<Integer,String>(getTotalProperties(), 1.0f);
211 		int i = 0;
212 		for (String key : defaultCSSproperties.keySet()) {
213 			ords.put(key, i);
214 			ordsRev.put(i, key);
215 			i++;
216 		}
217 		this.ordinals = ords;
218 		this.ordinalsRev = ordsRev;
219 	}
220 }
