166 	@Override
167 	public void run() throws SaxonApiException {
168 		super.run();
169 		Function0<Void> evictTempTranslator = Functions.noOp; // any temporary translators that are created specially
170 		                                                      // for this conversion need to be destroyed afterwards
171 		try {
172 			
173 			XdmNode obflNode = source.read();
174 			String styleType = getOption(_style_type, "");
175 			String mode = getOption(_mode).getString();
176 			String locale = getOption(_locale).getString();
177 			String brailleCharset = getOption(_braille_charset, "");
178 
179 			if (brailleCharset != null && !"".equals(brailleCharset)) {
180 				// if braille-charset is specified we can assume that mode is in query format
181 				Query modeQuery;
182 				try {
183 					modeQuery = query(mode);
184 				} catch (Exception e) {
185 					throw new IllegalArgumentException("Expected mode in query format: " + mode, e);
186 				}
187 				mode = mutableQuery(modeQuery).add("braille-charset", brailleCharset).toString();
188 			}
189 
190 			if ("text/css".equals(styleType)) {
191 				
192 				// We're assuming that no other translators than the DAISY Pipeline implementations
193 				// support text/css.
194 				Query mainQuery;
195 				try {
196 					mainQuery = query(mode);
197 				} catch (Exception e) {
198 					throw new IllegalArgumentException("Expected mode in query format: " + mode, e);
199 				}
200 				mainQuery = mutableQuery(mainQuery).add("input", "text-css").add("output", "braille");
201 				mode = mainQuery.toString();
202 				if (locale != null && !"und".equals(locale))
203 					mainQuery = mutableQuery(mainQuery).add("document-locale", locale);
204 				BrailleTranslator mainTranslator;
205 				try {
206 					mainTranslator = brailleTranslatorRegistry.get(mainQuery).iterator().next();
207 				} catch (NoSuchElementException e) {
208 					throw new XProcException(
209 						step.getNode(),
210 						"No translator available for mode '" + mode + "' and locale '" + locale + "' "
211 						+ "that supports style type " + styleType);
212 				}
213 				BrailleTranslator translator = mainTranslator;
214 				String textTransformDefinitions = getOption(_css_text_transform_definitions, "");
215 				if (!"".equals(textTransformDefinitions)) {
216 					Map<String,Query> subQueries = TextTransformParser.getBrailleTranslatorQueries(textTransformDefinitions,
217 					                                                                               obflNode.getBaseURI(),
218 					                                                                               mainQuery);
219 					if (subQueries != null && !subQueries.isEmpty()) {
220 						BrailleTranslator defaultTranslator = mainTranslator;
221 						Query defaultQuery = subQueries.remove("auto");
222 						if (defaultQuery != null && !defaultQuery.equals(mainQuery))
223 							try {
224 								defaultTranslator = brailleTranslatorRegistry.get(defaultQuery).iterator().next();
225 							} catch (NoSuchElementException e) {
226 								throw new XProcException(
227 									step.getNode(), "No translator available for " + defaultQuery + "");
228 							}
229 						if (defaultTranslator != mainTranslator || !subQueries.isEmpty()) {
230 							Map<String,Supplier<BrailleTranslator>> subTranslators
231 								= Maps.transformValues(
232 									subQueries,
233 									q -> () -> brailleTranslatorRegistry.get(q).iterator().next());
234 							translator = new CompoundBrailleTranslator(defaultTranslator, subTranslators);
235 						}
236 					}
237 				}
238 				String counterStyleDefinitions = getOption(_css_counter_style_definitions, "");
239 				Map<String,CounterStyle> counterStyles = "".equals(counterStyleDefinitions)
240 					? null
241 					: CounterStyle.parseCounterStyleRules(
242 						Iterables.filter(new InlineStyle(counterStyleDefinitions), RuleCounterStyle.class));
243 				// handle text-transform: -dotify-counter
244 				translator = new CounterHandlingBrailleTranslator(translator, counterStyles);
245 				evictTempTranslator = temporaryBrailleTranslatorProvider.provideTemporarily(translator);
246 				mode = mutableQuery().add("id", translator.getIdentifier()).toString();
247 				locale = "und";
248 			} else if (!"".equals(styleType)) {
249 				throw new XProcException(step.getNode(), "Value of style-type option not recognized: " + styleType);
250 			}
251 			
252 			// Read OBFL
253 			ByteArrayOutputStream s = new ByteArrayOutputStream();
254 			Serializer serializer = runtime.getProcessor().newSerializer();
255 			serializer.setOutputStream(s);
256 			serializer.setCloseOnCompletion(true);
257 			serializer.serializeNode(obflNode);
258 			serializer.close();
259 			InputStream obflStream = new ByteArrayInputStream(s.toByteArray());
260 			s.close();
261 			
262 			// Convert
263 			String identifier = getOption(_identifier, "");
264 			boolean markCapitalLetters; {
265 				String p = params.get("mark-capital-letters");
266 				markCapitalLetters = (p == null) ? true : !p.equalsIgnoreCase("false"); }
267 			boolean hyphenate; {
268 				String p = params.get("hyphenate");
269 				hyphenate = (p == null) ? true : !p.equalsIgnoreCase("false"); }
270 			boolean allowTextOverflowTrimming; {
271 				String p = params.get("allow-text-overflow-trimming");
272 				allowTextOverflowTrimming = (p == null) ? false : p.equalsIgnoreCase("true"); }
273 			boolean removeStyles; {
274 				String p = params.get("remove-styles");
275 				removeStyles = (p == null) ? false : p.equalsIgnoreCase("true"); }
276 			boolean allowEndingPageOnHyphen;
277 			boolean allowEndingVolumeOnHyphen; {
278 				String p = params.get("hyphenation-at-page-breaks");
279 				if ("false".equalsIgnoreCase(p)) {
280 					allowEndingPageOnHyphen = false;
281 					allowEndingVolumeOnHyphen = false;
282 				} else if ("except-at-volume-breaks".equalsIgnoreCase(p)) {
283 					allowEndingPageOnHyphen = true;
284 					allowEndingVolumeOnHyphen = false;
285 				} else { // true (default)
286 					allowEndingPageOnHyphen = true;
287 					// Follow the OBFL standard which says that "when volume-transition is
288 					// present, the last page or sheet in each volume may be modified so that
289 					// the volume break occurs earlier than usual: preferably between two
290 					// blocks, or if that is not possible, between words"
291 					// (http://braillespecs.github.io/obfl/obfl-specification.html#L8701). In
292 					// other words, volumes should by default not be allowed to end on a hyphen.
293 					allowEndingVolumeOnHyphen = !getOption(_has_volume_transition, false);
294 				}
295 			}
296 			FormatterConfiguration.Builder config = FormatterConfiguration.with(locale, mode)
297 				.markCapitalLetters(markCapitalLetters)
298 				.hyphenate(hyphenate)
299 				.allowsTextOverflowTrimming(allowTextOverflowTrimming)
300 				.allowsEndingPageOnHyphen(allowEndingPageOnHyphen)
301 				.allowsEndingVolumeOnHyphen(allowEndingVolumeOnHyphen);
302 			if (removeStyles)
303 				config.ignoreStyle("em").ignoreStyle("strong");
304 			Table brailleCharsetTable = "".equals(brailleCharset)
305 				? null
306 				: tableRegistry.get(mutableQuery().add("id", brailleCharset)).iterator().next();
307 			FormatterEngine engine = newFormatterEngine(config.build(),
308 			                                            newPEFWriter(identifier, brailleCharsetTable),
309 			                                            brailleCharsetTable == null
310 			                                                ? null
311 			                                                : brailleCharsetTable.newBrailleConverter());
312 			s = new ByteArrayOutputStream();
313 			engine.convert(obflStream, s);
314 			obflStream.close();
315 			InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
316 			s.close();
317 			
318 			// Write PEF
319 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(pefStream)));
320 			pefStream.close(); }
321 		
322 		catch (Throwable e) {
323 			String msg = e.getMessage();
324 			if (msg != null)
325 				if ((msg.contains("Cannot fit") && msg.contains("into a margin-region of size"))
326 				    || msg.contains("Failed to solve table"))
327 					throw new XProcException(DOTIFY_FAILURE, step, e);
328 			throw XProcStep.raiseError(e, step); }
329 		finally {
330 			evictTempTranslator.apply(); }
331 	}

333 	private PagedMediaWriter newPEFWriter(String identifier, Table brailleCharset) throws PagedMediaWriterConfigurationException {
334 		PagedMediaWriter writer = null;
335 		List<MetaDataItem> meta = new ArrayList<MetaDataItem>();
336 		if (brailleCharset != null)
337 			meta.add(new MetaDataItem(new javax.xml.namespace.QName("http://www.daisy.org/ns/pipeline/",
338 			                                                        "ascii-braille-charset",
339 			                                                        "dp2"),
340 			                          brailleCharset.getIdentifier()));
341 		if (!"".equals(identifier))
342 			meta.add(new MetaDataItem(new javax.xml.namespace.QName("http://purl.org/dc/elements/1.1/", "identifier", "dc"),
343 			                          identifier));
344 		writer = new PEFWriter(brailleCharset);
345 		if (!meta.isEmpty())
346 			writer.prepare(meta);
347 		return writer;
348 	}

350 	private FormatterEngine newFormatterEngine(FormatterConfiguration config,
351 	                                           PagedMediaWriter writer,
352 	                                           BrailleConverter brailleCharset) {
353 		// HACK: If a BrailleConverter is specified, we create a new TextBorderFactoryMakerService
354 		// that uses the default TextBorderFactoryService to create Unicode braille patterns and
355 		// encodes them with the given BrailleConverter. We then make the default
356 		// FormatterEngineFactoryService, FormatterFactory and ObflParserFactoryService use this
357 		// TextBorderFactoryMakerService by using reflection. This assumes that the
358 		// FormatterEngineFactoryService is a
359 		// org.daisy.dotify.formatter.impl.engine.LayoutEngineFactoryImpl, the FormatterFactory is a
360 		// org.daisy.dotify.formatter.impl.FormatterFactoryImpl, and the ObflParserFactoryService is
361 		// a org.daisy.dotify.formatter.impl.obfl.ObflParserFactoryImpl, and that these are the only
362 		// classes that bind a TextBorderFactoryMakerService. If no BrailleConverter is specified,
363 		// we need to reset the TextBorderFactoryMakerService of the default
364 		// FormatterEngineFactoryService, FormatterFactory and ObflParserFactoryService.
365 		TextBorderFactoryMakerService asciiTextBorderFactoryMakerService = new TextBorderFactoryMakerService() {
366 				public TextBorderStyle newTextBorderStyle(Map<String,Object> features)
367 						throws TextBorderConfigurationException {
368 					TextBorderFactory f = textBorderFactoryService.newFactory();
369 					for (String k : features.keySet())
370 						f.setFeature(k, features.get(k));
371 					TextBorderStyle style = f.newTextBorderStyle();
372 					if (brailleCharset == null)
373 						return style;
374 					return new TextBorderStyle.Builder()
375 						.topLeftCorner    (brailleCharset.toText(style.getTopLeftCorner()))
376 						.topBorder        (brailleCharset.toText(style.getTopBorder()))
377 						.topRightCorner   (brailleCharset.toText(style.getTopRightCorner()))
378 						.leftBorder       (brailleCharset.toText(style.getLeftBorder()))
379 						.rightBorder      (brailleCharset.toText(style.getRightBorder()))
380 						.bottomLeftCorner (brailleCharset.toText(style.getBottomLeftCorner()))
381 						.bottomBorder     (brailleCharset.toText(style.getBottomBorder()))
382 						.bottomRightCorner(brailleCharset.toText(style.getBottomRightCorner()))
383 						.build();
384 				}
385 			};
386 		for (Object object : new Object[]{formatterEngineFactoryService,
387 		                                  formatterFactory,
388 		                                  obflParserFactoryService}) {
389 			try {
390 				object.getClass()
391 					.getMethod(object == formatterFactory ? "setTextBorderFactory" : "setTextBorderFactoryMaker",
392 					           TextBorderFactoryMakerService.class)
393 					.invoke(object, asciiTextBorderFactoryMakerService);
394 			} catch (NoSuchMethodException |
395 			         SecurityException |
396 			         IllegalAccessException |
397 			         IllegalArgumentException |
398 			         InvocationTargetException e) {
399 				throw new RuntimeException(e);
400 			}
401 		}
402 		return formatterEngineFactoryService.newFormatterEngine(config, writer);
403 	}
