134 	@Override
135 	public void run() throws SaxonApiException {
136 		super.run();
137 		Function0<Void> evictTempTranslator = Functions.noOp; // any temporary translators that are created specially
138 		                                                      // for this conversion need to be destroyed afterwards
139 		try {
140 			
141 			XdmNode obflNode = source.read();
142 			String styleType = getOption(_style_type, "");
143 			String mode = getOption(_mode).getString();
144 			String locale = getOption(_locale).getString();
145 			
146 			if ("text/css".equals(styleType)) {
147 				
148 				// We're assuming that no other translators than the DAISY Pipeline implementations
149 				// support text/css.
150 				Query mainQuery;
151 				try {
152 					mainQuery = query(mode);
153 				} catch (Exception e) {
154 					throw new IllegalArgumentException("Expected mode in query format: " + mode, e);
155 				}
156 				if (locale != null && !"und".equals(locale))
157 					mainQuery = mutableQuery(mainQuery).add("locale", locale);
158 				BrailleTranslator mainTranslator;
159 				try {
160 					mainTranslator = brailleTranslatorProvider.get(mainQuery).iterator().next();
161 				} catch (NoSuchElementException e) {
162 					throw new XProcException(
163 						step.getNode(),
164 						"No translator available for mode '" + mode + "' and locale '" + locale + "' "
165 						+ "that supports style type " + styleType);
166 				}
167 				String textTransformDefinitions = getOption(_css_text_transform_definitions, "");
168 				if (!"".equals(textTransformDefinitions)) {
169 					Map<String,Query> subQueries = TextTransformParser.getBrailleTranslatorQueries(textTransformDefinitions,
170 					                                                                               obflNode.getBaseURI(),
171 					                                                                               mainQuery);
172 					if (subQueries != null && !subQueries.isEmpty()) {
173 						BrailleTranslator defaultTranslator = mainTranslator;
174 						Query defaultQuery = subQueries.remove("auto");
175 						if (defaultQuery != null && !defaultQuery.equals(mainQuery))
176 							try {
177 								defaultTranslator = brailleTranslatorProvider.get(defaultQuery).iterator().next();
178 							} catch (NoSuchElementException e) {
179 								throw new XProcException(
180 									step.getNode(), "No translator available for " + defaultQuery + "");
181 							}
182 						if (defaultTranslator != mainTranslator || !subQueries.isEmpty()) {
183 							Map<String,Supplier<BrailleTranslator>> subTranslators
184 								= Maps.transformValues(
185 									subQueries,
186 									q -> () -> brailleTranslatorProvider.get(q).iterator().next());
187 							BrailleTranslator compoundTranslator = new CompoundTranslator(defaultTranslator, subTranslators);
188 							evictTempTranslator = temporaryBrailleTranslatorProvider.provideTemporarily(compoundTranslator);
189 							mode = mutableQuery().add("id", compoundTranslator.getIdentifier()).toString();
190 							locale = "und";
191 						}
192 					}
193 				}
194 			} else if (!"".equals(styleType)) {
195 				throw new XProcException(step.getNode(), "Value of style-type option not recognized: " + styleType);
196 			}
197 			
198 			// Read OBFL
199 			ByteArrayOutputStream s = new ByteArrayOutputStream();
200 			Serializer serializer = runtime.getProcessor().newSerializer();
201 			serializer.setOutputStream(s);
202 			serializer.setCloseOnCompletion(true);
203 			serializer.serializeNode(obflNode);
204 			serializer.close();
205 			InputStream obflStream = new ByteArrayInputStream(s.toByteArray());
206 			s.close();
207 			
208 			// Convert
209 			// FIXME: duplication with DotifyTaskSystem! => use that class in
210 			// here (like in XMLToOBFL) when it supports setting of the mode
211 			String identifier = getOption(_identifier, "");
212 			boolean markCapitalLetters; {
213 				String p = params.get("mark-capital-letters");
214 				markCapitalLetters = (p == null) ? true : !p.equalsIgnoreCase("false"); }
215 			boolean hyphenate; {
216 				String p = params.get("hyphenate");
217 				hyphenate = (p == null) ? true : !p.equalsIgnoreCase("false"); }
218 			boolean allowTextOverflowTrimming; {
219 				String p = params.get("allow-text-overflow-trimming");
220 				allowTextOverflowTrimming = (p == null) ? false : p.equalsIgnoreCase("true"); }
221 			boolean removeStyles; {
222 				String p = params.get("remove-styles");
223 				removeStyles = (p == null) ? false : p.equalsIgnoreCase("true"); }
224 			boolean allowEndingVolumeOnHyphen; {
225 				String p = params.get("allow-ending-volume-on-hyphen");
226 				allowEndingVolumeOnHyphen = (p == null) ? true : p.equalsIgnoreCase("true"); }
227 			FormatterConfiguration.Builder config = FormatterConfiguration.with(locale, mode)
228 				.markCapitalLetters(markCapitalLetters)
229 				.hyphenate(hyphenate)
230 				.allowsTextOverflowTrimming(allowTextOverflowTrimming)
231 				.allowsEndingVolumeOnHyphen(allowEndingVolumeOnHyphen);
232 			if (removeStyles)
233 				config.ignoreStyle("em").ignoreStyle("strong");
234 			
235 			FormatterEngine engine = newFormatterEngine(config.build(), identifier);
236 			s = new ByteArrayOutputStream();
237 			engine.convert(obflStream, s);
238 			obflStream.close();
239 			InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
240 			s.close();
241 			
242 			// Write PEF
243 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(pefStream)));
244 			pefStream.close(); }
245 		
246 		catch (Throwable e) {
247 			if (e.getMessage().contains("Failed to solve table"))
248 				throw new XProcException(DOTIFY_ERROR, step, e);
249 			throw XProcStep.raiseError(e, step); }
250 		finally {
251 			evictTempTranslator.apply(); }
252 	}

254 	private PagedMediaWriter newPagedMediaWriter(String target) throws PagedMediaWriterConfigurationException {
255 		target = target.toLowerCase();
256 		for (PagedMediaWriterFactoryService s : writerFactoryServices)
257 			if (s.supportsMediaType(target))
258 				return s.newFactory(target).newPagedMediaWriter();
259 		throw new RuntimeException("Cannot find a PagedMediaWriter factory for " + target);
260 	}
261 	
262 	private PagedMediaWriter newPEFWriter(String identifier) throws PagedMediaWriterConfigurationException {
263 		PagedMediaWriter ret = newPagedMediaWriter(MediaTypes.PEF_MEDIA_TYPE);
264 		if (!"".equals(identifier)) {
265 			List<MetaDataItem> meta = new ArrayList<MetaDataItem>();
266 			javax.xml.namespace.QName name = new javax.xml.namespace.QName("http://purl.org/dc/elements/1.1/", "identifier", "dc");
267 			meta.add(new MetaDataItem(name, identifier));
268 			ret.prepare(meta);
269 		}
270 		return ret;
271 	}
