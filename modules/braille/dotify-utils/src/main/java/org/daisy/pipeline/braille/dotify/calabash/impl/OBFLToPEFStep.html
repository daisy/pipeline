158 	@Override
159 	public void run() throws SaxonApiException {
160 		super.run();
161 		Function0<Void> evictTempTranslator = Functions.noOp; // any temporary translators that are created specially
162 		                                                      // for this conversion need to be destroyed afterwards
163 		try {
164 			
165 			XdmNode obflNode = source.read();
166 			String styleType = getOption(_style_type, "");
167 			String mode = getOption(_mode).getString();
168 			String locale = getOption(_locale).getString();
169 			String brailleCharset = getOption(_braille_charset, "");
170 
171 			if (brailleCharset != null && !"".equals(brailleCharset)) {
172 				// if braille-charset is specified we can assume that mode is in query format
173 				Query modeQuery;
174 				try {
175 					modeQuery = query(mode);
176 				} catch (Exception e) {
177 					throw new IllegalArgumentException("Expected mode in query format: " + mode, e);
178 				}
179 				mode = mutableQuery(modeQuery).add("braille-charset", brailleCharset).toString();
180 			}
181 
182 			if ("text/css".equals(styleType)) {
183 				
184 				// We're assuming that no other translators than the DAISY Pipeline implementations
185 				// support text/css.
186 				Query mainQuery;
187 				try {
188 					mainQuery = query(mode);
189 				} catch (Exception e) {
190 					throw new IllegalArgumentException("Expected mode in query format: " + mode, e);
191 				}
192 				mainQuery = mutableQuery(mainQuery).add("input", "text-css").add("output", "braille");
193 				mode = mainQuery.toString();
194 				if (locale != null && !"und".equals(locale))
195 					mainQuery = mutableQuery(mainQuery).add("document-locale", locale);
196 				BrailleTranslator mainTranslator;
197 				try {
198 					mainTranslator = brailleTranslatorRegistry.get(mainQuery).iterator().next();
199 				} catch (NoSuchElementException e) {
200 					throw new XProcException(
201 						step.getNode(),
202 						"No translator available for mode '" + mode + "' and locale '" + locale + "' "
203 						+ "that supports style type " + styleType);
204 				}
205 				String textTransformDefinitions = getOption(_css_text_transform_definitions, "");
206 				if (!"".equals(textTransformDefinitions)) {
207 					Map<String,Query> subQueries = TextTransformParser.getBrailleTranslatorQueries(textTransformDefinitions,
208 					                                                                               obflNode.getBaseURI(),
209 					                                                                               mainQuery);
210 					if (subQueries != null && !subQueries.isEmpty()) {
211 						BrailleTranslator defaultTranslator = mainTranslator;
212 						Query defaultQuery = subQueries.remove("auto");
213 						if (defaultQuery != null && !defaultQuery.equals(mainQuery))
214 							try {
215 								defaultTranslator = brailleTranslatorRegistry.get(defaultQuery).iterator().next();
216 							} catch (NoSuchElementException e) {
217 								throw new XProcException(
218 									step.getNode(), "No translator available for " + defaultQuery + "");
219 							}
220 						if (defaultTranslator != mainTranslator || !subQueries.isEmpty()) {
221 							Map<String,Supplier<BrailleTranslator>> subTranslators
222 								= Maps.transformValues(
223 									subQueries,
224 									q -> () -> brailleTranslatorRegistry.get(q).iterator().next());
225 							BrailleTranslator compoundTranslator = new CompoundBrailleTranslator(defaultTranslator, subTranslators);
226 							evictTempTranslator = temporaryBrailleTranslatorProvider.provideTemporarily(compoundTranslator);
227 							mode = mutableQuery().add("id", compoundTranslator.getIdentifier()).toString();
228 							locale = "und";
229 						}
230 					}
231 				}
232 			} else if (!"".equals(styleType)) {
233 				throw new XProcException(step.getNode(), "Value of style-type option not recognized: " + styleType);
234 			}
235 			
236 			// Read OBFL
237 			ByteArrayOutputStream s = new ByteArrayOutputStream();
238 			Serializer serializer = runtime.getProcessor().newSerializer();
239 			serializer.setOutputStream(s);
240 			serializer.setCloseOnCompletion(true);
241 			serializer.serializeNode(obflNode);
242 			serializer.close();
243 			InputStream obflStream = new ByteArrayInputStream(s.toByteArray());
244 			s.close();
245 			
246 			// Convert
247 			// FIXME: duplication with DotifyTaskSystem! => use that class in
248 			// here (like in XMLToOBFL) when it supports setting of the mode
249 			String identifier = getOption(_identifier, "");
250 			boolean markCapitalLetters; {
251 				String p = params.get("mark-capital-letters");
252 				markCapitalLetters = (p == null) ? true : !p.equalsIgnoreCase("false"); }
253 			boolean hyphenate; {
254 				String p = params.get("hyphenate");
255 				hyphenate = (p == null) ? true : !p.equalsIgnoreCase("false"); }
256 			boolean allowTextOverflowTrimming; {
257 				String p = params.get("allow-text-overflow-trimming");
258 				allowTextOverflowTrimming = (p == null) ? false : p.equalsIgnoreCase("true"); }
259 			boolean removeStyles; {
260 				String p = params.get("remove-styles");
261 				removeStyles = (p == null) ? false : p.equalsIgnoreCase("true"); }
262 			boolean allowEndingVolumeOnHyphen; {
263 				String p = params.get("allow-ending-volume-on-hyphen");
264 				allowEndingVolumeOnHyphen = (p == null) ? true : p.equalsIgnoreCase("true"); }
265 			FormatterConfiguration.Builder config = FormatterConfiguration.with(locale, mode)
266 				.markCapitalLetters(markCapitalLetters)
267 				.hyphenate(hyphenate)
268 				.allowsTextOverflowTrimming(allowTextOverflowTrimming)
269 				.allowsEndingVolumeOnHyphen(allowEndingVolumeOnHyphen);
270 			if (removeStyles)
271 				config.ignoreStyle("em").ignoreStyle("strong");
272 			Table brailleCharsetTable = "".equals(brailleCharset)
273 				? null
274 				: tableRegistry.get(mutableQuery().add("id", brailleCharset)).iterator().next();
275 			FormatterEngine engine = newFormatterEngine(config.build(),
276 			                                            newPEFWriter(identifier, brailleCharsetTable),
277 			                                            brailleCharsetTable == null
278 			                                                ? null
279 			                                                : brailleCharsetTable.newBrailleConverter());
280 			s = new ByteArrayOutputStream();
281 			engine.convert(obflStream, s);
282 			obflStream.close();
283 			InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
284 			s.close();
285 			
286 			// Write PEF
287 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(pefStream)));
288 			pefStream.close(); }
289 		
290 		catch (Throwable e) {
291 			String msg = e.getMessage();
292 			if (msg != null)
293 				if ((msg.contains("Cannot fit") && msg.contains("into a margin-region of size"))
294 				    || msg.contains("Failed to solve table"))
295 					throw new XProcException(DOTIFY_FAILURE, step, e);
296 			throw XProcStep.raiseError(e, step); }
297 		finally {
298 			evictTempTranslator.apply(); }
299 	}

301 	private PagedMediaWriter newPEFWriter(String identifier, Table brailleCharset) throws PagedMediaWriterConfigurationException {
302 		PagedMediaWriter writer = null;
303 		List<MetaDataItem> meta = new ArrayList<MetaDataItem>();
304 		if (brailleCharset != null)
305 			meta.add(new MetaDataItem(new javax.xml.namespace.QName("http://www.daisy.org/ns/pipeline/",
306 			                                                        "ascii-braille-charset",
307 			                                                        "dp2"),
308 			                          brailleCharset.getIdentifier()));
309 		if (!"".equals(identifier))
310 			meta.add(new MetaDataItem(new javax.xml.namespace.QName("http://purl.org/dc/elements/1.1/", "identifier", "dc"),
311 			                          identifier));
312 		writer = new PEFWriter(brailleCharset);
313 		if (!meta.isEmpty())
314 			writer.prepare(meta);
315 		return writer;
316 	}

318 	private FormatterEngine newFormatterEngine(FormatterConfiguration config,
319 	                                           PagedMediaWriter writer,
320 	                                           BrailleConverter brailleCharset) {
321 		if (brailleCharset != null) {
322 			// HACK: We create a new TextBorderFactoryMakerService that uses the default
323 			// TextBorderFactoryService to create Unicode braille patterns and encodes them with the
324 			// given BrailleConverter. We then make the default FormatterEngineFactoryService,
325 			// FormatterFactory and ObflParserFactoryService use this TextBorderFactoryMakerService
326 			// by using reflection. This assumes that the FormatterEngineFactoryService is a
327 			// org.daisy.dotify.formatter.impl.engine.LayoutEngineFactoryImpl, the FormatterFactory
328 			// is a org.daisy.dotify.formatter.impl.FormatterFactoryImpl, and the
329 			// ObflParserFactoryService is a
330 			// org.daisy.dotify.formatter.impl.obfl.ObflParserFactoryImpl, and that these are the
331 			// only classes that bind a TextBorderFactoryMakerService.
332 			TextBorderFactoryMakerService asciiTextBorderFactoryMakerService = new TextBorderFactoryMakerService() {
333 					public TextBorderStyle newTextBorderStyle(Map<String,Object> features)
334 							throws TextBorderConfigurationException {
335 						TextBorderFactory f = textBorderFactoryService.newFactory();
336 						for (String k : features.keySet())
337 							f.setFeature(k, features.get(k));
338 						TextBorderStyle style = f.newTextBorderStyle();
339 						return new TextBorderStyle.Builder()
340 							.topLeftCorner    (brailleCharset.toText(style.getTopLeftCorner()))
341 							.topBorder        (brailleCharset.toText(style.getTopBorder()))
342 							.topRightCorner   (brailleCharset.toText(style.getTopRightCorner()))
343 							.leftBorder       (brailleCharset.toText(style.getLeftBorder()))
344 							.rightBorder      (brailleCharset.toText(style.getRightBorder()))
345 							.bottomLeftCorner (brailleCharset.toText(style.getBottomLeftCorner()))
346 							.bottomBorder     (brailleCharset.toText(style.getBottomBorder()))
347 							.bottomRightCorner(brailleCharset.toText(style.getBottomRightCorner()))
348 							.build();
349 					}
350 				};
351 			for (Object object : new Object[]{formatterEngineFactoryService,
352 			                                  formatterFactory,
353 			                                  obflParserFactoryService}) {
354 				try {
355 					object.getClass()
356 						.getMethod(object == formatterFactory ? "setTextBorderFactory" : "setTextBorderFactoryMaker",
357 						           TextBorderFactoryMakerService.class)
358 						.invoke(object, asciiTextBorderFactoryMakerService);
359 				} catch (NoSuchMethodException |
360 				         SecurityException |
361 				         IllegalAccessException |
362 				         IllegalArgumentException |
363 				         InvocationTargetException e) {
364 					throw new RuntimeException(e);
365 				}
366 			}
367 		}
368 		return formatterEngineFactoryService.newFormatterEngine(config, writer);
369 	}
