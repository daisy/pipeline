35  /**
36   * @see <a href="../../../../../../../README.md">Documentation</a>
37   */
38  public interface LiblouisMathMLTransform {
39  	
40  	public enum MathCode {
41  		NEMETH, UKMATHS, MARBURG, WOLUWE
42  	}
43  	
44  	@Component(
45  		name = "org.daisy.pipeline.braille.liblouis.impl.LiblouisMathMLTransform.Provider",
46  		service = {
47  			TransformProvider.class
48  		}
49  	)
50  	public class Provider extends AbstractTransformProvider<Transform> {
51  		
52  		private URI href;
53  		
54  		@Activate
55  		protected void activate(final Map<?,?> properties) {
56  			href = URLs.asURI(URLs.getResourceFromJAR("xml/translate-mathml.xpl", LiblouisMathMLTransform.class));
57  		}
58  		
59  		private final static Iterable<Transform> empty = Iterables.<Transform>empty();
60  		
61  		private final static List<String> supportedOutput = ImmutableList.of("braille");
62  		
63  		protected Iterable<Transform> _get(final Query query) {
64  			try {
65  				if ("mathml".equals(query.getOnly("input").getValue().get())) {
66  					for (Feature f : query.get("output"))
67  						if (!supportedOutput.contains(f.getValue().get()))
68  							return empty;
69  					if (query.containsKey("locale")) {
70  						Locale locale; {
71  							try {
72  								locale = parseLocale(query.getOnly("locale").getValue().get()); }
73  							catch (IllegalArgumentException e) {
74  								return empty; }
75  						}
76  						MathCode code = mathCodeFromLocale(locale);
77  						if (code != null)
78  							return of(logCreate((Transform)new TransformImpl(code))); }}}
79  			catch (IllegalStateException e) {}
80  			return empty;
81  		}
82  		
83  		private class TransformImpl extends AbstractTransform implements XProcStepProvider {
84  			
85  			private final MathCode code;
86  			private final Map<String,String> options;
87  			
88  			private TransformImpl(MathCode code) {
89  				this.code = code;
90  				options = ImmutableMap.of("math-code", code.name());
91  			}
92  			
93  			@Override
94  			public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
95  				return new CxEvalBasedTransformer(href, null, options).newStep(runtime, step);
96  			}
97  			
98  			@Override
99  			public ToStringHelper toStringHelper() {
100 				return MoreObjects.toStringHelper("o.d.p.b.liblouis.impl.LiblouisMathMLTransform$Provider$TransformImpl")
101 					.add("mathCode", code);
102 			}
103 		}
104 		
105 		private static MathCode mathCodeFromLocale(Locale locale) {
106 			String language = locale.getLanguage().toLowerCase();
107 			String country = locale.getCountry().toUpperCase();
108 			if (language.equals("en")) {
109 				if (country.equals("GB"))
110 					return MathCode.UKMATHS;
111 				else
112 					return MathCode.NEMETH; }
113 			else if (language.equals("de"))
114 				return MathCode.MARBURG;
115 			else if (language.equals("nl"))
116 				return MathCode.WOLUWE;
117 			else
118 				return null;
119 		}
120 		
121 		@Override
122 		public ToStringHelper toStringHelper() {
123 			return MoreObjects.toStringHelper(LiblouisMathMLTransform.Provider.class.getName());
124 		}
125 	}
126 }
