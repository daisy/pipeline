1   package org.daisy.pipeline.braille.liblouis.impl;
2   
3   import java.net.URL;
4   import java.net.URI;
5   import java.util.Hashtable;
6   
7   import org.daisy.common.file.URLs;
8   import org.daisy.pipeline.braille.common.NativePath;
9   
10  import org.osgi.framework.BundleActivator;
11  import org.osgi.framework.BundleContext;
12  
13  import org.osgi.service.component.annotations.Activate;
14  import org.osgi.service.component.annotations.Component;
15  
16  @Component(
17  	name = "org.daisy.pipeline.braille.liblouis.impl.LiblouisExternalNativePath",
18  	service = { NativePath.class },
19  	property = {
20  		"identifier:String=http://www.liblouis.org/native/EXTERNAL"
21  	}
22  )
23  public class LiblouisExternalNativePath implements NativePath {
24  	
25  	final static boolean LIBLOUIS_EXTERNAL = Boolean.getBoolean("org.daisy.pipeline.braille.liblouis.external");
26  	
27  	/**
28  	 * @throws RuntimeException if using the Liblouis library present on the system is not allowed
29  	 */
30  	@Activate
31  	protected void activate() throws RuntimeException {
32  		if (!LIBLOUIS_EXTERNAL)
33  			throw new RuntimeException("Using external Liblouis is not allowed");
34  	}
35  	
36  	public URI getIdentifier() { return URLs.asURI("http://www.liblouis.org/native/EXTERNAL"); }
37  	public URI canonicalize(URI resource) { throw new UnsupportedOperationException(); }
38  	public URL resolve(URI resource) { throw new UnsupportedOperationException(); }
39  	public Iterable<URI> get(String query) { throw new UnsupportedOperationException(); }
40  	
41  	// Throwing an exception from the activate method, although permitted by the spec
42  	// (https://osgi.org/specification/osgi.cmpn/7.0.0/service.component.html#service.component-activate.method),
43  	// breaks bindings with cardinality 0..1 in Felix. For this reason the DS annotations are used
44  	// only to create the SPI wrapper, and for OSGi we use an activator.
45  	public static class Activator implements BundleActivator {
46  		
47  		public void start(BundleContext context) throws Exception {
48  			if (LIBLOUIS_EXTERNAL) {
49  				NativePath dummy = new LiblouisExternalNativePath();
50  				Hashtable<String,Object> properties = new Hashtable<String,Object>();
51  				properties.put("identifier", "http://www.liblouis.org/native/EXTERNAL");
52  				context.registerService(NativePath.class.getName(), dummy, properties);
53  			}
54  		}
55  		
56  		public void stop(BundleContext context) throws Exception {}
57  		
58  	}
59  }
