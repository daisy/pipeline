136 	@Override
137 	public void run() throws SaxonApiException {
138 		
139 		super.run();
140 		
141 		try {
142 			
143 			Map<String,String> settings = new HashMap<String,String>();
144 			settings.put("lineEnd", "\\n");
145 			settings.put("pageEnd", "\\f");
146 			
147 			// Get options
148 			if (getOption(_paged) != null)
149 				settings.put("braillePages",  getOption(_paged).getBoolean() ? "yes" : "no");
150 			if (params.containsKey(louis_page_width))
151 				settings.put("cellsPerLine", params.get(louis_page_width).getString());
152 			if (params.containsKey(louis_page_height))
153 				settings.put("linesPerPage", params.get(louis_page_height).getString());
154 			if (params.containsKey(louis_braille_page_position)) {
155 				String position = params.get(louis_braille_page_position).getString();
156 				if (position.equals("top-right") || position.equals("bottom-right")) {
157 					settings.put("braillePageNumberAt", position.replace("-right", ""));
158 					settings.put("numberBraillePages", "yes"); }
159 				else if (position.equals("none"))
160 					settings.put("numberBraillePages", "no"); }
161 			if (params.containsKey(louis_print_page_position)) {
162 				String position = params.get(louis_print_page_position).getString();
163 				if (position.equals("top-right") || position.equals("bottom-right")) {
164 					settings.put("printPageNumberAt", position.replace("-right", ""));
165 					settings.put("printPages", "yes"); }
166 				else if (position.equals("none"))
167 					settings.put("printPages", "no"); }
168 			if (params.containsKey(louis_page_break_separator)) {
169 				boolean separator = params.get(louis_page_break_separator).getBoolean();
170 				settings.put("pageSeparator", separator ? "yes" : "no");
171 				settings.put("pageSeparatorNumber", separator ? "yes" : "no"); }
172 			if (params.containsKey(louis_braille_page_begin))
173 				settings.put("beginningPageNumber",
174 				             params.get(louis_braille_page_begin).getString());
175 			if (params.containsKey(louis_braille_pages_in_toc))
176 				settings.put("braillePageNumbersInContents",
177 				             params.get(louis_braille_pages_in_toc).getBoolean() ? "yes" : "no");
178 			if (params.containsKey(louis_print_pages_in_toc))
179 				settings.put("printPageNumbersInContents",
180 				             params.get(louis_print_pages_in_toc).getBoolean() ? "yes" : "no");
181 			if (params.containsKey(louis_toc_leader_pattern)) {
182 				String pattern = params.get(louis_toc_leader_pattern).getString();
183 				if (pattern.length() == 1)
184 					settings.put("lineFill", pattern); }
185 			
186 			File tempDir = asFile(getOption(_temp_dir).getString());
187 			URI configPath = null;
188 			
189 			// Get configuration files
190 			List<String> configFileNames = new ArrayList<String>();
191 			if (styles != null) {
192 				while(styles.moreDocuments()) {
193 					XdmNode fileset = (XdmNode)styles.read().axisIterator(Axis.CHILD, d_fileset).next();
194 					URI baseURI = fileset.getBaseURI();
195 					XdmSequenceIterator files = fileset.axisIterator(Axis.CHILD, d_file);
196 					while (files != null && files.hasNext()) {
197 						URI uri = baseURI.resolve(((XdmNode)files.next()).getAttributeValue(_href));
198 						URI path = uri.resolve(".");
199 						if (configPath == null)
200 							configPath = path;
201 						else if (!configPath.equals(path))
202 							throw new XProcException(step.getNode(),
203 									"All configuration files and semantic action files must be placed in " + configPath);
204 						configFileNames.add(decode(path.relativize(uri).toString())); }}}
205 			
206 			// Get semantic action files
207 			List<String> semanticFileNames = new ArrayList<String>();
208 			if (semantics != null) {
209 				while(semantics.moreDocuments()) {
210 					XdmNode fileset = (XdmNode)semantics.read().axisIterator(Axis.CHILD, d_fileset).next();
211 					URI baseURI = fileset.getBaseURI();
212 					XdmSequenceIterator files = fileset.axisIterator(Axis.CHILD, d_file);
213 					while (files != null && files.hasNext()) {
214 						URI uri = baseURI.resolve(((XdmNode)files.next()).getAttributeValue(_href));
215 						URI path = uri.resolve(".");
216 						if (configPath == null)
217 							configPath = path;
218 						else if (!configPath.equals(path))
219 							throw new XProcException(step.getNode(),
220 									"All configuration files and semantic action files must be placed in " + configPath);
221 						semanticFileNames.add(decode(path.relativize(uri).toString())); }}}
222 			
223 			LiblouisTable table = null;
224 			if (getOption(_table) != null)
225 				table = new LiblouisTable(getOption(_table).getString());
226 			
227 			// Write XML document to file
228 			XdmNode xml = source.read();
229 			File xmlFile = File.createTempFile("liblouisutdml.", ".xml", tempDir);
230 			Serializer serializer = runtime.getProcessor().newSerializer();
231 			serializer.setOutputStream(new FileOutputStream(xmlFile));
232 			serializer.setCloseOnCompletion(true);
233 			serializer.serializeNode(xml);
234 			serializer.close();
235 			
236 			File bodyTempFile = new File(tempDir, "lbx_body.temp");
237 			bodyTempFile.delete();
238 			
239 			// Convert using file2brl
240 			File brailleFile = File.createTempFile("liblouisutdml.", ".txt", tempDir);
241 			liblouisutdml.translateFile(configFileNames, semanticFileNames, table,
242 					settings, xmlFile, brailleFile, configPath, tempDir);
243 			
244 			// Read the braille document and wrap it in a new XML document
245 			LiblouisResultReader reader = new LiblouisResultReader(brailleFile, bodyTempFile);
246 			while (true) {
247 				StringBuilder section = new StringBuilder();
248 				String page;
249 				int pageCount = 0;
250 				while ((page = reader.readPage()) != null) {
251 					section.append(page).append('\f');
252 					pageCount++; }
253 				writeDocument(xml.getBaseURI(), section.toString(), pageCount);
254 				if (!reader.nextSection())
255 					break; }
256 			reader.close(); }
257 		
258 		catch (Exception e) {
259 			throw new XProcException(step.getNode(), e); }
260 	}
261 	
262 	private void writeDocument(URI baseURI, String textContent, int pageCount) {
263 		TreeWriter treeWriter = new TreeWriter(runtime);
264 		treeWriter.startDocument(baseURI);
265 		treeWriter.addStartElement(louis_result);
266 		treeWriter.addAttribute(_pages, String.valueOf(pageCount));
267 		treeWriter.startContent();
268 		treeWriter.addText(textContent);
269 		treeWriter.addEndElement();
270 		treeWriter.endDocument();
271 		result.write(treeWriter.getResult());
272 	}
