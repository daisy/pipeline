137 	@Override
138 	public void run() throws SaxonApiException {
139 		
140 		super.run();
141 		
142 		try {
143 			
144 			Map<String,String> settings = new HashMap<String,String>();
145 			settings.put("lineEnd", "\\n");
146 			settings.put("pageEnd", "\\f");
147 			
148 			// Get options
149 			if (getOption(_paged) != null)
150 				settings.put("braillePages",  getOption(_paged).getBoolean() ? "yes" : "no");
151 			if (params.containsKey(louis_page_width))
152 				settings.put("cellsPerLine", params.get(louis_page_width).getString());
153 			if (params.containsKey(louis_page_height))
154 				settings.put("linesPerPage", params.get(louis_page_height).getString());
155 			if (params.containsKey(louis_braille_page_position)) {
156 				String position = params.get(louis_braille_page_position).getString();
157 				if (position.equals("top-right") || position.equals("bottom-right")) {
158 					settings.put("braillePageNumberAt", position.replace("-right", ""));
159 					settings.put("numberBraillePages", "yes"); }
160 				else if (position.equals("none"))
161 					settings.put("numberBraillePages", "no"); }
162 			if (params.containsKey(louis_print_page_position)) {
163 				String position = params.get(louis_print_page_position).getString();
164 				if (position.equals("top-right") || position.equals("bottom-right")) {
165 					settings.put("printPageNumberAt", position.replace("-right", ""));
166 					settings.put("printPages", "yes"); }
167 				else if (position.equals("none"))
168 					settings.put("printPages", "no"); }
169 			if (params.containsKey(louis_page_break_separator)) {
170 				boolean separator = params.get(louis_page_break_separator).getBoolean();
171 				settings.put("pageSeparator", separator ? "yes" : "no");
172 				settings.put("pageSeparatorNumber", separator ? "yes" : "no"); }
173 			if (params.containsKey(louis_braille_page_begin))
174 				settings.put("beginningPageNumber",
175 				             params.get(louis_braille_page_begin).getString());
176 			if (params.containsKey(louis_braille_pages_in_toc))
177 				settings.put("braillePageNumbersInContents",
178 				             params.get(louis_braille_pages_in_toc).getBoolean() ? "yes" : "no");
179 			if (params.containsKey(louis_print_pages_in_toc))
180 				settings.put("printPageNumbersInContents",
181 				             params.get(louis_print_pages_in_toc).getBoolean() ? "yes" : "no");
182 			if (params.containsKey(louis_toc_leader_pattern)) {
183 				String pattern = params.get(louis_toc_leader_pattern).getString();
184 				if (pattern.length() == 1)
185 					settings.put("lineFill", pattern); }
186 			
187 			File tempDir = asFile(getOption(_temp_dir).getString());
188 			URI configPath = null;
189 			
190 			// Get configuration files
191 			List<String> configFileNames = new ArrayList<String>();
192 			if (styles != null) {
193 				while(styles.moreDocuments()) {
194 					XdmNode fileset = (XdmNode)styles.read().axisIterator(Axis.CHILD, d_fileset).next();
195 					URI baseURI = fileset.getBaseURI();
196 					XdmSequenceIterator files = fileset.axisIterator(Axis.CHILD, d_file);
197 					while (files != null && files.hasNext()) {
198 						URI uri = URLs.resolve(baseURI, URLs.asURI(((XdmNode)files.next()).getAttributeValue(_href)));
199 						URI path = URLs.resolve(uri, URLs.asURI("."));
200 						if (configPath == null)
201 							configPath = path;
202 						else if (!configPath.equals(path))
203 							throw new XProcException(step,
204 									"All configuration files and semantic action files must be placed in " + configPath);
205 						configFileNames.add(URLs.relativize(path, uri).getPath()); }}}
206 			
207 			// Get semantic action files
208 			List<String> semanticFileNames = new ArrayList<String>();
209 			if (semantics != null) {
210 				while(semantics.moreDocuments()) {
211 					XdmNode fileset = (XdmNode)semantics.read().axisIterator(Axis.CHILD, d_fileset).next();
212 					URI baseURI = fileset.getBaseURI();
213 					XdmSequenceIterator files = fileset.axisIterator(Axis.CHILD, d_file);
214 					while (files != null && files.hasNext()) {
215 						URI uri = URLs.resolve(baseURI, URLs.asURI(((XdmNode)files.next()).getAttributeValue(_href)));
216 						URI path = URLs.resolve(uri, URLs.asURI("."));
217 						if (configPath == null)
218 							configPath = path;
219 						else if (!configPath.equals(path))
220 							throw new XProcException(step,
221 									"All configuration files and semantic action files must be placed in " + configPath);
222 						semanticFileNames.add(path.relativize(uri).getPath()); }}}
223 			
224 			LiblouisTable table = null;
225 			if (getOption(_table) != null)
226 				table = new LiblouisTable(getOption(_table).getString());
227 			
228 			// Write XML document to file
229 			XdmNode xml = source.read();
230 			File xmlFile = File.createTempFile("liblouisutdml.", ".xml", tempDir);
231 			Serializer serializer = runtime.getProcessor().newSerializer();
232 			serializer.setOutputStream(new FileOutputStream(xmlFile));
233 			serializer.setCloseOnCompletion(true);
234 			serializer.serializeNode(xml);
235 			serializer.close();
236 			
237 			File bodyTempFile = new File(tempDir, "lbx_body.temp");
238 			bodyTempFile.delete();
239 			
240 			// Convert using file2brl
241 			File brailleFile = File.createTempFile("liblouisutdml.", ".txt", tempDir);
242 			liblouisutdml.translateFile(configFileNames, semanticFileNames, table,
243 					settings, xmlFile, brailleFile, configPath, tempDir);
244 			
245 			// Read the braille document and wrap it in a new XML document
246 			LiblouisResultReader reader = new LiblouisResultReader(brailleFile, bodyTempFile);
247 			while (true) {
248 				StringBuilder section = new StringBuilder();
249 				String page;
250 				int pageCount = 0;
251 				while ((page = reader.readPage()) != null) {
252 					section.append(page).append('\f');
253 					pageCount++; }
254 				writeDocument(xml.getBaseURI(), section.toString(), pageCount);
255 				if (!reader.nextSection())
256 					break; }
257 			reader.close(); }
258 		
259 		catch (Throwable e) {
260 			throw XProcStep.raiseError(e, step); }
261 	}
262 	
263 	private void writeDocument(URI baseURI, String textContent, int pageCount) {
264 		TreeWriter treeWriter = new TreeWriter(runtime);
265 		treeWriter.startDocument(baseURI);
266 		treeWriter.addStartElement(louis_result);
267 		treeWriter.addAttribute(_pages, String.valueOf(pageCount));
268 		treeWriter.startContent();
269 		treeWriter.addText(textContent);
270 		treeWriter.addEndElement();
271 		treeWriter.endDocument();
272 		result.write(treeWriter.getResult());
273 	}
