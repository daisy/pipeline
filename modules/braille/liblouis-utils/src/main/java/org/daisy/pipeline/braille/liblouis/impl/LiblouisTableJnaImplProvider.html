143 	private void registerTableResolver() {
144 		Louis.setTableResolver(new TableResolver() {
145 				private final Map<String,URL> aggregatorTables = new HashMap<String,URL>();
146 				@Override
147 				public URL resolve(String table, URL base) {
148 					logger.debug("Resolving " + table + (base != null ? " against base " + base : ""));
149 					// if we are resolving an include rule from a generated aggregator table, resolve without base
150 					if (aggregatorTables.containsValue(base))
151 						base = null;
152 					File baseFile = base == null ? null : asFile(base); // base is expected to be a file
153 					File[] resolved = tableRegistry.resolveLiblouisTable(new LiblouisTable(table), baseFile);
154 					if (resolved != null) {
155 						logger.debug("Resolved to " + join(resolved, ","));
156 						if (resolved.length == 1)
157 							return URLs.asURL(resolved[0]);
158 						else {
159 							// if it is a comma separated table list, create a single file that includes all the sub-tables
160 							if (aggregatorTables.containsKey(table)) {
161 								URL u = aggregatorTables.get(table);
162 								logger.debug("... aggregated into " + u);
163 								return u;
164 							}
165 							try {
166 								StringBuilder b = new StringBuilder();
167 								for (File f : resolved)
168 									b.append("include ").append(URLs.asURI(f.getCanonicalFile()).toASCIIString()).append('\n');
169 								InputStream in = new ByteArrayInputStream(b.toString().getBytes(StandardCharsets.UTF_8));
170 								File f = createTempFile("aggregator-", ".tbl");
171 								f.delete();
172 								Files.copy(in, f.toPath());
173 								f = f.getCanonicalFile();
174 								URL u = URLs.asURL(f);
175 								aggregatorTables.put(table, u);
176 								logger.debug("... aggregated into " + u);
177 								return u;
178 							} catch (IOException e) {
179 								throw new RuntimeException(e); // should not happen
180 							}
181 						}
182 					}
183 					logger.error("Table could not be resolved");
184 					return null;
185 				}
186 				@Override
187 				public Set<String> list() {
188 					return newHashSet(
189 						transform(
190 							tableRegistry.listAllTableFiles(),
191 							toStringFunction()));
192 				}
193 			}
194 		);

200 	protected void activate() {
201 		logger.debug("Loading liblouis service");
202 		try {
203 			tableRegistry.onPathChange(
204 				new Function<LiblouisTableRegistry,Void>() {
205 					public Void apply(LiblouisTableRegistry r) {
206 						// re-register table resolver so that liblouis-java re-indexes tables
207 						registerTableResolver();
208 						invalidateCache();
209 						return null; }});
210 			registerTableResolver();
211 			// invoke after table resolver registered because otherwise default tables will be unpacked for no reason
212 			logger.debug("liblouis version: {}", Louis.getVersion());
213 			File unicodeDisFile = new File(makeUnpackDir(), "unicode.dis");
214 			unpack(
215 				URLs.getResourceFromJAR("/tables/unicode.dis", LiblouisTableJnaImplProvider.class),
216 				unicodeDisFile);
217 			unicodeDisplayTable = DisplayTable.fromTable("" + URLs.asURI(unicodeDisFile), Fallback.MASK);
218 			spacesFile = new File(makeUnpackDir(), "spaces.cti");
219 			unpack(
220 				URLs.getResourceFromJAR("/tables/spaces.cti", LiblouisTableJnaImplProvider.class),
221 				spacesFile);
222 			spacesDisFile = new File(makeUnpackDir(), "spaces.dis");
223 			unpack(
224 				URLs.getResourceFromJAR("/tables/spaces.dis", LiblouisTableJnaImplProvider.class),
225 				spacesDisFile);
226 			unicodeDisplayTableWithNoBreakSpace = DisplayTable.fromTable(

229 					@Override
230 					public void log(Level level, String message) {
231 						switch (level) {
232 						case ALL: logger.trace(message); break;
233 						case DEBUG: logger.debug(message); break;
234 						case INFO: logger.debug("INFO: " + message); break;
235 						case WARN: logger.debug("WARN: " + message); break;
236 						// FIXME: capture these and include them into CompilationException or TranslationException
237 						case ERROR: logger.error(message); break;
238 						case FATAL: logger.error(message); break; }}}); }
239 		catch (Throwable e) {
240 			logger.error("liblouis service could not be loaded", e);
241 			throw e; }
242 	}

302 	private TransformProvider<LiblouisTableJnaImpl> _provider
303 	= varyLocale(
304 		new AbstractTransformProvider<LiblouisTableJnaImpl>() {
305 			public Iterable<LiblouisTableJnaImpl> _get(final Query query) {
306 				return Iterables.of(
307 					new WithSideEffect<LiblouisTableJnaImpl,Logger>() {
308 						public LiblouisTableJnaImpl _apply() {
309 							MutableQuery q = mutableQuery(query);
310 							String table = null;
311 							String charset = null;
312 							TableInfo tableInfo = null;
313 							boolean whiteSpace = false;
314 							String dotsForUndefinedChar = null;
315 							Locale documentLocale = null;
316 							if (q.containsKey("white-space")) {
317 								q.removeOnly("white-space");
318 								whiteSpace = true; }
319 							if (q.containsKey("dots-for-undefined-char")) {
320 								dotsForUndefinedChar = q.removeOnly("dots-for-undefined-char").getValue().get();
321 								if (!dotsForUndefinedChar.matches("[\u2800-\u28FF]+")) {
322 									logger.warn(dotsForUndefinedChar + " is not a valid dot pattern string.");
323 									throw new NoSuchElementException();
324 								}
325 							}
326 							if (q.containsKey("document-locale"))
327 								documentLocale = parseLocale(q.removeOnly("document-locale").getValue().get());
328 							if (q.containsKey("charset") || q.containsKey("braille-charset"))
329 								charset = q.containsKey("charset")
330 									? q.removeOnly("charset").getValue().get()
331 									: q.removeOnly("braille-charset").getValue().get();
332 							if (q.containsKey("table") || q.containsKey("liblouis-table")) {
333 								table = q.containsKey("table")
334 									? q.removeOnly("table").getValue().get()
335 									: q.removeOnly("liblouis-table").getValue().get();
336 								tableInfo = new TableInfo(table);
337 								if (q.containsKey("locale")) {
338 									// locale is shorthand for language + region
339 									String locale = q.removeOnly("locale").getValue().get();
340 									q.add("language", locale);
341 									q.add("region", locale);
342 								}
343 								for (Feature f : q)
344 									if (!f.getValue().orElse("yes").equals(tableInfo.get(f.getKey()))) {
345 										logger.warn("Table " + table + " does not match " + f);
346 										throw new NoSuchElementException(); }
347 							} else {
348 								if (documentLocale != null && !q.containsKey("locale")) {
349 									// Liblouis table selection happens based on a language tag (primary target language
350 									// of the braille code) and an optional region tag (region or community in which the
351 									// braille code applies).
352 									if (!documentLocale.equals(new Locale(documentLocale.getLanguage(), documentLocale.getCountry()))) {
353 										// If the document locale has other subtags than language and region, we
354 										// interpret the locale as a language.
355 										if (!q.containsKey("language"))
356 											q.add("language", documentLocale.toLanguageTag());
357 									} else if (q.containsKey("region")) {
358 										// If the region is already specified in the query, we ignore the region subtag
359 										// of the document locale.
360 										if (!q.containsKey("language"))
361 											q.add("language", documentLocale.getLanguage());
362 									} else {
363 										// Otherwise we use the language subtag of the document locale as the language,
364 										// and the region subtag (if specified) as the region in which the braille code
365 										// applies.
366 										if (!q.containsKey("language"))
367 											q.add("language", documentLocale.getLanguage());
368 										if (!"".equals(documentLocale.getCountry()))
369 											q.add("region", documentLocale.toLanguageTag());
370 									}
371 								}
372 								if (q.isEmpty())
373 									throw new NoSuchElementException();
374 								StringBuilder b = new StringBuilder();
375 								
376 								// FIXME: if query does not contain (type:display), need to match for absence of "type:display"
377 								// -> i.e. Liblouis query syntax must support negation!
378 								// -> this used to be solved by matching "type:translation" but the downside of this
379 								//    is that this feature had to be added to every table which is not desired
380 								// -> another solution would be to let Liblouis return a list of possible matches and
381 								//    select the first match that does not end with ".dis" (or that does not have the
382 								//    feature "type:display")
383 								
384 								for (Feature f : q) {
385 									String k = f.getKey();
386 									if (!k.matches("[a-zA-Z0-9_-]+")) {
387 										__apply(
388 											warn("Invalid syntax for feature key: " + k));
389 										throw new NoSuchElementException(); }
390 									b.append(k);
391 									if (f.hasValue()) {
392 										String v = f.getValue().get();
393 										if (!v.matches("[a-zA-Z0-9_-]+")) {
394 											__apply(
395 												warn("Invalid syntax for feature value: " + v));
396 											throw new NoSuchElementException(); }
397 										b.append(":" + v); }
398 									b.append(" "); }
399 								try {
400 									Table t = Table.find(b.toString());
401 									table = t.getIdentifier();
402 									tableInfo = t.getInfo(); }
403 								catch (IllegalArgumentException e) {}
404 								catch (NoSuchElementException e) {}}
405 							if (table != null) {
406 								if (whiteSpace)
407 									table = URLs.asURI(spacesFile) + "," + table;
408 								DisplayTable displayTable = null;
409 								if (charset == null)
410 									displayTable = whiteSpace ? unicodeDisplayTableWithNoBreakSpace : unicodeDisplayTable;
411 								else
412 									try {
413 										if (whiteSpace)
414 											charset = "" + URLs.asURI(spacesDisFile) + "," + charset;
415 										// using Translator.asDisplayTable() and not DisplayTable.fromTable() so we can
416 										// catch CompilationException
417 										displayTable = new Translator(charset).asDisplayTable(); }
418 									catch (CompilationException e) {
419 										// the specified table is not a Liblouis table
420 										throw new NoSuchElementException(); }
421 								if (dotsForUndefinedChar != null) {
422 									try {
423 										File undefinedFile = createTempFile("undefined-", ".uti");
424 										undefinedFile.createNewFile();
425 										FileOutputStream writer = new FileOutputStream(undefinedFile);
426 										String dotPattern; {
427 											StringBuilder b = new StringBuilder();
428 											for (char c : dotsForUndefinedChar.toCharArray()) {
429 												b.append("-");
430 												c &= (char)0xFF;
431 												if (c == 0)
432 													b.append("0");
433 												else
434 													for (int k = 1; k <= 8; k++) {
435 														if ((c & (char)1) != 0)
436 															b.append(k);
437 														c = (char)(c >> 1); }}
438 											dotPattern = b.toString().substring(1); }
439 										writer.write(("undefined " + dotPattern + "\n").getBytes());
440 										writer.flush();
441 										writer.close();
442 										// adding the "undefined" rule to the end overwrites any previous rules
443 										table = table + "," + URLs.asURI(undefinedFile);
444 									} catch (IOException e) {
445 										throw new RuntimeException(e);
446 									}
447 								}
448 								try {
449 									return new LiblouisTableJnaImpl(new Translator(table), displayTable, tableInfo); }
450 								catch (CompilationException e) {
451 									__apply(
452 										warn("Could not compile table " + table));
453 									logger.warn("Could not compile table", e); }}
454 							throw new NoSuchElementException();
455 						}
456 					}
457 				);
458 			}
459 		}
460 	);

493 	private void createDatatype() throws ParserConfigurationException, DOMException {
494 		Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder()
495 		                                     .getDOMImplementation().createDocument(null, "choice", null);
496 		List<String> values = new ArrayList<>();
497 		Element choice = doc.getDocumentElement();
498 		String defaultValue = "";
499 		values.add(defaultValue);
500 		choice.appendChild(doc.createElement("value"))
501 		      .appendChild(doc.createTextNode(defaultValue));
502 		choice.appendChild(doc.createElementNS("http://relaxng.org/ns/compatibility/annotations/1.0", "documentation"))
503 		      .appendChild(doc.createTextNode("-"));
504 		List<Table> tables = new ArrayList<>();
505 		tables.addAll(Louis.listTables());
506 		Collections.sort(tables,
507 		                 new Comparator<Table>() {
508 				public int compare(Table o1, Table o2) {
509 					String s1 = o1.getInfo().get("index-name");
510 					String s2 = o2.getInfo().get("index-name");
511 					if (s1 == null)
512 						if (s2 == null)
513 							return 0;
514 						else
515 							return -1;
516 					else if (s2 == null)
517 						return 1;
518 					else
519 						return s1.compareTo(s2); }});
520 		for (Table table : tables) {
521 			// We can only do this because we know the identifier can be used to construct a new
522 			// Translator, but this is not an official feature. A more correct solution would be to
523 			// cache the Table objects and use Table.getTranslator() to get the Translator.
524 			String value = "(liblouis-table:\"" + table.getIdentifier().replace("http://www.liblouis.org/tables/", "") + "\")";
525 			values.add(value);
526 			choice.appendChild(doc.createElement("value"))
527 			      .appendChild(doc.createTextNode(value));
528 			String indexName = table.getInfo().get("index-name");
529 			if (indexName != null) {
530 				choice.appendChild(doc.createElementNS("http://relaxng.org/ns/compatibility/annotations/1.0", "documentation"))
531 				      .appendChild(doc.createTextNode(indexName));
532 			}
533 		}
534 		xmlDefinition = doc;
535 		enumerationValues = values;
536 	}
