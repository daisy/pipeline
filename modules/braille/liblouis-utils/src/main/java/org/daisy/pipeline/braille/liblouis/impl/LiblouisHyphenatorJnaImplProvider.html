76  	private final static Iterable<LiblouisHyphenator> empty = Optional.<LiblouisHyphenator>absent().asSet();
77  	
78  	private Provider.util.MemoizingProvider<Query,LiblouisHyphenator> provider
79  	= new Provider.util.Memoize<Query,LiblouisHyphenator>() {
80  		public Iterable<LiblouisHyphenator> _get(Query query) {
81  			MutableQuery q = mutableQuery(query);
82  			if (q.containsKey("hyphenator"))
83  				if (!"liblouis".equals(q.removeOnly("hyphenator").getValue().orElse(null)))
84  					return empty;
85  			String table = null;
86  			if (q.containsKey("liblouis-table"))
87  				table = q.removeOnly("liblouis-table").getValue().get();
88  			if (q.containsKey("table"))
89  				if (table != null) {
90  					logger.warn("A query with both 'table' and 'liblouis-table' never matches anything");
91  					return empty; }
92  				else
93  					table = q.removeOnly("table").getValue().get();
94  			String v = null;
95  			v = null;
96  			if (q.containsKey("locale"))
97  				v = q.removeOnly("locale").getValue().get();
98  			final String locale = v;
99  			if (table != null && !q.isEmpty()) {
100 				logger.warn("A query with both 'table' or 'liblouis-table' and '"
101 				            + q.iterator().next().getKey() + "' never matches anything");
102 				return empty; }
103 			if (table != null)
104 				q.add("table", table);
105 			if (locale != null)
106 				try {
107 					q.add("locale", parseLocale(locale).toLanguageTag()); }
108 				catch (IllegalArgumentException e) {
109 					logger.error("Invalid locale", e);
110 					return empty; }
111 			Iterable<LiblouisTableJnaImpl> tables = tableProvider.get(q.asImmutable());
112 			return filter(
113 				transform(
114 					tables,
115 					new Function<LiblouisTableJnaImpl,LiblouisHyphenator>() {
116 						public LiblouisHyphenator apply(LiblouisTableJnaImpl table) {
117 							try {
118 								return new LiblouisHyphenatorImpl(table.getTranslator()); }
119 							catch (IllegalArgumentException e) {
120 								return null; }}}),
121 				Predicates.notNull());
122 		}
123 	};

125 	private static class LiblouisHyphenatorImpl extends AbstractHyphenator implements LiblouisHyphenator {
126 		
127 		private final LiblouisTable table;
128 		private final Translator translator;
129 		
130 		private LiblouisHyphenatorImpl(Translator translator) {
131 			this.table = new LiblouisTable(translator.getTable());
132 			this.translator = translator;
133 
134 			// The table is not guaranteed to contain a hyphenation (.dic) sub-table. We
135 			// can find out by hyphenating a test string.
136 			try {
137 				translator.hyphenate("foobar");
138 			} catch (TranslationException e) {
139 				// Note that Liblouis raises this error but does not log an error
140 				// message. We assume that the reason of the failure is because
141 				// `table->hyphenStatesArray == 0' (in other words, because no hyphenation
142 				// sub-table was encountered).
143 				throw new IllegalArgumentException("Table does not support hyphenation: " + translator.getTable());
144 			}
145 		}
146 		
147 		public LiblouisTable asLiblouisTable() {
148 			return table;
149 		}
150 		
151 		@Override
152 		public FullHyphenator asFullHyphenator() {
153 			return fullHyphenator;
154 		}
155 		
156 		private final FullHyphenator fullHyphenator = new DefaultFullHyphenator() {
157 			protected boolean isCodePointAware() { return true; }
158 			protected byte[] getHyphenationOpportunities(String textWithoutHyphens) throws RuntimeException {
159 				try {
160 					return translator.hyphenate(textWithoutHyphens); }
161 				catch (TranslationException e) {
162 					throw new RuntimeException(e); }
163 			}
164 		};
165 		
166 		@Override
167 		public ToStringHelper toStringHelper() {
168 			return MoreObjects.toStringHelper("o.d.p.b.liblouis.impl.LiblouisHyphenatorJnaImplProvider$LiblouisHyphenatorImpl")
169 				.add("table", table);
170 		}
171 	}
172 	
173 	@Override
174 	public String toString() {
175 		return MoreObjects.toStringHelper(LiblouisHyphenatorJnaImplProvider.class.getName()).toString();
176 	}
