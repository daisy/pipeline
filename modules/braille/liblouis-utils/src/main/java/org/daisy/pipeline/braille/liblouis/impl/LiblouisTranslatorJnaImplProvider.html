141 	protected final Iterable<LiblouisTranslator> _get(Query query) {
142 		MutableQuery q = mutableQuery(query);
143 		for (Feature f : q.removeAll("input"))
144 			if (!supportedInput.contains(f.getValue().get()))
145 				return empty;
146 		if (q.containsKey("output")) {
147 			String v = q.removeOnly("output").getValue().get();
148 			if ("braille".equals(v)) {}
149 			else
150 				return empty; }
151 		if (q.containsKey("translator"))
152 			if (!"liblouis".equals(q.removeOnly("translator").getValue().get()))
153 				return empty;
154 		String table = null;
155 		if (q.containsKey("liblouis-table"))
156 			table = q.removeOnly("liblouis-table").getValue().get();
157 		if (q.containsKey("table"))
158 			if (table != null) {
159 				logger.warn("A query with both 'table' and 'liblouis-table' never matches anything");
160 				return empty; }
161 			else
162 				table = q.removeOnly("table").getValue().get();
163 		String v = null;
164 		if (q.containsKey("handle-non-standard-hyphenation"))
165 			v = q.removeOnly("handle-non-standard-hyphenation").getValue().get();
166 		else
167 			v = "ignore";
168 		final int handleNonStandardHyphenation = v.equalsIgnoreCase("fail") ?
169 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_FAIL : v.equalsIgnoreCase("defer") ?
170 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_DEFER :
171 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_IGNORE;
172 		if (table != null)
173 			q.add("table", table);
174 		q.add("white-space");
175 		Iterable<LiblouisTranslator> translators = memoize(
176 			getSimpleTranslator(
177 				q.asImmutable(),
178 				handleNonStandardHyphenation));
179 		if (translators.apply(NOP_LOGGER).iterator().hasNext()) {
180 			// all translators use the same display table
181 			// FIXME: table has already been computed in the getSimpleTranslator() call above
182 			LiblouisTableJnaImpl t = tableProvider.withContext(NOP_LOGGER).get(q).iterator().next();
183 			BrailleTranslator unityTranslator = new UnityBrailleTranslator(
184 				t.usesCustomDisplayTable()
185 					? new LiblouisDisplayTableBrailleConverter(t.getDisplayTable())
186 					: null,
187 				false);
188 			return Iterables.transform(
189 				translators,
190 				new Function<LiblouisTranslator,LiblouisTranslator>() {
191 					public LiblouisTranslator _apply(LiblouisTranslator t) {
192 						return __apply(logCreate(new HandleTextTransformNone(t, unityTranslator))); }});
193 		} else
194 			return translators;
195 	}

395 			final FromStyledTextToBraille fullTranslator;

397 			protected LineBreaker(FromStyledTextToBraille fullTranslator) {
398 				// Note that `displayTable.encode('\u2800')` is always a space because of the addition of spaces.dis
399 				super(displayTable.encode('\u2800'),
400 				      displayTable.encode('\u2824'),
401 				      new LiblouisDisplayTableBrailleConverter(displayTable),
402 				      logger);
403 				this.fullTranslator = fullTranslator;

406 			protected BrailleStream translateAndHyphenate(java.lang.Iterable<CSSStyledText> styledText, int from, int to) {
407 				java.lang.Iterable<CSSStyledText> braille;
408 				try {
409 					braille = fullTranslator.transform(styledText); }
410 				catch (NonStandardHyphenationException e) {
411 					return new BrailleStreamImpl(styledText,
412 					                             from,
413 					                             to); }
414 				// style is mutated and may not be empty
415 				List<String> brailleWithPreservedWS = new ArrayList<>(); {
416 					for (CSSStyledText b : braille) {
417 						String t = b.getText();
418 						// the only property expected in the output is white-space
419 						// ignore other properties
420 						SimpleInlineStyle s = b.getStyle();
421 						if (s != null) {
422 							Hyphens h = s.getProperty("hyphens");
423 							if (h == Hyphens.NONE || h == Hyphens.MANUAL) {
424 								if (h == Hyphens.NONE)
425 									t = t.replaceAll("[\u00AD\u200B]","");
426 								s.removeProperty("hyphens"); }
427 							WhiteSpace ws = s.getProperty("white-space");
428 							if (ws != null) {
429 								if (ws == WhiteSpace.PRE_WRAP)
430 									t = t.replaceAll("[\\x20\t\\u2800]+", "$0"+ZWSP)
431 									     .replaceAll("[\\x20\t\\u2800]", ""+NBSP);
432 								if (ws == WhiteSpace.PRE_WRAP || ws == WhiteSpace.PRE_LINE)
433 									t = t.replaceAll("[\\n\\r]", ""+LS);
434 								s.removeProperty("white-space"); }
435 							TextTransform tt = s.getProperty("text-transform");
436 							if (tt == TextTransform.NONE)
437 								s.removeProperty("text-transform");
438 							s.removeProperty("braille-charset");
439 							for (String prop : s.getPropertyNames())
440 								logger.warn("{} not supported", s.get(prop));
441 						}
442 						brailleWithPreservedWS.add(t);
443 					}
444 				}
445 				StringBuilder joined = new StringBuilder();
446 				int fromChar = 0;
447 				int toChar = to >= 0 ? 0 : -1;
448 				for (String s : brailleWithPreservedWS) {
449 					joined.append(s);
450 					if (--from == 0)
451 						fromChar = joined.length();
452 					if (--to == 0)
453 						toChar = joined.length();
454 				}
455 				return new FullyHyphenatedAndTranslatedString(joined.toString(), fromChar, toChar);
456 			}

514 					int size = size(styledText);
515 					if (to < 0) to = size;
516 					this.to = to;

519 					String[] text = new String[size];
520 					SimpleInlineStyle[] styles = new SimpleInlineStyle[size];
521 					languages = new Locale[size]; {
522 						int i = 0;
523 						for (CSSStyledText t : styledText) {
524 							text[i] = t.getText();
525 							styles[i] = t.getStyle();
526 							if (styles[i] != null)
527 								styles[i] = (SimpleInlineStyle)styles[i].clone();
528 							languages[i] = t.getLanguage();
529 							i++; }}
530 					
531 					// perform Unicode normalization
532 					if (unicodeNormalization != null)
533 						for (int k = 0; k < text.length; k++)
534 							text[k] = Normalizer.normalize(text[k], unicodeNormalization);

537 						typeform = new Typeform[size];
538 						hyphenate = new boolean[size];
539 						preserveLines = new boolean[size];
540 						preserveSpace = new boolean[size];
541 						letterSpacing = new int[size];
542 						for (int i = 0; i < size; i++) {
543 							typeform[i] = Typeform.PLAIN_TEXT;
544 							hyphenate[i] = false;
545 							preserveLines[i] = preserveSpace[i] = false;
546 							letterSpacing[i] = 0;
547 							SimpleInlineStyle style = styles[i];
548 							if (style != null) {
549 								CSSProperty val = style.getProperty("white-space");
550 								if (val != null) {
551 									if (val == WhiteSpace.PRE_WRAP)
552 										preserveLines[i] = preserveSpace[i] = true;
553 									else if (val == WhiteSpace.PRE_LINE)
554 										preserveLines[i] = true;
555 									style.removeProperty("white-space"); }
556 								val = style.getProperty("text-transform");
557 								if (val != null) {
558 									if (val == TextTransform.NONE) {
559 										// "text-transform: none" is handled by HandleTextTransformNone, but
560 										// HandleTextTransformNone is a CompoundBrailleTranslator and
561 										// CompoundBrailleTranslator puts "text-transform: none" on (already translated)
562 										// context segments. We assume that all Liblouis tables correctly handle Unicode
563 										// braille. If this is not the case, it is not the end of the words because this
564 										// is a context segment.
565 										val = style.getProperty("braille-charset");
566 										if (val != null) {
567 											if (val == BrailleCharset.CUSTOM)
568 												// translate to Unicode braille
569 												text[i] = displayTable.decode(text[i]);
570 											style.removeProperty("braille-charset"); }
571 										style.removeProperty("text-transform");
572 										continue; }
573 									else if (val == TextTransform.AUTO) {}
574 									else if (val == TextTransform.list_values) {
575 										TermList values = style.getValue(TermList.class, "text-transform");
576 										text[i] = textFromTextTransform(text[i], values);
577 										typeform[i] = typeform[i].add(typeformFromTextTransform(values, translator, supportedTypeforms)); }
578 									style.removeProperty("text-transform"); }
579 								val = style.getProperty("hyphens");
580 								if (val != null) {
581 									if (val == Hyphens.AUTO)
582 										hyphenate[i] = true;
583 									else if (val == Hyphens.NONE)
584 										text[i] = extractHyphens(text[i], false, SHY, ZWSP)._1;
585 									style.removeProperty("hyphens"); }
586 								val = style.getProperty("letter-spacing");
587 								if (val != null) {
588 									if (val == LetterSpacing.length) {
589 										letterSpacing[i] = style.getValue(TermInteger.class, "letter-spacing").getIntValue();
590 										if (letterSpacing[i] < 0) {
591 											logger.warn("{} not supported, must be non-negative", val);
592 											letterSpacing[i] = 0; }}
593 									style.removeProperty("letter-spacing"); }
594 								typeform[i] = typeform[i].add(typeformFromInlineCSS(style, translator, supportedTypeforms));
595 								for (String prop : style.getPropertyNames())
596 									logger.warn("{} not supported", style.get(prop)); }}
597 					}
598 					{ // compute preserved white space segments (textWithWs, textWithWsMapping, pre)
599 						List<String> l1 = new ArrayList<String>();
600 						List<Boolean> l2 = new ArrayList<Boolean>();
601 						List<Integer> l3 = new ArrayList<Integer>();
602 						for (int i = 0; i < text.length; i++) {
603 							String t = text[i];
604 							if (t.isEmpty()) {
605 								l1.add(t);
606 								l2.add(false);
607 								l3.add(i); }
608 							else {
609 								Pattern ws;
610 								if (preserveSpace[i])
611 									ws = ON_SPACE_SPLITTER;
612 								else if (preserveLines[i])
613 									ws = LINE_SPLITTER;
614 								else
615 									ws = ON_NBSP_SPLITTER;
616 								boolean p = false;
617 								for (String s : splitInclDelimiter(t, ws)) {
618 									if (!s.isEmpty()) {
619 										l1.add(s);
620 										l2.add(p);
621 										l3.add(i); }
622 									p = !p; }}}
623 						int len = l1.size();
624 						textWithWs = new String[len];
625 						pre = new boolean[len];
626 						textWithWsMapping = new int[len];
627 						for (int i = 0; i < len; i++) {
628 							textWithWs[i] = l1.get(i);
629 							pre[i] = l2.get(i);
630 							textWithWsMapping[i] = l3.get(i); }
631 					}
632 					{ // compute joined text and manual hyphens array (joinedText, joinedTextMapping, manualHyphens)
633 						String[] textWithWsReplaced = new String[textWithWs.length];
634 						for (int i = 0; i < textWithWs.length; i++)
635 							textWithWsReplaced[i] = pre[i] ? ""+NBSP : textWithWs[i];
636 						Tuple2<String,byte[]> t = extractHyphens(join(textWithWsReplaced, RS), true, SHY, ZWSP);
637 						manualHyphens = t._2;
638 						String[] nohyph = toArray(SEGMENT_SPLITTER.split(t._1), String.class);
639 						joinedTextMapping = new int[lengthByCodePoints(join(nohyph))];
640 						int i = 0;
641 						int j = 0;
642 						for (String s : nohyph) {
643 							int l = lengthByCodePoints(s);
644 							for (int k = 0; k < l; k++)
645 								joinedTextMapping[i++] = j;
646 							j++; }
647 						t = extractHyphens(manualHyphens, t._1, true, null, null, null, RS);
648 						joinedText = t._1;
649 					}
650 					{ // compute initial curPos and endPos from from and to
651 						int fromChar = -1;
652 						int toChar = -1;
653 						for (int i = 0; i < joinedTextMapping.length; i++) {
654 							if (fromChar < 0 || (toChar < 0 && to >= 0)) {
655 								int indexInText = textWithWsMapping[joinedTextMapping[i]];
656 								if (fromChar < 0 && indexInText >= from)
657 									fromChar = i;
658 								if (toChar < 0 && indexInText >= to)
659 									toChar = i;
660 							} else
661 								break;
662 						}
663 						if (toChar < 0) toChar = joinedTextMapping.length;
664 						this.curPos = fromChar;
665 						this.endPos = toChar;
666 					}
667 				}

669 				public String next(final int limit, final boolean force, boolean allowHyphens) {
670 					String next = "";
671 					if (limit > 0) {
672 					int available = limit;
673 				  segments: while (true) {
674 						if (curPos == endPos)
675 							break;
676 						if (joinedBraille == null)
677 							updateBraille();
678 						int curSegment = joinedTextMapping[curPos];
679 						int curSegmentEnd; {
680 							int i = curPos;
681 							for (; i < endPos; i++)
682 								if (joinedTextMapping[i] > curSegment)
683 									break;
684 							curSegmentEnd = i; }
685 						int curSegmentEndInBraille = positionInBraille(curSegmentEnd);
686 						if (curSegmentEndInBraille == curPosInBraille)
687 							continue segments;
688 						String segment = substringByCodePoints(joinedText, curPos, curSegmentEnd);
689 						String segmentInBraille = joinedBraille.substring(curPosInBraille, curSegmentEndInBraille);
690 						byte[] segmentManualHyphens = manualHyphens != null
691 							? Arrays.copyOfRange(manualHyphens, curPos, curSegmentEnd - 1)
692 							: null;
693 						
694 						// restore preserved white space segments
695 						if (pre[curSegment]) {
696 							Matcher m = Pattern.compile("\\xA0([\\xAD\\u200B]*)").matcher(segmentInBraille);
697 							if (m.matches()) {
698 								String restoredSpace = segment.replaceAll("[\\x20\t\\u2800]", ""+NBSP)
699 								                              .replaceAll("[\\n\\r]", ""+LS) + m.group(1);
700 								next += restoredSpace;
701 								available -= lengthByCodePoints(restoredSpace);
702 								curPos = curSegmentEnd;
703 								curPosInBraille = curSegmentEndInBraille;
704 								continue segments; }}
705 						
706 						// don't hyphenate if the segment fits in the available space
707 						// FIXME: we don't know for sure that the segment fits because there might be letter spacing!
708 						if (segmentInBraille.length() <= available) {
709 							segmentInBraille = addLetterSpacing(segment, segmentInBraille, curPos, curPosInBraille,
710 							                                    letterSpacing[textWithWsMapping[curSegment]]);
711 							next += segmentInBraille;
712 							available -= segmentInBraille.length();
713 							curPos = curSegmentEnd;
714 							curPosInBraille = curSegmentEndInBraille;
715 							continue segments; }
716 						
717 						// don't hyphenate if hyphenation is disabled for this segment, but do break compound words with a hyphen
718 						Locale language = languages[textWithWsMapping[curSegment]];
719 						if (!hyphenate[textWithWsMapping[curSegment]]) {
720 							segmentInBraille = addHyphensAndLetterSpacing(compoundWordHyphenator,
721 							                                              segment, segmentInBraille, curPos, curPosInBraille,
722 							                                              segmentManualHyphens, language,
723 							                                              letterSpacing[textWithWsMapping[curSegment]]);
724 							next += segmentInBraille;
725 							available -= segmentInBraille.length();
726 							curPos = curSegmentEnd;
727 							curPosInBraille = curSegmentEndInBraille;
728 							continue segments; }
729 						
730 						// try standard hyphenation of the whole segment
731 						if (fullHyphenator != null) {
732 							if (fullHyphenator == compoundWordHyphenator)
733 								logger.warn("hyphens: auto not supported");
734 							try {
735 								segmentInBraille = addHyphensAndLetterSpacing(fullHyphenator, segment, segmentInBraille, curPos, curPosInBraille,
736 								                                              segmentManualHyphens, language,
737 								                                              letterSpacing[textWithWsMapping[curSegment]]);
738 								next += segmentInBraille;
739 								available -= segmentInBraille.length();
740 								curPos = curSegmentEnd;
741 								curPosInBraille = curSegmentEndInBraille;
742 								continue segments; }
743 							catch (NonStandardHyphenationException e) {}}
744 						
745 						// loop over words in segment
746 						Matcher m = WORD_SPLITTER.matcher(segment);
747 						int segmentStart = curPos;
748 						boolean foundSpace;
749 						while ((foundSpace = m.find()) || curPos < curSegmentEnd) {
750 							int wordEnd = foundSpace ? segmentStart + m.start() : curSegmentEnd;
751 							if (wordEnd > curPos) {
752 								int wordEndInBraille = positionInBraille(wordEnd);
753 								if (wordEndInBraille > curPosInBraille) {
754 									String word = substringByCodePoints(joinedText, curPos, wordEnd);
755 									String wordInBraille = joinedBraille.substring(curPosInBraille, wordEndInBraille);
756 									byte[] wordManualHyphens = manualHyphens != null
757 										? Arrays.copyOfRange(manualHyphens, curPos, wordEnd - 1)
758 										: null;
759 									
760 									// don't hyphenate if word fits in the available space
761 									if (wordInBraille.length() <= available) {
762 										next += wordInBraille;
763 										available -= wordInBraille.length();
764 										curPos = wordEnd;
765 										curPosInBraille = wordEndInBraille; }
766 									else {
767 										
768 										// try standard hyphenation of the whole word
769 										try {
770 											if (fullHyphenator == null) throw new NonStandardHyphenationException();
771 											wordInBraille = addHyphensAndLetterSpacing(fullHyphenator, word, wordInBraille, curPos, curPosInBraille,
772 											                                           wordManualHyphens, language,
773 											                                           letterSpacing[textWithWsMapping[curSegment]]);
774 											next += wordInBraille;
775 											available -= wordInBraille.length();
776 											curPos = wordEnd;
777 											curPosInBraille = wordEndInBraille; }
778 										catch (NonStandardHyphenationException ee) {
779 											
780 											// before we try non-standard hyphenation, return what we have already
781 											// we do this because we are not sure that the value of "available" is accurate
782 											// this way we leave the responsibility of white space normalisation to DefaultLineBreaker
783 											// FIXME: remove the "available" variable
784 											if (!next.isEmpty())
785 												break segments;
786 											
787 											// try non-standard hyphenation
788 											if (lineBreaker == null) throw ee;
789 											Hyphenator.LineIterator lines = lineBreaker.transform(word, language);
790 											
791 											// do a binary search for the optimal break point
792 											LineBreakSolution bestSolution = null;
793 											int left = 1;
794 											int right = lengthByCodePoints(word) - 1;
795 											int textAvailable = available;
796 											if (textAvailable > right)
797 												textAvailable = right;
798 											if (textAvailable < left)
799 												break segments;
800 											while (true) {
801 												String line = lines.nextLine(textAvailable, force && next.isEmpty(), allowHyphens);
802 												String replacementWord = line + lines.remainder();
803 												if (updateInput(curPos, wordEnd, replacementWord)) {
804 													wordEnd = curPos + lengthByCodePoints(replacementWord);
805 													updateBraille(); }
806 												int lineEnd = curPos + lengthByCodePoints(line);
807 												int lineEndInBraille = positionInBraille(lineEnd);
808 												String lineInBraille = joinedBraille.substring(curPosInBraille, lineEndInBraille);
809 												lineInBraille = addLetterSpacing(line, lineInBraille, curPos, curPosInBraille,
810 												                                 letterSpacing[textWithWsMapping[curSegment]]);
811 												int lineInBrailleLength = lineInBraille.length();
812 												if (lines.lineHasHyphen()) {
813 													lineInBraille += "\u00ad";
814 													lineInBrailleLength++; }
815 												if (lineInBrailleLength == available) {
816 													bestSolution = new LineBreakSolution(); {
817 														bestSolution.line = line;
818 														bestSolution.replacementWord = replacementWord;
819 														bestSolution.lineInBraille = lineInBraille;
820 														bestSolution.lineInBrailleLength = lineInBrailleLength; }
821 													left = textAvailable + 1;
822 													right = textAvailable - 1; }
823 												else if (lineInBrailleLength < available) {
824 													left = textAvailable + 1;
825 													if (bestSolution == null || lineInBrailleLength > bestSolution.lineInBrailleLength) {
826 														bestSolution = new LineBreakSolution(); {
827 															bestSolution.line = line;
828 															bestSolution.replacementWord = replacementWord;
829 															bestSolution.lineInBraille = lineInBraille;
830 															bestSolution.lineInBrailleLength = lineInBrailleLength; }}}
831 												else
832 													right = textAvailable - 1;
833 												lines.reset();
834 												textAvailable = (right + left) / 2;
835 												if (textAvailable < left || textAvailable > right) {
836 													if (bestSolution != null) {
837 														next += bestSolution.lineInBraille;
838 														available = 0;
839 														if (updateInput(curPos, wordEnd, bestSolution.replacementWord))
840 															updateBraille();
841 														curPos += lengthByCodePoints(bestSolution.line);
842 														curPosInBraille = positionInBraille(curPos); }
843 													else if (force && next.isEmpty()) {
844 														next = wordInBraille;
845 														available = 0;
846 														curPos = wordEnd;
847 														curPosInBraille = wordEndInBraille; }
848 													break segments; } }}}}}
849 							if (foundSpace) {
850 								int spaceEnd = segmentStart + m.end();
851 								int spaceEndInBraille = positionInBraille(spaceEnd);
852 								if (spaceEndInBraille > curPosInBraille) {
853 									String spaceInBraille = joinedBraille.substring(curPosInBraille, spaceEndInBraille);
854 									next += spaceInBraille;
855 									available -= spaceInBraille.length();
856 									curPos = spaceEnd;
857 									curPosInBraille = spaceEndInBraille; }}}}
858 					}
859 					if (lastPeek != null && !next.isEmpty() && next.charAt(0) != lastPeek)
860 						throw new IllegalStateException();
861 					lastPeek = null;
862 					return next;
863 				}

919 					int posInBraille = curPosInBraille;

927 				private String addHyphensAndLetterSpacing(FullHyphenator fullHyphenator,
928 				                                          String segment,
929 				                                          String segmentInBraille,
930 				                                          int curPos,
931 				                                          int curPosInBraille,
932 				                                          byte[] manualHyphens,
933 				                                          Locale language,
934 				                                          int letterSpacing) {
935 					byte[] hyphens = fullHyphenator.hyphenate(
936 						// insert manual hyphens first so that hyphenator knows which words to skip
937 						insertHyphens(segment, manualHyphens, true, SHY, ZWSP),
938 						language);
939 					// FIXME: don't hard-code the number 4
940 					byte[] hyphensAndLetterBoundaries
941 						= (letterSpacing > 0) ? detectLetterBoundaries(hyphens, segment, (byte)4) : hyphens;
942 					if (hyphensAndLetterBoundaries == null && manualHyphens == null)
943 						return segment;
944 					byte[] hyphensAndLetterBoundariesInBraille = new byte[segmentInBraille.length() - 1];
945 					if (hyphensAndLetterBoundaries != null) {
946 						for (int i = 0; i < hyphensAndLetterBoundariesInBraille.length; i++) {
947 							int pos = interCharacterIndicesInBraille[curPosInBraille + i] - 1;
948 							if (pos >= 0)
949 								hyphensAndLetterBoundariesInBraille[i] = hyphensAndLetterBoundaries[pos - curPos];
950 						}
951 					}
952 					String r = insertHyphens(segmentInBraille, hyphensAndLetterBoundariesInBraille, false, SHY, ZWSP, US);
953 					return (letterSpacing > 0) ? applyLetterSpacing(r, letterSpacing) : r;
954 				}

956 				private String addLetterSpacing(String segment,
957 				                                String segmentInBraille,
958 				                                int curPos,
959 				                                int curPosInBraille,
960 				                                int letterSpacing) {
961 					if (letterSpacing > 0) {
962 						// FIXME: don't hard-code the number 1
963 						byte[] letterBoundaries = detectLetterBoundaries(null, segment, (byte)1);
964 						byte[] letterBoundariesInBraille = new byte[segmentInBraille.length() - 1];
965 						for (int i = 0; i < letterBoundariesInBraille.length; i++) {
966 							int pos = interCharacterIndicesInBraille[curPosInBraille + i] - 1;
967 							if (pos >= 0)
968 								letterBoundariesInBraille[i] = letterBoundaries[pos - curPos];
969 						}
970 						return applyLetterSpacing(insertHyphens(segmentInBraille, letterBoundariesInBraille, false, US), letterSpacing); }
971 					else
972 						return segmentInBraille;
973 				}

975 				private boolean updateInput(int start, int end, String replacement) {
976 					if (substringByCodePoints(joinedText, start, end).equals(replacement))
977 						return false;
978 					joinedText = substringByCodePoints(joinedText, 0, start) + replacement + substringByCodePoints(joinedText, end);
979 					{ // recompute joinedTextMapping

1449 								if ((outputAttrs[j] & 4) == 4)
1450 									b.append(US); }}
1451 						brailleWithWs[l] = b.toString();
1452 						if (pre[l])
1453 							if (brailleWithWs[l].equals(""+NBSP))
1454 								brailleWithWs[l] = textWithWs[l];
1455 							else
1456 								wsLost = true;
1457 						l++;
1458 						while (l < kmax) {
1459 							if (pre[l])
1460 								wsLost = true;
1461 							brailleWithWs[l++] = ""; }
1462 						if (wsLost) {
1463 							logger.warn("White space was not preserved: " + joinedText.replaceAll("\\s+"," "));
1464 							logger.debug("White space was lost in the output.\n"
1465 							             + "Input: " + Arrays.toString(textWithWs) + "\n"
1466 							             + "Output: " + Arrays.toString(brailleWithWs)); }
1467 					}
1468 				}
1469 			} catch (TranslationException e) {
1470 				throw new RuntimeException(e);
1471 			} catch (DisplayException e) {
1472 				throw new RuntimeException(e); }
1473 			
1474 			// recombine white space segments with other segments
1475 			String braille[] = new String[text.length];
1476 			for (int i = 0; i < braille.length; i++)
1477 				braille[i] = "";
1478 			for (int j = 0; j < brailleWithWs.length; j++)
1479 				braille[textWithWsMapping[j]] += brailleWithWs[j];
1480 			
1481 			// apply letter spacing
1482 			if (someLetterSpacing)
1483 				for (int i = 0; i < braille.length; i++)
1484 					braille[i] = applyLetterSpacing(braille[i], letterSpacing[i]);
1485 			
1486 			return braille;
1487 		}

1489 		/*
1490 		 * Detect where letter boundaries meet. Length of addTo must be one less than length of text.
1491 		 */
1492 		private byte[] detectLetterBoundaries(byte[] addTo, String text, byte val) {
1493 			if (addTo == null)
1494 				addTo = new byte[lengthByCodePoints(text) - 1];
1495 			int i = 0;
1496 			int prev = -1;
1497 			for (int c : text.codePoints().toArray()) {
1498 				if (i > 0 && ((Character.isLetter(c) && Character.isLetter(prev)) ||
1499 				              c == '-' ||
1500 				              prev == '-'))
1501 					addTo[i - 1] |= val;
1502 				if (i < addTo.length && c == '\u00ad') // SHY is not actual character, so boundary only after SHY
1503 					addTo[i] |= val;
1504 				prev = c;
1505 				i++;
1506 			}
1507 			return addTo;
1508 		}

1700 	 * * -louis-...

1712 					typeform = typeform.add(Typeform.COMPUTER);

1715 						emphClass= "italic";
1716 					else if (emphClass.equals("under"))
1717 						emphClass = "underline";
1718 					Typeform tf = supportedTypeforms.get(emphClass);
1719 					if (tf != null)
1720 						typeform = typeform.add(tf);
1721 					else
1722 						logger.warn("text-transform: {} not supported: emphclass '{}' not defined in table {}",
1723 						            tt,
1724 						            emphClass,
1725 						            table.getTable());

1727 				continue;
1728 			} else if (tt.equals("uppercase") || tt.equals("lowercase")) {
