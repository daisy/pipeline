159 	protected final Iterable<LiblouisTranslator> _get(Query query) {
160 		MutableQuery q = mutableQuery(query);
161 		for (Feature f : q.removeAll("input"))
162 			if (!supportedInput.contains(f.getValue().get()))
163 				return empty;
164 		boolean asciiBraille = false;
165 		if (q.containsKey("output")) {
166 			String v = q.removeOnly("output").getValue().get();
167 			if ("braille".equals(v)) {}
168 			else if ("ascii".equals(v))
169 				asciiBraille = true;
170 			else
171 				return empty; }
172 		if (q.containsKey("translator"))
173 			if (!"liblouis".equals(q.removeOnly("translator").getValue().get()))
174 				return empty;
175 		String table = null;
176 		if (q.containsKey("liblouis-table"))
177 			table = q.removeOnly("liblouis-table").getValue().get();
178 		if (q.containsKey("table"))
179 			if (table != null) {
180 				logger.warn("A query with both 'table' and 'liblouis-table' never matches anything");
181 				return empty; }
182 			else
183 				table = q.removeOnly("table").getValue().get();
184 		String v = null;
185 		if (q.containsKey("hyphenator"))
186 			v = q.removeOnly("hyphenator").getValue().get();
187 		else
188 			v = "auto";
189 		final String hyphenator = v;
190 		v = null;
191 		if (q.containsKey("locale"))
192 			v = q.removeAll("locale").iterator().next().getValue().get();
193 		final String locale = v;
194 		v = null;
195 		if (q.containsKey("handle-non-standard-hyphenation"))
196 			v = q.removeOnly("handle-non-standard-hyphenation").getValue().get();
197 		else
198 			v = "ignore";
199 		final int handleNonStandardHyphenation = v.equalsIgnoreCase("fail") ?
200 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_FAIL : v.equalsIgnoreCase("defer") ?
201 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_DEFER :
202 			LiblouisTranslatorImpl.NON_STANDARD_HYPH_IGNORE;
203 		if (table != null && !q.isEmpty()) {
204 			logger.warn("A query with both 'table' or 'liblouis-table' and '"
205 			            + q.iterator().next().getKey() + "' never matches anything");
206 			return empty; }
207 		if (table != null)
208 			q.add("table", table);
209 		if (locale != null)
210 			try {
211 				q.add("locale", parseLocale(locale).toLanguageTag()); }
212 			catch (IllegalArgumentException e) {
213 				logger.error("Invalid locale", e);
214 				return empty; }
215 		if (!asciiBraille)
216 			q.add("unicode");
217 		q.add("white-space");
218 		Iterable<LiblouisTranslator> translators = getSimpleTranslator(
219 			q.asImmutable(),
220 			locale,
221 			hyphenator,
222 			handleNonStandardHyphenation);
223 		if (translators.iterator().hasNext()) {
224 			String contraction = q.containsKey("contraction")
225 				? q.removeAll("contraction").iterator().next().getValue().get()
226 				: null;
227 			boolean computer = q.containsKey("type")
228 				&& "computer".equals(q.get("type").iterator().next().getValue().orElse(null));
229 			if (!computer && !"no".equals(contraction)) {
230 				q.add("contraction", "no");
231 				q.removeAll("grade");
232 				Iterable<LiblouisTranslator> nonContractingTranslators = Iterables.memoize(
233 					getSimpleTranslator(
234 						q.asImmutable(),
235 						locale,
236 						hyphenator,
237 						handleNonStandardHyphenation));
238 				if (nonContractingTranslators.iterator().hasNext())
239 					return concat(
240 						Iterables.transform(
241 							combinations(
242 								Maps.toMap(
243 									ImmutableList.<Boolean>of(Boolean.TRUE, Boolean.FALSE),
244 									contracted -> contracted ? translators : nonContractingTranslators)),
245 							new Function<Map<Boolean,WithSideEffect<LiblouisTranslator,Logger>>,LiblouisTranslator>() {
246 								public LiblouisTranslator _apply(Map<Boolean,WithSideEffect<LiblouisTranslator,Logger>> translators)
247 									throws NoSuchElementException {
248 									LiblouisTranslator t;
249 									try {
250 										t = __apply(translators.get(true));
251 									} catch (NoSuchElementException e) {
252 										// make sure all elements that we get from an iterator are also
253 										// dereferenced (because AbstractTransformProvider.util.concat
254 										// requires it)
255 										__apply(translators.get(false));
256 										throw e;
257 									}
258 									return new HandleTextTransformUncontracted(t, __apply(translators.get(false)));
259 								}
260 							}),
261 						translators);
262 			}
263 		}
264 		return translators;
265 	}

267 	private Iterable<LiblouisTranslator> getSimpleTranslator(Query query,
268 	                                                         String locale,
269 	                                                         String hyphenator,
270 	                                                         int handleNonStandardHyphenation) {
271 		return concat(
272 			transform(
273 				logSelect(query, tableProvider),
274 				new Function<LiblouisTableJnaImpl,Iterable<LiblouisTranslator>>() {
275 					public Iterable<LiblouisTranslator> _apply(final LiblouisTableJnaImpl table) {
276 						Iterable<LiblouisTranslator> translators = empty;
277 						Normalizer.Form unicodeNormalization = table.getUnicodeNormalizationForm();
278 						if (!"none".equals(hyphenator)) {
279 							if ("liblouis".equals(hyphenator) || "auto".equals(hyphenator))
280 								for (URI t : table.asURIs())
281 									if (t.toString().endsWith(".dic")) {
282 										translators = Iterables.of(
283 											logCreate((LiblouisTranslator)new LiblouisTranslatorHyphenatorImpl(
284 													table.getTranslator(),
285 													handleNonStandardHyphenation,
286 													locale,
287 													unicodeNormalization))
288 										);
289 										break; }
290 							if (!"liblouis".equals("hyphenator")) {
291 								MutableQuery hyphenatorQuery = mutableQuery();
292 								if (!"auto".equals(hyphenator))
293 									hyphenatorQuery.add("hyphenator", hyphenator);
294 								if (locale != null)
295 									hyphenatorQuery.add("locale", locale);
296 								Iterable<Hyphenator> hyphenators = logSelect(hyphenatorQuery.asImmutable(), hyphenatorProvider);
297 								if (locale != null && !"auto".equals(hyphenator)) {
298 									// also search without locale because locale might only refer to translator itself
299 									hyphenatorQuery.removeAll("locale");
300 									hyphenators = concat(
301 										hyphenators,
302 										logSelect(hyphenatorQuery.asImmutable(), hyphenatorProvider)); }
303 								translators = concat(
304 									translators,
305 									transform(
306 										hyphenators,
307 										new Function<Hyphenator,LiblouisTranslator>() {
308 											public LiblouisTranslator _apply(Hyphenator hyphenator) {
309 												return __apply(
310 													logCreate(
311 														(LiblouisTranslator)new LiblouisTranslatorImpl(
312 																table.getTranslator(),
313 																hyphenator,
314 																handleNonStandardHyphenation,
315 																locale,
316 																unicodeNormalization))); }}));
317 								}}
318 						if ("none".equals(hyphenator) || "auto".equals(hyphenator))
319 							translators = concat(
320 								translators,
321 								logCreate((LiblouisTranslator)new LiblouisTranslatorImpl(
322 										table.getTranslator(),
323 										null,
324 										handleNonStandardHyphenation,
325 										locale,
326 										unicodeNormalization)));
327 						return translators;
328 					}
329 				}
330 			)
331 		);
332 	}

497 			protected BrailleStream translateAndHyphenate(java.lang.Iterable<CSSStyledText> styledText, int from, int to) {
498 				// styledText is cloned because we are mutating the style objects
499 				java.lang.Iterable<CSSStyledText> styledTextCopy
500 					= org.daisy.pipeline.braille.common.util.Iterables.clone(styledText);
501 				java.lang.Iterable<String> braille;
502 				try {
503 					braille = fullTranslator.transform(styledTextCopy); }
504 				catch (Exception e) {
505 					return new BrailleStreamImpl(liblouisTranslator,
506 					                             supportedTypeforms,
507 					                             lineBreaker,
508 					                             fullHyphenator,
509 					                             unicodeNormalization,
510 					                             styledText,
511 					                             from,
512 					                             to); }
513 				// style is mutated and may not be empty
514 				Iterator<SimpleInlineStyle> style = Iterators.transform(styledTextCopy.iterator(), CSSStyledText::getStyle);
515 				StringBuilder brailleString = new StringBuilder();
516 				int fromChar = 0;
517 				int toChar = to >= 0 ? 0 : -1;
518 				for (String s : braille) {
519 					// the only property expected in the output is white-space
520 					// ignore other properties
521 					SimpleInlineStyle st = style.next();

524 						if (ws != null) {

526 								s = s.replaceAll("[\\x20\t\\u2800]+", "$0\u200B")
527 									.replaceAll("[\\x20\t\\u2800]", "\u00A0");

919 														next += bestSolution.lineInBraille;
920 														available = 0;

922 															updateBraille();

930 													break segments; } }}}}}

938 									curPos = spaceEnd;

940 					}

942 						throw new IllegalStateException();
943 					lastPeek = null;
944 					return next;
945 				}

1492 						// split at all positions where the segment number is increased in the output
1493 						TranslationResult r = translator.translate(joinedText, _typeform, inputSegmentNumbers, null);

1502 						int k = joinedTextMapping[0];

1504 						while (l < k)
1505 							brailleWithWs[l++] = "";

1507 							if (outputSegmentNumbers[j] > l) {
1508 								brailleWithWs[l] = b.toString();
1509 								b = new StringBuffer();
1510 								// FIXME: don't hard-code the number 8
1511 								if (j > 0 && (outputAttrs[j - 1] & 8) == 8) {
1512 									if (pre[l]) {
1513 										Matcher m = Pattern.compile("\\xA0([\\xAD\\u200B]*)").matcher(brailleWithWs[l]);
1514 										if (m.matches())
1515 											brailleWithWs[l] = textWithWs[l] + m.group(1);
1516 										else
1517 											wsLost = true; }}
1518 								else {
1519 									if (pre[l])
1520 										wsLost = true;
1521 									if (l <= kmax && pre[l + 1]) {
1522 										pre[l + 1] = false;
1523 										wsLost = true; }}
1524 								l++;
1525 								while (outputSegmentNumbers[j] > l) {
1526 									brailleWithWs[l] = "";
1527 									if (pre[l])
1528 										wsLost = true;
1529 									l++; }}
1530 							b.append(joinedBrailleWithoutHyphens.charAt(j));
1531 							if (j < jmax - 1) {
1532 								// FIXME: don't hard-code these numbers
1533 								if ((outputAttrs[j] & 1) == 1)
1534 									b.append(SHY);
1535 								if ((outputAttrs[j] & 2) == 2)
1536 									b.append(ZWSP);
1537 								if ((outputAttrs[j] & 4) == 4)

1539 						brailleWithWs[l] = b.toString();

1541 							if (brailleWithWs[l].equals(""+NBSP))
1542 								brailleWithWs[l] = textWithWs[l];
1543 							else
1544 								wsLost = true;
1545 						l++;
1546 						while (l < kmax) {
1547 							if (pre[l])
1548 								wsLost = true;
1549 							brailleWithWs[l++] = ""; }
1550 						if (wsLost) {
1551 							logger.warn("White space was not preserved: " + joinedText.replaceAll("\\s+"," "));
1552 							logger.debug("White space was lost in the output.\n"
1553 							             + "Input: " + Arrays.toString(textWithWs) + "\n"
1554 							             + "Output: " + Arrays.toString(brailleWithWs)); }
1555 					}
1556 				}
1557 			} catch (TranslationException e) {
1558 				throw new RuntimeException(e);
1559 			} catch (DisplayException e) {
1560 				throw new RuntimeException(e); }
1561 			
1562 			// recombine white space segments with other segments
1563 			String braille[] = new String[text.length];
1564 			for (int i = 0; i < braille.length; i++)
1565 				braille[i] = "";
1566 			for (int j = 0; j < brailleWithWs.length; j++)
1567 				braille[textWithWsMapping[j]] += brailleWithWs[j];
1568 			
1569 			// apply letter spacing
1570 			if (someLetterSpacing)
1571 				for (int i = 0; i < braille.length; i++)
1572 					braille[i] = applyLetterSpacing(braille[i], letterSpacing[i]);
1573 			
1574 			return braille;
1575 		}

1578 		 * Detect where letter boundaries meet. Length of addTo must be one less than length of text.
1579 		 */
1580 		private static byte[] detectLetterBoundaries(byte[] addTo, String text, byte val) {
1581 			if (addTo == null)
1582 				addTo = new byte[text.length() - 1];
1583 			for(int i = 0; i < addTo.length; i++){
1584 				if(Character.isLetter(text.charAt(i)) && Character.isLetter(text.charAt(i+1)))
1585 					addTo[i] |= val;
1586 				if((text.charAt(i) == '-') || (text.charAt(i+1) == '-'))
1587 					addTo[i] |= val;
1588 				if((text.charAt(i) == '\u00ad')) // SHY is not actual character, so boundary only after SHY
1589 					addTo[i] |= val;
1590 				}
1591 			return addTo;
1592 		}
