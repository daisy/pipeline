32  import org.slf4j.LoggerFactory;

34  @Component(
35  	name = "louis:hyphenate",
36  	service = { ExtensionFunctionDefinition.class }
37  )
38  @SuppressWarnings("serial")
39  public class HyphenateDefinition extends ExtensionFunctionDefinition {
40  	
41  	private static final StructuredQName funcname = new StructuredQName("louis",
42  			"http://liblouis.org/liblouis", "hyphenate");
43  	
44  	@Reference(
45  		name = "LiblouisHyphenatorProvider",
46  		unbind = "-",
47  		service = LiblouisHyphenator.Provider.class,
48  		cardinality = ReferenceCardinality.MANDATORY,
49  		policy = ReferencePolicy.STATIC
50  	)
51  	protected void bindLiblouisHyphenatorProvider(LiblouisHyphenator.Provider provider) {
52  		hyphenatorProvider = provider;
53  		logger.debug("Binding LiblouisHyphenator provider: {}", provider);
54  	}
55  	
56  	private LiblouisHyphenator.Provider hyphenatorProvider;
57  	
58  	public StructuredQName getFunctionQName() {
59  		return funcname;
60  	}
61  	
62  	@Override
63  	public int getMinimumNumberOfArguments() {
64  		return 2;
65  	}
66  	
67  	@Override
68  	public int getMaximumNumberOfArguments() {
69  		return 2;
70  	}
71  	
72  	public SequenceType[] getArgumentTypes() {
73  		return new SequenceType[] {
74  				SequenceType.SINGLE_STRING,
75  				SequenceType.SINGLE_STRING};
76  	}
77  	
78  	public SequenceType getResultType(SequenceType[] suppliedArgumentTypes) {
79  		return SequenceType.SINGLE_STRING;
80  	}
81  	
82  	public ExtensionFunctionCall makeCallExpression() {
83  		return new ExtensionFunctionCall() {
84  			public Sequence call(XPathContext context, Sequence[] arguments) throws XPathException {
85  				try {
86  					Query query = query(((AtomicSequence)arguments[0]).getStringValue());
87  					String[] text = sequenceToArray(arguments[1]);
88  					Hyphenator hyphenator;
89  					try {
90  						hyphenator = hyphenatorProvider.get(query).iterator().next(); }
91  					catch (NoSuchElementException e) {
92  						throw new XPathException("Could not find a hyphenator for query: " + query); }
93  					return arrayToSequence(hyphenator.asFullHyphenator().transform(text));}
94  				catch (XPathException e) {
95  					throw e; }
96  				catch (Throwable e) {
97  					throw new XPathException("Unexpected error in louis:hyphenate", e); }
98  			}
99  		};
100 	}
101 	
102 	private static String[] sequenceToArray(Sequence seq) throws XPathException {
103 		List<String> list = new ArrayList<String>();
104 		SequenceIterator iterator = seq.iterate();
105 		for (Item item = iterator.next(); item != null; item = iterator.next())
106 			list.add(item.getStringValue());
107 		return list.toArray(new String[list.size()]);
108 	}
109 	
110 	private static Sequence arrayToSequence(String[] array) {
111 		List<StringValue> list = new ArrayList<StringValue>();
112 		for (String s : array)
113 			list.add(new StringValue(s));
114 		return new SequenceExtent(list);
115 	}
116 	
117 	private static final Logger logger = LoggerFactory.getLogger(HyphenateDefinition.class);
118 }
