1   package org.daisy.pipeline.braille.liblouis;
2   
3   import java.net.URI;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   import com.google.common.base.Function;
8   import static com.google.common.collect.Iterables.transform;
9   
10  import org.daisy.pipeline.braille.common.BundledResourcePath;
11  
12  public class LiblouisTablePath extends BundledResourcePath {
13  	
14  	private static final String MANIFEST = "manifest";
15  	
16  	@Override
17  	protected void activate(Map<?,?> properties, Class<?> context) {
18  		if (properties.get(UNPACK) != null)
19  			throw new IllegalArgumentException(UNPACK + " property not supported");
20  		if (properties.get(MANIFEST) != null)
21  			throw new IllegalArgumentException(MANIFEST + " property not supported");
22  		Map<Object,Object> props = new HashMap<Object,Object>(properties);
23  		props.put(BundledResourcePath.UNPACK, true);
24  		super.activate(props, context);
25  	}
26  	
27  	public Iterable<URI> listTableFiles() {
28  		return transform(
29  			listResources(),
30  			new Function<URI,URI>() {
31  				public URI apply(URI resource) {
32  					return getIdentifier().resolve(resource); }});
33  	}
34  }
