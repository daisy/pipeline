1   package org.daisy.pipeline.braille.liblouis.impl;

3   import java.net.URI;
4   import java.util.List;
5   import java.util.Locale;
6   import java.util.Map;
7   
8   import com.google.common.base.Objects;
9   import com.google.common.base.Objects.ToStringHelper;
10  import com.google.common.collect.ImmutableList;
11  import com.google.common.collect.ImmutableMap;
12  
13  import org.daisy.pipeline.braille.common.AbstractTransform;
14  import org.daisy.pipeline.braille.common.AbstractTransformProvider;
15  import org.daisy.pipeline.braille.common.AbstractTransformProvider.util.Iterables;
16  import static org.daisy.pipeline.braille.common.AbstractTransformProvider.util.Iterables.of;
17  import static org.daisy.pipeline.braille.common.AbstractTransformProvider.util.logCreate;
18  import org.daisy.pipeline.braille.common.Query;
19  import org.daisy.pipeline.braille.common.Query.Feature;
20  import org.daisy.pipeline.braille.common.Transform;

22  import static org.daisy.pipeline.braille.common.util.Locales.parseLocale;
23  import static org.daisy.pipeline.braille.common.util.URIs.asURI;
24  
25  import org.osgi.service.component.annotations.Activate;
26  import org.osgi.service.component.annotations.Component;
27  import org.osgi.service.component.ComponentContext;
28  
29  public interface LiblouisMathMLTransform {
30  	
31  	public enum MathCode {
32  		NEMETH, UKMATHS, MARBURG, WOLUWE
33  	}
34  	
35  	@Component(
36  		name = "org.daisy.pipeline.braille.liblouis.impl.LiblouisMathMLTransform.Provider",
37  		service = {
38  			TransformProvider.class
39  		}
40  	)
41  	public class Provider extends AbstractTransformProvider<Transform> {
42  		
43  		private URI href;
44  		
45  		@Activate
46  		private void activate(ComponentContext context, final Map<?,?> properties) {
47  			href = asURI(context.getBundleContext().getBundle().getEntry("xml/translate-mathml.xpl"));
48  		}
49  		
50  		private final static Iterable<Transform> empty = Iterables.<Transform>empty();
51  		
52  		private final static List<String> supportedOutput = ImmutableList.of("braille");
53  		
54  		protected Iterable<Transform> _get(final Query query) {
55  			try {
56  				if ("mathml".equals(query.getOnly("input").getValue().get())) {
57  					for (Feature f : query.get("output"))
58  						if (!supportedOutput.contains(f.getValue().get()))
59  							return empty;
60  					if (query.containsKey("locale")) {
61  						Locale locale; {
62  							try {
63  								locale = parseLocale(query.getOnly("locale").getValue().get()); }
64  							catch (IllegalArgumentException e) {
65  								return empty; }
66  						}
67  						MathCode code = mathCodeFromLocale(locale);
68  						if (code != null)
69  							return of(logCreate((Transform)new TransformImpl(code))); }}}
70  			catch (IllegalStateException e) {}
71  			return empty;
72  		}
73  		
74  		private class TransformImpl extends AbstractTransform {
75  			
76  			private final MathCode code;
77  			private final XProc xproc;
78  			
79  			private TransformImpl(MathCode code) {
80  				this.code = code;
81  				Map<String,String> options = ImmutableMap.of("math-code", code.name());
82  				xproc = new XProc(href, null, options);
83  			}
84  			
85  			@Override
86  			public XProc asXProc() {
87  				return xproc;
88  			}
89  			
90  			@Override
91  			public ToStringHelper toStringHelper() {
92  				return Objects.toStringHelper("o.d.p.b.liblouis.impl.LiblouisMathMLTransform$Provider$TransformImpl")
93  					.add("mathCode", code);
94  			}
95  		}
96  		
97  		private static MathCode mathCodeFromLocale(Locale locale) {
98  			String language = locale.getLanguage().toLowerCase();
99  			String country = locale.getCountry().toUpperCase();
100 			if (language.equals("en")) {
101 				if (country.equals("GB"))
102 					return MathCode.UKMATHS;
103 				else
104 					return MathCode.NEMETH; }
105 			else if (language.equals("de"))
106 				return MathCode.MARBURG;
107 			else if (language.equals("nl"))
108 				return MathCode.WOLUWE;
109 			else
110 				return null;
111 		}
112 		
113 	}
114 }
