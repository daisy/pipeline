46  		public interface MemoizingProvider<T extends Transform>
47  			extends Provider.util.MemoizingProvider<Query,T>, TransformProvider<T> {}
48  		
49  		public static <T extends Transform> MemoizingProvider<T> memoize(TransformProvider<T> provider) {
50  			return new MemoizeFromProvider<T>(provider);
51  		}

53  		public static abstract class Memoize<T extends Transform>
54  				extends Provider.util.Memoize<Query,T> implements MemoizingProvider<T> {
55  			
56  			private final Map<Logger,TransformProvider<T>> providerCache;
57  			private final boolean idOnly;
58  			
59  			public Memoize() {
60  				this(false);
61  			}
62  			
63  			/**
64  			 * @param idOnly whether to memoize {@code (id:...)} lookups only.
65  			 */
66  			public Memoize(boolean idOnly) {
67  				providerCache = new HashMap<Logger,TransformProvider<T>>();
68  				providerCache.put(null, this);
69  				this.idOnly = idOnly;
70  			}
71  			
72  			/**
73  			 * Create a new {@link Memoize} object that is based on the same underlying caches.
74  			 */
75  			protected Memoize(Memoize<T> shareCacheWith) {
76  				super(shareCacheWith);
77  				this.providerCache = shareCacheWith.providerCache;
78  				this.idOnly = shareCacheWith.idOnly;
79  			}
80  			
81  			protected boolean skip(Query query) {
82  				return !idOnly || !(query.containsKey("id") && Iterables.size(query) == 1);
83  			}
84  			
85  			@Override
86  			public Iterable<T> get(Query query) {
87  				MutableQuery q = mutableQuery(query);
88  				if (q.containsKey("id")) {
89  					Feature f = q.removeOnly("id");
90  					if (q.isEmpty()) {
91  						T t = fromId(f.getValue().get());
92  						if (t != null)
93  							return Collections.singleton(t); }}
94  				return rememberId(super.get(query));
95  			}
96  			
97  			private final Map<String,T> fromId = new HashMap<>();
98  			
99  			private T fromId(String id) {
100 				return fromId.get(id);
101 			}
102 			
103 			private T rememberId(T t) {
104 				fromId.put(t.getIdentifier(), t);
105 				return t;
106 			}
107 			
108 			private Iterable<T> rememberId(final Iterable<T> iterable) {
109 				return new java.lang.Iterable<T>() {
110 					public Iterator<T> iterator() {
111 						return new Iterator<T>() {
112 							Iterator<T> i = iterable.iterator();
113 							public boolean hasNext() {
114 								return i.hasNext();
115 							}
116 							public T next() {
117 								return rememberId(i.next());
118 							}
119 							public void remove() {
120 								i.remove();
121 							}
122 						};
123 					}
124 				};
125 			}
126 			
127 			/**
128 			 * Create a new {@link TransformProvider} with the given context that is also memoizing
129 			 * and is based on the same underlying caches.
130 			 */
131 			protected abstract TransformProvider<T> _withContext(Logger context);
132 			
133 			public TransformProvider<T> withContext(Logger context) {
134 				if (providerCache.containsKey(context))
135 					return providerCache.get(context);
136 				TransformProvider<T> provider = _withContext(context);
137 				providerCache.put(context, provider);
138 				return provider;
139 			}
140 		}
141 		
142 		private static class MemoizeFromProvider<T extends Transform> extends Memoize<T> {
143 			private final TransformProvider<T> provider;
144 			private MemoizeFromProvider(TransformProvider<T> provider) {
145 				this.provider = provider;
146 			}
147 			private MemoizeFromProvider(MemoizeFromProvider<T> provider) {
148 				super(provider);
149 				this.provider = provider.provider;
150 			}
151 			protected Iterable<T> _get(Query query) {
152 				return provider.get(query);
153 			}
154 			protected TransformProvider<T> _withContext(Logger context) {
155 				return new MemoizeFromProvider<T>(provider.withContext(context));
156 			}
157 			@Override
158 			public String toString() {
159 				return "memoize( " + provider + " )";
160 			}
161 		}

167 		public static abstract class Dispatch<T extends Transform>
168 				extends Provider.util.Dispatch<Query,T> implements TransformProvider<T> {
169 			
170 			private final Logger context;
171 			
172 			public Dispatch(Logger context) {
173 				this.context = context;
174 			}
175 			
176 			protected abstract Iterable<TransformProvider<T>> _dispatch();
177 			
178 			public final Iterable<Provider<Query,T>> dispatch() {
179 				return Iterables.transform(
180 					_dispatch(),
181 					new Function<TransformProvider<T>,Provider<Query,T>>() {
182 						public Provider<Query,T> apply(TransformProvider<T> provider) {
183 							return provider.withContext(context); }});
184 			}
185 		}
186 		
187 		@SuppressWarnings(
188 			"unchecked" // safe cast to Iterable<Provider<Q,X>>
189 		)
190 		public static <T extends Transform> TransformProvider<T> dispatch(Iterable<? extends TransformProvider<T>> dispatch) {
191 			return new DispatchFromProviderIterable<T>((Iterable<TransformProvider<T>>)dispatch, null);
192 		}
193 		
194 		private static class DispatchFromProviderIterable<T extends Transform> extends Dispatch<T> {
195 			private final Iterable<TransformProvider<T>> dispatch;
196 			private DispatchFromProviderIterable(Iterable<TransformProvider<T>> dispatch, Logger context) {
197 				super(context);
198 				this.dispatch = dispatch;
199 			}
200 			protected Iterable<TransformProvider<T>> _dispatch() {
201 				return dispatch;
202 			}
203 			public TransformProvider<T> withContext(Logger context) {
204 				return new DispatchFromProviderIterable<T>(dispatch, context);
205 			}
206 			@Override
207 			public String toString() {
208 				return "dispatch( " + join(_dispatch(), ", ") + " )";
209 			}
210 		}

220 		private static class Partial<T extends Transform> implements TransformProvider<T> {
221 			private final Query partialQuery;
222 			private final TransformProvider<T> provider;
223 			public Partial(Query partialQuery, TransformProvider<T> provider) {
224 				this.partialQuery = partialQuery;
225 				this.provider = provider;
226 			}
227 			public final Iterable<T> get(Query query) {
228 				return provider.get(mutableQuery().addAll(partialQuery).addAll(query));
229 			}
230 			public TransformProvider<T> withContext(Logger context) {
231 				return new Partial<T>(partialQuery, provider.withContext(context));
232 			}
233 			@Override
234 			public String toString() {
235 				return "partial( " + partialQuery + ", " + provider + " )";
236 			}

248 		public static <T extends Transform> Iterable<T> logSelect(final Query query,
249 		                                                          final TransformProvider<T> provider,
250 		                                                          final Logger context) {
251 			return new Iterable<T>() {
252 				public Iterator<T> iterator() {
253 					return new AbstractIterator<T>() {
254 						Iterator<T> i = provider.get(query).iterator();
255 						boolean first = true;
256 						public T computeNext() {
257 							if (!i.hasNext()) {
258 								if (first)
259 									context.trace("No match for query " + query);
260 								return endOfData(); }
261 							T t = i.next();
262 							context.debug("Selected " + t + " for query " + query);
263 							first = false;
264 							return t;
265 						}
266 					};
267 				}
268 			};
269 		}

279 		private static class VaryLocale<T extends Transform>
280 				extends Provider.util.VaryLocale<Query,T> implements TransformProvider<T> {
281 			
282 			private final TransformProvider<T> delegate;
283 			private final Logger context;
284 			private VaryLocale(TransformProvider<T> delegate, Logger context) {
285 				this.delegate = delegate;
286 				this.context = context;
287 			}
288 			public Iterable<T> _get(Query query) {
289 				return delegate.withContext(context).get(query);
290 			}
291 			public Locale getLocale(Query query) {
292 				if (query.containsKey("document-locale"))
293 					try {
294 						return parseLocale(query.getOnly("document-locale").getValue().get()); }
295 					catch (IllegalArgumentException e) {
296 						logger.warn("Invalid locale", e);
297 						return null; }
298 				else
299 					return null;
300 			}
301 			public Query assocLocale(Query query, Locale locale) {
302 				MutableQuery q = mutableQuery(query);
303 				q.removeAll("document-locale");
304 				q.add("document-locale", locale.toLanguageTag());
305 				return q.asImmutable();
306 			}
307 			public TransformProvider<T> withContext(Logger context) {
308 				return new VaryLocale<T>(delegate, context);
309 			}
310 			@Override
311 			public String toString() {
312 				return "varyLocale( " + delegate + " )";
313 			}
314 		}
