1   package org.daisy.pipeline.braille.common;
2   
3   import java.io.File;
4   import java.net.URI;
5   import java.net.URL;
6   import static java.nio.file.Files.createTempDirectory;
7   import java.util.ArrayList;
8   import java.util.Collection;
9   import java.util.Map;
10  
11  import com.google.common.base.Function;
12  import com.google.common.base.Predicate;
13  import com.google.common.base.Predicates;
14  import static com.google.common.base.Predicates.alwaysTrue;
15  import com.google.common.collect.Collections2;
16  import com.google.common.collect.ImmutableList;
17  
18  import org.daisy.common.file.URLs;
19  import static org.daisy.pipeline.braille.common.util.Iterators.partition;
20  import static org.daisy.pipeline.braille.common.util.Predicates.matchesGlobPattern;
21  import static org.daisy.pipeline.braille.common.util.Tuple2;
22  import static org.daisy.pipeline.braille.common.util.Files.normalize;
23  
24  public class BundledResourcePath extends AbstractResourcePath {
25  	
26  	protected static final String IDENTIFIER = "identifier";
27  	protected static final String PATH = "path";
28  	protected static final String UNPACK = "unpack";
29  	protected static final String EXECUTABLES = "executables";
30  	protected static final String INCLUDES = "includes";
31  	
32  	private URI identifier = null;
33  	private URL path = null;
34  	private File unpackDir = null;
35  	private boolean unpacking = false;
36  	private boolean executables = false;
37  	
38  	/* The included resources as relative paths */
39  	private Collection<URI> resources = null;
40  	
41  	protected void activate(final Map<?,?> properties, Class<?> context) throws IllegalArgumentException {
42  		if (properties.get(IDENTIFIER) == null || properties.get(IDENTIFIER).toString().isEmpty()) {
43  			throw new IllegalArgumentException(IDENTIFIER + " property must not be empty"); }
44  		String identifierAsString = properties.get(IDENTIFIER).toString();
45  		if (!identifierAsString.endsWith("/")) identifierAsString += "/";
46  		try { identifier = URLs.asURI(identifierAsString); }
47  		catch (Exception e) {
48  			throw new IllegalArgumentException(IDENTIFIER + " could not be parsed into a URI"); }
49  		if (!identifier.isAbsolute())
50  			throw new IllegalArgumentException(IDENTIFIER + " must be an absolute URI");
51  		if (properties.get(PATH) == null || properties.get(PATH).toString().isEmpty()) {
52  			throw new IllegalArgumentException(PATH + " property must not be empty"); }
53  		String pathAsRelativeFilePath = properties.get(PATH).toString();
54  		path = URLs.getResourceFromJAR(pathAsRelativeFilePath, context);
55  		if (path == null)
56  			throw new IllegalArgumentException("Resource path at location " + pathAsRelativeFilePath + " could not be found");
57  		final Predicate<URI> includes =
58  			(properties.get(INCLUDES) != null
59  			 && !properties.get(INCLUDES).toString().isEmpty()
60  			 && !properties.get(INCLUDES).toString().equals("*")) ?
61  				Predicates.compose(
62  					matchesGlobPattern(properties.get(INCLUDES).toString()),
63  					URI::getPath) :
64  				alwaysTrue();
65  		Function<String,Collection<String>> getFilePaths = new Function<String,Collection<String>>() {
66  			public Collection<String> apply(String path) {
67  				Tuple2<Collection<String>,Collection<String>> entries = partition(
68  					URLs.listResourcesFromJAR(path, context),
69  					s -> s.endsWith("/"));
70  				Collection<String> files = new ArrayList<String>();
71  				files.addAll(entries._2);
72  				for (String folder : entries._1) files.addAll(apply(folder));
73  				return files; }};
74  		resources = new ImmutableList.Builder<URI>()
75  			.addAll(
76  				Collections2.<URI>filter(
77  					Collections2.<String,URI>transform(
78  						getFilePaths.apply(pathAsRelativeFilePath),
79  						s -> URLs.relativize(URLs.asURI(path), URLs.asURI(URLs.getResourceFromJAR(s, context)))),
80  					includes))
81  			.build();
82  		if (properties.get(UNPACK) != null && (Boolean)properties.get(UNPACK))
83  			unpacking = true;
84  		if (properties.get(EXECUTABLES) != null && (Boolean)properties.get(EXECUTABLES))
85  			executables = true;
86  	}
87  	
88  	public URI getIdentifier() {
89  		return identifier;
90  	}
91  	
92  	protected boolean containsResource(URI resource) {
93  		return resources.contains(resource);
94  	}
95  	
96  	protected Collection<URI> listResources() {
97  		return resources;
98  	}
99  	
100 	protected URL getBasePath() {
101 		return path;
102 	}
103 	
104 	protected boolean isUnpacking() {
105 		return unpacking;
106 	}
107 	
108 	@Override
109 	protected File makeUnpackDir() {
110 		if (!unpacking)
111 			return null;
112 		if (unpackDir == null) {
113 			File tmpDirectory; {
114 				try {
115 					tmpDirectory = createTempDirectory("pipeline-").toFile(); }
116 				catch (Exception e) {
117 					throw new RuntimeException("Could not create temporary directory", e); }
118 				tmpDirectory.deleteOnExit();
119 			}
120 			unpackDir = normalize(tmpDirectory); }
121 		return unpackDir;
122 	}
123 	
124 	protected boolean isExecutable(URI resources) {
125 		return executables;
126 	}
127 }
