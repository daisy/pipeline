1   package org.daisy.pipeline.braille.common;
2   
3   import org.daisy.dotify.api.translator.BrailleTranslatorResult;
4   import org.daisy.pipeline.braille.css.CSSStyledText;
5   
6   public interface BrailleTranslator extends Transform {
7   
8   	/**
9   	 * Return a new {@link BrailleTranslator} that uses the given {@link Hyphenator} to perform
10  	 * hyphenation.
11  	 *
12  	 * The returned object should be selectable based on its identifier from the same {@link
13  	 * BrailleTranslatorProvider} that provided this {@link BrailleTranslator}.
14  	 */
15  	public BrailleTranslator withHyphenator(Hyphenator hyphenator) throws UnsupportedOperationException;
16  	
17  	public FromStyledTextToBraille fromStyledTextToBraille() throws UnsupportedOperationException;
18  	
19  	public LineBreakingFromStyledText lineBreakingFromStyledText() throws UnsupportedOperationException;
20  	
21  	/* ------------------------- */
22  	/* fromStyledTextToBraille() */
23  	/* ------------------------- */
24  	
25  	public interface FromStyledTextToBraille {
26  		
27  		default public Iterable<String> transform(Iterable<CSSStyledText> styledText) throws TransformationException {
28  			return transform(styledText, 0, -1);
29  		}
30  		
31  		public Iterable<String> transform(Iterable<CSSStyledText> styledText, int from, int to) throws TransformationException;
32  	}
33  	
34  	/* ---------------------------- */
35  	/* lineBreakingFromStyledText() */
36  	/* ---------------------------- */
37  	
38  	public interface LineBreakingFromStyledText {
39  		
40  		/**
41  		 * Transform a sequence of {@link CSSStyledText} to braille, in the form of a {@link
42  		 * LineIterator}, which is an interface for laying out text into lines.
43  		 */
44  		default public LineIterator transform(Iterable<CSSStyledText> styledText) throws TransformationException {
45  			return transform(styledText, 0, -1);
46  		}
47  		
48  		/**
49  		 * Transform the part of the input sequence between index <code>from</code> (included) and
50  		 * index <code>to</code> (not included). If <code>to</code> is a negative number it means
51  		 * transform up until the last segment.
52  		 */
53  		public LineIterator transform(Iterable<CSSStyledText> styledTextWithContext, int from, int to) throws TransformationException;
54  	}
55  	
56  	/* ------------ */
57  	/* LineIterator */
58  	/* ------------ */
59  	
60  	public interface LineIterator extends BrailleTranslatorResult {
61  	}
62  	
63  }
