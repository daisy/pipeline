26  	private Map<Query,Iterable<T>> transformCache = new HashMap<Query,Iterable<T>>();
27  	
28  	private Map<Logger,TransformProvider<T>> providerCache = new HashMap<Logger,TransformProvider<T>>();
29  	
30  	protected abstract Iterable<T> _get(Query query);
31  	
32  	// WARNING: using "(id:<ID>)" in a query where <ID> is the result of a
33  	// call to AbstractTransform#getIdentifier() will only work when that
34  	// transformer came from the exact same provider and its cache is not
35  	// cleared in the meantime.
36  	private final java.lang.Iterable<T> get(Query query, Logger context) {
37  		MutableQuery q = mutableQuery(query);
38  		if (q.containsKey("id")) {
39  			Feature f = q.removeOnly("id");
40  			if (q.isEmpty())
41  				return Optional.fromNullable(fromId(f.getValue().get())).asSet(); }
42  		Iterable<T> i;
43  		if (transformCache.containsKey(query))
44  			i = transformCache.get(query);
45  		else {
46  			// memoize() doesn't make sense
47  			i = util.Iterables.memoize(_get(query));
48  			transformCache.put(query, i); }
49  		return rememberId(i.apply(context));
50  	}
51  	
52  	public java.lang.Iterable<T> get(Query query) {
53  		return get(query, null);
54  	}
55  	
56  	public final TransformProvider<T> withContext(Logger context) {
57  		if (providerCache.containsKey(context))
58  			return providerCache.get(context);
59  		TransformProvider<T> provider = new DerivativeProvider(context);
60  		providerCache.put(context, provider);
61  		return provider;
62  	}
63  	
64  	public void invalidateCache() {
65  		transformCache.clear();
66  	}
67  	
68  	public ToStringHelper toStringHelper() {
69  		return MoreObjects.toStringHelper(this);
70  	}
71  	
72  	@Override
73  	public String toString() {
74  		return toStringHelper().add("context", null).toString();
75  	}

100 	private java.lang.Iterable<T> rememberId(final java.lang.Iterable<T> iterable) {
101 		return new java.lang.Iterable<T>() {
102 			public Iterator<T> iterator() {
103 				return new Iterator<T>() {
104 					Iterator<T> i = iterable.iterator();
105 					public boolean hasNext() {
106 						return i.hasNext();
107 					}
108 					public T next() {
109 						T t = i.next();
110 						fromId.put(t.getIdentifier(), t);
111 						return t;
112 					}
113 					public void remove() {
114 						i.remove();
115 					}
116 				};
117 			}
118 		};
119 	}
120 	
121 	protected Function0<Void> provideTemporarily(T t) {
122 		// assumes t is not in cache yet and is unique to this call
123 		fromId.put(t.getIdentifier(), t);
124 		return () -> {
125 			fromId.remove(t.getIdentifier());
126 			// also remove from transformCache
127 			transformCache.remove(mutableQuery().add("id", t.getIdentifier()));
128 			return null;
129 		};
130 	}

156 		public static <T extends Transform> Iterable<T> logSelect(final Query query,
157 		                                                          final TransformProvider<T> provider) {
158 			// not using provider.withContext() because memoizing only makes sense if sub-providers
159 			// have no side-effects and provided transformers have no context
160 			return logSelect(query, provider.withContext(null).get(query));
161 		}
162 		
163 		public static <T extends Transform> Iterable<T> logSelect(final Query query,
164 		                                                          final java.lang.Iterable<T> iterable) {
165 			return Iterables.of(
166 				new java.lang.Iterable<WithSideEffect<T,Logger>>() {
167 					public Iterator<WithSideEffect<T,Logger>> iterator() {
168 						return new Iterator<WithSideEffect<T,Logger>>() {
169 							Iterator<T> i = iterable.iterator();
170 							boolean first = true;
171 							public boolean hasNext() {
172 								if (first)
173 									return true;
174 								return i.hasNext();
175 							}
176 							public WithSideEffect<T,Logger> next() {
177 								final T t;
178 								if (first) {
179 									first = false;
180 									try { t = i.next(); }
181 									catch (NoSuchElementException e) {
182 										return new WithSideEffect<T,Logger>() {
183 											public T _apply() throws NoSuchElementException {
184 												__apply(debug("No match for query " + query));
185 												throw e;
186 											}
187 										};
188 									}
189 								} else
190 									t = i.next();
191 								return new WithSideEffect<T,Logger>() {
192 									public T _apply() {
193 										__apply(info("Selected " + t + " for query " + query));
194 										return t;
195 									}
196 								};
197 							}
198 							public void remove() {
199 								throw new UnsupportedOperationException();
200 							}
201 						};
202 					}
203 				}
204 			);
205 		}

357 			public static <T> Iterable<T> concat(final Iterable<? extends Iterable<T>> inputs) {
358 				return new Concat<T>() {
359 					protected Iterator<? extends Iterable<T>> iterator(Logger context) {
360 						return inputs.apply(context).iterator();
361 					}
362 					public Iterator<WithSideEffect<T,Logger>> iterator() {
363 						return new AbstractIterator<WithSideEffect<T,Logger>>() {
364 							Iterator<? extends WithSideEffect<? extends Iterable<T>,Logger>> iterableIterator = inputs.iterator();
365 							Iterator<WithSideEffect<T,Logger>> current;
366 							WithSideEffect<T,Logger> nextEvaluate;
367 							boolean evaluated = true;
368 							protected WithSideEffect<T,Logger> computeNext() {
369 								if (!evaluated)
370 									throw new RuntimeException("Previous element must be evaluated first");
371 								if (current != null && current.hasNext())
372 									return current.next();
373 								else if (!iterableIterator.hasNext())
374 									return endOfData();
375 								nextEvaluate = new WithSideEffect<T,Logger>() {
376 									public T _apply() throws NoSuchElementException {
377 										if (nextEvaluate == this)
378 											evaluated = true;
379 										while (current == null || !current.hasNext()) {
380 											WithSideEffect<? extends Iterable<T>,Logger> nextIterable = iterableIterator.next();
381 											try {
382 												current = __apply(nextIterable).iterator();
383 												break; }
384 											catch (NoSuchElementException e) {
385 												continue; }}
386 										if (current == null)
387 											throw new NoSuchElementException();
388 										T result = __apply(current.next());
389 										return result;
390 									}
391 								};
392 								evaluated = false;
393 								return nextEvaluate;
394 							}
395 						};
396 					}
397 				};
398 			}

406 				return of(
407 					new java.lang.Iterable<WithSideEffect<T,Logger>>() {
408 						public Iterator<WithSideEffect<T,Logger>> iterator() {
409 							return new AbstractIterator<WithSideEffect<T,Logger>>() {
410 								Iterator<WithSideEffect<T,Logger>> itrA = a.iterator();
411 								Iterator<WithSideEffect<T,Logger>> itrB = b.iterator();
412 								Set<T> returned = new HashSet<>();
413 								Set<T> setB = new HashSet<>();
414 								protected WithSideEffect<T,Logger> computeNext() {
415 									if (!itrA.hasNext())
416 										return endOfData();
417 									return new WithSideEffect<T,Logger>() {
418 										public T _apply() throws NoSuchElementException {
419 											while (itrA.hasNext()) {
420 												T nextA; {
421 													try {
422 														nextA = __apply(itrA.next()); }
423 													catch (NoSuchElementException e) {
424 														continue; }}
425 												if (returned.contains(nextA))
426 													continue;
427 												if (setB.contains(nextA)) {
428 													returned.add(nextA);
429 													return nextA; }
430 												while (itrB.hasNext())
431 													try {
432 														T nextB = __apply(itrB.next());
433 														setB.add(nextB);
434 														if (Objects.equal(nextA, nextB)) {
435 															returned.add(nextA);
436 															return nextA; }}
437 													catch (NoSuchElementException e) {}
438 											}
439 											throw new NoSuchElementException();
440 										}
441 									};
442 								}
443 							};
444 						}
445 					}
446 				);
