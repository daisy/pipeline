12  /**
13   * Identification of a braille system, including details on contraction levels, specialized braille
14   * systems, and additional elements relevant to braille production.
15   *
16   * For more info see the <a
17   * href="https://daisy.github.io/ebraille/published/registries/codes/">eBraille Braille Codes
18   * Registry</a>.
19   */
20  public class BrailleCode {
21  
22  	/**
23  	 * The level of contraction or complexity in the braille code
24  	 */
25  	public enum Grade {
26  		GRADE0,
27  		GRADE1,
28  		GRADE1_2, // not in spec!
29  		GRADE1_3, // not in spec!
30  		GRADE1_4, // not in spec!
31  		GRADE1_5, // not in spec!
32  		GRADE2,
33  		GRADE3,   // not in spec!
34  		GRADE4,   // not in spec!
35  		/** code does not use contractions or a grading system */
36  		NO_GRADE;
37  
38  		@Override
39  		public String toString() {
40  			return name().toLowerCase().replaceAll("(?<=[0-9])_(?=[0-9])", ".")
41  			                           .replace("_", "-");
42  		}
43  	}
44  
45  	/**
46  	 * Specialized systems
47  	 */
48  	public enum Specialization {
49  		/** 8-dot computer braille */
50  		COMP8,
51  		/** 6-dot computer braille */
52  		COMP6,
53  		/** code is only concerned with mathematics */
54  		MATH,
55  		/** code is only concerned with music */
56  		MUSIC,
57  		/** code is only concerned with phonetics */
58  		PHONETIC;
59  
60  		@Override
61  		public String toString() {
62  			return name().toLowerCase();
63  		}
64  	}

98  	/**
99  	 * Return a IETF BCP 47 language tag with the braille code identifier integrated as a "t"
100 	 * (transformed content) extension (RFC 6497) with a "x0" (private use transform) part.
101 	 *
102 	 * @param sourceLanguage The language tag of the source (untransformed) content.
103 	 * @return the language tag of the braille content, which contains the "Brai" script subtag and
104 	 *         the braille code as a "t" extention.
105 	 */
106 	public Locale toLanguageTag(Locale sourceLanguage) {
107 		Locale.Builder b = new Locale.Builder();
108 		b.setLocale(sourceLanguage);
109 		b.setScript("Brai");
110 		b.setExtension('t', sourceLanguage.toLanguageTag().toLowerCase() + "-x0-" + idForLanguageTag);
111 		BrailleCode prev = idForLanguageTagCache.put(idForLanguageTag, this);
112 		if (prev != null && !prev.equals(this))
113 			throw new IllegalStateException(); // collision
114 		return b.build();
115 	}
116 
117 	/**
118 	 * Get braille code from a IETF BCP 47 language tag. The language tag is expected to contain a
119 	 * "t" extension (RFC 6497) with a "x0" (private use transform) part, and the subtags after the
120 	 * "x0" subtag are expected to form the identifier of the braille code.
121 	 */
122 	public static BrailleCode fromLanguageTag(Locale language) throws IllegalArgumentException {
123 		String t = language.getExtension('t');
124 		if (t == null)
125 			throw new IllegalArgumentException("Language tag does not contain 't' extension: " + language);
126 		int i = t.indexOf("-x0-");
127 		if (i < 0)
128 			throw new IllegalArgumentException("Language tag does not contain 't' extension with 'x0' part: " + language);
129 		t = t.substring(i + 4);
130 		// HACK!
131 		// For now, because the BrailleCode feature is only used in the context of
132 		// dtbook-to-ebraille to pass braille code information from LiblouisTranslatorImpl through
133 		// language tags, and these language tags do not end up in the final eBraille publication,
134 		// we can put whatever we want in them. To simplify things, we generate the language tag
135 		// using a simple but lossy algorithm. For this reason, we don't try to parse it again, but
136 		// rather cache it when it is returned from toLanguageTag() (and hope there are no
137 		// collisions).
138 		BrailleCode cached = idForLanguageTagCache.get(t);
139 		if (cached == null)
140 			throw new IllegalArgumentException(); // should not happen if tag came from toLanguageTag()
141 		return cached;
142 	}

144 	public static BrailleCode parse(String identifier) throws IllegalArgumentException {
145 		List<String> parts = new ArrayList<>();
146 		for (String part : identifier.split(" "))
147 			if (part.length() == 0)
148 				throw new IllegalArgumentException();
149 			else
150 				parts.add(part);
151 		if (parts.size() < 2)
152 			throw new IllegalArgumentException("Missing grade");
153 		String lastPart = parts.get(parts.size() - 1);
154 		parts.remove(parts.size() - 1);
155 		Grade grade;
156 		Specialization specialization = null; {
157 			try {
158 				grade = Grade.valueOf(lastPart.replaceAll("[-.]", "_").toUpperCase());
159 			} catch (IllegalArgumentException e) {
160 				if (parts.size() < 2)
161 					throw new IllegalArgumentException("Could not parse grade: " + lastPart);
162 				String secondToLastPart = parts.get(parts.size() - 1);
163 				parts.remove(parts.size() - 1);
164 				try {
165 					grade = Grade.valueOf(secondToLastPart.replaceAll("[-.]", "_").toUpperCase());
166 				} catch (IllegalArgumentException ee) {
167 					throw new IllegalArgumentException("Could not parse grade: " + secondToLastPart + " or " + lastPart);
168 				}
169 				try {
170 					specialization = Specialization.valueOf(lastPart.toUpperCase());
171 				} catch (IllegalArgumentException ee) {
172 					throw new IllegalArgumentException("Could not parse specialization: " + lastPart);
173 				}
174 			}
175 		}
176 		return new BrailleCode(Strings.join(parts, " "), grade, specialization);
177 	}

179 	private final String id;
180 	private final String system;
181 	private final Grade grade;
182 	private final Optional<Specialization> specialization;
183 	private final String idForLanguageTag; // for inclusion in language tag, see https://github.com/daisy/ebraille/issues/291
184 	private static final Map<String,BrailleCode> idForLanguageTagCache = new HashMap<>();
185 
186 	/**
187 	 * Note that the {@code system} parameter, as well as the combination with {@code grade} and
188 	 * {@code specialization}, is currently not validated, as there are no registered braille codes
189 	 * at this point.
190 	 */
191 	public BrailleCode(String system, Grade grade, Specialization specialization) {
192 		if (system == null || grade == null)
193 			throw new IllegalArgumentException();
194 		this.system = system;
195 		this.grade = grade;
196 		this.specialization  = Optional.ofNullable(specialization);
197 		this.id = system + " "
198 			+ grade
199 			+ (specialization == null ? "" : (" " + specialization));
200 		List<String> languageSubtags = new ArrayList<>(); {
201 			for (String part : system.split(" ")) {
202 				part = part.toLowerCase();
203 				part = part.replaceAll("[^a-z0-9]", "");
204 				if (part.length() < 3) // too short to make into a language subtag
205 					throw new IllegalArgumentException();
206 				if (part.length() > 8) // too long to make into a language subtag
207 					part = part.substring(0, 8);
208 				languageSubtags.add(part);
209 			}
210 			languageSubtags.add(grade.toString().replace(".", "")   // grade1.5 -->  grade15
211 			                                    .replace("-", "")); // no-grade --> nograde
212 			if (specialization != null)
213 				languageSubtags.add(specialization.toString());
214 		}
215 		this.idForLanguageTag = Strings.join(languageSubtags, "-");
216 	}
