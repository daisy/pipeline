8   public abstract class LazyValue<V> implements Function0<V>, Iterable<V> {
9   	
10  	public Iterator<V> iterator() {
11  		return new Iterator<V>() {
12  			boolean hasNext = true;
13  			public boolean hasNext() {
14  				return hasNext;
15  			}
16  			public V next() {
17  				if (!hasNext())
18  					throw new NoSuchElementException();
19  				hasNext = false;
20  				return apply();
21  			}
22  			public void remove() {
23  				throw new UnsupportedOperationException();
24  			}
25  		};
26  	}
27  	
28  	public static <V> LazyValue<V> from(final Function0<V> value) {
29  		return new LazyValue<V>() {
30  			public V apply() {
31  				return value.apply();
32  			}
33  		};
34  	}
35  	
36  	public static abstract class ImmutableLazyValue<V> extends LazyValue<V> {
37  		
38  		private V value = null;
39  		protected boolean computed = false;
40  		
41  		public final V apply() {
42  			if (!computed) {
43  				value = _apply();
44  				computed = true; }
45  			return value;
46  		}
47  		
48  		protected abstract V _apply();
49  		
50  		public static <V> LazyValue<V> from(final Function0<V> value) {
51  			return new ImmutableLazyValue<V>() {
52  				public V _apply() {
53  					return value.apply();
54  				}
55  			};
56  		}
57  	}
