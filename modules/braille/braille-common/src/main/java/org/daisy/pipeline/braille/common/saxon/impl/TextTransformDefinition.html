42  	private static final StructuredQName funcname = new StructuredQName("pf",
43  			"http://www.daisy.org/ns/pipeline/functions", "text-transform");
44  	
45  	@Reference(
46  		name = "BrailleTranslatorRegistry",
47  		unbind = "-",
48  		service = BrailleTranslatorRegistry.class,
49  		cardinality = ReferenceCardinality.MANDATORY,

52  	protected void bindBrailleTranslatorRegistry(BrailleTranslatorRegistry registry) {
53  		translatorRegistry = registry;
54  		logger.debug("Binding BrailleTranslator registry: {}", registry);
55  	}

86  	public ExtensionFunctionCall makeCallExpression() {
87  		return new ExtensionFunctionCall() {
88  			public Sequence call(XPathContext context, Sequence[] arguments) throws XPathException {
89  				try {
90  					Query query = query(arguments[0].head().getStringValue());
91  					List<String> text = sequenceToList(arguments[1]);
92  					List<CSSStyledText> styledText = new ArrayList<CSSStyledText>();
93  					if (arguments.length > 2) {
94  						List<String> style = sequenceToList(arguments[2]);
95  						if (style.size() != text.size())
96  							throw new XPathException("Lengths of text and style sequences must match");
97  						if (arguments.length > 3) {
98  							List<String> lang = sequenceToList(arguments[3]);
99  							if (lang.size() != text.size())
100 								throw new XPathException("Lengths of text and lang sequences must match");
101 							for (int i = 0; i < text.size(); i++) {
102 								Map<String,String> attrs = new HashMap<String,String>();
103 								attrs.put("lang", lang.get(i));
104 								styledText.add(new CSSStyledText(text.get(i), style.get(i), attrs)); }}
105 						else
106 							for (int i = 0; i < text.size(); i++)
107 								styledText.add(new CSSStyledText(text.get(i), style.get(i))); }
108 					else
109 						for (int i = 0; i < text.size(); i++)
110 							styledText.add(new CSSStyledText(text.get(i)));
111 					for (BrailleTranslator t : translatorRegistry.get(query)) {
112 						FromStyledTextToBraille fsttb;
113 						try {
114 							fsttb = t.fromStyledTextToBraille(); }
115 						catch (UnsupportedOperationException e) {
116 							logger.trace("Translator does not implement the FromStyledTextToBraille interface: " + t);
117 							continue; }
118 						try {
119 							return iterableToSequence(fsttb.transform(styledText)); }
120 						catch (Exception e) {
121 							logger.debug("Failed to translate string with translator " + t);
122 							throw e; }
123 					}
124 					throw new XPathException("Could not find a BrailleTranslator for query: " + query); }
125 				catch (XPathException e) {
126 					throw e; }
127 				catch (Throwable e) {
128 					throw new XPathException("Unexpected error in pf:text-transform", e); }
129 			}
130 		};
131 	}
132 	
133 	private static List<String> sequenceToList(Sequence seq) throws XPathException {
134 		List<String> list = new ArrayList<String>();
135 		SequenceIterator iterator = seq.iterate();
136 		for (Item item = iterator.next(); item != null; item = iterator.next())
137 			list.add(item.getStringValue());
138 		return list;
139 	}
140 	
141 	private static Sequence iterableToSequence(Iterable<String> iterable) {
142 		List<StringValue> list = new ArrayList<StringValue>();
143 		for (String s : iterable)
144 			list.add(new StringValue(s));
145 		return new SequenceExtent(list);
146 	}
