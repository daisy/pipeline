32  /**
33   * {@link BrailleTranslator} that dispatches to sub-translators based on text-transform values.
34   */
35  public class CompoundBrailleTranslator extends AbstractBrailleTranslator {
36  
37  	private final Map<String,BrailleTranslator> translators;
38  	private final boolean implementsFromStyledTextToBraille;
39  	private final boolean implementsLineBreakingFromStyledText;
40  
41  	public CompoundBrailleTranslator(BrailleTranslator mainTranslator, Map<String,Supplier<BrailleTranslator>> subTranslators) {
42  		if (subTranslators.containsKey("auto") || mainTranslator == null)
43  			throw new IllegalArgumentException();
44  		translators = new HashMap<>();
45  		translators.put("auto", mainTranslator);
46  		for (Map.Entry<String,Supplier<BrailleTranslator>> t : subTranslators.entrySet())
47  			try {
48  				translators.put(t.getKey(), t.getValue().get());
49  			} catch (NoSuchElementException e) {
50  				logger.warn("No braille translator found for handling text-transform '" + t.getKey() + "'");
51  			}
52  		implementsFromStyledTextToBraille = Iterables.all(
53  			translators.values(),
54  			t -> {
55  				try {
56  					t.fromStyledTextToBraille();
57  					return true; }
58  				catch (UnsupportedOperationException e) {
59  					return false; }} );
60  		implementsLineBreakingFromStyledText = Iterables.all(
61  			translators.values(),
62  			t -> {
63  				try {
64  					t.lineBreakingFromStyledText();
65  					return true; }
66  				catch (UnsupportedOperationException e) {
67  					return false; }} );
68  	}
69  
70  	protected CompoundBrailleTranslator(CompoundBrailleTranslator from) {
71  		super(from);
72  		this.translators = from.translators;
73  		this.implementsFromStyledTextToBraille = from.implementsFromStyledTextToBraille;
74  		this.implementsLineBreakingFromStyledText = from.implementsLineBreakingFromStyledText;
75  	}
76  
77  	@Override
78  	public ToStringHelper toStringHelper() {
79  		return MoreObjects.toStringHelper("CompoundBrailleTranslator")
80  			.add("translators", translators);
81  	}

83  	/**
84  	 * @throws UnsupportedOperationException if the main translator or one of the sub-translators's
85  	 *                                       {@code #withHyphenator()} method throws {@link UnsupportedOperationException}
86  	 */
87  	public CompoundBrailleTranslator _withHyphenator(Hyphenator hyphenator) {
88  		Map<String,BrailleTranslator> translatorsWithHyphenator = new HashMap<>(); {
89  			for (String k : translators.keySet())
90  				translatorsWithHyphenator.put(k, translators.get(k).withHyphenator(hyphenator));
91  		}
92  		return new CompoundBrailleTranslator(translatorsWithHyphenator.remove("auto"),
93  		                                     Maps.transformValues(translatorsWithHyphenator, x -> (() -> x)));
94  	}

96  	private static abstract class TransformImpl<T> {
97  
98  		abstract Iterable<T> transform(Iterable<CSSStyledText> styledText, int from, int to, String textTransform);
99  		abstract Iterable<String> transformContext(Iterable<CSSStyledText> styledText, int from, int to, String textTransform);
100 		
101 		abstract boolean supports(String textTransform);
102 
103 		List<T> transform(Iterable<CSSStyledText> styledText, int from, int to) {
104 			if (to < 0)
105 				to = Iterables.size(styledText);
106 			if (from < 0 || from > to)
107 				throw new IndexOutOfBoundsException();
108 			List<T> transformed = new ArrayList<>();
109 			if (from == to) return transformed;
110 			List<CSSStyledText> styledTextList = new ArrayList<>();
111 			styledText.forEach(styledTextList::add);
112 			// segments with same text-transform to be transformed next
113 			List<CSSStyledText> buffer = new ArrayList<CSSStyledText>();
114 			// already transformed segments (text-transform: none) to be used as context for next transformation
115 			List<CSSStyledText> context = new ArrayList<CSSStyledText>();
116 			String textTransform = "auto";
117 			for (int i = styledTextList.size(); i >= 0; i--) {
118 				CSSStyledText next;
119 				String nextTextTransform; {
120 					if (i == 0) {
121 						next = null;
122 						nextTextTransform = null;
123 					} else {
124 						next = styledTextList.get(i - 1);
125 						nextTextTransform = "auto";
126 						SimpleInlineStyle style = next.getStyle();
127 						if (style != null) {
128 							CSSProperty val = style.getProperty("text-transform");
129 							if (val != null) {
130 								if (val == TextTransform.NONE) {
131 									if (supports("none")) {
132 										nextTextTransform = "none";
133 										style.removeProperty("text-transform"); }
134 								} else if (val == TextTransform.list_values) {
135 									TermList values = style.getValue(TermList.class, "text-transform");
136 
137 									// According to the spec values should be "applied" from left to right, and
138 									// values of inner elements always come before values of outer elements (see
139 									// http://braillespecs.github.io/braille-css/#the-text-transform-property).
140 									// The way it works here is that the sub-translator that maps to the first
141 									// text-transform value (starting from left) is used to translate the segment,
142 									// and all values to the left and right are passed to the sub-translator.
143 									Iterator<Term<?>> it = values.iterator();
144 									while (it.hasNext()) {
145 										String tt = ((TermIdent)it.next()).getValue();
146 										if (supports(tt)) {
147 											nextTextTransform = tt;
148 											it.remove();
149 											break; }}
150 									if (values.isEmpty())
151 										style.removeProperty("text-transform"); }}}
152 					}
153 				}
154 				if (next == null || (nextTextTransform != textTransform && buffer.size() > 0)) {
155 					if (i < to) {
156 						int j = 0;
157 						for (T t : transform(org.daisy.pipeline.braille.common.util.Iterables.clone(
158 						                         Iterables.concat(buffer, context)),
159 						                     from > i ? from - i : 0,
160 						                     to < i + buffer.size() ? to - i : buffer.size(),
161 						                     textTransform))
162 							transformed.add(j++, t);
163 					}
164 					if (i <= from)
165 						return transformed;
166 					else {
167 						// set context for next transform
168 						try {
169 							int j = 0;
170 							for (String s : transformContext(org.daisy.pipeline.braille.common.util.Iterables.clone(
171 							                                     Iterables.concat(buffer, context)),
172 							                                 from > i ? from - i : 0,
173 							                                 buffer.size(),
174 							                                 textTransform))
175 								context.add(j++, new CSSStyledText(s, "text-transform: none; braille-charset: custom"));
176 							buffer.clear();
177 						} catch (NonStandardHyphenationException e) {
178 							// try without hyphenation
179 							for (int j = buffer.size() - 1; j >= 0; j--) {
180 								CSSStyledText st = (CSSStyledText)buffer.get(j).clone();
181 								SimpleInlineStyle style = st.getStyle();
182 								if (style != null && style.getProperty("hyphens") == Hyphens.AUTO)
183 									style.removeProperty("hyphens");
184 								buffer.set(j, st);
185 								context.add(0,
186 								            new CSSStyledText(
187 								                transformContext(org.daisy.pipeline.braille.common.util.Iterables.clone(
188 								                                     Iterables.concat(buffer, context)),
189 								                                 j,
190 								                                 j + 1,
191 								                                 textTransform).iterator().next(),
192 								                "text-transform: none; braille-charset: custom"));
193 								buffer.remove(j);
194 							}
195 						}
196 					}
197 				}
198 				if (next != null) {
199 					buffer.add(0, next);
200 					textTransform = nextTextTransform;
201 				}
202 			}
203 			return transformed;
204 		}
205 	}

207 	private FromStyledTextToBraille fromStyledTextToBraille = null;
208 
209 	/**
210 	 * @throws UnsupportedOperationException if the main translator or one of the sub-translators's
211 	 *                                       {@code #withHyphenator()} method throws {@link UnsupportedOperationException}
212 	 */
213 	@Override
214 	public FromStyledTextToBraille fromStyledTextToBraille() {
215 		if (!implementsFromStyledTextToBraille)
216 			throw new UnsupportedOperationException();
217 		if (fromStyledTextToBraille == null)
218 			fromStyledTextToBraille = new FromStyledTextToBraille() {
219 					TransformImpl<String> impl = new TransformImpl<String>() {
220 							Iterable<String> transform(Iterable<CSSStyledText> styledText, int from, int to, String textTransform) {
221 								return translators.get(textTransform).fromStyledTextToBraille().transform(styledText, from, to);
222 							}
223 							Iterable<String> transformContext(Iterable<CSSStyledText> styledText, int from, int to, String textTransform) {
224 								return transform(styledText, from, to, textTransform);
225 							}
226 							boolean supports(String textTransform) {
227 								return translators.containsKey(textTransform);
228 							}
229 						};
230 					public Iterable<String> transform(Iterable<CSSStyledText> styledText, int from, int to) {
231 						return impl.transform(styledText, from, to);
232 					}
233 					@Override
234 					public String toString() {
235 						return CompoundBrailleTranslator.this.toString();
236 					}
237 				};
238 		return fromStyledTextToBraille;
239 	}

241 	private LineBreakingFromStyledText lineBreakingFromStyledText = null;
242 
243 	/**
244 	 * @throws UnsupportedOperationException if the main translator or one of the sub-translators's
245 	 *                                       {@code #withHyphenator()} method throws {@link UnsupportedOperationException}
246 	 */
247 	@Override
248 	public LineBreakingFromStyledText lineBreakingFromStyledText() {
249 		if (!implementsLineBreakingFromStyledText)
250 			throw new UnsupportedOperationException();
251 		if (lineBreakingFromStyledText == null)
252 			lineBreakingFromStyledText = new LineBreakingFromStyledText() {
253 					TransformImpl<LineIterator> impl = new TransformImpl<LineIterator>() {
254 							Iterable<LineIterator> transform(Iterable<CSSStyledText> styledText, int from, int to, String textTransform) {
255 								return Collections.singleton(
256 									translators.get(textTransform).lineBreakingFromStyledText().transform(styledText, from, to));
257 							}
258 							Iterable<String> transformContext(Iterable<CSSStyledText> styledText, int from, int to, String textTransform) {
259 								return translators.get(textTransform).fromStyledTextToBraille().transform(styledText, from, to);
260 							}
261 							boolean supports(String textTransform) {
262 								return translators.containsKey(textTransform);
263 							}
264 						};
265 					public LineIterator transform(Iterable<CSSStyledText> styledText, int from, int to) {
266 						return concatLineIterators(impl.transform(styledText, from, to));
267 					}
268 					@Override
269 					public String toString() {
270 						return CompoundBrailleTranslator.this.toString();
271 					}
272 				};
273 		return lineBreakingFromStyledText;
274 	}

276 	/*
277 	 * Note that this function can not be used to concatenate any LineIterator. It is assumed that
278 	 * the LineIterator are translations of consecutive segments of the same string, where each
279 	 * translation has taken into account the whole context.
280 	 */
281 	static BrailleTranslator.LineIterator concatLineIterators(List<BrailleTranslator.LineIterator> iterators) {
282 		if (iterators.size() == 0)
283 			return new BrailleTranslator.LineIterator() {
284 				public String nextTranslatedRow(int limit, boolean force, boolean wholeWordsOnly) {
285 					return "";
286 				}
287 				public String getTranslatedRemainder() {
288 					return "";
289 				}
290 				public int countRemaining() {
291 					return 0;
292 				}
293 				public boolean hasNext() {
294 					return false;
295 				}
296 				public BrailleTranslator.LineIterator copy() {
297 					return this;
298 				}
299 				public boolean supportsMetric(String metric) {
300 					return false;
301 				}
302 				public double getMetric(String metric) {
303 					throw new UnsupportedMetricException("Metric not supported: " + metric);
304 				}
305 			};
306 		else if (iterators.size() == 1 && iterators.get(0) != null)
307 			return iterators.get(0);
308 		else
309 			return new ConcatLineIterators(iterators);
310 	}

312 	private static class ConcatLineIterators implements BrailleTranslator.LineIterator {
313 
314 		final List<BrailleTranslator.LineIterator> iterators;
315 		BrailleTranslator.LineIterator current;
316 		int currentIndex = 0;
317 
318 		ConcatLineIterators(List<BrailleTranslator.LineIterator> iterators) {
319 			this.iterators = iterators;
320 			currentIndex = -1;
321 			current = null;
322 			computeCurrent();
323 		}
324 
325 		void computeCurrent() {
326 			while (current == null || !current.hasNext())
327 				if (currentIndex + 1 < iterators.size())
328 					current = iterators.get(++currentIndex);
329 				else {
330 					current = null;
331 					break; }
332 		}
333 
334 		public String nextTranslatedRow(int limit, boolean force, boolean wholeWordsOnly) {
335 			String row = "";
336 			BrailleTranslator.LineIterator prev = null;
337 			while (limit > row.length()) {
338 				if (current == null || current == prev) break;
339 				String s = current.nextTranslatedRow(limit - row.length(), force, wholeWordsOnly);
340 				if (s.isEmpty() && current.hasNext())
341 					break;
342 				else
343 					force = false;
344 				row += s;
345 				prev = current;
346 				computeCurrent(); }
347 			return row;
348 		}
349 
350 		public String getTranslatedRemainder() {
351 			String remainder = "";
352 			if (current == null) return remainder;
353 			for (int i = currentIndex; i < iterators.size(); i++)
354 				if (iterators.get(i) != null)
355 					remainder += iterators.get(i).getTranslatedRemainder();
356 			return remainder;
357 		}
358 
359 		public int countRemaining() {
360 			int remaining = 0;
361 			if (current == null) return remaining;
362 			for (int i = currentIndex; i < iterators.size(); i++)
363 				if (iterators.get(i) != null)
364 					remaining += iterators.get(i).countRemaining();
365 			return remaining;
366 		}
367 
368 		public boolean hasNext() {
369 			computeCurrent();
370 			return current != null;
371 		}
372 
373 		public ConcatLineIterators copy() {
374 			List<BrailleTranslator.LineIterator> iteratorsCopy = new ArrayList<>(iterators.size() - currentIndex);
375 			for (int i = currentIndex; i < iterators.size(); i++)
376 				if (iterators.get(i) != null)
377 					iteratorsCopy.add((BrailleTranslator.LineIterator)iterators.get(i).copy());
378 			return new ConcatLineIterators(iteratorsCopy);
379 		}
380 
381 		public boolean supportsMetric(String metric) {
382 			return false;
383 		}
384 
385 		public double getMetric(String metric) {
386 			throw new UnsupportedMetricException("Metric not supported: " + metric);
387 		}
388 	}
