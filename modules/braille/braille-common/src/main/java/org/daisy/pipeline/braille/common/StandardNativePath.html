1   package org.daisy.pipeline.braille.common;
2   
3   import java.net.URL;
4   import java.net.URI;
5   import java.util.ArrayList;
6   import java.util.List;
7   import java.util.NoSuchElementException;
8   
9   import static com.google.common.collect.Iterables.find;
10  import com.google.common.base.Optional;
11  import com.google.common.base.Predicate;
12  
13  import static org.daisy.common.file.URLs.asURI;
14  import static org.daisy.pipeline.braille.common.util.OS;
15  
16  /**
17   * NativePath implementation from a ResourcePath which is unpacking and in
18   * which the binaries are named according to the following convention:
19   *
20   * - x86/foo.so
21   * - x86/foo.dll
22   * - x86/bar
23   * - x86/bar.exe
24   * - x86_64/foo.so
25   * - x86_64/foo.dll
26   * - x86_64/bar
27   * - x86_64/bar.exe
28   * - linux/x86/foo.so
29   * - linux/x86/bar
30   * - linux/x86_64/foo.so
31   * - linux/x86_64/bar
32   * - macosx/x86/foo.dylib
33   * - macosx/x86/bar
34   * - macosx/x86_64/foo.dylib
35   * - macosx/x86_64/bar
36   * - macosx/aarch64/foo.dylib
37   * - macosx/aarch64/bar
38   * - windows/x86/foo.dll
39   * - windows/x86/bar.exe
40   * - windows/x86_64/foo.dll
41   * - windows/x86_64/bar.exe
42   */
43  public abstract class StandardNativePath implements NativePath {
44  	
45  	protected abstract ResourcePath delegate();
46  	
47  	public URI getIdentifier() {
48  		return delegate().getIdentifier();
49  	}
50  	
51  	public URL resolve(URI resource) {
52  		return delegate().resolve(resource);
53  	}
54  	
55  	public URI canonicalize(URI resource) {
56  		return delegate().canonicalize(resource);
57  	}
58  	
59  	/**
60  	 * Get an executable or a shared library.
61  	 * @param name The name (without the extension)
62  	 * @return A collection of executables or libraries.
63  	 */
64  	public Iterable<URI> get(String name) {
65  		URI path;
66  		if ((path = getExecutable(name)) != null) {}
67  		else if ((path = getSharedLibrary(name)) != null) {}
68  		else { return Optional.<URI>absent().asSet(); }
69  		return Optional.of(canonicalize(path)).asSet();
70  	}
71  	
72  	private URI getExecutable(String name) {
73  		String fileName = OS.isWindows() ? name + ".exe" : name;
74  		String os = OS.getFamily().toString().toLowerCase();
75  		String arch = OS.getArch();
76  		if (!OS.is64Bit()) arch = "x86";
77  		else if ("amd64".equals(arch)) arch = "x86_64";
78  		else if ("arm64".equals(arch)) arch = "aarch64";
79  		List<URI> possiblePaths = new ArrayList<URI>();
80  		possiblePaths.add(asURI(arch + "/" + fileName));
81  		possiblePaths.add(asURI(os + "/" + arch + "/" + fileName));
82  		if (OS.is64Bit() && OS.isWindows()) {
83  			possiblePaths.add(asURI("x86/" + fileName));
84  			possiblePaths.add(asURI(os + "/x86/" + fileName)); }
85  		try {
86  			return find(
87  				possiblePaths,
88  				new Predicate<URI>() { public boolean apply(URI p) { return (delegate().resolve(p) != null); }}); }
89  		catch (NoSuchElementException e) { return null; }
90  	}
91  	
92  	private URI getSharedLibrary(String name) {
93  		String fileName = name + (OS.isWindows() ? ".dll" : OS.isMacOSX() ? ".dylib" : ".so");
94  		String os = OS.getFamily().toString().toLowerCase();
95  		String arch = OS.getArch();
96  		if (!OS.is64Bit()) arch = "x86";
97  		else if ("amd64".equals(arch)) arch = "x86_64";
98  		else if ("arm64".equals(arch)) arch = "aarch64";
99  		List<URI> possiblePaths = new ArrayList<URI>();
100 		possiblePaths.add(asURI(arch + "/" + fileName));
101 		possiblePaths.add(asURI(os + "/" + arch + "/" + fileName));
102 		try {
103 			return find(
104 				possiblePaths,
105 				new Predicate<URI>() { public boolean apply(URI p) { return (delegate().resolve(p) != null); }}); }
106 		catch (NoSuchElementException e) { return null; }
107 	}
108 }
