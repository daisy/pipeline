129 	private static Iterable<BrailleTranslator> getWithHyphenator(Provider<Query,BrailleTranslator> translatorProvider,
130 	                                                             Provider<Query,Hyphenator> hyphenatorProvider,
131 	                                                             Query query,
132 	                                                             Logger context) {
133 		if (query.containsKey("id"))
134 			return translatorProvider.get(query);
135 		MutableQuery q = mutableQuery(query);
136 
137 		// select hyphenator
138 		MutableQuery hyphenatorQuery = mutableQuery();
139 		String hyphenator = q.containsKey("hyphenator")
140 			? q.removeOnly("hyphenator").getValue().get()
141 			: "auto";
142 		if (!"auto".equals(hyphenator))
143 			hyphenatorQuery.add("hyphenator", hyphenator);
144 		String documentLocale; {
145 			try {
146 				documentLocale = q.containsKey("document-locale")
147 					? parseLocale(q.getOnly("document-locale").getValue().get()).toLanguageTag()
148 					: null; }
149 			catch (IllegalArgumentException e) {
150 				logger.error("Invalid locale", e);
151 				return Collections.emptyList(); }
152 		}
153 		if (documentLocale != null)
154 			hyphenatorQuery.add("document-locale", documentLocale);
155 		Iterable<Hyphenator> hyphenators; {
156 			Iterable<Hyphenator> h = hyphenatorProvider.get(hyphenatorQuery.asImmutable());
157 			if (documentLocale != null && !"auto".equals(hyphenator)) {
158 				// also search without locale because "hyphenator" feature might be an ID
159 				hyphenatorQuery.removeAll("document-locale");
160 				h = Iterables.concat(h, hyphenatorProvider.get(hyphenatorQuery.asImmutable())); }
161 			hyphenators = h;
162 		}
163 
164 		// select translator and bind hyphenator
165 		Iterable<BrailleTranslator> translators = translatorProvider.get(q);
166 		return Iterables.concat(
167 			Iterables.concat(
168 				Iterables.transform(
169 					translators,
170 					t -> Iterables.filter(
171 						Iterables.transform(
172 							hyphenators,
173 							h -> {
174 								try {
175 									return logCreate(t.withHyphenator(h), context);
176 								} catch (UnsupportedOperationException e) {
177 									logger.debug("Could not set hyphenator: " + e.getMessage());
178 									return null;
179 								}}),
180 						Predicates.notNull()))),
181 			translators);
182 	}

184 	/**
185 	 * Select {@link BrailleTranslator}s based on a query and a CSS style sheet possibly containing
186 	 * {@code @text-transform} rules.
187 	 *
188 	 * Contrary to {@link #get(Query)}, this method is not memoized, and the returned objects may
189 	 * not be selectable based on their identifier.
190 	 *
191 	 * @param baseURI Base URI for resolving relative paths in CSS against.
192 	 * @param forceMainTranslator if {@code true}, the translator defined in {@code query} is used
193 	 *                            even if a default translator has been defined in CSS.
194 	 */
195 	public Iterable<BrailleTranslator> get(Query query, String style, URI baseURI, boolean forceMainTranslator) {
196 		if (style != null) {
197 			Map<String,Query> subQueries = TextTransformParser.getBrailleTranslatorQueries(style, baseURI, query);
198 			if (subQueries != null && !subQueries.isEmpty()) {
199 				Query mainQuery = subQueries.remove("auto");
200 				if (mainQuery == null || forceMainTranslator)
201 					mainQuery = query;
202 				if (subQueries.isEmpty())
203 					return get(mainQuery);
204 				else {
205 					Map<String,Supplier<BrailleTranslator>> subTranslators
206 						= Maps.transformValues(
207 							subQueries,
208 							// using getWithHyphenator() because @text-transform rules may contain "hyphenator"
209 							// descriptor (though it should be deprecated)
210 							q -> () -> BrailleTranslatorRegistry.this.getWithHyphenator(q).iterator().next());
211 					return Iterables.transform(
212 						get(mainQuery),
213 						t -> logCreate(new CompoundBrailleTranslator(t, subTranslators), context));
214 				}
215 			}
216 		}
217 		return get(query);
218 	}

220 	/**
221 	 * Select {@link BrailleTranslator}s with {@link Hyphenator} based on a single query and a CSS
222 	 * style sheet possibly containing {@code @text-transform} and {@code @hyphenation-resource}
223 	 * rules.
224 	 *
225 	 * Contrary to {@link #get(Query)} and similar to {@link #get(Query, String, URI, boolean)},
226 	 * this method is not memoized, and the returned objects may not be selectable based on their
227 	 * identifier.
228 	 */
229 	public Iterable<BrailleTranslator> getWithHyphenator(Query query, String style, URI baseURI, boolean forceMainTranslator) {
230 		return getWithHyphenator(
231 			q -> get(q, style, baseURI, forceMainTranslator),
232 			q -> hyphenatorRegistry.get(q, style, baseURI),
233 			query,
234 			context);
235 	}
