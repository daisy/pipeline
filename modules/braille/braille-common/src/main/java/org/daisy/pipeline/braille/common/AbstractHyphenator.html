39  		private final static Pattern ON_SPACE_SPLITTER = Pattern.compile("\\s+");
40  		private final static Pattern COMPOUND_WORD_HYPHEN = Pattern.compile("[\\p{L}\\p{N}]-(?=[\\p{L}\\p{N}])");
41  
42  		public static abstract class DefaultFullHyphenator implements FullHyphenator {
43  
44  			/**
45  			 * Whether the length of the array returned by {@link
46  			 * #getHyphenationOpportunities(String)} is based on the number of code points in the
47  			 * input or the number of characters.
48  			 */
49  			protected abstract boolean isCodePointAware();
50  
51  			/**
52  			 * Get hyphenation opportunities as a byte array (1 = SHY, 2 = ZWSP)
53  			 *
54  			 * @param textWithoutHyphens text that does not contain SHY and ZWSP characters (and from
55  			 *                           which no SHY and ZWSP characters were extracted either)
56  			 */
57  			protected abstract byte[] getHyphenationOpportunities(String textWithoutHyphens) throws NonStandardHyphenationException;
58  
59  			private final static char SHY = '\u00AD';
60  			private final static char ZWSP = '\u200B';
61  			private final static char US = '\u001F';
62  			private final static Splitter SEGMENT_SPLITTER = Splitter.on(US);
63  
64  			public Iterable<CSSStyledText> transform(Iterable<CSSStyledText> text) throws NonStandardHyphenationException {
65  				List<CSSStyledText> result = new ArrayList<>();
66  				List<CSSProperty> hyphenate = new ArrayList<>();
67  				boolean someHyphenate = false;
68  				for (CSSStyledText t : text) {
69  					t = t.clone();
70  					SimpleInlineStyle style = t.getStyle();
71  					CSSProperty h = style != null ? style.getProperty("hyphens") : null;
72  					if (h == null) h = Hyphens.MANUAL;
73  					hyphenate.add(h);
74  					if (h == Hyphens.AUTO)
75  						someHyphenate = true;
76  					if (style != null)
77  						style.removeProperty("hyphens");
78  					result.add(t);
79  				}
80  				if (result.size() == 0)
81  					return result;
82  				Tuple2<String,byte[]> t = extractHyphens(
83  					join(Iterables.transform(text, CSSStyledText::getText), US), isCodePointAware(), SHY, ZWSP);
84  				List<String> textWithoutHyphens = SEGMENT_SPLITTER.splitToList(t._1);
85  				t = extractHyphens(t._2, t._1, isCodePointAware(), null, null, null, US);
86  				String joinedTextWithoutHyphens = t._1;
87  				byte[] manualHyphensAndSegmentBoundaries = t._2;
88  				// insert zero-width space after hard hyphens ("-" followed and preceded by a letter
89  				// or number)
90  				byte[] hardHyphens; {
91  					int len = joinedTextWithoutHyphens.length();
92  					if (len > 1) {
93  						hardHyphens = new byte[len - 1]; {
94  							Matcher m = COMPOUND_WORD_HYPHEN.matcher(joinedTextWithoutHyphens);
95  							while (m.find())
96  								hardHyphens[m.start() + 1] = 4; }
97  						if (isCodePointAware())
98  							hardHyphens = extractHyphens(
99  								insertHyphens(joinedTextWithoutHyphens, hardHyphens, false, null, null, ZWSP),
100 								true, null, null, ZWSP)._2;
101 					} else
102 						hardHyphens = null;
103 				}
104 				if (hardHyphens != null) {
105 					if (manualHyphensAndSegmentBoundaries == null)
106 						manualHyphensAndSegmentBoundaries = hardHyphens;
107 					else
108 						for (int k = 0; k < hardHyphens.length - 1; k++)
109 							manualHyphensAndSegmentBoundaries[k] |= hardHyphens[k];
110 				}
111 				byte[] hyphensAndSegmentBoundaries = someHyphenate
112 					? transform(manualHyphensAndSegmentBoundaries, joinedTextWithoutHyphens)
113 					: manualHyphensAndSegmentBoundaries;
114 				List<String> textWithHyphensAuto =
115 					SEGMENT_SPLITTER.splitToList(
116 						insertHyphens(
117 							joinedTextWithoutHyphens, hyphensAndSegmentBoundaries, isCodePointAware(), SHY, ZWSP, ZWSP, US)
118 						.replace("" + ZWSP + ZWSP, "" + ZWSP));
119 				List<String> textWithHyphensManual =
120 					SEGMENT_SPLITTER.splitToList(
121 						insertHyphens(
122 							joinedTextWithoutHyphens, manualHyphensAndSegmentBoundaries, isCodePointAware(), SHY, ZWSP, ZWSP, US)
123 						.replace("" + ZWSP + ZWSP, "" + ZWSP));
124 				List<String> textWithHyphensNone =
125 					SEGMENT_SPLITTER.splitToList(
126 						insertHyphens(
127 							joinedTextWithoutHyphens, manualHyphensAndSegmentBoundaries, isCodePointAware(), null, null, ZWSP, US));
128 				int j = 0;
129 				for (int i = 0; i < result.size(); i++) {
130 					if (textWithoutHyphens.get(i).isEmpty())
131 						result.set(i, new CSSStyledText("", result.get(i).getStyle()));
132 					else {
133 						CSSProperty h = hyphenate.get(i);
134 						if (h == Hyphens.AUTO)
135 							result.set(i, new CSSStyledText(textWithHyphensAuto.get(j++), result.get(i).getStyle()));
136 						else if (h == Hyphens.MANUAL)
137 							result.set(i, new CSSStyledText(textWithHyphensManual.get(j++), result.get(i).getStyle()));
138 						else // h == Hyphens.NONE
139 							// FIXME: better would be to only remove SHY and ZWSP within words, but the
140 							// issue with this is that what constitutes a "word" is language dependent
141 							// and finding words requires NLP
142 							result.set(i, new CSSStyledText(textWithHyphensNone.get(j++), result.get(i).getStyle()));
143 					}
144 				}
145 				return result;
146 			}
147 
148 			/**
149 			 * @param manualHyphens      SHY, ZWSP and US characters that were extracted from the original
150 			 *                           text, which resulted in <code>textWithoutHyphens</code>
151 			 * @param textWithoutHyphens text without SHY, ZWSP and US characters
152 			 */
153 			protected final byte[] transform(byte[] manualHyphens, String textWithoutHyphens) throws NonStandardHyphenationException {
154 				if (textWithoutHyphens.length() == 0)
155 					return manualHyphens;
156 				boolean hasManualHyphens = false; {
157 					if (manualHyphens != null)
158 						for (byte b : manualHyphens)
159 							if (b == (byte)1 || b == (byte)2) {
160 								hasManualHyphens = true;
161 								break; }}
162 				if (hasManualHyphens) {
163 					// input contains SHY or ZWSP; hyphenate only the words without SHY or ZWSP
164 					// FIXME: for simplicity a "word" means a sequence of non white space here, but
165 					// a better definition is needed
166 					byte[] hyphens = Arrays.copyOf(manualHyphens, manualHyphens.length);
167 					boolean word = true;
168 					int pos = 0;
169 					for (String segment : splitInclDelimiter(textWithoutHyphens, ON_SPACE_SPLITTER)) {
170 						int len = isCodePointAware()
171 							? segment.codePointCount(0, segment.length())
172 							: segment.length();
173 						if (word && len > 0) {
174 							boolean wordHasManualHyphens = false; {
175 								for (int k = 0; k < len - 1; k++)
176 									if (hyphens[pos + k] != 0) {
177 										wordHasManualHyphens = true;
178 										break; }}
179 							if (!wordHasManualHyphens) {
180 								byte[] wordHyphens = getHyphenationOpportunities(segment);
181 								if (wordHyphens != null)
182 									for (int k = 0; k < len - 1; k++)
183 										hyphens[pos + k] |= wordHyphens[k];
184 							}
185 						}
186 						pos += len;
187 						word = !word;
188 					}
189 					return hyphens;
190 				} else {
191 					byte[] hyphens = getHyphenationOpportunities(textWithoutHyphens);
192 					if (manualHyphens != null) {
193 						if (hyphens == null)
194 							hyphens = manualHyphens;
195 						else
196 							for (int k = 0; k < hyphens.length; k++)
197 								hyphens[k] |= manualHyphens[k]; }
198 					return hyphens;
199 				}
200 			}
201 		}

204 		 * {@link FullHyphenator} that does not provide any hyphenation opportunities (only soft
205 		 * wrap opportunities outside words).

207 		public static class NoHyphenator extends DefaultFullHyphenator {
208 			
209 			public Iterable<CSSStyledText> transform(java.lang.Iterable<CSSStyledText> text) {
210 				if (COMPOUND_WORD_HYPHEN.matcher(
211 					    join(com.google.common.collect.Iterables.transform(text, CSSStyledText::getText))).find())
212 					return super.transform(text);
213 				else
214 					return text;
215 			}

218 			protected boolean isLanguageAdaptive() { return false; }

221 				return null;
222 			}

225 		// TODO: caching?
226 		public static abstract class DefaultLineBreaker implements LineBreaker {
227 			
228 			/**
229 			 * This method should be overriden. The default behavior is that a word is only broken
230 			 * after hard hyphens, or if the <code>force</code> argument is <code>true</code>.
231 			 */
232 			protected Break breakWord(String word, int limit, boolean force) {
233 				if (word.length() <= limit)
234 					return new Break(word, limit, false);
235 				// break after hard hyphens
236 				Matcher m = COMPOUND_WORD_HYPHEN.matcher(word);
237 				if (m.find()) {
238 					int len = m.start() + 2;
239 					if (len <= limit)
240 						return new Break(word, len, false);
241 				}
242 				if (force)
243 					return new Break(word, limit, false);
244 				return new Break(word, 0, false);
245 			}
246 			
247 			protected static class Break {
248 				private final String text;
249 				private final int position;
250 				private final boolean hyphen;
251 				public Break(String text, int position, boolean hyphen) {
252 					this.text = text;
253 					this.position = position;
254 					this.hyphen = hyphen;
255 				}
256 				private String firstLine() {
257 					return text.substring(0, position);
258 				}
259 				private String secondLine() {
260 					return text.substring(position);
261 				}
262 				@Override
263 				public String toString() {
264 					return firstLine() + "=" + secondLine();
265 				}
266 			}
267 			
268 			public LineIterator transform(final String text) {
269 				
270 				return new LineIterator() {
271 					
272 					String remainder = text;
273 					String remainderAtMark = text;
274 					boolean lineHasHyphen = false;
275 					boolean lineHasHyphenAtMark = false;
276 					boolean started = false;
277 					boolean startedAtMark = false;
278 					
279 					public String nextLine(int limit, boolean force) {
280 						return nextLine(limit, force, true);
281 					}
282 					
283 					public String nextLine(int limit, boolean force, boolean allowHyphens) {
284 						started = true;
285 						String line = "";
286 						lineHasHyphen = false;
287 						if (remainder != null) {
288 							if (remainder.length() <= limit) {
289 								line += remainder;
290 								remainder = null; }
291 							else {
292 								String r = "";
293 								int available = limit;
294 								boolean word = true;
295 								for (String segment : splitInclDelimiter(remainder, ON_SPACE_SPLITTER)) {
296 									if (available == 0)
297 										r += segment;
298 									else if (segment.length() <= available) {
299 										line += segment;
300 										available -= segment.length();
301 										word = !word; }
302 									else if (word && allowHyphens) {
303 										Break brokenWord = breakWord(segment, available, force && (available == limit));
304 										line += brokenWord.firstLine();
305 										lineHasHyphen = brokenWord.hyphen;
306 										r += brokenWord.secondLine();
307 										available = 0; }
308 									else {
309 										r += segment;
310 										available = 0; }}
311 								remainder = r.isEmpty() ? null : r; }}
312 						return line;
313 					}
314 					
315 					public boolean hasNext() {
316 						return remainder != null;
317 					}
318 					
319 					public boolean lineHasHyphen() {
320 						if (!started)
321 							throw new RuntimeException("nextLine must be called first.");
322 						return lineHasHyphen;
323 					}
324 					
325 					public String remainder() {
326 						return remainder;
327 					}
328 					
329 					public void mark() {
330 						remainderAtMark = remainder;
331 						lineHasHyphenAtMark = lineHasHyphen;
332 						startedAtMark = started;
333 					}
334 					
335 					public void reset() {
336 						remainder = remainderAtMark;
337 						lineHasHyphen = lineHasHyphenAtMark;
338 						started = startedAtMark;
339 					}
340 				};
341 			}
342 		}
