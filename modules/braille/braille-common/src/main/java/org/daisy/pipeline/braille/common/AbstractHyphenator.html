5   import static org.daisy.pipeline.braille.common.util.Strings.splitInclDelimiter;

7   public abstract class AbstractHyphenator extends AbstractTransform implements Hyphenator {
8   	
9   	public FullHyphenator asFullHyphenator() throws UnsupportedOperationException {
10  		throw new UnsupportedOperationException();
11  	}
12  	
13  	public LineBreaker asLineBreaker() throws UnsupportedOperationException {
14  		throw new UnsupportedOperationException();
15  	}
16  	
17  	/* ================== */
18  	/*       UTILS        */
19  	/* ================== */
20  	
21  	public static abstract class util {
22  		
23  		// TODO: caching?
24  		public static abstract class DefaultLineBreaker implements LineBreaker {
25  			
26  			protected abstract Break breakWord(String word, int limit, boolean force);
27  			
28  			protected static class Break {
29  				private final String text;
30  				private final int position;
31  				private final boolean hyphen;
32  				public Break(String text, int position, boolean hyphen) {
33  					this.text = text;
34  					this.position = position;
35  					this.hyphen = hyphen;
36  				}
37  				private String firstLine() {
38  					return text.substring(0, position);
39  				}
40  				private String secondLine() {
41  					return text.substring(position);
42  				}
43  				@Override
44  				public String toString() {
45  					return firstLine() + "=" + secondLine();
46  				}
47  			}
48  			
49  			private final static Pattern ON_SPACE_SPLITTER = Pattern.compile("\\s+");
50  			
51  			public LineIterator transform(final String text) {
52  				
53  				return new LineIterator() {
54  					
55  					String remainder = text;
56  					String remainderAtMark = text;
57  					boolean lineHasHyphen = false;
58  					boolean lineHasHyphenAtMark = false;
59  					boolean started = false;
60  					boolean startedAtMark = false;
61  					
62  					public String nextLine(int limit, boolean force) {
63  						return nextLine(limit, force, true);
64  					}
65  					
66  					public String nextLine(int limit, boolean force, boolean allowHyphens) {
67  						started = true;
68  						String line = "";
69  						lineHasHyphen = false;
70  						if (remainder != null) {
71  							if (remainder.length() <= limit) {
72  								line += remainder;
73  								remainder = null; }
74  							else {
75  								String r = "";
76  								int available = limit;
77  								boolean word = true;
78  								for (String segment : splitInclDelimiter(remainder, ON_SPACE_SPLITTER)) {
79  									if (available == 0)
80  										r += segment;
81  									else if (segment.length() <= available) {
82  										line += segment;
83  										available -= segment.length();
84  										word = !word; }
85  									else if (word && allowHyphens) {
86  										Break brokenWord = breakWord(segment, available, force && (available == limit));
87  										line += brokenWord.firstLine();
88  										lineHasHyphen = brokenWord.hyphen;
89  										r += brokenWord.secondLine();
90  										available = 0; }
91  									else {
92  										r += segment;
93  										available = 0; }}
94  								remainder = r.isEmpty() ? null : r; }}
95  						return line;
96  					}
97  					
98  					public boolean hasNext() {
99  						return remainder != null;
100 					}
101 					
102 					public boolean lineHasHyphen() {
103 						if (!started)
104 							throw new RuntimeException("nextLine must be called first.");
105 						return lineHasHyphen;
106 					}
107 					
108 					public String remainder() {
109 						return remainder;
110 					}
111 					
112 					public void mark() {
113 						remainderAtMark = remainder;
114 						lineHasHyphenAtMark = lineHasHyphen;
115 						startedAtMark = started;
116 					}
117 					
118 					public void reset() {
119 						remainder = remainderAtMark;
120 						lineHasHyphen = lineHasHyphenAtMark;
121 						started = startedAtMark;
122 					}
123 				};
124 			}
125 		}
126 	}
127 }
