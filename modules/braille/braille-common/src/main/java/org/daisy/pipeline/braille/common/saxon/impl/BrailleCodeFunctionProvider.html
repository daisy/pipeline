1   package org.daisy.pipeline.braille.common.saxon.impl;
2   
3   import java.util.Locale;
4   import java.util.Map;
5   import java.util.Optional;
6   
7   import org.daisy.common.xpath.saxon.ExtensionFunctionProvider;
8   import org.daisy.common.xpath.saxon.ReflexiveExtensionFunctionProvider;
9   import org.daisy.pipeline.braille.common.BrailleCode;
10  import org.daisy.pipeline.braille.common.BrailleTranslator;
11  import org.daisy.pipeline.braille.common.BrailleTranslatorRegistry;
12  import org.daisy.pipeline.braille.common.Query;
13  import static org.daisy.pipeline.braille.common.Query.util.mutableQuery;
14  
15  import org.osgi.service.component.annotations.Component;
16  import org.osgi.service.component.annotations.Reference;
17  import org.osgi.service.component.annotations.ReferenceCardinality;
18  import org.osgi.service.component.annotations.ReferencePolicy;
19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  @Component(
24  	name = "BrailleCode",
25  	service = { ExtensionFunctionProvider.class }
26  )
27  public class BrailleCodeFunctionProvider extends ReflexiveExtensionFunctionProvider {
28  
29  	public BrailleCodeFunctionProvider() {
30  		super();
31  		addExtensionFunctionDefinitionsFromClass(BrailleCodeFunctions.class, new BrailleCodeFunctions());
32  	}
33  
34  	public class BrailleCodeFunctions {
35  
36  		public String getSystem(BrailleCode code) {
37  			return code.getSystem();
38  		}
39  
40  		public String getGrade(BrailleCode code) {
41  			return code.getGrade().toString();
42  		}
43  
44  		public Optional<String> getSpecialization(BrailleCode code) {
45  			return code.getSpecialization().map(BrailleCode.Specialization::toString);
46  		}
47  
48  		public BrailleCode fromLanguageTag(Locale language) throws IllegalArgumentException {
49  			return BrailleCode.fromLanguageTag(language);
50  		}
51  
52  		/**
53  		 * Get information about the braille code implementation (e.g. Liblouis table).
54  		 *
55  		 * @throws RuntimeException if no implementation can be found of the braille code.
56  		 */
57  		public Map<String,String> getImplementationInfo(Object code) {
58  			if (code instanceof String)
59  				return getImplementationInfo(BrailleCode.parse((String)code));
60  			else if (code instanceof BrailleCode)
61  				return getImplementationInfo((BrailleCode)code);
62  			else
63  				throw new IllegalArgumentException("Expected BrailleCode or String");
64  		}
65  
66  		private Map<String,String> getImplementationInfo(BrailleCode code) {
67  			return getImplementation(code).getInfo();
68  		}
69  
70  		private BrailleTranslator getImplementation(BrailleCode code) {
71  			for (BrailleTranslator t : translatorRegistry.get(mutableQuery().add("braille-code", code.toString())))
72  				return t;
73  			throw new RuntimeException("No implementation found of braille code '" + code + "'");
74  		}
75  	}
76  
77  	private BrailleTranslatorRegistry translatorRegistry;
78  
79  	@Reference(
80  		name = "BrailleTranslatorRegistry",
81  		unbind = "-",
82  		service = BrailleTranslatorRegistry.class,
83  		cardinality = ReferenceCardinality.MANDATORY,
84  		policy = ReferencePolicy.STATIC
85  	)
86  	protected void bindBrailleTranslatorRegistry(BrailleTranslatorRegistry registry) {
87  		translatorRegistry = registry.withContext(logger);
88  		logger.debug("Binding BrailleTranslator registry: {}", registry);
89  	}
90  
91  	private static final Logger logger = LoggerFactory.getLogger(BrailleCodeFunctionProvider.class);
92  }
