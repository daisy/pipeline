125 		public StandardLineBreaker(FromStyledTextToBraille fromStyledTextToBraille, char blankChar, char defaultHyphenChar, BrailleConverter brailleCharset) {

130 		protected BrailleStream translateAndHyphenate(Iterable<CSSStyledText> styledText, int from, int to) {
131 			List<String> braille = new ArrayList<>();
132 			Iterator<CSSStyledText> style = styledText.iterator();
133 			for (String s : fromStyledTextToBraille.transform(styledText)) {
134 				SimpleInlineStyle st = style.next().getStyle();
135 				if (st != null) {
136 					if (st.getProperty("hyphens") == Hyphens.NONE) {
137 						s = s.replaceAll("[\u00AD\u200B]","");
138 						st.removeProperty("hyphens"); }
139 					CSSProperty ws = st.getProperty("white-space");
140 					if (ws != null) {
141 						if (ws == WhiteSpace.PRE_WRAP)
142 							s = s.replaceAll("[\\x20\t\\u2800]+", "$0\u200B")
143 								.replaceAll("[\\x20\t\\u2800]", "\u00A0");
144 						if (ws == WhiteSpace.PRE_WRAP || ws == WhiteSpace.PRE_LINE)
145 							s = s.replaceAll("[\\n\\r]", "\u2028");
146 						st.removeProperty("white-space"); }}
147 				braille.add(s);
148 			}
149 			StringBuilder brailleString = new StringBuilder();
150 			int fromChar = 0;
151 			int toChar = to >= 0 ? 0 : -1;
152 			for (String s : braille) {
153 				brailleString.append(s);
154 				if (--from == 0)
155 					fromChar = brailleString.length();
156 				if (--to == 0)
157 					toChar = brailleString.length();
158 			}
159 			return new FullyHyphenatedAndTranslatedString(brailleString.toString(), fromChar, toChar);
160 		}
161 	}

217 			 * href="http://braillespecs.github.io/braille-css/master/index.html#dfn-braille-character">braille

260 			public BrailleTranslator.LineIterator transform(final Iterable<CSSStyledText> text, int from, int to)
261 					throws TransformationException {
262 				List<Character> hyphenChars = new ArrayList<>();
263 				int wordSpacing; {
264 					wordSpacing = -1;
265 					for (CSSStyledText st : text) {
266 						SimpleInlineStyle style = st.getStyle();
267 						char hyphenChar = defaultHyphenChar;
268 						int spacing = 1;
269 						if (style != null) {
270 							CSSProperty val = style.getProperty("hyphenate-character");
271 							if (val != null) {
272 								if (val == HyphenateCharacter.braille_string) {
273 									String s = style.getValue(TermString.class, "hyphenate-character").getValue();
274 									if (s.length() == 1) {
275 										hyphenChar = s.charAt(0);
276 										if (brailleCharset != null)
277 											hyphenChar = brailleCharset.toText("" + hyphenChar).toCharArray()[0];
278 									} else
279 										logger.warn("The 'hyphenate-character' property must be a single character, "
280 										            + "but got {}", s); }
281 								style.removeProperty("hyphenate-character"); }
282 							val = style.getProperty("word-spacing");
283 							if (val != null) {
284 								if (val == WordSpacing.length) {
285 									spacing = style.getValue(TermInteger.class, "word-spacing").getIntValue();
286 									if (spacing < 0) {
287 										if (logger != null)
288 											logger.warn("word-spacing: {} not supported, must be non-negative", val);
289 										spacing = 1; }}
290 								
291 								// FIXME: assuming style is mutable and text.iterator() does not create copies
292 								style.removeProperty("word-spacing"); }}
293 						hyphenChars.add(hyphenChar);
294 						if (wordSpacing < 0)
295 							wordSpacing = spacing;
296 						else if (wordSpacing != spacing)
297 							throw new TransformationException("word-spacing must be constant, but both "
298 							                                  + wordSpacing + " and " + spacing + " specified"); }
299 					if (wordSpacing < 0) wordSpacing = 1;
300 				}
301 				List<BrailleTranslator.LineIterator> lineIterators = new ArrayList<>();
302 				Character hyphenChar = null;
303 				if (to < 0) to = hyphenChars.size();
304 				int i = from;
305 				while (i < to) {
306 					Character nextHyphenChar = hyphenChars.get(i);
307 					if (hyphenChar != nextHyphenChar) {
308 						if (i > from) {
309 							lineIterators.add(
310 								new LineIterator(translateAndHyphenate(text, from, i), blankChar, hyphenChar, wordSpacing));
311 							from = i; }
312 						hyphenChar = nextHyphenChar; }
313 					i++; }
314 				if (i > from)
315 					lineIterators.add(
316 						new LineIterator(translateAndHyphenate(text, from, i), blankChar, hyphenChar, wordSpacing));
317 				return CompoundBrailleTranslator.concatLineIterators(lineIterators);
318 			}

320 			protected static class FullyHyphenatedAndTranslatedString implements BrailleStream {
321 				private String next;
322 				private String lastWordPart; // if the last word continues in the next segment
323 				private String lastWordOtherPart;
324 				private final boolean precedingSpace;
325 				private final boolean alwaysEmpty;
326 				private final char hyphenChar;
327 				// SPACE, TAB, LF, CR, BLANK or LS
328 				private final static Pattern WORD_BOUNDARY = Pattern.compile("[\\x20\t\\n\\r\\u2800\u2028]");
329 				public FullyHyphenatedAndTranslatedString(String string) {
330 					this(string, 0, -1);
331 				}
332 				public FullyHyphenatedAndTranslatedString(String string, int from, int to) {
333 					this(string, from, to, SHY); // LineIterator converts SHY at the end of a line to a hyphen character
334 				}
335 				public FullyHyphenatedAndTranslatedString(String string, int from, int to, char hyphenChar) {
336 					// if there are no preceding segments, assume that we are at the beginning of a line,
337 					// so leading space can be stripped
338 					// FIXME: we can not make this assumption! see for example FormatterCoreContext,
339 					// which translates a space in order to obtain a value to use as margin
340 					// character, and it expects it to be a non-empty string
341 					//if (from == 0) precedingSpace = true; else
342 					precedingSpace = DefaultLineBreaker.hasPrecedingSpace(string, from);
343 					int len = string.length();
344 					if (from < 0 || from > len)
345 						throw new IllegalArgumentException();
346 					next = string.substring(from);
347 					lastWordPart = lastWordOtherPart = null;
348 					if (to >= 0 && to != len) {
349 						if (to > len || to < from)
350 							throw new IllegalArgumentException();
351 						to -= from;
352 						len -= from;
353 						int lastWordStart = lastIndexOf((" " + next.substring(0, to)), WORD_BOUNDARY);
354 						if (lastWordStart == to) {
355 							next = next.substring(0, to);
356 						} else {
357 							int lastWordEnd = to + indexOf((next.substring(to, len) + " "), WORD_BOUNDARY);
358 							if (lastWordEnd == to) {
359 								next = next.substring(0, to);
360 							} else {
361 								lastWordPart = next.substring(lastWordStart, to);
362 								lastWordOtherPart = next.substring(to, lastWordEnd);
363 								next = next.substring(0, lastWordStart);
364 								// ZWSP is not considered a WORD_BOUNDARY, but a word should not start with a ZWSP
365 								lastWordPart = lastWordPart.replaceAll("^\u200b*", "");
366 								if (lastWordPart.isEmpty())
367 									lastWordPart = lastWordOtherPart = null;
368 							}
369 						}
370 					} else {
371 						// remove SHY at end of stream because it would be converted to hyphen character
372 						next = next.replaceAll("\u00ad$", "");
373 					}
374 					if (next.replaceAll("[\u00ad\u200b]","").isEmpty())
375 						next = null;
376 					alwaysEmpty = (next == null);
377 					this.hyphenChar = hyphenChar;
378 				}
379 				public boolean hasNext() {
380 					return next != null || lastWordPart != null;
381 				}
382 				public String next(int limit, boolean force, boolean allowHyphens) {
383 					if (next != null) {
384 						String n = next;
385 						next = null;
386 						return n; }
387 					else if (lastWordPart != null) {
388 						if (force) {
389 							String n = lastWordPart;
390 							lastWordPart = null;
391 							return n; }
392 						Tuple2<String,byte[]> t = extractHyphens(lastWordPart + RS + lastWordOtherPart, false, SHY, ZWSP, RS);
393 						String lastWord = t._1;
394 						// if last word fits
395 						if (lastWord.length() <= limit) {
396 							String n = lastWordPart;
397 							lastWordPart = null;
398 							return n; }
399 						// else if the word can be hyphenated
400 						// assuming that if allowHyphens is true for this segment it will be true for the next segment
401 						else if (allowHyphens) {
402 							byte[] hyphens = t._2;
403 							boolean inFirstPart = false;
404 							String nextLastWordPart = null;
405 							for (int i = limit; i > 0; i--) {
406 								// FIXME: don't hard-code the number 4
407 								if ((hyphens[i - 1] & 4) == 4) {
408 									inFirstPart = true;
409 									nextLastWordPart = lastWord.substring(0, i); }
410 								// FIXME: don't hard-code these numbers
411 								if ((((hyphens[i - 1] & 1) == 1) && (i < limit)) || ((hyphens[i - 1] & 2) == 2)) {
412 									if (!inFirstPart) {
413 										// break point in second part of word, or in first part and first part does not fit on line
414 										// in both cases the implicit line breaking will break correctly
415 										String n = lastWordPart;
416 										lastWordPart = null;
417 										return n; }
418 									else {
419 										// break point in first part of word and first part fits on line
420 										// need to switch to "explicit" mode because otherwise the first part would not be broken
421 										String n = lastWord.substring(0, i);
422 										// FIXME: don't hard-code the number 1
423 										if ((hyphens[i - 1] & 1) == 1)
424 											n += hyphenChar;
425 										lastWordPart = nextLastWordPart.substring(i);
426 										if (lastWordPart.isEmpty()) lastWordPart = null;
427 										return n; }}}}
428 						// else move the word to the next line
429 						return "";
430 					}
431 					else
432 						throw new NoSuchElementException();
433 				}
434 				public Character peek() {
435 					if (next != null)
436 						return next.charAt(0);
437 					else if (lastWordPart != null)
438 						return lastWordPart.charAt(0);
439 					else
440 						throw new NoSuchElementException();
441 				}
442 				public String remainder() {
443 					if (next == null && lastWordPart == null)
444 						throw new NoSuchElementException();
445 					StringBuilder remainder = new StringBuilder();
446 					if (next != null)
447 						remainder.append(next);
448 					if (lastWordPart != null)
449 						remainder.append(lastWordPart);
450 					return remainder.toString();
451 				}
452 				public boolean hasPrecedingSpace() {
453 					if (next == null && !alwaysEmpty)
454 						throw new UnsupportedOperationException();
455 					else
456 						return precedingSpace;
457 				}

516 				/**
517 				 * Array with soft wrap opportunity info
518 				 * - SPACE, LF, CR, TAB and ZWSP create normal soft wrap opportunities

520 				 * - normal soft wrap opportunities override soft wrap opportunities that insert a hyphen glyph

522 				 * @see <a href="http://braillespecs.github.io/braille-css/#h3_line-breaking">Braille CSS   9.4 Line Breaking</a>
523 				 */
524 				// byte at index i corresponds with boundary between characters i and i+1 of charBuffer
525 				// or end of string if i == charBuffer.length()
526 				private ArrayList<Byte> wrapInfo = new ArrayList<Byte>();
527 				// from lowest to highest precedence:
528 				private final static byte NO_SOFT_WRAP = (byte)0x0;
529 				private final static byte SOFT_WRAP_WITH_HYPHEN = (byte)0x1;     //    x
530 				private final static byte SOFT_WRAP_WITHOUT_HYPHEN = (byte)0x3;  //   xx
531 				private final static byte SOFT_WRAP_AFTER_SPACE = (byte)0x7;     //  xxx
532 				private final static byte HARD_WRAP = (byte)0x15;                // xxxx
533 				
534 				// for collapsing spaces
535 				private boolean lastCharIsSpace = false;
536 				private int forcedBreakCount = 0;
537 				
538 				/**
539 				 * Fill the character (charBuffer) and soft wrap opportunity (wrapInfo) buffers while normalising and collapsing spaces
540 				 * - until the buffers are at least 'limit' long
541 				 * - or until the current row is full according to the input feed (BrailleStream)
542 				 * - and while the remaining input starts with SPACE, LF, CR, TAB, NBSP, BRAILLE PATTERN BLANK, SHY, ZWSP or LS
543 				 */
544 				private void fillRow(int limit, boolean force, boolean allowHyphens) {
545 					int bufSize = charBuffer.length();
546 					while (true) {
547 						if (inputBuffer != null && !inputBuffer.hasNext()) {
548 							// there is a break opportunity after each string returned by the stream
549 							// we keep soft hyphens at the end of the string (also if we are at the very end of the stream)
550 							if (bufSize > 0 && wrapInfo.get(bufSize - 1) != SOFT_WRAP_WITH_HYPHEN)
551 								wrapInfo.set(bufSize - 1, (byte)(wrapInfo.get(bufSize - 1) | SOFT_WRAP_WITHOUT_HYPHEN));
552 							inputBuffer = null;
553 						}
554 						if (inputBuffer == null) {
555 							if (!inputStream.hasNext()) // end of stream
556 								return;
557 							if (bufSize < limit) {
558 								String next = inputStream.next(limit - bufSize, force && (bufSize == 0), allowHyphens);
559 								if (next.isEmpty()) // row full according to input feed
560 									return;
561 								inputBuffer = peekingIterator(charactersOf(next).iterator()); }
562 							if (inputBuffer == null) {
563 								switch (inputStream.peek()) {
564 								case SHY:    case TAB:
565 								case ZWSP:   case BLANK:
566 								case SPACE:  case NBSP:
567 								case LF:     case LS:
568 								case CR:
569 									String next = inputStream.next(1, true, allowHyphens);
570 									if (next.isEmpty())
571 										throw new RuntimeException("coding error");
572 									inputBuffer = peekingIterator(charactersOf(next).iterator());
573 									break;
574 								default:
575 									return; }}}
576 						char next = inputBuffer.peek();
577 						switch (next) {
578 						case SHY:
579 							if (bufSize > 0)
580 								wrapInfo.set(bufSize - 1, (byte)(wrapInfo.get(bufSize - 1) | SOFT_WRAP_WITH_HYPHEN));
581 							lastCharIsSpace = false;
582 							break;
583 						case ZWSP:
584 							if (bufSize > 0)
585 								wrapInfo.set(bufSize - 1, (byte)(wrapInfo.get(bufSize - 1) | SOFT_WRAP_WITHOUT_HYPHEN));
586 							break;
587 						case SPACE:
588 						case LF:
589 						case CR:
590 						case TAB:
591 						case BLANK:
592 							if (lastCharIsSpace)
593 								break;
594 							if (bufSize > 0)
595 								wrapInfo.set(bufSize - 1, (byte)(wrapInfo.get(bufSize - 1) | SOFT_WRAP_WITHOUT_HYPHEN));
596 							for (int i = 0; i < wordSpacing; i++) {
597 								charBuffer.append(blankChar);
598 								bufSize ++;
599 								wrapInfo.add(SOFT_WRAP_AFTER_SPACE);
600 							}
601 							lastCharIsSpace = true;
602 							break;
603 						case NBSP:
604 							charBuffer.append(blankChar);
605 							bufSize ++;
606 							wrapInfo.add(NO_SOFT_WRAP);
607 							lastCharIsSpace = false;
608 							break;
609 						case LS:
610 							if (bufSize > 0 && (wrapInfo.get(bufSize - 1) & HARD_WRAP) != HARD_WRAP)
611 								wrapInfo.set(bufSize - 1, (byte)(wrapInfo.get(bufSize - 1) | HARD_WRAP));
612 							else {
613 								// add a blank to attach the HARD_WRAP info to
614 								// otherwise we can't preserve empty lines
615 								charBuffer.append(blankChar);
616 								bufSize ++;
617 								wrapInfo.add(HARD_WRAP);
618 							}
619 							lastCharIsSpace = true;
620 							break;
621 						default:
622 							if (bufSize >= limit) return;
623 							charBuffer.append(next);
624 							bufSize ++;
625 							wrapInfo.add(NO_SOFT_WRAP);
626 							lastCharIsSpace = false; }
627 						inputBuffer.next();
628 					}
629 				}
630 				
631 				/**
632 				 * Flush the first 'size' elements of the character and soft wrap opportunity buffers
633 				 * Assumes that 'size &lt;= charBuffer.length()'
634 				 */
635 				private void flushBuffers(int size) {
636 					charBuffer = new StringBuilder(charBuffer.substring(size));
637 					wrapInfo = new ArrayList<Byte>(wrapInfo.subList(size, wrapInfo.size()));
638 				}
639 				
640 				/**
641 				 * @param limit specifies the maximum number of characters allowed in the result
642 				 * @param force specifies if the translator should force a break at the limit
643 				 *              if no natural break point is found
644 				 * @param wholeWordsOnly specifies that the row may not end on a break point inside a word.
645 				 * @return returns the translated string preceding the row break, including a translated hyphen
646 				 *                 at the end if needed.
647 				 */
648 				public String nextTranslatedRow(int limit, boolean force, boolean wholeWordsOnly) {
649 					fillRow(limit, force, !wholeWordsOnly);
650 					int bufSize = charBuffer.length();
651 					
652 					// charBuffer may be empty (even if hasNext() was true)
653 					if (bufSize == 0)
654 						return "";
655 					
656 					// always break at preserved line breaks
657 					for (int i = 0; i < min(bufSize, limit); i++) {
658 						if (wrapInfo.get(i) == HARD_WRAP) {
659 							int cut = i + 1;
660 							
661 							// strip trailing SPACE/LF/CR/TAB/BLANK/NBSP (all except NBSP are already collapsed into one)
662 							while (cut > 0 && charBuffer.charAt(cut - 1) == blankChar) cut--;
663 							
664 							// preserve if at beginning of stream
665 							if (cut == 0)
666 								cut = i + 1;
667 							String rv = charBuffer.substring(0, cut);
668 							
669 							// strip leading SPACE/LF/CR/TAB/BLANK in remaining text (are already collapsed into one)
670 							cut = i + 1;
671 							while (cut < bufSize && wrapInfo.get(cut) == SOFT_WRAP_AFTER_SPACE) cut++;
672 							flushBuffers(cut);
673 							return rv; }}
674 					
675 					// no need to break if remaining text is shorter than line
676 					if (bufSize < limit) {
677 						String rv = charBuffer.substring(0, bufSize);
678 							
679 						// replace soft hyphen with real hyphen
680 						if (wrapInfo.get(rv.length() - 1) == SOFT_WRAP_WITH_HYPHEN)
681 							rv += hyphenChar;
682 						charBuffer.setLength(0);
683 						wrapInfo.clear();
684 						
685 						// strip trailing SPACE/LF/CR/TAB/BLANK/NBSP (all except NBSP are already collapsed into one)
686 						int cut = bufSize;
687 						while (cut > 0 && rv.charAt(cut - 1) == blankChar) cut--;
688 						
689 						// preserve if at beginning of stream or end of stream
690 						if (cut > 0 && cut < bufSize && hasNext())
691 							rv = rv.substring(0, cut);
692 						return rv; }
693 					
694 					// return nothing if limit is 0 (limit should not be less than 0, but check anyway)
695 					if (limit <= 0) {
696 						
697 						// strip leading SPACE/LF/CR/TAB/BLANK in remaining text (are already collapsed into one)
698 						int cut = 0;
699 						while (cut < bufSize && wrapInfo.get(cut) == SOFT_WRAP_AFTER_SPACE) cut++;
700 						flushBuffers(cut);
701 						return ""; }
702 					
703 					// break at SPACE or ZWSP
704 					// FIXME: ignore ZWSP if it comes from hyphenation (how to find out?) and wholeWordsOnly==true
705 					if ((wrapInfo.get(limit - 1) & SOFT_WRAP_WITHOUT_HYPHEN) == SOFT_WRAP_WITHOUT_HYPHEN) {
706 						int cut = limit;
707 						
708 						// strip trailing SPACE/LF/CR/TAB/BLANK/NBSP (all except NBSP are already collapsed into one)
709 						while (cut > 0 && charBuffer.charAt(cut - 1) == blankChar) cut--;
710 						
711 						// strip leading SPACE/LF/CR/TAB/BLANK in remaining text (are already collapsed into one)
712 						int cut2 = limit;
713 						while (cut2 < bufSize && wrapInfo.get(cut2) == SOFT_WRAP_AFTER_SPACE) cut2++;
714 						String rv = charBuffer.substring(0, cut2);
715 						flushBuffers(cut2);
716 						
717 						// preserve if at beginning of stream or end of stream and not overflowing
718 						if (cut > 0 && cut < cut2 && hasNext())
719 							rv = rv.substring(0, cut);
720 						else if (cut2 > limit)
721 							rv = rv.substring(0, limit);
722 						return rv; }
723 					
724 					// try to break later if the overflowing characters are blank
725 					for (int i = limit + 1; i - 1 < bufSize && charBuffer.charAt(i - 1) == blankChar; i++)
726 						if ((wrapInfo.get(i - 1) & SOFT_WRAP_WITHOUT_HYPHEN) == SOFT_WRAP_WITHOUT_HYPHEN) {
727 							
728 							// strip trailing SPACE/LF/CR/TAB/BLANK/NBSP (all except NBSP are already collapsed into one)
729 							int cut = limit;
730 							while (cut > 0 && charBuffer.charAt(cut - 1) == blankChar) cut--;
731 							
732 							// strip leading SPACE/LF/CR/TAB/BLANK in remaining text (are already collapsed into one)
733 							int cut2 = i;
734 							while (cut2 < bufSize && wrapInfo.get(cut2) == SOFT_WRAP_AFTER_SPACE) cut2++;
735 							String rv = charBuffer.substring(0, cut2);
736 							flushBuffers(cut2);
737 							
738 							// preserve if at end of stream and not overflowing
739 							if (cut < cut2 && hasNext())
740 								rv = rv.substring(0, cut);
741 							else if (cut2 > limit)
742 								rv = rv.substring(0, limit);
743 							return rv; }
744 					
745 					// try to break sooner
746 					for (int i = limit - 1; i > 0; i--) {
747 						
748 						// break at SPACE, ZWSP or SHY
749 						// FIXME: ignore ZWSP if it comes from hyphenation (how to find out?) and wholeWordsOnly==true
750 						if ((wrapInfo.get(i - 1) & SOFT_WRAP_WITHOUT_HYPHEN) == SOFT_WRAP_WITHOUT_HYPHEN
751 						    || (!wholeWordsOnly && (wrapInfo.get(i - 1) == SOFT_WRAP_WITH_HYPHEN))) {
752 							int cut = i;
753 							String rv;
754 							
755 							// insert hyphen glyph at SHY
756 							if (wrapInfo.get(cut - 1) == 0x1) {
757 								rv = charBuffer.substring(0, cut);
758 								rv += hyphenChar; }
759 							else {
760 								
761 								// strip trailing SPACE/LF/CR/TAB/BLANK/NBSP (all except NBSP are already collapsed into one)
762 								while (cut > 0 && charBuffer.charAt(cut - 1) == blankChar) cut--;
763 								
764 								// preserve if at beginning of stream
765 								if (cut == 0)
766 									cut = i;
767 								rv = charBuffer.substring(0, cut); }
768 							
769 							// strip leading SPACE/LF/CR/TAB/BLANK in remaining text (are already collapsed into one)
770 							// FIXME: breaks cases where space after SHY is not from letter-spacing
771 							cut = i;
772 							while(cut < bufSize && charBuffer.charAt(cut) == blankChar) cut++;
773 							flushBuffers(cut);
774 							return rv; }}
775 					
776 					// force hard break
777 					if (force) {
778 						forcedBreakCount++;
779 						String rv = charBuffer.substring(0, limit);
780 						flushBuffers(limit);
781 						return rv; }
782 					
783 					return "";
784 				}

787 				 * @return returns true if there are characters not yet extracted with nextTranslatedRow
788 				 */
789 				public boolean hasNext() {
790 					return charBuffer.length() > 0
791 						|| (inputBuffer != null && inputBuffer.hasNext())
792 						|| inputStream.hasNext();
793 				}

795 				/**
796 				 * @return returns the characters not yet extracted with nextTranslatedRow
797 				 */
798 				public String getTranslatedRemainder() {
799 					
800 					// Note that I could use clone() here, but the current code is slightly more efficient
801 					BrailleStream save_inputStream = inputStream;
802 					inputStream = emptyBrailleStream;
803 					List<Character> save_inputBuffer = inputBuffer != null ? ImmutableList.copyOf(inputBuffer) : null;
804 					if (save_inputBuffer != null) {
805 						if (save_inputStream.hasNext()) {
806 							inputBuffer = peekingIterator(
807 									concat(save_inputBuffer.iterator(), charactersOf(save_inputStream.remainder()).iterator()));
808 						} else {
809 							inputBuffer = peekingIterator(save_inputBuffer.iterator());
810 						}
811 					} else if (save_inputStream.hasNext()) {
812 						inputBuffer = peekingIterator(charactersOf(save_inputStream.remainder()).iterator());
813 					} else {
814 						inputBuffer = null;
815 					}
816 					StringBuilder save_charBuffer = charBuffer;
817 					charBuffer = new StringBuilder(charBuffer.toString());
818 					ArrayList<Byte> save_wrapInfo = wrapInfo;
819 					wrapInfo = new ArrayList<Byte>(wrapInfo);
820 					boolean save_lastCharIsSpace = lastCharIsSpace;
821 					int save_forcedBreakCount = forcedBreakCount;
822 					forcedBreakCount = 0;
823 					fillRow(Integer.MAX_VALUE, true, false);
824 					String remainder = charBuffer.toString();
825 					inputStream = save_inputStream;
826 					inputBuffer = save_inputBuffer != null ? peekingIterator(save_inputBuffer.iterator()) : null;
827 					charBuffer = save_charBuffer;
828 					wrapInfo = save_wrapInfo;
829 					lastCharIsSpace = save_lastCharIsSpace;
830 					forcedBreakCount = save_forcedBreakCount;
831 					return remainder;
832 				}

851 					LineIterator clone; {
852 						try {

855 							throw new InternalError("coding error");
856 						}
857 					}

862 						clone.inputBuffer = peekingIterator(list.iterator());
863 					}
864 					clone.charBuffer = new StringBuilder(charBuffer.toString());
865 					clone.wrapInfo = new ArrayList<Byte>(wrapInfo);
866 					return clone;

892 			/**
893 			 * Whether there is a space immediately before the substring starting at <code>before</code>.
894 			 */
895 			protected static boolean hasPrecedingSpace(String string, int before) {
896 				while (before > 0)
897 					switch (string.charAt(--before)) {
898 					case SPACE:
899 					case BLANK:
900 					case CR:
901 					case LF:
902 					case TAB:
903 					case LS:
904 						return true;
905 					case ZWSP:
906 						continue;
907 					default:
908 						return false;
909 					}
910 				return false;
911 			}
