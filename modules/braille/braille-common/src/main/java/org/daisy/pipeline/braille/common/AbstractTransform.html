1   package org.daisy.pipeline.braille.common;
2   
3   import java.util.NoSuchElementException;
4   import java.util.Queue;
5   import java.util.concurrent.atomic.AtomicLong;
6   import java.util.concurrent.ConcurrentLinkedQueue;
7   
8   import com.google.common.base.MoreObjects;
9   import com.google.common.base.MoreObjects.ToStringHelper;
10  
11  public abstract class AbstractTransform implements Transform {
12  	
13  	private static final AtomicLong idCounter = new AtomicLong(0);
14  	private static final Queue<Long> availableIds = new ConcurrentLinkedQueue<>();
15  	
16  	private final long idNumber;
17  	private final String id;
18  	
19  	protected AbstractTransform() {
20  		idNumber = getUniqueId();
21  		id = "transform" + idNumber;
22  	}
23  	
24  	public String getIdentifier() {
25  		return id;
26  	}
27  	
28  	private static long getUniqueId() {
29  		try {
30  			return availableIds.remove();
31  		} catch (NoSuchElementException e) {
32  			return idCounter.incrementAndGet();
33  		}
34  	}
35  	
36  	public ToStringHelper toStringHelper() {
37  		return MoreObjects.toStringHelper(this);
38  	}
39  	
40  	@Override
41  	public String toString() {
42  		return toStringHelper().add("id", id).toString();
43  	}
44  	
45  	public void finalize() {
46  		availableIds.add(idNumber);
47  	}
48  }
