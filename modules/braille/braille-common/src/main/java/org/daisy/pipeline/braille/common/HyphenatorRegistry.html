1   package org.daisy.pipeline.braille.common;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import static org.daisy.pipeline.braille.common.TransformProvider.util.dispatch;
7   import org.daisy.pipeline.braille.common.TransformProvider.util.Memoize;
8   import org.daisy.pipeline.braille.common.util.Strings;
9   
10  import org.osgi.service.component.annotations.Component;
11  import org.osgi.service.component.annotations.Reference;
12  import org.osgi.service.component.annotations.ReferenceCardinality;
13  import org.osgi.service.component.annotations.ReferencePolicy;
14  
15  import org.slf4j.Logger;
16  
17  @Component(
18  	name = "hyphenator-registry",
19  	service = { HyphenatorRegistry.class }
20  )
21  public class HyphenatorRegistry extends Memoize<Hyphenator> implements HyphenatorProvider<Hyphenator> {
22  
23  	public HyphenatorRegistry() {
24  		super();
25  	}
26  
27  	private final List<TransformProvider<Hyphenator>> providers = new ArrayList<>();
28  	private final TransformProvider<Hyphenator> dispatch = dispatch(providers);
29  
30  	@Reference(
31  		name = "HyphenatorProvider",
32  		unbind = "-",
33  		service = HyphenatorProvider.class,
34  		cardinality = ReferenceCardinality.MULTIPLE,
35  		policy = ReferencePolicy.STATIC
36  	)
37  	public void addProvider(HyphenatorProvider p) {
38  		providers.add(p);
39  	}
40  
41  	public Iterable<Hyphenator> _get(Query q) {
42  		return dispatch.get(q);
43  	}
44  
45  	public TransformProvider<Hyphenator> _withContext(Logger context) {
46  		return dispatch.withContext(context);
47  	}
48  
49  	@Override
50  	public String toString() {
51  		return "memoize(dispatch( " + Strings.join(providers, ", ") + " ))";
52  	}
53  }
