30  /**
31   * @see <a href="../../../../../../../README.md">Documentation</a>
32   */
33  @Component(
34  	name = "org.daisy.pipeline.braille.pef.impl.LocaleTableProvider",
35  	service = {
36  		TableProvider.class
37  		// org.daisy.dotify.api.table.TableProvider.class
38  	}
39  )
40  public class LocaleBasedTableProvider extends AbstractTableProvider {
41  	
42  	private static Set<String> supportedFeatures = ImmutableSet.of("locale", "document-locale");
43  	private static Map<String,String> tableFromLocale = new HashMap<String,String>();
44  	
45  	public LocaleBasedTableProvider() {
46  		tableFromLocale.put("en", "org.daisy.braille.impl.table.DefaultTableProvider.TableType.EN_US");
47  		tableFromLocale.put("de", "org_daisy.EmbosserTableProvider.TableType.DE_DE");
48  		tableFromLocale.put("nl", "com_braillo.BrailloTableProvider.TableType.BRAILLO_6DOT_031_01");
49  	}
50  
51  	/**
52  	 * Recognized features:
53  	 *
54  	 * - locale: A locale that is mapped to a specific table
55  	 *     that is a sane default for that locale.
56  	 */
57  	protected Iterable<Table> _get(Query query) {
58  		for (Feature feature : query)
59  			if (!supportedFeatures.contains(feature.getKey())) {
60  				logger.debug("Unsupported feature: " + feature);
61  				return empty; }
62  		Iterable<Table> table = empty;
63  		MutableQuery q = mutableQuery(query);
64  		final String documentLocale = q.containsKey("document-locale")
65  			? q.removeOnly("document-locale").getValue().get()
66  			: null;
67  		if (q.containsKey("locale")) {
68  			String id = tableFromLocale.get(q.removeOnly("locale").getValue().get());
69  			if (id != null && q.isEmpty())
70  				table = get(id); }
71  		else if (documentLocale != null && q.isEmpty()) {
72  			String id = tableFromLocale.get(documentLocale);
73  			if (id != null)
74  				table = get(id); }
75  		return table;
76  	}
77  	
78  	private final static Iterable<Table> empty = Optional.<Table>absent().asSet();
79  	
80  	private final List<org.daisy.dotify.api.table.TableProvider> providers
81  	= new ArrayList<org.daisy.dotify.api.table.TableProvider>();
82  	
83  	@Reference(
84  		name = "TableProvider",
85  		unbind = "removeTableProvider",
86  		service = org.daisy.dotify.api.table.TableProvider.class,
87  		cardinality = ReferenceCardinality.MULTIPLE,
88  		policy = ReferencePolicy.DYNAMIC
89  	)
90  	public void addTableProvider(org.daisy.dotify.api.table.TableProvider provider) {
91  		providers.add(provider);
92  	}
93  	
94  	public void removeTableProvider(org.daisy.dotify.api.table.TableProvider provider) {
95  		providers.remove(provider);
96  	}
97  	
98  	private Iterable<Table> get(String id) {
99  		for (org.daisy.dotify.api.table.TableProvider p : providers)
100 			for (FactoryProperties fp : p.list())
101 				if (fp.getIdentifier().equals(id))
102 					return Optional.fromNullable(p.newFactory(id)).asSet();
103 		return empty;
104 	}
105 	
106 	private static final Logger logger = LoggerFactory.getLogger(LocaleBasedTableProvider.class);
107 	
108 }
