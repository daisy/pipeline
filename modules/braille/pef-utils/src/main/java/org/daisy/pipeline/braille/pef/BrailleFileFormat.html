18  public class BrailleFileFormat extends EmbossedMedium implements FileFormat {
19  
20  	private final FileFormat underlyingFormat;
21  	private final Area printableArea;
22  	private final boolean sheetsMultipleOfTwo;
23  	private final boolean blankLastPage;
24  	private final boolean overflowAllowed;
25  
26  	/**
27  	 * {@code printableArea}, {@code pageWidth} and {@code pageHeight} determine the dimensions of
28  	 * the medium as observed in media queries, but may or may not be linked to hard constraints
29  	 * such as a physical sheet of paper and embosser.
30  	 *
31  	 * @param printableArea The dimensions and position of the printable area of the page
32  	 * @param sheetsMultipleOfTwo whether volumes have a number of sheets that is a multiple of 2
33  	 * @param blankLastPage whether the last page of each volume should be left blank
34  	 */
35  	public BrailleFileFormat(FileFormat format, Area printableArea,
36  	                         Dimension pageWidth, Dimension pageHeight,
37  	                         double cellWidth, double cellHeight,
38  	                         boolean sheetsMultipleOfTwo, boolean blankLastPage, boolean overflowAllowed,
39  	                         MediumBuilder parser) {
40  		super(new Dimension(printableArea.getWidth(), Unit.MM),
41  		      new Dimension(printableArea.getHeight(), Unit.MM),
42  		      pageWidth,
43  		      pageHeight,
44  		      null,
45  		      cellWidth,
46  		      cellHeight,
47  		      parser);
48  		this.underlyingFormat = format;
49  		this.printableArea = printableArea;
50  		this.sheetsMultipleOfTwo = sheetsMultipleOfTwo;
51  		this.blankLastPage = blankLastPage;
52  		this.overflowAllowed = overflowAllowed;
53  	}
54  
55  	/**
56  	 * The printable area of the braille page
57  	 */
58  	public Area getPrintableArea() {
59  		return printableArea;
60  	}
61  
62  	/**
63  	 * Whether content is allowed to overflow the viewport.
64  	 *
65  	 * An example of when this could be true is when the file is not bound to a specific embosser
66  	 * and paper.
67  	 */
68  	public boolean isOverflowAllowed() {
69  		return overflowAllowed;
70  	}
71  
72  	/* ////// */
73  	/* Medium */
74  	/* ////// */
75  
76  	@Override
77  	protected Object getNonStandardFeature(String feature) {
78  		// sheets-multiple-of-two and blank-last-page are handled in px:pef-store (and are not
79  		// expected to be features of the underlying FileFormat)
80  		if ("-daisy-sheets-multiple-of-two".equals(feature))
81  			return sheetsMultipleOfTwo;
82  		else if ("-daisy-blank-last-page".equals(feature))
83  			return blankLastPage;
84  		else {
85  			if (feature.startsWith("-daisy-"))
86  				feature = feature.substring("-daisy-".length());
87  			return getFeature(feature);
88  		}
89  	}
