80  	 * pdfbox does not support OpenType fonts, nor does it support the font-face properties
81  	 * ascent-override, descent-override and size-adjust. NotCourierSans-Bold.ttf was created from
82  	 * NotCourierSans-Bold.otf using the Font Squirrel Webfont Generator tool
83  	 * (https://www.fontsquirrel.com/tools/webfont-generator). I used these settings:
84  	 *
85  	 * - Upload Fonts: NotCourierSans-Bold.otf
86  	 * - Font Formats: TrueType
87  	 * - Vertical Metrics: Custom Adjustment:
88  	 *   - Ascent: 70%

121 	@Override
122 	public void run() throws SaxonApiException {
123 		super.run();
124 		try {
125 			URI href = null; {
126 				try {
127 					href = URI.create(getOption(_HREF).getString());
128 				} catch (IllegalArgumentException e) {
129 					throw new IllegalArgumentException("Not a valid URI: " + href);
130 				}
131 			}
132 			File pdfFile = null; {
133 				try {
134 					pdfFile = new File(href);
135 				} catch (IllegalArgumentException e) {
136 					throw new IllegalArgumentException("Not a valid file URI: " + href);
137 				}
138 			}
139 			if (pdfFile.exists())
140 				throw new RuntimeException("File exists: " + pdfFile);
141 			Table table = null; {
142 				MutableQuery q = mutableQuery(query(getOption(_TABLE, "")));
143 				logger.debug("Finding table for query: " + q);
144 				try {
145 					table = tableRegistry.get(q).iterator().next();
146 				} catch (NoSuchElementException e) {
147 					try {
148 						table = tableRegistry.get(mutableQuery().add("id", DEFAULT_TABLE)).iterator().next();
149 						logger.warn("Can not find a table for query: " + q);
150 						logger.warn("Falling back to default");
151 					} catch (NoSuchElementException ee) {
152 						throw new IllegalArgumentException("Could not find a table for query: " + q);
153 					}
154 				}
155 			}
156 			logger.debug("Storing PEF to PDF using table: " + table);
157 			new PEF2PDF(pdfFile, table).transform(
158 				ImmutableMap.of(_SOURCE, new XMLCalabashInputValue(source)),
159 				ImmutableMap.of()
160 			).run();
161 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(cResultDocument(pdfFile.toURI().toString()))));
162 		} catch (Throwable e) {
163 			throw XProcStep.raiseError(e, step);
164 		}
165 	}

222 		@Override
223 		public Runnable transform(Map<QName,InputValue<?>> input, Map<QName,OutputValue<?>> output) {
224 			input = XMLTransformer.validateInput(input, ImmutableMap.of(_SOURCE, InputType.MANDATORY_NODE_SEQUENCE));
225 			output = XMLTransformer.validateOutput(output, null);
226 			XMLInputValue<?> source = (XMLInputValue<?>)input.get(_SOURCE);
227 			return () -> {
228 				try {
229 					org.daisy.common.stax.BaseURIAwareXMLStreamReader pef = source.asXMLStreamReader();
230 					ByteArrayOutputStream htmlBytes = new ByteArrayOutputStream();
231 					Writer html = new OutputStreamWriter(htmlBytes, StandardCharsets.UTF_8);
232 					BrailleConverter bc = table.newBrailleConverter();
233 					boolean tableMatchesBrailleCharset = false;
234 					int maxColumns = 0; // in cells per line
235 					int maxRows = 0; // in lines
236 					int marginLeft = 10; // in mm
237 					int marginTop = 10; // in mm
238 					LinkedList<QName> elementStack = new LinkedList<>();
239 					LinkedList<Boolean> duplexStack = new LinkedList<>();
240 					LinkedList<Integer> colsStack = new LinkedList<>();
241 					LinkedList<Integer> rowsStack = new LinkedList<>();
242 					LinkedList<Integer> rowgapStack = new LinkedList<>();
243 					int volumeCount = 0;
244 					int pagesInSection = 0;
245 					events: while (true)
246 						try {
247 							int event = pef.next();
248 							switch (event) {
249 							case START_ELEMENT: {
250 								QName elemName = pef.getName();
251 								switch (elementStack.size()) {
252 								case 0:
253 									if (PEF_PEF.equals(elemName)) {
254 										html.write(
255 											"<!DOCTYPE html>\n" +
256 											"<html xmlns=\"http://www.w3.org/1999/xhtml\">\n" +
257 											"	<head>\n" +
258 											"		<meta http-equiv=\"Content-Type\" content=\"text/html; charset=utf-8\" />\n" +
259 											"		<style type=\"text/css\">\n" +
260 											"		/*<![CDATA[*/\n" +
261 											"@font-face {\n" +
262 											"	font-family: odt2braille;\n" +
263 											"	src: url(data:font/truetype;charset=utf8;base64,"
264 											         // since Java 9 there is also InputStream.readAllBytes()
265 											         + Base64.getEncoder().encodeToString(ByteStreams.toByteArray(font.openStream())) + ")\n" +
266 											"	     format(\"truetype\");\n" +
267 											"}\n" +
268 											"@font-face {\n" +
269 											"	font-family: NotCourierSans;\n" +
270 											"	src: url(data:font/truetype;charset=utf8;base64,"
271 											         + Base64.getEncoder().encodeToString(ByteStreams.toByteArray(fallbackFont.openStream())) + ")\n" +
272 											"	     format(\"truetype\");\n" +
273 											"}\n" +
274 											"@page {\n" +
275 											String.format("	margin-left: %dmm;\n", marginLeft) +
276 											String.format("	margin-top: %dmm;\n", marginTop) +
277 											"	margin-right: 0mm;\n" +
278 											"	margin-bottom: 0mm;\n" +
279 											"}\n" +
280 											"body {\n" +
281 											"	font-family: 'Helvetica Neue', Helvetica, Arial, sans-serif;\n" +
282 											"	margin: 0;\n" +
283 											"	font-size: 25px;\n" +
284 											"}\n" +
285 											"h1.bookmark {\n" +
286 											"	position: absolute;\n" +
287 											"	left: -1000mm;\n" +
288 											"	font-size: 1px;\n" +
289 											"}\n" +
290 											".page {\n" +
291 											"	page-break-before: always;\n" +
292 											"}\n" +
293 											".row {\n" +
294 											"    font-family: odt2braille, NotCourierSans;\n" +
295 											"    letter-spacing: 0px;\n" +
296 											"    white-space: pre;\n" +
297 											"    letter-spacing: 0px;\n" +
298 											"    font-size: 125%;\n" +
299 											"    height: 1em;\n" +
300 											"}\n" +
301 											".row[rowgap=\"1\"] {\n" +
302 											"    height: 1.25em;\n" +
303 											"}\n" +
304 											".row[rowgap=\"2\"] {\n" +
305 											"    height: 1.5em;\n" +
306 											"}\n" +
307 											".row[rowgap=\"3\"] {\n" +
308 											"    height: 1.75em;\n" +
309 											"}\n" +
310 											".row[rowgap=\"4\"] {\n" +
311 											"    height: 2em;\n" +
312 											"}\n" +
313 											"		/*]]>*/\n" +
314 											"		</style>\n" +
315 											"	</head>\n" +
316 											"	<body>\n");
317 										elementStack.push(elemName);
318 										continue events;
319 									}
320 									break;
321 								case 1:
322 									if (PEF_HEAD.equals(elemName) || PEF_BODY.equals(elemName)) {
323 										elementStack.push(elemName);
324 										continue events;
325 									}
326 									break;
327 								case 2:
328 									if (elementStack.get(0).equals(PEF_HEAD)) {
329 										if (PEF_META.equals(elemName)) {
330 											elementStack.push(elemName);
331 											continue events;
332 										}
333 									} else { // BODY
334 										if (PEF_VOLUME.equals(elemName)) {
335 											duplexStack.push(getDuplex(pef, duplexStack));
336 											colsStack.push(getCols(pef, colsStack));
337 											rowsStack.push(getRows(pef, rowsStack));
338 											rowgapStack.push(getRowgap(pef, rowgapStack));
339 											elementStack.push(elemName);
340 											volumeCount++;
341 											// FIXME: bookmarks supported by wkhtmltopdf but not by openhtmltopdf
342 											html.write("		<div class=\"volume\">\n" +
343 											           "			<h1 class=\"bookmark\">Volume " + volumeCount + "</h1>\n");
344 											continue events;
345 										}
346 									}
347 									break;
348 								case 3:
349 									if (elementStack.get(0).equals(PEF_META)) {
350 										if (DP2_ASCII_BRAILLE_CHARSET.equals(elemName)) {
351 											String charset = pef.getElementText(); // consumes whole element
352 											if (charset.equals(table.getIdentifier()))
353 												tableMatchesBrailleCharset = true;
354 											continue events;
355 										}
356 									} else {// VOLUME
357 										if (PEF_SECTION.equals(elemName)) {
358 											pagesInSection = 0;
359 											int cols = getCols(pef, colsStack);
360 											if (cols > maxColumns)
361 												maxColumns = cols;
362 											int rows = getRows(pef, rowsStack);
363 											if (rows > maxRows)
364 												maxRows = rows;
365 											duplexStack.push(getDuplex(pef, duplexStack));
366 											rowgapStack.push(getRowgap(pef, rowgapStack));
367 											elementStack.push(elemName);
368 											continue events;
369 										}
370 									}
371 									break;
372 								case 4:
373 									if (elementStack.get(0).equals(PEF_SECTION)) {
374 										if (PEF_PAGE.equals(elemName)) {
375 											pagesInSection++;
376 											rowgapStack.push(getRowgap(pef, rowgapStack));
377 											elementStack.push(elemName);
378 											html.write("			<div class=\"page\">\n");
379 											continue events;
380 										}
381 									}
382 									break;
383 								case 5:
384 									if (elementStack.get(0).equals(PEF_PAGE)) {
385 										if (PEF_ROW.equals(elemName)) {
386 											int rowgap = getRowgap(pef, rowgapStack);
387 											String row = pef.getElementText(); // consumes whole element
388 											if (tableMatchesBrailleCharset) {
389 												String attr = pef.getAttributeValue(DP2_ASCII.getNamespaceURI(), DP2_ASCII.getLocalPart());
390 												if (attr != null)
391 													row = attr;
392 												else
393 													row = bc.toText(row);
394 											} else
395 												row = bc.toText(row);
396 											html.write("				<div class=\"row\" rowgap=\"" + rowgap + "\">");
397 											html.write(row.replace("&", "&amp;")
398 											              .replace("<", "&lt;")
399 											              .replace(">", "&gt;") + "</div>\n");
400 											continue events;
401 										}
402 									}
403 									break;
404 								default:
405 								}
406 								pef.getElementText(); // consume whole element
407 								break;
408 							}
409 							case END_ELEMENT: {
410 								QName elemName = elementStack.pop();
411 								if (PEF_PAGE.equals(elemName)) {
412 									rowgapStack.pop();
413 									html.write("			</div>\n");
414 								} else if (PEF_SECTION.equals(elemName)) {
415 									if (duplexStack.pop()) {
416 										if (pagesInSection % 2 != 0) {
417 											// insert empty page
418 											html.write("			<div class=\"page\">\n");
419 											html.write("			</div>\n");
420 										}
421 									}
422 									rowgapStack.pop();
423 								} else if (PEF_VOLUME.equals(elemName)) {
424 									duplexStack.pop();
425 									colsStack.pop();
426 									rowsStack.pop();
427 									rowgapStack.pop();
428 									html.write("		</div>\n");
429 								} else if (PEF_PEF.equals(elemName)) {
430 									html.write("	</body>\n" +
431 									           "</html>\n");
432 								}
433 								break;
434 							}
435 							default:
436 							}
437 						} catch (NoSuchElementException e) {
438 							break;
439 						}
440 					html.flush();
441 					pdf.getParentFile().mkdirs();
442 					int pageWidth = (int)Math.ceil(4.2 * maxColumns) + 2 * marginLeft;
443 					int pageHeight = (int)Math.ceil(8.44 * maxRows) + 2 * marginTop;
444 					try (OutputStream os = new FileOutputStream(pdf)) {
445 						new PdfRendererBuilder()
446 							.withHtmlContent(new String(htmlBytes.toByteArray(), StandardCharsets.UTF_8),
447 							                 pef.getBaseURI().toString())
448 							.useDefaultPageSize(pageWidth, pageHeight, PageSizeUnits.MM)
449 							.toStream(os)
450 							.run();
451 					}
452 				} catch (Throwable e) {
453 					throw new TransformerException(e);
454 				}
455 			};
456 		}

458 		public boolean getDuplex(XMLStreamReader pef, LinkedList<Boolean> stack) {
459 			// not using XMLStreamReader#getAttributeValue(String, String) because it is buggy
460 			// String attr = pef.getAttributeValue(_DUPLEX.getNamespaceURI(), _DUPLEX.getLocalPart());
461 			for (int i = 0; i < pef.getAttributeCount(); i++)
462 				if (_DUPLEX.equals(pef.getAttributeName(i)))
463 					return Boolean.parseBoolean(pef.getAttributeValue(i)); // assume "true" or "false"
464 			return stack.getFirst(); // throws NoSuchElementException if stack is empty, which can not
465 			                         // happen if PEF is valid
466 		}
467 
468 		public int getCols(XMLStreamReader pef, LinkedList<Integer> stack) {
469 			for (int i = 0; i < pef.getAttributeCount(); i++)
470 				if (_COLS.equals(pef.getAttributeName(i)))
471 					return Integer.parseInt(pef.getAttributeValue(i)); // assume positive integer
472 			return stack.getFirst(); // throws NoSuchElementException if stack is empty, which can not
473 			                         // happen if PEF is valid
474 		}
475 
476 		public int getRows(XMLStreamReader pef, LinkedList<Integer> stack) {
477 			for (int i = 0; i < pef.getAttributeCount(); i++)
478 				if (_ROWS.equals(pef.getAttributeName(i)))
479 					return Integer.parseInt(pef.getAttributeValue(i)); // assume positive integer
480 			return stack.getFirst(); // throws NoSuchElementException if stack is empty, which can not
481 			                         // happen if PEF is valid
482 		}
483 
484 		public int getRowgap(XMLStreamReader pef, LinkedList<Integer> stack) {
485 			for (int i = 0; i < pef.getAttributeCount(); i++)
486 				if (_ROWGAP.equals(pef.getAttributeName(i)))
487 					return Integer.parseInt(pef.getAttributeValue(i)); // assume non-negative integer
488 			return stack.getFirst(); // throws NoSuchElementException if stack is empty, which can not
489 			                         // happen if PEF is valid
490 		}
