1   package org.daisy.pipeline.braille.pef.impl;
2   
3   import java.io.IOException;
4   import java.nio.charset.Charset;
5   
6   import org.daisy.dotify.api.embosser.EmbosserWriter;
7   import org.daisy.dotify.api.embosser.LineBreaks;
8   import org.daisy.dotify.api.table.BrailleConverter;
9   
10  /*
11   * Initially copied from AbstractEmbosserWriter because it is in a private package
12   */
13  public abstract class BRFWriter implements EmbosserWriter {
14  	
15  	public static enum Padding {
16  		BOTH,
17  		BEFORE,
18  		AFTER,
19  		NONE
20  	};
21  	
22  	public interface PageBreaks {
23  		public String getString();
24  	}
25  	
26  	private int rowgap;
27  	private boolean isOpen = false;
28  	private boolean isClosed = false;
29  	private boolean currentDuplex;
30  	private int currentPage;
31  	private int charsOnRow;
32  	private int rowsOnPage;
33  	
34  	public abstract LineBreaks getLinebreakStyle();
35  	public abstract PageBreaks getPagebreakStyle();
36  	public abstract Padding getPaddingStyle();
37  	public abstract Charset getCharset();
38  	public abstract BrailleConverter getTable();
39  	protected abstract void add(byte b) throws IOException;
40  	protected abstract void addAll(byte[] b) throws IOException;
41  	
42  	@Override
43  	public void newLine() throws IOException {
44  		for (int i=0; i<((rowgap / 4)+1); i++) {
45  			lineFeed();
46  		}
47  	}
48  	
49  	@Override
50  	public void setRowGap(int value) {
51  		if (value<0) {
52  			throw new IllegalArgumentException("Non negative integer expected.");
53  		} else {
54  			rowgap = value;
55  		}
56  	}
57  	
58  	@Override
59  	public int getRowGap() {
60  		return rowgap;
61  	}
62  	
63  	@Override
64  	public void open(boolean duplex) throws IOException {
65  		charsOnRow = 0;
66  		rowsOnPage = 0;
67  		rowgap = 0;
68  		currentPage = 1;
69  		isOpen=true;
70  		currentDuplex = duplex;
71  	}
72  	
73  	@Override
74  	public void close() throws IOException {
75  		isClosed=true;
76  		isOpen=false;
77  	}
78  
79  	@Override
80  	public void write(String braille) throws IOException {
81  		charsOnRow += braille.length();
82  		addAll(String.valueOf(getTable().toText(braille)).getBytes(getCharset().name()));
83  	}
84  	
85  	private void lineFeed() throws IOException {
86  		rowsOnPage++;
87  		charsOnRow = 0;
88  		addAll(getLinebreakStyle().getString().getBytes());
89  	}
90  	
91  	private void formFeed() throws IOException {
92  		rowsOnPage++;
93  		switch (getPaddingStyle()) {
94  			case BEFORE:
95  				lineFeed();
96  			case NONE:
97  				addAll(getPagebreakStyle().getString().getBytes());
98  				break;
99  			case BOTH:
100 				lineFeed();
101 			case AFTER:
102 				addAll(getPagebreakStyle().getString().getBytes());
103 				lineFeed();
104 				break;
105 		}
106 		currentPage++;
107 		rowsOnPage = 0;
108 		charsOnRow = 0;
109 	}
110 	
111 	@Override
112 	public void newPage() throws IOException {
113 		if (!currentDuplex && (currentPage % 2)==1) {
114 			formFeed();
115 		}
116 		formFeed();
117 	}
118 
119 	@Override
120 	public void newSectionAndPage(boolean duplex) throws IOException {
121 		if ((currentPage % 2)==1) {
122 			formFeed();
123 		}
124 		newPage();
125 		currentDuplex = duplex;
126 	}
127 
128 	@Override
129 	public void newVolumeSectionAndPage(boolean duplex) throws IOException {
130 		newSectionAndPage(duplex);
131 	}
132 
133 	@Override
134 	public boolean isOpen() {
135 		return isOpen;
136 	}
137 
138 	@Override
139 	public boolean isClosed() {
140 		return isClosed;
141 	}
142 
143 	@Override
144 	public int getMaxWidth() {
145 		return Integer.MAX_VALUE;
146 	}
147 
148 	@Override
149 	public boolean supportsAligning() {
150 		return false;
151 	}
152 }
