1   package org.daisy.pipeline.braille.pef.impl;
2   
3   import java.io.IOException;
4   import java.nio.charset.Charset;
5   import org.daisy.braille.api.embosser.Contract;
6   import org.daisy.braille.api.embosser.ContractNotSupportedException;
7   import org.daisy.braille.api.embosser.EmbosserWriter;
8   import org.daisy.braille.api.embosser.LineBreaks;
9   import org.daisy.braille.api.table.BrailleConverter;
10  
11  /*
12   * Initially copied from AbstractEmbosserWriter because it is in a private package
13   */
14  public abstract class BRFWriter implements EmbosserWriter {
15  	
16  	public static enum Padding {
17  		BOTH,
18  		BEFORE,
19  		AFTER,
20  		NONE
21  	};
22  	
23  	public interface PageBreaks {
24  		public String getString();
25  	}
26  	
27  	private int rowgap;
28  	private boolean isOpen = false;
29  	private boolean isClosed = false;
30  	private boolean currentDuplex;
31  	private int currentPage;
32  	private int charsOnRow;
33  	private int rowsOnPage;
34  	
35  	public abstract LineBreaks getLinebreakStyle();
36  	public abstract PageBreaks getPagebreakStyle();
37  	public abstract Padding getPaddingStyle();
38  	public abstract Charset getCharset();
39  	public abstract BrailleConverter getTable();
40  	protected abstract void add(byte b) throws IOException;
41  	protected abstract void addAll(byte[] b) throws IOException;
42  	
43  	public void newLine() throws IOException {
44  		for (int i=0; i<((rowgap / 4)+1); i++) {
45  			lineFeed();
46  		}
47  	}
48  	
49  	public void setRowGap(int value) {
50  		if (value<0) {
51  			throw new IllegalArgumentException("Non negative integer expected.");
52  		} else {
53  			rowgap = value;
54  		}
55  	}
56  	
57  	public int getRowGap() {
58  		return rowgap;
59  	}
60  	
61  	public void open(boolean duplex) throws IOException {
62  		try {
63  			open(duplex, new Contract.Builder().build());
64  		} catch (ContractNotSupportedException e) {
65  			IOException ex = new IOException("Could not open embosser.");
66  			ex.initCause(e);
67  			throw ex;
68  		}
69  	}
70  	
71  	public void open(boolean duplex, Contract contract) throws IOException, ContractNotSupportedException {
72  		charsOnRow = 0;
73  		rowsOnPage = 0;
74  		rowgap = 0;
75  		currentPage = 1;
76  		isOpen=true;
77  		currentDuplex = duplex;
78  	}
79  	
80  	public int currentPage() {
81  		return currentPage;
82  	}
83  	
84  	public boolean pageIsEmpty() {
85  		return (charsOnRow+rowsOnPage)==0;
86  	}
87  	
88  	public void close() throws IOException {
89  		isClosed=true;
90  		isOpen=false;
91  	}
92  
93  	public void write(String braille) throws IOException {
94  		charsOnRow += braille.length();
95  		addAll(String.valueOf(getTable().toText(braille)).getBytes(getCharset().name()));
96  	}
97  	
98  	protected void lineFeed() throws IOException {
99  		rowsOnPage++;
100 		charsOnRow = 0;
101 		addAll(getLinebreakStyle().getString().getBytes());
102 	}
103 	
104 	protected void formFeed() throws IOException {
105 		rowsOnPage++;
106 		switch (getPaddingStyle()) {
107 			case BEFORE:
108 				lineFeed();
109 			case NONE:
110 				addAll(getPagebreakStyle().getString().getBytes());
111 				break;
112 			case BOTH:
113 				lineFeed();
114 			case AFTER:
115 				addAll(getPagebreakStyle().getString().getBytes());
116 				lineFeed();
117 				break;
118 		}
119 		currentPage++;
120 		rowsOnPage = 0;
121 		charsOnRow = 0;
122 	}
123 	
124 	public void newPage() throws IOException {
125 		if (supportsDuplex() && !currentDuplex && (currentPage % 2)==1) {
126 			formFeed();
127 		}
128 		formFeed();
129 	}
130 
131 	public void newSectionAndPage(boolean duplex) throws IOException {
132 		if (supportsDuplex() && (currentPage % 2)==1) {
133 			formFeed();
134 		}
135 		newPage();
136 		currentDuplex = duplex;
137 	}
138 
139 	public void newVolumeSectionAndPage(boolean duplex) throws IOException {
140 		newSectionAndPage(duplex);
141 	}
142 
143 	public boolean isOpen() {
144 		return isOpen;
145 	}
146 
147 	public boolean isClosed() {
148 		return isClosed;
149 	}
150 
151 	public int getMaxHeight() {
152 		return Integer.MAX_VALUE;
153 	}
154 
155 	public int getMaxWidth() {
156 		return Integer.MAX_VALUE;
157 	}
158 
159 	public boolean supports8dot() {
160 		return false;
161 	}
162 
163 	public boolean supportsAligning() {
164 		return false;
165 	}
166 
167 	public boolean supportsDuplex() {
168 		return true;
169 	}
170 
171 	public boolean supportsVolumes() {
172 		return false;
173 	}
174 	
175 	public boolean supportsZFolding() {
176 		return false;
177 	}
178 	
179 	public boolean supportsPrintMode(PrintMode mode) {
180 		return false;
181 	}
182 }
