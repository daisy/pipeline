213 	public static Color web(String colorString, double opacity) {
214 		if (colorString == null) {
215 			throw new NullPointerException(
216 					"The color components or name must be specified");
217 		}
218 		if (colorString.isEmpty()) {
219 			throw new IllegalArgumentException("Invalid color specification");
220 		}
221 
222 		String color = colorString.toLowerCase(Locale.ROOT);
223 
224 		if (color.startsWith("#")) {
225 			color = color.substring(1);
226 		} else if (color.startsWith("0x")) {
227 			color = color.substring(2);
228 		} else if (color.startsWith("rgb")) {
229 			if (color.startsWith("(", 3)) {
230 				return parseRGBColor(color, 4, false, opacity);
231 			} else if (color.startsWith("a(", 3)) {
232 				return parseRGBColor(color, 5, true, opacity);
233 			}
234 		} else if (color.startsWith("hsl")) {
235 			if (color.startsWith("(", 3)) {
236 				return parseHSLColor(color, 4, false, opacity);
237 			} else if (color.startsWith("a(", 3)) {
238 				return parseHSLColor(color, 5, true, opacity);
239 			}
240 		} else {
241 			Color col = NamedColors.get(color);
242 			if (col != null) {
243 				if (opacity == 1.0) {
244 					return col;
245 				} else {
246 					return new Color(col.getRed(), col.getGreen(), col.getBlue(), (int)(255 * opacity + 0.5));
247 				}
248 			}
249 		}
250 
251 		int len = color.length();
252 
253 		try {
254 			int r;
255 			int g;
256 			int b;
257 			int a;
258 
259 			if (len == 3) {
260 				r = Integer.parseInt(color.substring(0, 1), 16);
261 				g = Integer.parseInt(color.substring(1, 2), 16);
262 				b = Integer.parseInt(color.substring(2, 3), 16);
263 				return new Color(r / 15.0f, g / 15.0f, b / 15.0f, (float)opacity);
264 			} else if (len == 4) {
265 				r = Integer.parseInt(color.substring(0, 1), 16);
266 				g = Integer.parseInt(color.substring(1, 2), 16);
267 				b = Integer.parseInt(color.substring(2, 3), 16);
268 				a = Integer.parseInt(color.substring(3, 4), 16);
269 				return new Color(r / 15.0f, g / 15.0f, b / 15.0f, (float)(opacity * a / 15.0));
270 			} else if (len == 6) {
271 				r = Integer.parseInt(color.substring(0, 2), 16);
272 				g = Integer.parseInt(color.substring(2, 4), 16);
273 				b = Integer.parseInt(color.substring(4, 6), 16);
274 				return new Color(r, g, b, (int)(opacity*255+0.5));
275 			} else if (len == 8) {
276 				r = Integer.parseInt(color.substring(0, 2), 16);
277 				g = Integer.parseInt(color.substring(2, 4), 16);
278 				b = Integer.parseInt(color.substring(4, 6), 16);
279 				a = Integer.parseInt(color.substring(6, 8), 16);
280 				return new Color(r, g, b, (int)(opacity * a + 0.5));
281 			}
282 		} catch (NumberFormatException nfe) {}
283 
284 		throw new IllegalArgumentException("Invalid color specification");
285 	}

287 	private static Color parseRGBColor(String color, int roff,
288 									   boolean hasAlpha, double a)
289 	{
290 		try {
291 			int rend = color.indexOf(',', roff);
292 			int gend = rend < 0 ? -1 : color.indexOf(',', rend+1);
293 			int bend = gend < 0 ? -1 : color.indexOf(hasAlpha ? ',' : ')', gend+1);
294 			int aend = hasAlpha ? (bend < 0 ? -1 : color.indexOf(')', bend+1)) : bend;
295 			if (aend >= 0) {
296 				double r = parseComponent(color, roff, rend, PARSE_COMPONENT);
297 				double g = parseComponent(color, rend+1, gend, PARSE_COMPONENT);
298 				double b = parseComponent(color, gend+1, bend, PARSE_COMPONENT);
299 				if (hasAlpha) {
300 					a *= parseComponent(color, bend+1, aend, PARSE_ALPHA);
301 				}
302 				return new Color((float)r, (float)g, (float)b, (float)a);
303 			}
304 		} catch (NumberFormatException nfe) {}
305 
306 		throw new IllegalArgumentException("Invalid color specification");
307 	}

309 	private static Color parseHSLColor(String color, int hoff,
310 									   boolean hasAlpha, double a)
311 	{
312 		try {
313 			int hend = color.indexOf(',', hoff);
314 			int send = hend < 0 ? -1 : color.indexOf(',', hend+1);
315 			int lend = send < 0 ? -1 : color.indexOf(hasAlpha ? ',' : ')', send+1);
316 			int aend = hasAlpha ? (lend < 0 ? -1 : color.indexOf(')', lend+1)) : lend;
317 			if (aend >= 0) {
318 				double h = parseComponent(color, hoff, hend, PARSE_ANGLE);
319 				double s = parseComponent(color, hend+1, send, PARSE_PERCENT);
320 				double l = parseComponent(color, send+1, lend, PARSE_PERCENT);
321 				if (hasAlpha) {
322 					a *= parseComponent(color, lend+1, aend, PARSE_ALPHA);
323 				}
324 				return hsb(h, s, l, a);
325 			}
326 		} catch (NumberFormatException nfe) {}
327 
328 		throw new IllegalArgumentException("Invalid color specification");
329 	}

331 	private static final int PARSE_COMPONENT = 0; // percent, or clamped to [0,255] => [0,1]
332 	private static final int PARSE_PERCENT = 1; // clamped to [0,100]% => [0,1]
333 	private static final int PARSE_ANGLE = 2; // clamped to [0,360]
334 	private static final int PARSE_ALPHA = 3; // clamped to [0.0,1.0]
335 	private static double parseComponent(String color, int off, int end, int type) {
336 		color = color.substring(off, end).trim();
337 		if (color.endsWith("%")) {
338 			if (type > PARSE_PERCENT) {
339 				throw new IllegalArgumentException("Invalid color specification");
340 			}
341 			type = PARSE_PERCENT;
342 			color = color.substring(0, color.length()-1).trim();
343 		} else if (type == PARSE_PERCENT) {
344 			throw new IllegalArgumentException("Invalid color specification");
345 		}
346 		double c = ((type == PARSE_COMPONENT)
347 				? Integer.parseInt(color)
348 				: Double.parseDouble(color));
349 		switch (type) {
350 			case PARSE_ALPHA:
351 				return (c < 0.0) ? 0.0 : ((c > 1.0) ? 1.0 : c);
352 			case PARSE_PERCENT:
353 				return (c <= 0.0) ? 0.0 : ((c >= 100.0) ? 1.0 : (c / 100.0));
354 			case PARSE_COMPONENT:
355 				return (c <= 0.0) ? 0.0 : ((c >= 255.0) ? 1.0 : (c / 255.0));
356 			case PARSE_ANGLE:
357 				return ((c < 0.0)
358 						? ((c % 360.0) + 360.0)
359 						: ((c > 360.0)
360 						? (c % 360.0)
361 						: c));
362 		}
363 
364 		throw new IllegalArgumentException("Invalid color specification");
365 	}

491 		int i = red;

493 		i = i | green;

1559     public static double[] HSBtoRGB(double hue, double saturation, double brightness) {
1560         // normalize the hue
1561         double normalizedHue = ((hue % 360) + 360) % 360;
1562         hue = normalizedHue/360;
1563 
1564         double r = 0, g = 0, b = 0;
1565         if (saturation == 0) {
1566             r = g = b = brightness;
1567         } else {
1568             double h = (hue - Math.floor(hue)) * 6.0;
1569             double f = h - java.lang.Math.floor(h);
1570             double p = brightness * (1.0 - saturation);
1571             double q = brightness * (1.0 - saturation * f);
1572             double t = brightness * (1.0 - (saturation * (1.0 - f)));
1573             switch ((int) h) {
1574                 case 0:
1575                     r = brightness;
1576                     g = t;
1577                     b = p;
1578                     break;
1579                 case 1:
1580                     r = q;
1581                     g = brightness;
1582                     b = p;
1583                     break;
1584                 case 2:
1585                     r = p;
1586                     g = brightness;
1587                     b = t;
1588                     break;
1589                 case 3:
1590                     r = p;
1591                     g = q;
1592                     b = brightness;
1593                     break;
1594                 case 4:
1595                     r = t;
1596                     g = p;
1597                     b = brightness;
1598                     break;
1599                 case 5:
1600                     r = brightness;
1601                     g = p;
1602                     b = q;
1603                     break;
1604             }
1605         }
1606         double[] f = new double[3];
1607         f[0] = r;
1608         f[1] = g;
1609         f[2] = b;
1610         return f;
1611     }

1613     public static double[] RGBtoHSB(double r, double g, double b) {
1614         double hue, saturation, brightness;
1615         double[] hsbvals = new double[3];
1616         double cmax = (r > g) ? r : g;
1617         if (b > cmax) cmax = b;
1618         double cmin = (r < g) ? r : g;
1619         if (b < cmin) cmin = b;
1620 
1621         brightness = cmax;
1622         if (cmax != 0)
1623             saturation = (double) (cmax - cmin) / cmax;
1624         else
1625             saturation = 0;
1626 
1627         if (saturation == 0) {
1628             hue = 0;
1629         } else {
1630             double redc = (cmax - r) / (cmax - cmin);
1631             double greenc = (cmax - g) / (cmax - cmin);
1632             double bluec = (cmax - b) / (cmax - cmin);
1633             if (r == cmax)
1634                 hue = bluec - greenc;
1635             else if (g == cmax)
1636                 hue = 2.0 + redc - bluec;
1637             else
1638                 hue = 4.0 + greenc - redc;
1639             hue = hue / 6.0;
1640             if (hue < 0)
1641                 hue = hue + 1.0;
1642         }
1643         hsbvals[0] = hue * 360;
1644         hsbvals[1] = saturation;
1645         hsbvals[2] = brightness;
1646         return hsbvals;
1647     }
