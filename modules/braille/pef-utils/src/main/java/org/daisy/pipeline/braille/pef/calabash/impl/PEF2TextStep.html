88  	@Override
89  	public void run() throws SaxonApiException {
90  		super.run();
91  		MutableQuery q = mutableQuery(query(getOption(_file_format, "")));
92  		q.removeAll("blank-last-page"); // has been handled in pef2text.xpl
93  		q.removeAll("sheets-multiple-of-two"); // has been handled in pef2text.xpl
94  		addOption(_line_breaks, q);
95  		addOption(_page_breaks, q);
96  		addOption(_pad, q);
97  		addOption(_charset, q);
98  		logger.debug("Finding file format for query: " + q);
99  		Iterable<FileFormat> fileFormats = fileFormatRegistry.get(q);
100 		if (!fileFormats.iterator().hasNext()) {
101 			throw new XProcException(step, "No file format found for query: " + q); }
102 		for (FileFormat fileFormat : fileFormats) {
103 			try {
104 				logger.debug("Storing PEF to file format: " + fileFormat);
105 				
106 				// Initialize output directory
107 				File textDir = new File(new URI(getOption(_output_dir).getString()));
108 				textDir.mkdirs();
109 				
110 				// Read source PEF
111 				ByteArrayOutputStream s = new ByteArrayOutputStream();
112 				Serializer serializer = runtime.getProcessor().newSerializer();
113 				serializer.setOutputStream(s);
114 				serializer.setCloseOnCompletion(true);
115 				serializer.setOutputProperty(Serializer.Property.INDENT, "yes");
116 				serializer.serializeNode(source.read());
117 				serializer.close();
118 				InputStream pefStream = new ByteArrayInputStream(s.toByteArray());
119 				s.close();
120 				
121 				// Parse pattern
122 				String singleVolumeName = getOption(_single_volume_name, "");
123 				String pattern = getOption(_name_pattern, "");
124 				if (pattern.isEmpty())
125 					pattern = "volume-{}";
126 				int match = pattern.indexOf("{}");
127 				if (match < 0 || match != pattern.lastIndexOf("{}")) {
128 					logger.error("name-pattern is invalid: '" + pattern + "'");
129 					if (singleVolumeName.isEmpty())
130 						throw new RuntimeException("name-pattern and single-volume-name may not both be empty");
131 				}
132 				if ((fileFormat.supportsVolumes() && !singleVolumeName.isEmpty())
133 				    || match < 0 || match != pattern.lastIndexOf("{}")) {
134 					// Output to single file
135 					convertPEF2Text(pefStream,
136 							new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
137 				} else {
138 					// Split PEF
139 					pattern = pattern.replaceAll("'", "''")
140 							.replaceAll("([0#\\.,;%\u2030\u00A4-]+)", "'$1'");
141 					// Recalculate after replacement
142 					match = pattern.indexOf("{}");
143 					File splitDir = new File(textDir, "split");
144 					splitDir.mkdir();
145 					// FIXME: to validating the result PEFs, get a PEFValidator instance
146 					// (implemented in dotify.task.impl) through the streamline API.
147 					PEFFileSplitter splitter = new PEFFileSplitter(x -> true);
148 					String prefix = PEFFileSplitter.PREFIX;
149 					String postfix = PEFFileSplitter.POSTFIX;
150 					splitter.split(pefStream, splitDir, prefix, postfix);
151 					File[] pefFiles = splitDir.listFiles();
152 					String formatPattern = pattern.substring(0, match);
153 					int nWidth; {
154 						try {
155 							nWidth = Integer.parseInt(getOption(_number_width, "")); }
156 						catch (NumberFormatException e) {
157 							nWidth = 0; }}
158 					if (nWidth == 0)
159 						formatPattern += "###"; // Assume max 999 volumes
160 					else
161 						while (nWidth > 0) { formatPattern += "0"; nWidth--; }
162 					formatPattern += pattern.substring(match + 2);
163 					NumberFormat format = new DecimalFormat(formatPattern);
164 					for (File pefFile : pefFiles) {
165 						InputStream is = new FileInputStream(pefFile);
166 						if (pefFiles.length == 1 && !singleVolumeName.isEmpty()) {
167 							// Output to single file
168 							convertPEF2Text(is, new File(textDir, singleVolumeName + fileFormat.getFileExtension()), fileFormat);
169 						} else {
170 							String pefName = pefFile.getName();
171 							if (pefName.length() <= prefix.length() + postfix.length()
172 							    || !pefName.substring(0, prefix.length()).equals(prefix)
173 							    || !pefName.substring(pefName.length() - postfix.length()).equals(postfix)) {
174 								is.close();
175 								throw new RuntimeException("Coding error");
176 							}
177 							String textName = format.format(
178 									Integer.parseInt(pefName.substring(prefix.length(), pefName.length() - postfix.length())));
179 							convertPEF2Text(is,
180 									new File(textDir, textName + fileFormat.getFileExtension()),
181 									fileFormat);
182 						}
183 						is.close();
184 						if (!pefFile.delete()) pefFile.deleteOnExit();
185 					}
186 					pefStream.close();
187 					if (!splitDir.delete()) splitDir.deleteOnExit();
188 				}
189 				return; }
190 			catch (Exception e) {
191 				logger.error("Storing PEF to file format '" + fileFormat + "' failed", e); }}
192 		throw new XProcException(step, "pef:pef2text failed");
193 	}

195 	private void convertPEF2Text(InputStream pefStream, File textFile, FileFormat fileFormat)
196 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
197 		OutputStream textStream = new FileOutputStream(textFile);
198 		if ("pef".equals(fileFormat.getIdentifier())) {
199 
200 			// just write pefStream to textFile without parsing it
201 			byte[] buf = new byte[153600];
202 			int length;
203 			while ((length = pefStream.read(buf)) > 0)
204 				textStream.write(buf, 0, length);
205 		} else {
206 			EmbosserWriter writer = fileFormat.newEmbosserWriter(textStream);
207 			PEFHandler.Builder builder = new PEFHandler.Builder(writer);
208 			builder.range(null).align(Alignment.LEFT).offset(0);
209 			parsePefFile(pefStream, builder.build());
210 		}
211 		textStream.close();
212 	}

247 	// copied from org.daisy.braille.facade.PEFConverterFacade because it is no longer static
248 	/**
249 	 * Parses the given input stream using the supplied PEFHandler.
250 	 * @param is the input stream
251 	 * @param ph the PEFHandler
252 	 * @throws ParserConfigurationException
253 	 * @throws SAXException
254 	 * @throws IOException
255 	 * @throws UnsupportedWidthException
256 	 */
257 	private static void parsePefFile(InputStream is, PEFHandler ph)
258 			throws ParserConfigurationException, SAXException, IOException, UnsupportedWidthException {
259 		
260 		SAXParserFactory spf = SAXParserFactory.newInstance();
261 		spf.setNamespaceAware(true);
262 		SAXParser sp = spf.newSAXParser();
263 		try {
264 			sp.parse(is, ph); }
265 		catch (SAXException e) {
266 			if (ph.hasWidthError())
267 				throw new UnsupportedWidthException(e);
268 			else
269 				throw e; }
270 	}
