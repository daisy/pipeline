1   package org.daisy.pipeline.braille.pef.impl;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import com.google.common.base.Optional;
7   
8   import org.daisy.braille.api.factory.FactoryProperties;
9   import org.daisy.braille.api.table.Table;
10  
11  import org.daisy.pipeline.braille.common.Query;
12  import org.daisy.pipeline.braille.common.Query.MutableQuery;
13  import static org.daisy.pipeline.braille.common.Query.util.mutableQuery;
14  import org.daisy.pipeline.braille.pef.TableProvider;
15  
16  import org.osgi.service.component.annotations.Component;
17  import org.osgi.service.component.annotations.Reference;
18  import org.osgi.service.component.annotations.ReferenceCardinality;
19  import org.osgi.service.component.annotations.ReferencePolicy;
20  
21  // Note that although the name suggests it, this class doesn't use instances
22  // of the TableCatalog interface. It uses instances of the more low-level
23  // org.daisy.braille.api.table.TableProvider directly. The name was chosen
24  // because it more or less provides the same functionality as a TableCatalog,
25  // except that it's based on the query syntax instead of ID's.
26  @Component(
27  	name = "org.daisy.pipeline.braille.pef.impl.BrailleUtilsTableCatalog",
28  	service = { TableProvider.class }
29  )
30  public class BrailleUtilsTableCatalog implements TableProvider {
31  	
32  	public Iterable<Table> get(Query query) {
33  		MutableQuery q = mutableQuery(query);
34  		if (q.containsKey("id")) {
35  			String id = q.removeOnly("id").getValue().get();
36  			if (q.isEmpty())
37  				return get(id); }
38  		return empty;
39  	}
40  	
41  	private Iterable<Table> get(String id) {
42  		for (org.daisy.braille.api.table.TableProvider p : providers)
43  			for (FactoryProperties fp : p.list())
44  				if (fp.getIdentifier().equals(id))
45  					return Optional.fromNullable(p.newFactory(id)).asSet();
46  		return empty;
47  	}
48  	
49  	private final static Iterable<Table> empty = Optional.<Table>absent().asSet();
50  	
51  	private final List<org.daisy.braille.api.table.TableProvider> providers
52  	= new ArrayList<org.daisy.braille.api.table.TableProvider>();
53  	
54  	@Reference(
55  		name = "TableProvider",
56  		unbind = "removeTableProvider",
57  		service = org.daisy.braille.api.table.TableProvider.class,
58  		cardinality = ReferenceCardinality.MULTIPLE,
59  		policy = ReferencePolicy.DYNAMIC
60  	)
61  	public void addTableProvider(org.daisy.braille.api.table.TableProvider provider) {
62  		providers.add(provider);
63  	}
64  	
65  	public void removeTableProvider(org.daisy.braille.api.table.TableProvider provider) {
66  		providers.remove(provider);
67  	}
68  }
