4   import java.util.Collection;
5   import java.util.List;
6   import java.util.stream.Collectors;

8   import com.google.common.base.Optional;

10  import org.daisy.dotify.api.factory.FactoryProperties;
11  import org.daisy.dotify.api.table.Table;
12  
13  import org.daisy.pipeline.braille.common.Query;
14  import org.daisy.pipeline.braille.common.Query.MutableQuery;
15  import static org.daisy.pipeline.braille.common.Query.util.mutableQuery;
16  import org.daisy.pipeline.braille.pef.TableProvider;
17  
18  import org.osgi.service.component.annotations.Component;
19  import org.osgi.service.component.annotations.Reference;
20  import org.osgi.service.component.annotations.ReferenceCardinality;
21  import org.osgi.service.component.annotations.ReferencePolicy;
22  
23  // Note that although the name suggests it, this class doesn't use instances
24  // of the TableCatalog interface. It uses instances of the more low-level
25  // org.daisy.dotify.api.table.TableProvider directly. The name was chosen
26  // because it more or less provides the same functionality as a TableCatalog,
27  // except that it's based on the query syntax instead of ID's.
28  @Component(
29  	name = "org.daisy.pipeline.braille.pef.impl.BrailleUtilsTableCatalog",
30  	service = { TableProvider.class }
31  )
32  public class BrailleUtilsTableCatalog implements TableProvider {
33  	
34  	public Iterable<Table> get(Query query) {
35  		MutableQuery q = mutableQuery(query);
36  		if (q.containsKey("id")) {
37  			String id = q.removeOnly("id").getValue().get();
38  			if (q.isEmpty())
39  				return get(id); }
40  		return empty;
41  	}
42  	
43  	private Iterable<Table> get(String id) {
44  		for (org.daisy.dotify.api.table.TableProvider p : providers)
45  			for (FactoryProperties fp : p.list())
46  				if (fp.getIdentifier().equals(id))
47  					return Optional.fromNullable(p.newFactory(id)).asSet();
48  		return empty;
49  	}
50  
51  	Collection<FactoryProperties> list() {
52  		return providers.stream().flatMap(p -> p.list().stream()).collect(Collectors.toList());
53  	}
54  
55  	private final static Iterable<Table> empty = Optional.<Table>absent().asSet();
56  	
57  	private final List<org.daisy.dotify.api.table.TableProvider> providers
58  	= new ArrayList<org.daisy.dotify.api.table.TableProvider>();
59  	
60  	@Reference(
61  		name = "TableProvider",
62  		unbind = "-",
63  		service = org.daisy.dotify.api.table.TableProvider.class,
64  		cardinality = ReferenceCardinality.MULTIPLE,
65  		policy = ReferencePolicy.STATIC
66  	)
67  	public void addTableProvider(org.daisy.dotify.api.table.TableProvider provider) {
68  		providers.add(provider);
69  	}
70  	
71  	public void removeTableProvider(org.daisy.dotify.api.table.TableProvider provider) {
72  		providers.remove(provider);
73  	}
74  }
