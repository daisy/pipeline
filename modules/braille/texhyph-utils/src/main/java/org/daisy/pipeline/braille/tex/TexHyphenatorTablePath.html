1   package org.daisy.pipeline.braille.tex;
2   
3   import java.net.URI;
4   import java.net.URL;
5   import java.util.Locale;
6   import java.util.Map;
7   
8   import org.daisy.common.file.URLs;
9   import org.daisy.pipeline.braille.common.BundledResourcePath;
10  import org.daisy.pipeline.braille.common.Provider;
11  import static org.daisy.pipeline.braille.common.util.Locales.parseLocale;
12  
13  public class TexHyphenatorTablePath extends BundledResourcePath implements Provider<Locale,URI> {
14  	
15  	private static final String MANIFEST = "manifest";
16  	
17  	@Override
18  	protected void activate(Map<?,?> properties, Class<?> context) {
19  		if (properties.get(UNPACK) != null)
20  			throw new IllegalArgumentException(UNPACK + " property not supported");
21  		super.activate(properties, context);
22  		if (properties.get(MANIFEST) != null) {
23  			String manifestPath = properties.get(MANIFEST).toString();
24  			final URL manifestURL = URLs.getResourceFromJAR(manifestPath, TexHyphenatorTablePath.class);
25  			if (manifestURL == null)
26  				throw new IllegalArgumentException("Manifest at location " + manifestPath + " could not be found");
27  			initProvider(manifestURL); }
28  	}
29  	
30  	public Iterable<URI> get(Locale locale) {
31  		return provider.get(locale);
32  	}
33  	
34  	private Provider<Locale,URI> provider = Provider.util.<Locale,URI>empty();
35  	
36  	private void initProvider(URL manifestURL) {
37  		provider = new Provider.util.SimpleMappingProvider<Locale,URI>(manifestURL) {
38  			public Locale parseKey(String locale) {
39  				return parseLocale(locale);
40  			}
41  			public URI parseValue(String table) {
42  				return canonicalize(URLs.asURI(table));
43  			}
44  		};
45  	}
46  }
