80  	private final static Iterable<TexHyphenator> empty = Iterables.<TexHyphenator>empty();
81  	
82  	public Iterable<TexHyphenator> _get(Query query) {
83  		MutableQuery q = mutableQuery(query);
84  		if (q.containsKey("hyphenator")) {
85  			String v = q.removeOnly("hyphenator").getValue().get();
86  			if (!"texhyph".equals(v) && !"tex".equals(v))
87  				return fromNullable(fromId(v)); }
88  		if (q.containsKey("table")) {
89  			String v = q.removeOnly("table").getValue().get();
90  			if (!q.isEmpty()) {
91  				logger.warn("A query with both 'table' and '" + q.iterator().next().getKey() + "' never matches anything");
92  				return empty; }
93  			return fromNullable(get(URLs.asURI(v))); }
94  		Locale locale; {
95  			String loc;
96  			if (q.containsKey("document-locale"))
97  				loc = q.removeOnly("document-locale").getValue().get();
98  			else
99  				loc = "und";
100 			try {
101 				locale = parseLocale(loc); }
102 			catch (IllegalArgumentException e) {
103 				logger.error("Invalid locale", e);
104 				return empty; }
105 		}
106 		if (!q.isEmpty()) {
107 			logger.warn("A query with '" + q.iterator().next().getKey() + "' never matches anything");
108 			return empty; }
109 		if (tableRegistry != null) {
110 			return transform(
111 				tableRegistry.get(locale),
112 				new Function<URI,TexHyphenator>() {
113 					public TexHyphenator _apply(URI table) {
114 						return get(table); }}); }
115 		return empty;
116 	}

118 	private TexHyphenator get(URI table) {
119 		try {
120 			URL resolved = resolveTable(table);
121 			Properties properties = new Properties();
122 			URI base = null;
123 			if (table.toString().endsWith(".tex")) {
124 				properties.setProperty(TexHyphenatorImpl.PATTERN_PATH_KEY, URLs.asURI(resolved).toASCIIString());
125 				properties.setProperty(TexHyphenatorImpl.MODE_KEY, TexHyphenatorImpl.BYTE_MODE); }
126 			else if (table.toString().endsWith(".properties")) {
127 				base = URLs.asURI(resolved);
128 				InputStream stream = resolved.openStream();
129 				properties.load(stream);
130 				stream.close(); }
131 			else if (table.toString().endsWith(".xml")) {
132 				base = URLs.asURI(resolved);
133 				InputStream stream = resolved.openStream();
134 				properties.loadFromXML(stream);
135 				stream.close(); }
136 			else
137 				return null;
138 			try {
139 				return new TexHyphenatorImpl(properties, base); }
140 			catch (Exception e) {
141 				logger.warn("Could not create a hyphenator for properties " + properties, e); }}
142 		catch (Exception e) {
143 			logger.warn("Could not create a hyphenator for table", e); }
144 		return null;
145 	}
146 	
147 	private URL resolveTable(URI table) {
148 		URL resolvedTable = isAbsoluteFile(table) ? URLs.asURL(table) : tableRegistry.resolve(table);
149 		if (resolvedTable == null)
150 			throw new RuntimeException("Hyphenation table " + table + " could not be resolved");
151 		return resolvedTable;
152 	}

157 	private class TexHyphenatorImpl extends AbstractHyphenator implements TexHyphenator {
158 		
159 		private final static String LEFT_HYPHEN_MIN_KEY = "beginLimit";
160 		private final static String RIGHT_HYPHEN_MIN_KEY = "endLimit";
161 		private final static String ENCODING_KEY = "encoding";
162 		private final static String PATTERN_PATH_KEY = "patternPath";
163 		private final static String MODE_KEY = "mode";
164 		private final static String BYTE_MODE = "byte";
165 		private final static String CHARACTER_MODE = "character";
166 		
167 		private final URI table;
168 		private final net.davidashen.text.Hyphenator hyphenator;
169 		private final int beginLimit;
170 		private final int endLimit;
171 		
172 		private TexHyphenatorImpl(Properties props, URI base) throws IOException, TexParserException {
173 			String patternPath = props.getProperty(PATTERN_PATH_KEY);
174 			if (patternPath == null)
175 				throw new RuntimeException("Required property named '" + PATTERN_PATH_KEY + "' missing.");
176 			table = base == null ? URLs.asURI(patternPath) : URLs.resolve(base, URLs.asURI(patternPath));
177 			hyphenator = new net.davidashen.text.Hyphenator();
178 			String leftHyphenMinStr = props.getProperty(LEFT_HYPHEN_MIN_KEY);
179 			if (leftHyphenMinStr != null)
180 				beginLimit = Integer.parseInt(leftHyphenMinStr);
181 			else
182 				beginLimit = 1;
183 			String rightHyphenMinStr = props.getProperty(RIGHT_HYPHEN_MIN_KEY);
184 			if (rightHyphenMinStr != null)
185 				endLimit = Integer.parseInt(rightHyphenMinStr);
186 			else
187 				endLimit = 1;
188 			String encoding = props.getProperty(ENCODING_KEY);
189 			String modeStr = props.getProperty(MODE_KEY);
190 			if (modeStr == null)
191 				throw new RuntimeException("Required property named '" + MODE_KEY + "' missing.");
192 			else if (modeStr.equals(BYTE_MODE)) {
193 				if (encoding != null)
194 					logger.warn("Configuration problem: Encoding has no effect in byte mode.");
195 				hyphenator.loadTable(URLs.asURL(table).openStream()); }
196 			else if (modeStr.equals(CHARACTER_MODE)) {
197 				if (encoding == null)
198 					logger.warn("Configuration problem: Encoding should be set in character mode.");
199 				hyphenator.loadTable(new InputStreamReader(URLs.asURL(table).openStream(), Charset.forName(encoding))); }
200 			else
201 				throw new RuntimeException("Unrecognized mode. Allowed values are " + BYTE_MODE + " and " + CHARACTER_MODE);
202 		}

214 			public String transform(String text) {
215 				return TexHyphenatorImpl.this.transform(text);

217 			public String[] transform(String[] text) {

222 		public String transform(String text) {
223 			try {
224 				return hyphenator.hyphenate(text, beginLimit, endLimit); }
225 			catch (Exception e) {
226 				throw new RuntimeException("Error during TeX hyphenation", e); }
227 		}
228 		
229 		public String[] transform(String[] text) {
230 			String[] hyphenated = new String[text.length];
231 			for (int i = 0; i < text.length; i++)
232 				try {
233 					hyphenated[i] = hyphenator.hyphenate(text[i], beginLimit, endLimit); }
234 				catch (Exception e) {
235 					throw new RuntimeException("Error during TeX hyphenation", e); }
236 			return hyphenated;
237 		}
238 	}
