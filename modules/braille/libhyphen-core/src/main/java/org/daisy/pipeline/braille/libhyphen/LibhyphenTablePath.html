1   package org.daisy.pipeline.braille.libhyphen;

3   import java.net.URI;
4   import java.util.HashMap;
5   import java.util.Locale;
6   import java.util.Map;
7   
8   import com.google.common.base.Functions;
9   import static com.google.common.base.Functions.toStringFunction;
10  import com.google.common.base.Optional;
11  import com.google.common.base.Predicates;
12  import static com.google.common.collect.Iterables.filter;
13  
14  import org.daisy.pipeline.braille.common.BundledResourcePath;
15  import org.daisy.pipeline.braille.common.Provider;
16  import static org.daisy.pipeline.braille.common.util.Predicates.matchesGlobPattern;
17  import static org.daisy.pipeline.braille.common.util.URLs.decode;
18  import static org.daisy.pipeline.braille.common.util.URIs.asURI;
19  
20  import org.osgi.service.component.ComponentContext;
21  
22  public class LibhyphenTablePath extends BundledResourcePath implements LibhyphenTableProvider {
23  	
24  	@Override
25  	protected void activate(ComponentContext context, Map<?,?> properties) throws Exception {
26  		if (properties.get(UNPACK) != null)
27  			throw new IllegalArgumentException(UNPACK + " property not supported");
28  		Map<Object,Object> props = new HashMap<Object,Object>(properties);
29  		props.put(BundledResourcePath.UNPACK, true);
30  		super.activate(context, props);
31  	}
32  	
33  	/**
34  	 * Lookup a table based on a locale, using the fact that table names are
35  	 * always in the form `hyph_language[_COUNTRY[_variant]].dic`
36  	 */
37  	public Iterable<URI> get(Locale locale) {
38  		return provider.get(locale);
39  	}
40  	
41  	private Provider<Locale,URI> provider = new Provider.util.VaryLocale<Locale,URI>() {
42  		public Iterable<URI> _get(Locale locale) {
43  			String language = locale.getLanguage().toLowerCase();
44  			String country = locale.getCountry().toUpperCase();
45  			String variant = locale.getVariant().toLowerCase();
46  			URI resource = null;
47  			if (!"".equals(variant))
48  				resource = asURI(String.format("hyph_%s_%s_%s.dic", language, country, variant));
49  			else if (!"".equals(country))
50  				resource = asURI(String.format("hyph_%s_%s.dic", language, country));
51  			else
52  				resource = asURI(String.format("hyph_%s.dic", language));
53  			if (containsResource(resource))
54  				return Optional.of(canonicalize(resource)).asSet();
55  			return Optional.<URI>absent().asSet();
56  		}
57  		@Override
58  		public Iterable<URI> fallback(Locale locale) {
59  			return filter(
60  				listResources(),
61  				Predicates.compose(
62  					matchesGlobPattern(String.format("hyph_%s_*.dic", locale.getLanguage().toLowerCase())),
63  					Functions.compose(decode, toStringFunction())));
64  		}
65  	};
66  }
