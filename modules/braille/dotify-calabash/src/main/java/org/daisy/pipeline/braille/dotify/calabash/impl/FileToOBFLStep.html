121 	@Override
122 	public void run() throws SaxonApiException {
123 		super.run();
124 		try {
125 			File inputFile = asFile(getOption(_source).getString());
126 			Map<String, Object> params = new HashMap<String, Object>();
127 			addOption(_template, params);
128 			addOption(_rows, params);
129 			addOption(_cols, params);
130 			addOption(_innerMargin, params);
131 			addOption(_outerMargin, params);
132 			addOption(_rowgap, params);
133 			addOption(_splitterMax, params);
134 			addOption(_identifier, params);
135 			
136 			params.putAll(parameters);
137 			
138 			RuntimeValue rv = getOption(_dotifyOptions);
139 			if (rv!=null) {
140 				for (Feature f : query(rv.getString())) {
141 					String key = f.getKey();
142 					Optional<String> val = f.getValue();
143 					//if there isn't a value, just repeat the key
144 					params.put(key, val.or(key));
145 				}
146 			}
147 			
148 			params.put("page-width", toInt(params.remove("cols"))+toInt(params.get("inner-margin"))+toInt(params.get("outer-margin")));
149 			params.put("page-height", toInt(params.remove("rows")));
150 			params.put("row-spacing", new BigDecimal(1+(toInt(params.remove("rowgap"))/4d)));
151 			
152 			String locale = getOption(_locale, Locale.getDefault().toString());
153 			String outputFormat = getOption(_format, "obfl");
154 			InputStream resultStream = convert(inputFile, outputFormat, locale, params);
155 			
156 			// Write result
157 			result.write(runtime.getProcessor().newDocumentBuilder().build(new StreamSource(resultStream)));
158 			resultStream.close();
159 		} catch (Exception e) {
160 			logger.error("dotify:file-to-obfl failed", e);
161 			throw new XProcException(step.getNode(), e);
162 		}
163 	}

176 	private InputStream convert(File input, String outputFormat, String locale, Map<String, Object> params) throws TaskSystemFactoryException, TaskSystemException, IOException {
177 		
178 		// FIXME: see https://github.com/joeha480/dotify/issues/205
179 		AnnotatedFile ai = identityService.identify(input);
180 
181 		String inputFormat = getFormatString(ai);
182 		if (!supportsInputFormat(inputFormat, taskGroupFactoryService.listAll())) {
183 			logger.debug("No input factory for " + inputFormat);
184 			logger.info("Note, the following detection code has been deprected. In future versions, an exception will be thrown if this point is reached."
185 						+ " To avoid this, use the IdentifierFactory interface to implement a detector for the file type.");
186 			// attempt to detect a supported type
187 			try {
188 				if (XMLTools.isWellformedXML(ai.getFile())) {
189 					ai = DefaultAnnotatedFile.with(ai).extension("xml").build();
190 					inputFormat = ai.getExtension();
191 					logger.info("Input is well-formed xml."); }}
192 			catch (XMLToolsException e) {
193 				logger.info("File is not well-formed xml: " + ai.getFile(), e);
194 			}
195 		} else {
196 			logger.info("Found an input factory for " + inputFormat);
197 		}
198 
199 		params.put("inputFormat", inputFormat);
200 		params.put("input", ai.getFile().getAbsolutePath());
201 		TaskSystem system = newTaskSystem(inputFormat, outputFormat, locale);
202 		List<InternalTask> tasks = system.compile(params);
203 		
204 		// Create a destination file
205 		File dest = File.createTempFile("file-to-obfl", ".tmp");
206 		dest.deleteOnExit();
207 		
208 		// Run tasks
209 		TaskRunner runner = TaskRunner.withName("dotify:file-to-obfl").build();
210 		runner.runTasks(ai, dest, tasks);
211 		
212 		// Return stream
213 		return new FileInputStream(dest);
214 	}

216 	private static boolean supportsInputFormat(String inputFormat, Set<TaskGroupInformation> specs) {
217 		for (TaskGroupInformation s : specs) {
218 			if (s.getInputFormat().equals(inputFormat)) {
219 				return true;
220 			}
221 		}
222 		return false;
223 	}
224 	
225 	private static String getFormatString(AnnotatedFile f) {
226 		// FIXME: see https://github.com/joeha480/dotify/issues/205
227 
228 		if (f.getFormatName()!=null) {
229 			return f.getFormatName();
230 		} else if (f.getExtension()!=null) {
231 			return f.getExtension();
232 		} else if (f.getMediaType()!=null) {
233 			return f.getMediaType();
234 		} else {
235 			return null;
236 		}
237 	}
