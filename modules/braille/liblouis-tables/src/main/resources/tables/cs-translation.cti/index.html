# ----------------------------------------------------------------------------------------------
# Braille indicator opcodes
# ----------------------------------------------------------------------------------------------

numsign  3456
capsign  6
begcaps  56
endcaps  5

# ----------------------------------------------------------------------------------------------
# Literary digits
# ----------------------------------------------------------------------------------------------

include litdigits6Dots.uti

# ----------------------------------------------------------------------------------------------
# Decimal points, hyphens
# ----------------------------------------------------------------------------------------------

decpoint  \x002C  2
decpoint  \x002E  3
hyphen    \x002D  36

# ----------------------------------------------------------------------------------------------
# Letter prefix in numbers
# ----------------------------------------------------------------------------------------------

class    digitletter                 abcdefghij
context  $d[]%digitletter            @5
context  $d"."[]%digitletter         @5
context  $d","[]%digitletter         @5
