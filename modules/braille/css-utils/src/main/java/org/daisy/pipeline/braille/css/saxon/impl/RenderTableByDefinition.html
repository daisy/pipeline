172 		private TableAsList(String axes) {
173 			this.axes = new ArrayList<String>(AXIS_SPLITTER.splitToList(axes));
174 			if (this.axes.remove("auto"))
175 				if (!this.axes.isEmpty())
176 					throw new RuntimeException();

185 		public void transform(Iterator<BaseURIAwareXMLStreamReader> input, Supplier<BaseURIAwareXMLStreamWriter> output) {
186 			XMLStreamReader reader = Iterators.getOnlyElement(input);
187 			XMLStreamWriter writer = output.get();
188 			
189 			try {
190 			
191 			writeActionsBefore = new ArrayList<WriterEvent>();
192 			writeActionsAfter = new ArrayList<WriterEvent>();
193 			cells = new ArrayList<TableCell>();
194 			coveredCoordinates = new HashSet<CellCoordinates>();
195 			List<WriterEvent> writeActions = writeActionsBefore;
196 			int depth = 0;
197 			TableCell withinCell = null;
198 			TableCell.RowType rowType = TableCell.RowType.TBODY;
199 			int rowGroup = 1;
200 			int row = 1;
201 			int col = 1;
202 		  loop: while (true)
203 				try {
204 					switch (reader.next()) {
205 					case START_ELEMENT: {
206 						QName name = reader.getName();
207 						depth++;
208 						boolean isCell = false;
209 						if (depth == 1) {
210 							if (!isHTMLorDTBookElement(TABLE, name))
211 								throw new RuntimeException("Expected table element (html|dtb).");
212 							if (XMLNS_HTML.equals(name.getNamespaceURI()))
213 								ns = XMLNS_HTML;
214 							else if (XMLNS_DTB.equals(name.getNamespaceURI()))
215 								ns = XMLNS_DTB; }
216 						else if (isHTMLorDTBookElement(THEAD, name)) {
217 							rowType = TableCell.RowType.THEAD;
218 							// TODO: if style != default, warning that style on thead element is ignored
219 							break; }
220 						else if (isHTMLorDTBookElement(TFOOT, name)) {
221 							rowType = TableCell.RowType.TFOOT;
222 							// TODO: if style != default, warning that style on tfoot element is ignored
223 							break; }
224 						else if (isHTMLorDTBookElement(TBODY, name)) {
225 							rowType = TableCell.RowType.TBODY;
226 							// TODO: if style != default, warning that style on tbody element is ignored
227 							break; }
228 						else if (isHTMLorDTBookElement(TR, name)) {
229 							// TODO: if style != default, warning that style on tr element is ignored
230 							break; }
231 						else if (isHTMLorDTBookElement(COLGROUP, name) || isHTMLorDTBookElement(COL, name))
232 							throw new RuntimeException("Elements colgroup and col not supported yet.");
233 						else if (isHTMLorDTBookElement(TD, name) || isHTMLorDTBookElement(TH, name)) {
234 							isCell = true;
235 							withinCell = new TableCell();
236 							withinCell.row = row;
237 							withinCell.col = col;
238 							withinCell.rowGroup = rowGroup;
239 							withinCell.rowType = rowType;
240 							setCovered(row, col);
241 							cells.add(withinCell);
242 							if (isHTMLorDTBookElement(TH, name))
243 								withinCell.type = TableCell.CellType.TH;
244 							writeActions = withinCell.writeActions; }
245 						writeActions.add(w -> writeStartElement(w, name));
246 						for (int i = 0; i < reader.getNamespaceCount(); i++) {
247 							String prf = reader.getNamespacePrefix(i);
248 							String ns = reader.getNamespaceURI(i);
249 							writeActions.add(w -> w.writeNamespace(prf, ns)); }
250 						for (int i = 0; i < reader.getAttributeCount(); i++) {
251 							QName attrName = reader.getAttributeName(i);
252 							String attrValue = reader.getAttributeValue(i);
253 							if (CSS_TABLE_HEADER_POLICY.equals(attrName)) {
254 								if (isCell)
255 									if ("once".equals(attrValue))
256 										withinCell.headerPolicy = TableCell.HeaderPolicy.ONCE;
257 									else if ("always".equals(attrValue))
258 										withinCell.headerPolicy = TableCell.HeaderPolicy.ALWAYS;
259 									else if ("front".equals(attrValue))
260 										withinCell.headerPolicy = TableCell.HeaderPolicy.FRONT;
261 									else
262 										throw new RuntimeException(
263 											"Expected value once|always for table-header-policy property but got " + attrValue); }
264 							else if (isCell && _HEADERS.equals(attrName))
265 								withinCell.headers = HEADERS_SPLITTER.splitToList(attrValue);
266 							else if (isCell && _SCOPE.equals(attrName)) {
267 								if ("row".equals(attrValue))
268 									withinCell.scope = TableCell.Scope.ROW;
269 								else if ("col".equals(attrValue))
270 									withinCell.scope = TableCell.Scope.COL;
271 								else if ("colgroup".equals(attrValue) || "rowgroup".equals(attrValue))
272 									throw new RuntimeException(
273 											"Value " + attrValue + " for scope attribute not supported yet.");
274 								else
275 									throw new RuntimeException(
276 											"Expected value col|row|colgroup|rowgroup for scope attribute but got " + attrValue); }
277 							else if (isCell && _AXIS.equals(attrName))
278 								withinCell.axis = AXIS_SPLITTER.splitToList(attrValue);
279 							else if (isCell && _ROWSPAN.equals(attrName)) {
280 								int rowspan = nonNegativeInteger(attrValue);
281 								if (rowspan == 0)
282 									throw new RuntimeException("rowspan 0 not supported yet.");
283 								withinCell.rowspan = rowspan;
284 								for (int m = 1; m < rowspan; m++)
285 									for (int n = 0; n < withinCell.colspan; n++)
286 										setCovered(row + m, col + n); }
287 							else if (isCell && _COLSPAN.equals(attrName)) {
288 								int colspan = nonNegativeInteger(attrValue);
289 								if (colspan == 0)
290 									throw new RuntimeException("colspan 0 not supported yet.");
291 								withinCell.colspan = colspan;
292 								for (int m = 0; m < withinCell.rowspan; m++)
293 									for (int n = 1; n < withinCell.colspan; n++)
294 										setCovered(row + m, col + n); }
295 							
296 							// TODO: check that there are no duplicate IDs?
297 							else if (isCell && _ID.equals(attrName))
298 								withinCell.id = attrValue;
299 							else if (depth == 1 && _STYLE.equals(attrName)) {
300 								String newStyle; {
301 									InlineStyle style = new InlineStyle(attrValue);
302 									String builder = null;
303 									for (RuleBlock<?> block : style) {
304 										if (block instanceof RuleMainBlock)
305 											builder = joinRuleBlocks(builder, serializeRuleBlock(style.getMainStyle()));
306 										else if (block instanceof RuleRelativeBlock) {
307 											RuleRelativeBlock ruleblock = (RuleRelativeBlock)block;
308 											List<Selector> selector = ruleblock.getSelector();
309 											if (selector.size() > 0) { // should always be true
310 												// note that in the cases "&::list-item", "&::list-header" and "&::table-by(...)" we
311 												// are ignoring any following selector parts except those that are "stacked" onto the
312 												// pseudo element
313 												if (selector.get(0).size() > 0) { // should always be true
314 													if (selector.get(0).get(0) instanceof PseudoElementImpl) {
315 														// selector.get(0).size() should always be 1
316 														PseudoElementImpl pseudo = (PseudoElementImpl)selector.get(0).get(0);
317 														// selector size should be 1
318 														if ("list-item".equals(pseudo.getName()))
319 															addListItemStyle(
320 																pseudo.getPseudoClasses(),
321 																new ListItemStyle(pseudo.getStackedPseudoElement(), ruleblock));
322 														else if ("list-header".equals(pseudo.getName())) {
323 															if (pseudo.getPseudoClasses().isEmpty())
324 																addListHeaderStyle(
325 																	new ListItemStyle(pseudo.getStackedPseudoElement(), ruleblock)); }
326 														else if ("table-by".equals(pseudo.getName())) {
327 															String axis = pseudo.getArguments()[0];
328 															if (pseudo.getPseudoClasses().isEmpty()) {
329 																if (pseudo.hasStackedPseudoElement()) {
330 																	pseudo = pseudo.getStackedPseudoElement();
331 																	if ("list-item".equals(pseudo.getName()))
332 																		getTableByStyle(axis).addListItemStyle(
333 																			pseudo.getPseudoClasses(),
334 																			new ListItemStyle(pseudo.getStackedPseudoElement(), ruleblock));
335 																	else if ("list-header".equals(pseudo.getName())) {
336 																		if (pseudo.getPseudoClasses().isEmpty())
337 																			getTableByStyle(axis).addListHeaderStyle(
338 																				new ListItemStyle(pseudo.getStackedPseudoElement(), ruleblock)); }
339 																	else
340 																		getTableByStyle(axis).addRuleBlock(pseudo, ruleblock); }
341 																else
342 																	getTableByStyle(axis).addRuleBlock(ruleblock); }}
343 														else
344 															builder = joinRuleBlocks(builder, serializeRuleBlock(ruleblock)); }
345 													else
346 														builder = joinRuleBlocks(builder, serializeRuleBlock(ruleblock)); }
347 												else
348 													builder = joinRuleBlocks(builder, serializeRuleBlock(ruleblock)); }}
349 										else
350 											throw new RuntimeException("Unexpected style " + block); }
351 									newStyle = builder; }
352 								if (newStyle != null)
353 									writeActions.add(w -> writeAttribute(w, attrName, newStyle)); }
354 							else
355 								writeActions.add(w -> writeAttribute(w, attrName, attrValue)); }
356 						break; }
357 					case CHARACTERS:
358 						String chars = reader.getText();
359 						writeActions.add(w -> w.writeCharacters(chars));
360 						break;
361 					case END_ELEMENT: {
362 						QName name = reader.getName();
363 						depth--;
364 						if (isHTMLorDTBookElement(THEAD, name)
365 						    || isHTMLorDTBookElement(TFOOT, name)
366 						    || isHTMLorDTBookElement(TBODY, name)) {
367 							rowGroup++;
368 							break; }
369 						else if (isHTMLorDTBookElement(TR, name)) {
370 							row++;
371 							col = 1;
372 							while (isCovered(row, col)) col++;
373 							break; }
374 						writeActions.add(w -> w.writeEndElement());
375 						if (isHTMLorDTBookElement(TD, name) || isHTMLorDTBookElement(TH, name)) {
376 							withinCell = null;
377 							writeActions = writeActionsAfter;
378 							while (isCovered(row, col)) col++; }
379 						break; }
380 					case END_DOCUMENT:
381 						break loop; }}
382 				catch (NoSuchElementException e) {
383 					break; }
384 			
385 			// handle colspan and rowspan on data cells by simply splitting them into several identical ones for now
386 			List<TableCell> moreCells = new ArrayList<TableCell>();
387 			for (TableCell c : cells)
388 				if (!isHeader(c)) {
389 					if (c.rowspan > 1) {
390 						int span = c.rowspan;
391 						c.rowspan = 1;
392 						for (int i = 1; i < span; i++) {
393 							TableCell dup = c.clone();
394 							dup.row = c.row + i;
395 							moreCells.add(dup); }}
396 					if (c.colspan > 1) {
397 						int span = c.colspan;
398 						c.colspan = 1;
399 						for (int i = 1; i < span; i++) {
400 							TableCell dup = c.clone();
401 							dup.col = c.col + i;
402 							moreCells.add(dup); }}}
403 			cells.addAll(moreCells);
404 			
405 			// rearrange row groups and order cells by row
406 			sort(cells, compose(sortByRowType, sortByRow, sortByColumn));
407 			rowGroup = 0;
408 			row = 0;
409 			int newRowGroup = rowGroup = 0;
410 			int newRow = row = 0;
411 			for (TableCell c : cells) {
412 				if (c.rowGroup != rowGroup) {
413 					rowGroup = c.rowGroup;
414 					newRowGroup++; }
415 				if (c.row != row) {
416 					row = c.row;
417 					newRow++; }
418 				c.rowGroup = newRowGroup;
419 				c.row = newRow; }
420 			
421 			writer.writeStartDocument();
422 			write(writer);
423 			writer.writeEndDocument();
424 			
425 			} catch (XMLStreamException e) {
426 				throw new RuntimeException(e);
427 			}
428 		}

430 		private boolean isHTMLorDTBookElement(String element, QName name) {
431 			return ((XMLNS_HTML.equals(name.getNamespaceURI())
432 			        || XMLNS_DTB.equals(name.getNamespaceURI()))
433 			        && name.getLocalPart().equalsIgnoreCase(element));
434 		}

436 		private void setCovered(int row, int col) {
437 			CellCoordinates coords = new CellCoordinates(row, col);
438 			if (coveredCoordinates.contains(coords))
439 				throw new RuntimeException("Table structure broken: cells overlap");
440 			coveredCoordinates.add(coords);
441 		}
442 		
443 		private boolean isCovered(int row, int col) {
444 			return coveredCoordinates.contains(new CellCoordinates(row, col));
445 		}

447 		private void write(XMLStreamWriter writer) throws XMLStreamException {
448 			for (WriterEvent action : writeActionsBefore)
449 				action.writeTo(writer);
450 			List<TableCell> dataCells = new ArrayList<TableCell>();
451 			for (TableCell c : cells)
452 				if (!isHeader(c))
453 					dataCells.add(c);
454 			new TableCellGroup(dataCells, axes.iterator()).write(writer);
455 			for (WriterEvent action : writeActionsAfter)
456 				action.writeTo(writer);
457 		}

467 		private class SingleTableCell extends TableCellCollection {
468 			
469 			private final TableCell cell;
470 			private final TableCellGroup parent;
471 			private final SingleTableCell precedingSibling;
472 			private final boolean columnAppliedBeforeRow;
473 			
474 			private SingleTableCell(TableCell cell, TableCellGroup parent, SingleTableCell precedingSibling, boolean columnAppliedBeforeRow) {
475 				this.cell = cell;
476 				this.parent = parent;
477 				this.precedingSibling = precedingSibling;
478 				this.columnAppliedBeforeRow = columnAppliedBeforeRow;
479 			}
480 			
481 			private List<TableCell> newlyAppliedHeaders;
482 			public List<TableCell> newlyAppliedHeaders() {
483 				if (newlyAppliedHeaders == null) {
484 					newlyAppliedHeaders = new ArrayList<TableCell>();
485 					for (TableCell h : findHeaders(cell, !columnAppliedBeforeRow))
486 						if (!parent.appliedHeaders().contains(h))
487 							newlyAppliedHeaders.add(h); }
488 				return newlyAppliedHeaders;
489 			}
490 			
491 			private List<TableCell> newlyRenderedOrPromotedHeaders;
492 			private List<TableCell> newlyRenderedOrPromotedHeaders() {
493 				if (newlyRenderedOrPromotedHeaders == null) {
494 					newlyRenderedOrPromotedHeaders = new ArrayList<TableCell>();
495 					Iterator<TableCell> lastAppliedHeaders = (
496 						precedingSibling != null ? precedingSibling.newlyAppliedHeaders() : emptyList
497 					).iterator();
498 					boolean canOmit = true;
499 					for (TableCell h : parent.deferredHeaders()) {
500 						newlyRenderedOrPromotedHeaders.add(h);
501 						canOmit = false; }
502 					for (TableCell h : newlyAppliedHeaders()) {
503 						if (canOmit
504 						    && h.headerPolicy != TableCell.HeaderPolicy.ALWAYS
505 						    && lastAppliedHeaders.hasNext() && lastAppliedHeaders.next().equals(h))
506 							continue;
507 						newlyRenderedOrPromotedHeaders.add(h);
508 						canOmit = false; }}
509 				return newlyRenderedOrPromotedHeaders;
510 			}
511 			
512 			private List<TableCell> newlyPromotedHeaders;
513 			public List<TableCell> newlyPromotedHeaders() {
514 				if (newlyPromotedHeaders == null) {
515 					newlyPromotedHeaders = new ArrayList<TableCell>();
516 					int i = newlyRenderedOrPromotedHeaders().size() - 1;
517 					while (i >= 0 && newlyRenderedOrPromotedHeaders().get(i).headerPolicy != TableCell.HeaderPolicy.FRONT)
518 						i--;
519 					while (i >= 0)
520 						newlyPromotedHeaders.add(0, newlyRenderedOrPromotedHeaders().get(i--)); }
521 				return newlyPromotedHeaders;
522 			}

581 				String firstAxis = axes.hasNext() ? axes.next() : null;
582 				List<String> nextAxes = ImmutableList.copyOf(axes);
583 				if (firstAxis != null) {
584 					Map<TableCell,List<TableCell>> categories = new LinkedHashMap<TableCell,List<TableCell>>();
585 					List<TableCell> uncategorized = null;
586 					for (TableCell c : cells) {
587 						boolean categorized = false;
588 						for (TableCell h : findHeaders(c, !columnAppliedBeforeRow))
589 							if (h.axis != null && h.axis.contains(firstAxis)) {
590 								List<TableCell> category = categories.get(h);
591 								if (category == null) {
592 									category = new ArrayList<TableCell>();
593 									categories.put(h, category); }
594 								category.add(c);

596 						if (!categorized) {
597 							if (uncategorized == null)
598 								uncategorized = new ArrayList<TableCell>();

600 					if (!categories.isEmpty()) {
601 						List<TableCellCollection> children = new ArrayList<TableCellCollection>();
602 						TableCellGroup child = null;
603 						for (TableCell h : categories.keySet()) {

605 							                           rowApplied, columnAppliedBeforeRow);
606 							children.add(child); }
607 						if (uncategorized != null) {
608 							child = new TableCellGroup(uncategorized, nextAxes.iterator(), this, null, firstAxis, child,
609 							                           rowApplied, columnAppliedBeforeRow);
610 							children.add(child); }
611 						return children; }
612 					else if ("row".equals(firstAxis)) {
613 						List<TableCellCollection> children = new ArrayList<TableCellCollection>();
614 						TableCellGroup child = null;
615 						Map<Integer,List<TableCell>> rows = new LinkedHashMap<Integer,List<TableCell>>();
616 						for (TableCell c : cells) {
617 							List<TableCell> row = rows.get(c.row);
618 							if (row == null) {
619 								row = new ArrayList<TableCell>();
620 								rows.put(c.row, row); }
621 							row.add(c); }
622 						for (List<TableCell> row : rows.values()) {
623 							child = new TableCellGroup(row, nextAxes.iterator(), this, null, firstAxis, child, true, columnAppliedBeforeRow);
624 							children.add(child); }
625 						return children; }
626 					else if ("column".equals(firstAxis) || "col".equals(firstAxis)) {
627 						List<TableCellCollection> children = new ArrayList<TableCellCollection>();
628 						TableCellGroup child = null;
629 						Map<Integer,List<TableCell>> columns = new LinkedHashMap<Integer,List<TableCell>>();
630 						for (TableCell c : cells) {
631 							List<TableCell> column = columns.get(c.col);
632 							if (column == null) {
633 								column = new ArrayList<TableCell>();
634 								columns.put(c.col, column); }
635 							column.add(c); }
636 						for (List<TableCell> column : columns.values()) {
637 							child = new TableCellGroup(column, nextAxes.iterator(), this, null, firstAxis, child, rowApplied, !rowApplied);
638 							children.add(child); }
639 						return children; }
640 					else if ("row-group".equals(firstAxis)) {
641 						List<TableCellCollection> children = new ArrayList<TableCellCollection>();
642 						TableCellGroup child = null;
643 						Map<Integer,List<TableCell>> rowGroups = new LinkedHashMap<Integer,List<TableCell>>();
644 						for (TableCell c : cells) {
645 							List<TableCell> rowGroup = rowGroups.get(c.rowGroup);
646 							if (rowGroup == null) {
647 								rowGroup = new ArrayList<TableCell>();
648 								rowGroups.put(c.rowGroup, rowGroup); }
649 							rowGroup.add(c); }
650 						for (List<TableCell> rowGroup : rowGroups.values()) {
651 							child = new TableCellGroup(rowGroup, nextAxes.iterator(), this, null, firstAxis, child,
652 							                           rowApplied, columnAppliedBeforeRow);
653 							children.add(child); }
654 						return children; }
655 					else
656 						return groupCellsBy(cells, nextAxes.iterator()); }
657 				else {
658 					List<TableCellCollection> children = new ArrayList<TableCellCollection>();
659 					SingleTableCell child = null;
660 					for (TableCell c : cells) {
661 						child = new SingleTableCell(c, this, child, columnAppliedBeforeRow);
662 						children.add(child); }
663 					return children; }
664 			}

769 				try {
770 					List<List<TableCell>> promotedHeaders = null;
771 					int i = 0;
772 					for (TableCellCollection c : children) {
773 						if (c instanceof TableCellGroup) {
774 							TableCellGroup g = (TableCellGroup)c;
775 							int j = 0;
776 							for (TableCellCollection cc : g.children) {
777 								if (!cc.newlyPromotedHeaders().isEmpty()) {
778 									if (promotedHeaders == null) {
779 										if (i == 0 && j == 0) {
780 											writeStartElement(writer, new QName(ns, "_"));
781 											writeStyleAttribute(writer,
782 											                    groupingAxis != null ? getTableByStyle(groupingAxis).getListHeaderStyle()
783 											                                         : getListHeaderStyle());
784 											writeStartElement(writer, new QName(ns, "_"));
785 											writeStyleAttribute(writer, getTableByStyle(g.groupingAxis));
786 											promotedHeaders = new ArrayList<List<TableCell>>(); }
787 										else
788 											throw new RuntimeException("Some headers of children promoted but not all children have a promoted header."); }
789 									if (i == 0) {
790 										writeStartElement(writer, new QName(ns, "_"));
791 										Predicate<PseudoClass> matcher = matchesPosition(j + 1, g.children.size());
792 										writeStyleAttribute(writer, getTableByStyle(g.groupingAxis).getListItemStyle(matcher));
793 										for (TableCell h : cc.newlyPromotedHeaders())
794 											h.write(writer);
795 										writer.writeEndElement();
796 										promotedHeaders.add(cc.newlyPromotedHeaders()); }
797 									else if (!promotedHeaders.get(j).equals(cc.newlyPromotedHeaders()))
798 										throw new RuntimeException("Headers of children promoted but not the same as promoted headers of sibling groups."); }
799 								else if (promotedHeaders != null)
800 									throw new RuntimeException("Some headers of children promoted but not all children have a promoted header.");
801 								j++; }
802 							if (promotedHeaders != null && promotedHeaders.size() != j) {
803 								throw new RuntimeException("Headers of children promoted but not the same as promoted headers of sibling groups."); }}
804 						else if (promotedHeaders != null)
805 							throw new RuntimeException("Coding error");
806 						i++; }
807 					if (promotedHeaders != null) {
808 						writer.writeEndElement();
809 						writer.writeEndElement(); }
810 					i = 0;
811 					for (TableCellCollection c : children) {
812 						writeStartElement(writer, new QName(ns, "_"));
813 						Predicate<PseudoClass> matcher = matchesPosition(i + 1, children.size());
814 						writeStyleAttribute(writer,
815 						                    groupingAxis != null ? getTableByStyle(groupingAxis).getListItemStyle(matcher)
816 						                                         : getListItemStyle(matcher));
817 						for (TableCell h : c.newlyRenderedHeaders())
818 							h.write(writer);
819 						if (c instanceof TableCellGroup) {
820 							TableCellGroup g = (TableCellGroup)c;
821 							writeStartElement(writer, new QName(ns, "_"));
822 							writeStyleAttribute(writer, getTableByStyle(g.groupingAxis)); }
823 						c.write(writer);
824 						if (c instanceof TableCellGroup)
825 							writer.writeEndElement();
826 						writer.writeEndElement();
827 						i++; }}
828 				catch (XMLStreamException e) {
829 					throw new RuntimeException(e); }
830 			}

991 		// see https://www.w3.org/TR/REC-html40/struct/tables.html#h-11.4.3
992 		private List<TableCell> findHeaders(TableCell cell, boolean firstLeftThenUpward) {
993 			List<TableCell> headers = new ArrayList<TableCell>();
994 			if (isHeader(cell))
995 				headers.add(cell);
996 			findHeaders(headers, 0, cell, firstLeftThenUpward);
997 			return headers;
998 		}

1000 		private int findHeaders(List<TableCell> headers, int index, TableCell cell, boolean firstLeftThenUpward) {
1001 			
1002 			// headers attribute
1003 			if (cell.headers != null) {
1004 				for (String id : cell.headers)
1005 					index = recurAddHeader(headers, index, getById(id), firstLeftThenUpward);
1006 				return index; }
1007 			
1008 			// scope attribute can be used instead of headers (they should not be used in same table)
1009 			List<TableCell> rowHeaders = new ArrayList<TableCell>();
1010 			List<TableCell> colHeaders = new ArrayList<TableCell>();
1011 			for (TableCell h : cells)
1012 				if (h != cell && h.scope != null) {
1013 					switch (h.scope) {
1014 					case ROW:
1015 						if (h.row <= (cell.row + cell.rowspan - 1) && cell.row <= (h.row + h.rowspan - 1))
1016 							rowHeaders.add(h);
1017 						break;
1018 					case COL:
1019 						if (h.col <= (cell.col + cell.colspan - 1) && cell.col <= (h.col + h.colspan - 1))
1020 							colHeaders.add(h);
1021 						break; }}
1022 			sort(rowHeaders, sortByColumnAndThenRow);
1023 			for (TableCell h : rowHeaders)
1024 				index = recurAddHeader(headers, index, h, firstLeftThenUpward);
1025 			sort(colHeaders, sortByRowAndThenColumn);
1026 			for (TableCell h : colHeaders)
1027 				index = recurAddHeader(headers, index, h, firstLeftThenUpward);
1028 			
1029 			if (!isHeader(cell)) {
1030 				int direction = (firstLeftThenUpward ? 0 : 1);
1031 			  outer: while (true) {
1032 					switch (direction) {
1033 					case 0: { // search left from the cell's position to find row header cells
1034 						int k = 0;
1035 						for (int i = 0; i < cell.rowspan; i++)
1036 							for (int j = cell.col - 1; j > 0;) {
1037 								boolean foundHeader = false;
1038 								TableCell c = getByCoordinates(cell.row + i, j);
1039 								if (c != null && isHeader(c)) {
1040 									foundHeader = true;
1041 									if (c.scope == null) {
1042 										int l = recurAddHeader(headers, index, c, firstLeftThenUpward) - index;
1043 										k += l;
1044 										if (l > 1)
1045 											break; }}
1046 								else if (foundHeader)
1047 									break;
1048 								if (c == null)
1049 									j--;
1050 								else
1051 									j = j - c.colspan; }
1052 						index += k;
1053 						if (!firstLeftThenUpward)
1054 							break outer; }
1055 					case 1: { // search upwards from the cell's position to find column header cells
1056 						int k = 0;
1057 						for (int i = 0; i < cell.colspan; i++)
1058 							for (int j = cell.row - 1; j > 0;) {
1059 								boolean foundHeader = false;
1060 								TableCell c = getByCoordinates(j, cell.col + i);
1061 								if (c != null && isHeader(c)) {
1062 									foundHeader = true;
1063 									if (c.scope == null) {
1064 										int l = recurAddHeader(headers, index, c, firstLeftThenUpward) - index;
1065 										k += l;
1066 										if (l > 1)
1067 											break; }}
1068 								else if (foundHeader)
1069 									break;
1070 								if (c == null)
1071 									j--;
1072 								else
1073 									j = j - c.rowspan; }
1074 						index += k;
1075 						if (!firstLeftThenUpward) {
1076 							direction--;
1077 							continue outer; }
1078 						break outer; }}}}
1079 			
1080 			return index;
1081 		}

1083 		private int recurAddHeader(List<TableCell> headers, int index, TableCell header, boolean firstLeftThenUpward) {
1084 			if (headers.contains(header))
1085 				throw new RuntimeException();
1086 			else {
1087 				headers.add(index, header);
1088 				index = findHeaders(headers, index, header, firstLeftThenUpward);
1089 				index ++; }
1090 			return index;
1091 		}

1336 		StringBuilder b = new StringBuilder();
1337 		boolean first = true;
1338 		for (Selector s : rule.getSelector()) {
1339 			Combinator c = s.getCombinator();
1340 			if (first) {

1342 					b.append("&");
1343 				else if (c != Combinator.CHILD)
1344 					b.append(c.value());
1345 				first = false;
1346 			} else if (c != null) // should always be true
1347 				b.append(c.value());
1348 			b = OutputUtil.appendList(b, s, OutputUtil.EMPTY_DELIM);
1349 		}
1350 		b.append(" { ");
1351 		b.append(serializeRuleBlock(rule, null));
1352 		b.append(" }");
1353 		return b.toString();

1357 		String b = null;

1361 					b = r;
1362 				else {
1363 					if (!(b.endsWith("}") || b.endsWith(";")))
1364 						b = b + ";";
1365 					b += " ";
1366 					b += r; }
1367 		return b;
