26  /*
27   * Translator that dispatches to sub-translator based on text-transform values.
28   */
29  public class CompoundTranslator extends AbstractBrailleTranslator {
30  
31  	private final Map<String,Supplier<BrailleTranslator>> translators;
32  
33  	public CompoundTranslator(BrailleTranslator mainTranslator, Map<String,Supplier<BrailleTranslator>> subTranslators) {
34  		if (subTranslators.containsKey("auto") || mainTranslator == null)
35  			throw new IllegalArgumentException();
36  		translators = ImmutableMap.<String,Supplier<BrailleTranslator>>builder()
37  		                          .put("auto", () -> mainTranslator)
38  		                          .putAll(subTranslators)
39  		                          .build();
40  	}

42  	private static abstract class TransformImpl<T> {
43  
44  		abstract Iterable<T> transform(Iterable<CSSStyledText> styledText, int from, int to, String textTransform);
45  
46  		abstract boolean supports(String textTransform);
47  
48  		List<T> transform(Iterable<CSSStyledText> styledText, int from, int to) {
49  			if (to < 0)
50  				to = Iterables.size(styledText);
51  			if (from < 0 || from > to)
52  				throw new IndexOutOfBoundsException();
53  			List<T> transformed = new ArrayList<>();
54  			if (from == to) return transformed;
55  			List<CSSStyledText> buffer = new ArrayList<CSSStyledText>();
56  			String curTextTransform = "auto";
57  			for (CSSStyledText st : styledText) {
58  				SimpleInlineStyle style = st.getStyle();
59  				String textTransform; {
60  					textTransform = "auto";
61  					if (style != null) {
62  						CSSProperty val = style.getProperty("text-transform");
63  						if (val != null) {
64  							if (val == TextTransform.list_values) {
65  								TermList values = style.getValue(TermList.class, "text-transform");
66  								
67  								// According to the spec values should be "applied" from left to right, and
68  								// values of inner elements always come before values of outer elements (see
69  								// http://braillespecs.github.io/braille-css/#the-text-transform-property).
70  								// The way it works here is that the sub-translator that maps to the first
71  								// text-transform value (starting from left) is used to translate the segment,
72  								// and all values to the right are ignored.
73  								Iterator<Term<?>> it = values.iterator();
74  								while (it.hasNext()) {
75  									String tt = ((TermIdent)it.next()).getValue();
76  									if (supports(tt)) {
77  										textTransform = tt;
78  										it.remove();
79  										while (it.hasNext()) {
80  											it.next();
81  											it.remove(); }
82  										break; }}
83  								if (values.isEmpty())
84  									style.removeProperty("text-transform"); }}}
85  				}
86  				if (textTransform != curTextTransform && !buffer.isEmpty()) {
87  					if (from < buffer.size())
88  						for (T s : transform(buffer, from, to < buffer.size() ? to : -1, curTextTransform))
89  							transformed.add(s);
90  					from -= buffer.size();
91  					if (from < 0) from = 0;
92  					if (to > 0) {
93  						to -= buffer.size();
94  						if (to <= 0)
95  							return transformed; }
96  					buffer = new ArrayList<CSSStyledText>(); }
97  				curTextTransform = textTransform;
98  				buffer.add(st); }
99  			if (!buffer.isEmpty() && from < buffer.size())
100 				for (T s : transform(buffer, from, to < buffer.size() ? to : -1, curTextTransform))
101 					transformed.add(s);
102 			return transformed;
103 		}
104 	}

111 	private final FromStyledTextToBraille fromStyledTextToBraille = new FromStyledTextToBraille() {
112 		TransformImpl<String> impl = new TransformImpl<String>() {
113 			Iterable<String> transform(Iterable<CSSStyledText> styledText, int from, int to, String textTransform) {
114 				return translators.get(textTransform).get().fromStyledTextToBraille().transform(styledText, from, to);
115 			}
116 			boolean supports(String textTransform) {
117 				if (translators.containsKey(textTransform))
118 					try {
119 						translators.get(textTransform).get();
120 						return true;
121 					} catch (NoSuchElementException e) {
122 					}
123 				return false;
124 			}
125 		};
126 		public Iterable<String> transform(Iterable<CSSStyledText> styledText, int from, int to) {
127 			return impl.transform(styledText, from, to);
128 		}
129 	};

157 	private static BrailleTranslator.LineIterator concatLineIterators(List<BrailleTranslator.LineIterator> iterators) {
158 		if (iterators.size() == 0)
159 			return new BrailleTranslator.LineIterator() {
160 				public String nextTranslatedRow(int limit, boolean force, boolean wholeWordsOnly) {
161 					return "";
162 				}
163 				public String getTranslatedRemainder() {
164 					return "";
165 				}
166 				public int countRemaining() {
167 					return 0;
168 				}
169 				public boolean hasNext() {
170 					return false;
171 				}
172 				public BrailleTranslator.LineIterator copy() {
173 					return this;
174 				}
175 				public boolean supportsMetric(String metric) {
176 					return false;
177 				}
178 				public double getMetric(String metric) {
179 					throw new UnsupportedMetricException("Metric not supported: " + metric);
180 				}
181 			};
182 		else if (iterators.size() == 1 && iterators.get(0) != null)
183 			return iterators.get(0);
184 		else
185 			return new ConcatLineIterators(iterators);
186 	}

188 	private static class ConcatLineIterators implements BrailleTranslator.LineIterator {
189 
190 		final List<BrailleTranslator.LineIterator> iterators;
191 		BrailleTranslator.LineIterator current;
192 		int currentIndex = 0;
193 
194 		ConcatLineIterators(List<BrailleTranslator.LineIterator> iterators) {
195 			this.iterators = iterators;
196 			currentIndex = -1;
197 			current = null;
198 			computeCurrent();
199 		}
200 
201 		void computeCurrent() {
202 			while (current == null || !current.hasNext())
203 				if (currentIndex + 1 < iterators.size())
204 					current = iterators.get(++currentIndex);
205 				else {
206 					current = null;
207 					break; }
208 		}
209 
210 		public String nextTranslatedRow(int limit, boolean force, boolean wholeWordsOnly) {
211 			String row = "";
212 			while (limit > row.length()) {
213 				if (current == null) break;
214 				row += current.nextTranslatedRow(limit - row.length(), force, wholeWordsOnly);
215 				computeCurrent(); }
216 			return row;
217 		}
218 
219 		public String getTranslatedRemainder() {
220 			String remainder = "";
221 			if (current == null) return remainder;
222 			for (int i = currentIndex; i < iterators.size(); i++)
223 				if (iterators.get(i) != null)
224 					remainder += iterators.get(i).getTranslatedRemainder();
225 			return remainder;
226 		}
227 
228 		public int countRemaining() {
229 			int remaining = 0;
230 			if (current == null) return remaining;
231 			for (int i = currentIndex; i < iterators.size(); i++)
232 				if (iterators.get(i) != null)
233 					remaining += iterators.get(i).countRemaining();
234 			return remaining;
235 		}
236 
237 		public boolean hasNext() {
238 			computeCurrent();
239 			return current != null;
240 		}
241 
242 		public ConcatLineIterators copy() {
243 			List<BrailleTranslator.LineIterator> iteratorsCopy = new ArrayList<>(iterators.size() - currentIndex);
244 			for (int i = currentIndex; i < iterators.size(); i++)
245 				if (iterators.get(i) != null)
246 					iteratorsCopy.add((BrailleTranslator.LineIterator)iterators.get(i).copy());
247 			return new ConcatLineIterators(iteratorsCopy);
248 		}
249 
250 		public boolean supportsMetric(String metric) {
251 			return false;
252 		}
253 
254 		public double getMetric(String metric) {
255 			throw new UnsupportedMetricException("Metric not supported: " + metric);
256 		}
257 	}
