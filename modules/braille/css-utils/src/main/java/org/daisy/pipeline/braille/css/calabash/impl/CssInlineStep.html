172 		importer = new Importer() {
173 			public Collection<Import> apply(String url, Import previous) {
174 				try {
175 					URI uri = asURI(url);
176 					URI base = previous.getAbsoluteUri();
177 					logger.debug("Importing SASS style sheet: " + uri + " (base = " + base + ")");
178 					URI abs = base.resolve(uri);
179 					InputStream is; {
180 						Source resolved; {
181 							try {
182 								resolved = _resolver.resolve(abs.toString(), ""); }
183 							catch (javax.xml.transform.TransformerException e) {
184 								throw new IOException(e); }}
185 						if (resolved != null && resolved instanceof StreamSource)
186 							is = ((StreamSource)resolved).getInputStream();
187 						else {
188 							if (resolved != null) {
189 								abs = asURI(resolved.getSystemId());
190 								logger.debug("Resolved to: " + abs); }
191 							is = asURL(abs).openStream(); }}
192 					try {
193 						return ImmutableList.of(
194 							new Import(uri, abs,
195 							           byteSource(is).asCharSource(StandardCharsets.UTF_8).read())); }
196 					catch (RuntimeException e) {
197 						throw new IOException(e); }}
198 				catch (IOException e) {
199 					if (!url.endsWith(".scss"))
200 						return apply(url + ".scss", previous);
201 					else
202 						throw new RuntimeException(e); }
203 			}
204 		};
205 		network = new DefaultNetworkProcessor() {
206 			@Override
207 			public InputStream fetch(URL url) throws IOException {
208 				logger.debug("Fetching CSS style sheet: " + url);
209 				InputStream is; {
210 					Source resolved; {
211 						try {
212 							resolved = _resolver.resolve(asURI(url).toString(), ""); }
213 						catch (javax.xml.transform.TransformerException e) {
214 							throw new IOException(e); }}
215 					if (resolved != null && resolved instanceof StreamSource)
216 						is = ((StreamSource)resolved).getInputStream();
217 					else {
218 						if (resolved != null) {
219 							url = new URL(resolved.getSystemId());
220 							logger.debug("Resolved to :" + url); }
221 						is = super.fetch(url); }}
222 				
223 				// skip BOM
224 				is = new BOMInputStream(is);
225 				if (url.toString().endsWith(".scss")) {
226 					Compiler sassCompiler = new Compiler();
227 					Options options = new Options();
228 					options.setIsIndentedSyntaxSrc(false);
229 					options.setOutputStyle(OutputStyle.EXPANDED);
230 					options.setSourceMapContents(false);
231 					options.setSourceMapEmbed(false);
232 					options.setSourceComments(false);
233 					options.setPrecision(5);
234 					options.setOmitSourceMapUrl(true);
235 					options.getImporters().add(importer);
236 					String scss = "";
237 					for (String var : sassVariables.keySet()) {
238 						String value = sassVariables.get(var);
239 						if (!value.matches(scssNumberColorString)) {
240 							// if value contains spaces or special characters that can mess up parsing; wrap it in single quotes
241 							logger.debug("scss variable '"+var+"' contains special characters: "+value);
242 							value = "'"+value.replaceAll("'", "\\\\'")+"'";
243 							logger.debug("scss variable '"+var+"' was quoted                 : "+value);
244 						} else {
245 							logger.debug("scss variable '"+var+"' contains no special characters: "+value);
246 						}
247 						scss += ("$" + var + ": " + value + ";\n");
248 					}
249 					scss += byteSource(is).asCharSource(StandardCharsets.UTF_8).read();
250 					try {
251 						Output result = sassCompiler.compileString(scss, StandardCharsets.UTF_8, asURI(url), null, options);
252 						if (result.getErrorStatus() != 0)
253 							throw new IOException("Could not compile SASS style sheet: " + result.getErrorMessage());
254 						String css = result.getCss();
255 						logger.debug(url + " compiled to:\n\n" + css);
256 						return new ByteArrayInputStream(css.getBytes(StandardCharsets.UTF_8)); }
257 					catch (CompilationException e) {
258 						throw new IOException("Could not compile SASS style sheet", e); }}
259 				else
260 					return is;
261 			}
262 		};

329 	static class InMemoryURIResolver implements URIResolver {
330 		private ReadablePipe documents = null;
331 		InMemoryURIResolver() {}
332 		void setContext(ReadablePipe documents) {
333 			this.documents = documents;
334 		}
335 		public Source resolve(String href, String base) throws TransformerException {
336 			if (documents == null) return null;
337 			try {
338 				URI uri = (base != null) ?
339 					new URI(base).resolve(new URI(href)) :
340 					new URI(href);
341 				uri = normalizeUri(uri);
342 				documents.resetReader();
343 				while (documents.moreDocuments()) {
344 					XdmNode doc = documents.read();
345 					URI docUri = normalizeUri(doc.getBaseURI());
346 					if (docUri.equals(uri)) {
347 						XdmNode root = S9apiUtils.getDocumentElement(doc);
348 						if (XProcConstants.c_result.equals(root.getNodeName())
349 						    && root.getAttributeValue(_content_type) != null
350 						    && root.getAttributeValue(_content_type).startsWith("text/"))
351 							return new StreamSource(new ByteArrayInputStream(doc.getStringValue().getBytes()),
352 							                        uri.toASCIIString());
353 						else if ("base64".equals(root.getAttributeValue(c_encoding)))
354 							return new StreamSource(new ByteArrayInputStream(Base64.decode(doc.getStringValue())),
355 							                        uri.toASCIIString());
356 						else if (XProcConstants.c_data.equals(root.getNodeName()))
357 							return new StreamSource(new ByteArrayInputStream(doc.getStringValue().getBytes()),
358 							                        uri.toASCIIString());
359 						else
360 							return doc.asSource(); }}}
361 			catch (URISyntaxException e) {
362 				e.printStackTrace();
363 				throw new TransformerException(e); }
364 			catch (SaxonApiException e) {
365 				e.printStackTrace();
366 				throw new TransformerException(e); }
367 			return null;
368 		}
369 	}

371 	private static URI normalizeUri(URI uri) {
372 		String s = uri.toASCIIString();
373 		
374 		// A URI that starts with "file:///" can be written as "file:/"
375 		if (s.startsWith("file:///"))
376 			s = "file:" + s.substring(7);
377 		
378 		// A URI that contains "!/" is a ZIP URI
379 		if (s.startsWith("file:") && s.contains("!/"))
380 			s = "zip:" + s;
381 		
382 		// The part of a ZIP URI after the "!" must start with "/"
383 		if (s.startsWith("zip:file:") && !s.contains("!/"))
384 			s = s.replaceFirst("!", "!/");
385 		try {
386 			return new URI(s); }
387 		catch (URISyntaxException e) {
388 			throw new RuntimeException(e); }
389 	}

448 	private static class CssInlineTransformer implements DOMToXMLStreamTransformer {
449 		
450 		private final NetworkProcessor network;
451 		private final String defaultStyleSheet;
452 		private final String medium;
453 		private final javax.xml.namespace.QName attributeName;
454 		private final List<CascadedStyle> styles;
455 		
456 		public CssInlineTransformer(NetworkProcessor network,
457 		                            String defaultStyleSheet,
458 		                            String medium,
459 		                            javax.xml.namespace.QName attributeName) {
460 			this.network = network;
461 			this.defaultStyleSheet = defaultStyleSheet;
462 			this.medium = medium;
463 			this.attributeName = attributeName;
464 			this.styles = new ArrayList<CascadedStyle>();
465 		}

467 		private BaseURIAwareXMLStreamWriter writer;
468 		
469 		public void transform(Iterator<Document> input, Supplier<BaseURIAwareXMLStreamWriter> output) throws TransformerException {
470 			
471 			Document document = Iterators.getOnlyElement(input);
472 			this.writer = output.get();
473 			
474 			try {
475 				URI baseURI = new URI(document.getBaseURI());
476 				URL baseURL; {
477 					if (baseURI == null || baseURI.toString().equals(""))
478 						// handling the case where base URI is empty, although this should in theory never happen
479 						// FIXME: find out why it happens
480 						baseURL = null;
481 					else
482 						baseURL = asURL(baseURI);
483 				}
484 				URL[] defaultSheets; {
485 					StringTokenizer t = new StringTokenizer(defaultStyleSheet);
486 					ArrayList<URL> l = new ArrayList<URL>();
487 					while (t.hasMoreTokens())
488 						l.add(asURL(baseURI.resolve(asURI(t.nextToken()))));
489 					defaultSheets = toArray(l, URL.class);
490 				}
491 				
492 				CascadedStyle style = new CascadedStyle();
493 				styles.add(style);
494 				StyleSheet stylesheet;
495 				if (medium.equals("embossed")) {
496 					stylesheet = (StyleSheet)brailleRuleFactory.createStyleSheet().unlock();
497 					if (defaultSheets != null)
498 						for (URL sheet : defaultSheets)
499 							stylesheet = brailleParserFactory.append(sheet, network, null, SourceType.URL, stylesheet, sheet);
500 					// CSSParserFactory injected via CSSFactory in CSSAssignTraversal.<init>
501 					CSSFactory.registerCSSParserFactory(brailleParserFactory);
502 					stylesheet = CSSFactory.getUsedStyles(document, null, baseURL, new MediaSpec(medium), network, stylesheet);
503 					// DeclarationTransformer injected via CSSFactory in SingleMapNodeData.<init>
504 					// SupportedCSS injected via CSSFactory in SingleMapNodeData.<init>, Repeater.assignDefaults, Variator.assignDefaults
505 					CSSFactory.registerDeclarationTransformer(brailleDeclarationTransformer);
506 					CSSFactory.registerSupportedCSS(brailleCSS);
507 					style.styleMap = new Analyzer(stylesheet).evaluateDOM(document, medium, false);
508 					style.pageRules = new HashMap<String,Map<String,RulePage>>(); {
509 						for (RulePage r : filter(stylesheet, RulePage.class)) {
510 							String name = firstNonNull(r.getName(), "auto");
511 							String pseudo = firstNonNull(r.getPseudo(), "");
512 							Map<String,RulePage> pageRule = style.pageRules.get(name);
513 							if (pageRule == null) {
514 								pageRule = new HashMap<String,RulePage>();
515 								style.pageRules.put(name, pageRule); }
516 							if (pageRule.containsKey(pseudo))
517 								pageRule.put(pseudo, makePageRule(name, "".equals(pseudo) ? null : pseudo,
518 								                                  ImmutableList.of(r, pageRule.get(pseudo))));
519 							else
520 								pageRule.put(pseudo, r);
521 						}
522 					}
523 					style.volumeRules = new HashMap<String,Map<String,RuleVolume>>(); {
524 						for (RuleVolume r : filter(stylesheet, RuleVolume.class)) {
525 							String name = "auto";
526 							String pseudo = firstNonNull(r.getPseudo(), "");
527 							Map<String,RuleVolume> volumeRule = style.volumeRules.get(name);
528 							if (volumeRule == null) {
529 								volumeRule = new HashMap<String,RuleVolume>();
530 								style.volumeRules.put(name, volumeRule); }
531 							if (volumeRule.containsKey(pseudo))
532 								volumeRule.put(pseudo, makeVolumeRule(name, "".equals(pseudo) ? null : pseudo,
533 								                                      ImmutableList.of(r, volumeRule.get(pseudo))));
534 							else
535 								volumeRule.put(pseudo, r);
536 						}
537 					}
538 					style.textTransformRules = filter(stylesheet, RuleTextTransform.class);
539 					style.otherAtRules = filter(stylesheet, AnyAtRule.class);
540 				} else if (medium.equals("print")) {
541 					stylesheet = (StyleSheet)printRuleFactory.createStyleSheet().unlock();
542 					if (defaultSheets != null)
543 						for (URL sheet : defaultSheets) {
544 							// RuleFactory injected via CSSFactory in SimplePreparator.<init>, CSSTreeParser.<init>
545 							CSSFactory.registerRuleFactory(printRuleFactory);
546 							stylesheet = printParserFactory.append(sheet, network, null, SourceType.URL, stylesheet, sheet); }
547 					// CSSParserFactory injected via CSSFactory in CSSAssignTraversal.<init>
548 					// RuleFactory injected via CSSFactory in SimplePreparator.<init>, CSSTreeParser.<init>
549 					CSSFactory.registerCSSParserFactory(printParserFactory);
550 					CSSFactory.registerRuleFactory(printRuleFactory);
551 					stylesheet = CSSFactory.getUsedStyles(document, null, baseURL, new MediaSpec(medium), network, stylesheet);
552 					// DeclarationTransformer injected via CSSFactory in SingleMapNodeData.<init>
553 					// SupportedCSS injected via CSSFactory in SingleMapNodeData.<init>, Repeater.assignDefaults, Variator.assignDefaults
554 					CSSFactory.registerDeclarationTransformer(printDeclarationTransformer);
555 					CSSFactory.registerSupportedCSS(printCSS);
556 					style.styleMap = new Analyzer(stylesheet).evaluateDOM(document, medium, false);
557 				} else {
558 					throw new RuntimeException("medium " + medium + " not supported");
559 				}
560 				
561 				writer.setBaseURI(baseURI);
562 				writer.writeStartDocument();
563 				traverse(document.getDocumentElement());
564 				writer.writeEndDocument();
565 				
566 			} catch (Exception e) {
567 				throw new TransformerException(e);
568 			}
569 		}

581 		private void traverse(Node node) throws XPathException, URISyntaxException, XMLStreamException {
582 			
583 			if (node.getNodeType() == Node.ELEMENT_NODE) {
584 				Element elem = (Element)node;
585 				writeStartElement(writer, elem);
586 				NamedNodeMap attributes = node.getAttributes();
587 				for (int i = 0; i < attributes.getLength(); i++) {
588 					Node attr = attributes.item(i);
589 					if (!(attr.getPrefix() == null && "style".equals(attr.getLocalName())))
590 						writeAttribute(writer, attr); }
591 				StringBuilder style = new StringBuilder();
592 				for (CascadedStyle cs : styles) {
593 					NodeData nodeData = cs.styleMap.get(elem);
594 					if (nodeData != null)
595 						insertStyle(style, nodeData);
596 					for (PseudoElement pseudo : sort(cs.styleMap.pseudoSet(elem), pseudoElementComparator)) {
597 						NodeData pseudoData = cs.styleMap.get(elem, pseudo);
598 						if (pseudoData != null)
599 							insertPseudoStyle(style, pseudoData, pseudo, cs.pageRules); }
600 					if (cs.pageRules != null) {
601 						Map<String,RulePage> pageRule = getPageRule(nodeData, cs.pageRules);
602 						if (pageRule != null) {
603 							insertPageStyle(style, pageRule, true); }
604 						else if (isRoot) {
605 							pageRule = getPageRule("auto", cs.pageRules);
606 							if (pageRule != null)
607 								insertPageStyle(style, pageRule, true); }}
608 					if (isRoot) {
609 						if (cs.volumeRules != null) {
610 							Map<String,RuleVolume> volumeRule = getVolumeRule("auto", cs.volumeRules);
611 							if (volumeRule != null)
612 								insertVolumeStyle(style, volumeRule, cs.pageRules); }
613 						if (cs.textTransformRules != null)
614 							for (RuleTextTransform r : cs.textTransformRules)
615 								insertTextTransformDefinition(style, r);
616 						if (cs.otherAtRules != null)
617 							for (AnyAtRule r : cs.otherAtRules) {
618 								if (style.length() > 0 && !style.toString().endsWith("} ")) {
619 									style.insert(0, "{ ");
620 									style.append("} "); }
621 								insertAtRule(style, r); }}}
622 				if (normalizeSpace(style).length() > 0) {
623 					writeAttribute(writer, attributeName, style.toString().trim()); }
624 				isRoot = false;
625 				for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling())
626 					traverse(child);
627 				writer.writeEndElement(); }
628 			else if (node.getNodeType() == Node.COMMENT_NODE)
629 				writeComment(writer, node);
630 			else if (node.getNodeType() == Node.TEXT_NODE)
631 				writeCharacters(writer, node);
632 			else if (node.getNodeType() == Node.PROCESSING_INSTRUCTION_NODE)
633 				writeProcessingInstruction(writer, node);
634 			else
635 				throw new UnsupportedOperationException("Unexpected node type");
636 		}

663 	static Function<Term,String> termToString = new Function<Term,String>() {
664 		public String apply(Term term) {
665 			if (term instanceof TermInteger) {
666 				TermInteger integer = (TermInteger)term;
667 				return "" + integer.getIntValue(); }
668 			else if (term instanceof TermNumber) {
669 				TermNumber number = (TermNumber)term;
670 				Double value = number.getValue().doubleValue();
671 				if (value == Math.floor(value))
672 					return "" + value.intValue();
673 				else
674 					return "" + value; }
675 			else if (term instanceof TermPercent) {
676 				TermPercent percent = (TermPercent)term;
677 				Double value = percent.getValue().doubleValue();
678 				if (value == Math.floor(value))
679 					return "" + value.intValue() + "%";
680 				else
681 					return "" + value + "%"; }
682 			else if (term instanceof TermList && !(term instanceof TermFunction)) {
683 				TermList list = (TermList)term;
684 				String s = "";
685 				for (Term<?> t : list) {
686 					if (!s.isEmpty()) {
687 						Term.Operator o = t.getOperator();
688 						if (o != null)
689 							switch (o) {
690 							case COMMA:
691 								s += ","; }
692 						s += " "; }
693 					s += termToString.apply(t); }
694 				return s; }
695 			else if (term instanceof TermPair) {
696 				TermPair<?,?> pair = (TermPair<?,?>)term;
697 				Object val = pair.getValue();
698 				return "" + pair.getKey() + " " + (val instanceof Term ? termToString.apply((Term)val) : val.toString()); }
699 			else if (term instanceof TermString) {
700 				TermString string = (TermString)term;
701 				return "'" + string.getValue().replaceAll("\n", "\\\\A") + "'"; }
702 			else
703 				return term.toString().replaceAll("^[,/ ]+", "");
704 		}
705 	};

708 		List<String> keys = new ArrayList<String>(nodeData.getPropertyNames());
709 		keys.remove("page");
710 		Collections.sort(keys);
711 		for(String key : keys) {
712 			builder.append(key).append(": ");
713 			Term<?> value = nodeData.getValue(key, true);
714 			if (value != null)
715 				builder.append(termToString.apply(value));
716 			else {
717 				CSSProperty prop = nodeData.getProperty(key);
718 				builder.append(prop); }
719 			builder.append("; "); }
720 	}

795 	private static Map<String,RulePage> getPageRule(String name, Map<String,Map<String,RulePage>> pageRules) {
796 		Map<String,RulePage> auto = pageRules == null ? null : pageRules.get("auto");
797 		Map<String,RulePage> named = null;
798 		if (!name.equals("auto"))
799 			named = pageRules.get(name);
800 		Map<String,RulePage> result = new HashMap<String,RulePage>();
801 		List<RulePage> from;
802 		RulePage r;
803 		Set<String> pseudos = new HashSet<String>();
804 		if (named != null)
805 			pseudos.addAll(named.keySet());
806 		if (auto != null)
807 			pseudos.addAll(auto.keySet());
808 		for (String pseudo : pseudos) {
809 			boolean noPseudo = "".equals(pseudo);
810 			from = new ArrayList<RulePage>();
811 			if (named != null) {
812 				r = named.get(pseudo);
813 				if (r != null) from.add(r);
814 				if (!noPseudo) {
815 					r = named.get("");
816 					if (r != null) from.add(r); }}
817 			if (auto != null) {
818 				r = auto.get(pseudo);
819 				if (r != null) from.add(r);
820 				if (!noPseudo) {
821 					r = auto.get("");
822 					if (r != null) from.add(r); }}
823 			result.put(pseudo, makePageRule(name, noPseudo ? null : pseudo, from)); }
824 		return result;
825 	}

827 	private static RulePage makePageRule(String name, String pseudo, List<RulePage> from) {
828 		RulePage pageRule = brailleRuleFactory.createPage().setName(name).setPseudo(pseudo);
829 		for (RulePage f : from)
830 			for (Rule<?> r : f)
831 				if (r instanceof Declaration) {
832 					Declaration d = (Declaration)r;
833 					String property = d.getProperty();
834 					if (getDeclaration(pageRule, property) == null)
835 						pageRule.add(r); }
836 				else if (r instanceof RuleMargin) {
837 					RuleMargin m = (RuleMargin)r;
838 					String marginArea = m.getMarginArea();
839 					RuleMargin marginRule = getRuleMargin(pageRule, marginArea);
840 					if (marginRule == null) {
841 						marginRule = brailleRuleFactory.createMargin(marginArea);
842 						pageRule.add(marginRule);
843 						marginRule.replaceAll(m); }
844 					else
845 						for (Declaration d : m)
846 							if (getDeclaration(marginRule, d.getProperty()) == null)
847 								marginRule.add(d); }
848 		return pageRule;
849 	}

946 	private static final Pattern FUNCTION = Pattern.compile("(nth|nth-last)\\(([1-9][0-9]*)\\)");

948 	private static RuleVolume makeVolumeRule(String name, String pseudo, List<RuleVolume> from) {
949 		String arg = null;
950 		if (pseudo != null) {
951 			Matcher m = FUNCTION.matcher(pseudo);
952 			if (m.matches()) {
953 				pseudo = m.group(1);
954 				arg = m.group(2); }}
955 		RuleVolume volumeRule = new RuleVolume(pseudo, arg);
956 		for (RuleVolume f : from)
957 			for (Rule<?> r : f)
958 				if (r instanceof Declaration) {
959 					Declaration d = (Declaration)r;
960 					String property = d.getProperty();
961 					if (getDeclaration(volumeRule, property) == null)
962 						volumeRule.add(r); }
963 				else if (r instanceof RuleVolumeArea) {
964 					RuleVolumeArea a = (RuleVolumeArea)r;
965 					String volumeArea = a.getVolumeArea().value;
966 					RuleVolumeArea volumeAreaRule = getRuleVolumeArea(volumeRule, volumeArea);
967 					if (volumeAreaRule == null) {
968 						volumeAreaRule = new RuleVolumeArea(volumeArea);
969 						volumeRule.add(volumeAreaRule);
970 						volumeAreaRule.replaceAll(a); }
971 					else
972 						for (Declaration d : filter(a, Declaration.class))
973 							if (getDeclaration(volumeAreaRule, d.getProperty()) == null)
974 								volumeAreaRule.add(d); }
975 		return volumeRule;
976 	}
