121 	protected final Iterable<LibhyphenHyphenator> _get(Query query) {
122 		MutableQuery q = mutableQuery(query);
123 		if (q.containsKey("hyphenator")) {
124 			String v = q.removeOnly("hyphenator").getValue().get();
125 			if (!("hyphen".equals(v) || "libhyphen".equals(v))) {
126 				Iterable<LibhyphenHyphenator> ret;
127 				LibhyphenHyphenator h = fromId(v);
128 				if (h != null)
129 					ret = fromNullable(h);
130 				else
131 					ret = of(get(URLs.asURI(v)));
132 				if (q.isEmpty())
133 					return ret;
134 				else
135 					return intersection(_get(q), ret); }}
136 		String table = null;
137 		if (q.containsKey("libhyphen-table"))
138 			table = q.removeOnly("libhyphen-table").getValue().get();
139 		if (q.containsKey("hyphen-table"))
140 			if (table != null) {
141 				logger.warn("A query with both 'libhyphen-table' and 'hyphen-table' never matches anything");
142 				return empty; }
143 			else
144 				table = q.removeOnly("hyphen-table").getValue().get();
145 		if (q.containsKey("table"))
146 			if (table != null) {
147 				logger.warn("A query with both 'table' and '(lib)hyphen-table' never matches anything");
148 				return empty; }
149 			else
150 				table = q.removeOnly("table").getValue().get();
151 		if (table != null) {
152 			if (!q.isEmpty()) {
153 				logger.warn("A query with both 'table' or '(lib)hyphen-table' and '"
154 				            + q.iterator().next().getKey() + "' never matches anything");
155 				return empty; }
156 			return of(get(URLs.asURI(table))); }
157 		Locale locale; {
158 			String loc = "und";
159 			if (q.containsKey("document-locale"))
160 				loc = q.removeOnly("document-locale").getValue().get();
161 			try {
162 				locale = parseLocale(loc); }
163 			catch (IllegalArgumentException e) {
164 				logger.error("Invalid locale", e);
165 				return empty; }
166 		}
167 		return transform(
168 			tableRegistry.get(locale),
169 			new Function<URI,LibhyphenHyphenator>() {
170 				public LibhyphenHyphenator _apply(URI table) {
171 					return __apply(get(table)); }});
172 	}

224 		private final LineBreaker lineBreaker = new DefaultLineBreaker() {
225 			protected Break breakWord(String word, int limit, boolean force) {
226 				Hyphenator.Break br = hyphenator.hyphenate(word, limit);
227 				if (force && br.getBreakPosition() == 0)
228 					return new Break(word, limit, false);
229 				else
230 					return new Break(br.getText(), br.getBreakPosition(), br.hasHyphen());
231 			}

234 		private String transform(String text) {
235 			if (text.length() == 0)
236 				return text;
237 			try {
238 				Tuple2<String,byte[]> t = extractHyphens(text, false, SHY, ZWSP);
239 				if (t._1.length() == 0)
240 					return text;
241 				return insertHyphens(t._1, transform(t._2, t._1), false, SHY, ZWSP); }
242 			catch (NonStandardHyphenationException e) {
243 				throw e; }
244 			catch (Exception e) {
245 				throw new RuntimeException("Error during libhyphen hyphenation", e); }
246 		}
247 		
248 		private String[] transform(String[] text) {
249 			try {
250 				Tuple2<String,byte[]> t = extractHyphens(join(text, US), false, SHY, ZWSP);
251 				String[] unhyphenated = toArray(SEGMENT_SPLITTER.split(t._1), String.class);
252 				t = extractHyphens(t._2, t._1, false, null, null, US);
253 				String _text = t._1;
254 				// This byte array is used not only to track the hyphen
255 				// positions but also the segment boundaries.
256 				byte[] positions = t._2;
257 				positions = transform(positions, _text);
258 				_text = insertHyphens(_text, positions, false, SHY, ZWSP, US);
259 				if (text.length == 1)
260 					return new String[]{_text};
261 				else {
262 					String[] rv = new String[text.length];
263 					int i = 0;
264 					for (String s : SEGMENT_SPLITTER.split(_text)) {
265 						while (unhyphenated[i].length() == 0)
266 							rv[i++] = "";
267 						rv[i++] = s; }
268 					while(i < text.length)
269 						rv[i++] = "";
270 					return rv; }}
271 			catch (NonStandardHyphenationException e) {
272 				throw e; }
273 			catch (Exception e) {
274 				throw new RuntimeException("Error during libhyphen hyphenation", e); }
275 		}
276 		
277 		private byte[] transform(byte[] manualHyphens, String textWithoutManualHyphens) {
278 			if (textWithoutManualHyphens.length() == 0)
279 				return manualHyphens;
280 			boolean hasManualHyphens = false; {
281 				if (manualHyphens != null)
282 					for (byte b : manualHyphens)
283 						if (b == (byte)1 || b == (byte)2) {
284 							hasManualHyphens = true;
285 							break; }}
286 			if (hasManualHyphens) {
287 				// input contains SHY or ZWSP; hyphenate only the words without SHY or ZWSP
288 				byte[] hyphens = Arrays.copyOf(manualHyphens, manualHyphens.length);
289 				boolean word = true;
290 				int pos = 0;
291 				for (String segment : splitInclDelimiter(textWithoutManualHyphens, ON_SPACE_SPLITTER)) {
292 					if (word && segment.length() > 0) {
293 						int len = segment.length();
294 						boolean wordHasManualHyphens = false; {
295 							for (int k = 0; k < len - 1; k++)
296 								if (hyphens[pos + k] != 0) {
297 									wordHasManualHyphens = true;
298 									break; }}
299 						if (!wordHasManualHyphens) {
300 							try {
301 								byte[] wordHyphens = hyphenator.hyphenate(segment);
302 								for (int k = 0; k < len - 1; k++)
303 									hyphens[pos + k] |= wordHyphens[k];
304 							} catch (StandardHyphenationException e) {
305 								throw new NonStandardHyphenationException(e);
306 							}
307 						}
308 					}
309 					pos += segment.length();
310 					word = !word;
311 				}
312 				return hyphens;
313 			} else
314 				try {
315 					return hyphenator.hyphenate(textWithoutManualHyphens);
316 				} catch (StandardHyphenationException e) {
317 					throw new NonStandardHyphenationException(e);
318 				}
319 		}

343 		URL resolvedTable = isAbsoluteFile(table) ? URLs.asURL(table) : tableRegistry.resolve(table);
