136 	protected final Iterable<LibhyphenHyphenator> _get(Query query) {
137 		MutableQuery q = mutableQuery(query);
138 		if (q.containsKey("hyphenator")) {
139 			String v = q.removeOnly("hyphenator").getValue().get();
140 			if (!("hyphen".equals(v) || "libhyphen".equals(v))) {
141 				Iterable<LibhyphenHyphenator> ret;
142 				LibhyphenHyphenator h = fromId(v);
143 				if (h != null)
144 					ret = fromNullable(h);
145 				else
146 					ret = of(get(URLs.asURI(v)));
147 				if (q.isEmpty())
148 					return ret;
149 				else
150 					return intersection(_get(q), ret); }}
151 		String table = null;
152 		if (q.containsKey("libhyphen-table"))
153 			table = q.removeOnly("libhyphen-table").getValue().get();
154 		if (q.containsKey("hyphen-table"))
155 			if (table != null) {
156 				logger.warn("A query with both 'libhyphen-table' and 'hyphen-table' never matches anything");
157 				return empty; }
158 			else
159 				table = q.removeOnly("hyphen-table").getValue().get();
160 		if (q.containsKey("table"))
161 			if (table != null) {
162 				logger.warn("A query with both 'table' and '(lib)hyphen-table' never matches anything");
163 				return empty; }
164 			else
165 				table = q.removeOnly("table").getValue().get();
166 		if (table != null) {
167 			if (!q.isEmpty()) {
168 				logger.warn("A query with both 'table' or '(lib)hyphen-table' and '"
169 				            + q.iterator().next().getKey() + "' never matches anything");
170 				return empty; }
171 			return of(get(URLs.asURI(table))); }
172 		if (tableProvider != null) {
173 			Locale locale; {
174 				String loc = "und";
175 				if (q.containsKey("locale"))
176 					loc = q.removeOnly("locale").getValue().get();
177 				try {
178 					locale = parseLocale(loc); }
179 				catch (IllegalArgumentException e) {
180 					logger.error("Invalid locale", e);
181 					return empty; }
182 			}
183 			return transform(
184 				tableProvider.get(locale),
185 				new Function<URI,LibhyphenHyphenator>() {
186 					public LibhyphenHyphenator _apply(URI table) {
187 						return __apply(get(table)); }}); }
188 		return empty;
189 	}

242 			protected Break breakWord(String word, int limit, boolean force) {
243 				Hyphenator.Break br = hyphenator.hyphenate(word, limit);
244 				if (force && br.getBreakPosition() == 0)

247 					return new Break(br.getText(), br.getBreakPosition(), br.hasHyphen());
248 			}

251 		private String transform(String text) {
252 			if (text.length() == 0)
253 				return text;
254 			try {
255 				Tuple2<String,byte[]> t = extractHyphens(text, SHY, ZWSP);
256 				if (t._1.length() == 0)
257 					return text;
258 				return insertHyphens(t._1, transform(t._2, t._1), SHY, ZWSP); }
259 			catch (Exception e) {
260 				throw new RuntimeException("Error during libhyphen hyphenation", e); }
261 		}
262 		
263 		private String[] transform(String[] text) {
264 			try {
265 				Tuple2<String,byte[]> t = extractHyphens(join(text, US), SHY, ZWSP);
266 				String[] unhyphenated = toArray(SEGMENT_SPLITTER.split(t._1), String.class);
267 				t = extractHyphens(t._2, t._1, null, null, US);
268 				String _text = t._1;
269 				// This byte array is used not only to track the hyphen
270 				// positions but also the segment boundaries.
271 				byte[] positions = t._2;
272 				positions = transform(positions, _text);
273 				_text = insertHyphens(_text, positions, SHY, ZWSP, US);
274 				if (text.length == 1)
275 					return new String[]{_text};
276 				else {
277 					String[] rv = new String[text.length];
278 					int i = 0;
279 					for (String s : SEGMENT_SPLITTER.split(_text)) {
280 						while (unhyphenated[i].length() == 0)
281 							rv[i++] = "";
282 						rv[i++] = s; }
283 					while(i < text.length)
284 						rv[i++] = "";
285 					return rv; }}
286 			catch (Exception e) {
287 				throw new RuntimeException("Error during libhyphen hyphenation", e); }
288 		}
289 		
290 		private byte[] transform(byte[] manualHyphens, String textWithoutManualHyphens) {
291 			if (textWithoutManualHyphens.length() == 0)
292 				return manualHyphens;
293 			boolean hasManualHyphens = false; {
294 				if (manualHyphens != null)
295 					for (byte b : manualHyphens)
296 						if (b == (byte)1 || b == (byte)2) {
297 							hasManualHyphens = true;
298 							break; }}
299 			if (hasManualHyphens) {
300 				// input contains SHY or ZWSP; hyphenate only the words without SHY or ZWSP
301 				byte[] hyphens = Arrays.copyOf(manualHyphens, manualHyphens.length);
302 				boolean word = true;
303 				int pos = 0;
304 				for (String segment : splitInclDelimiter(textWithoutManualHyphens, ON_SPACE_SPLITTER)) {
305 					if (word && segment.length() > 0) {
306 						int len = segment.length();
307 						boolean wordHasManualHyphens = false; {
308 							for (int k = 0; k < len - 1; k++)
309 								if (hyphens[pos + k] != 0) {
310 									wordHasManualHyphens = true;
311 									break; }}
312 						if (!wordHasManualHyphens) {
313 							byte[] wordHyphens = hyphenator.hyphenate(segment);
314 							for (int k = 0; k < len - 1; k++)
315 								hyphens[pos + k] |= wordHyphens[k];
316 						}
317 					}
318 					pos += segment.length();
319 					word = !word;
320 				}
321 				return hyphens;
322 			} else
323 				return hyphenator.hyphenate(textWithoutManualHyphens);
324 		}
