1   package org.daisy.pipeline.braille.libhyphen;
2   
3   import java.net.URI;
4   import java.util.HashMap;
5   import java.util.Locale;
6   import java.util.Map;
7   
8   import com.google.common.base.Optional;
9   import com.google.common.base.Predicates;
10  import static com.google.common.collect.Iterables.filter;
11  
12  import org.daisy.common.file.URLs;
13  import org.daisy.pipeline.braille.common.BundledResourcePath;
14  import org.daisy.pipeline.braille.common.Provider;
15  import static org.daisy.pipeline.braille.common.util.Predicates.matchesGlobPattern;
16  
17  public class LibhyphenTablePath extends BundledResourcePath implements LibhyphenTableProvider {
18  	
19  	@Override
20  	protected void activate(Map<?,?> properties, Class<?> context) throws IllegalArgumentException {
21  		if (properties.get(UNPACK) != null)
22  			throw new IllegalArgumentException(UNPACK + " property not supported");
23  		Map<Object,Object> props = new HashMap<Object,Object>(properties);
24  		props.put(BundledResourcePath.UNPACK, true);
25  		super.activate(props, context);
26  	}
27  	
28  	/**
29  	 * Lookup a table based on a locale, using the fact that table names are
30  	 * always in the form `hyph_language[_COUNTRY[_variant]].dic`
31  	 */
32  	public Iterable<URI> get(Locale locale) {
33  		return provider.get(locale);
34  	}
35  	
36  	private Provider<Locale,URI> provider = new Provider.util.VaryLocale<Locale,URI>() {
37  		public Iterable<URI> _get(Locale locale) {
38  			String language = locale.getLanguage().toLowerCase();
39  			String country = locale.getCountry().toUpperCase();
40  			String variant = locale.getVariant().toLowerCase();
41  			URI resource = null;
42  			if (!"".equals(variant))
43  				resource = URLs.asURI(String.format("hyph_%s_%s_%s.dic", language, country, variant));
44  			else if (!"".equals(country))
45  				resource = URLs.asURI(String.format("hyph_%s_%s.dic", language, country));
46  			else
47  				resource = URLs.asURI(String.format("hyph_%s.dic", language));
48  			if (containsResource(resource))
49  				return Optional.of(canonicalize(resource)).asSet();
50  			return Optional.<URI>absent().asSet();
51  		}
52  		@Override
53  		public Iterable<URI> fallback(Locale locale) {
54  			return filter(
55  				listResources(),
56  				Predicates.compose(
57  					matchesGlobPattern(String.format("hyph_%s_*.dic", locale.getLanguage().toLowerCase())),
58  					URI::getPath));
59  		}
60  	};
61  }
