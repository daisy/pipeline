1   package org.daisy.pipeline.braille.css.xpath;
2   
3   import java.util.Collections;
4   import java.util.IllformedLocaleException;
5   import java.util.Map;
6   import java.util.Optional;
7   import java.util.Locale;
8   
9   import org.daisy.braille.css.BrailleCSSParserFactory.Context;
10  import org.daisy.braille.css.SimpleInlineStyle;
11  import org.daisy.pipeline.braille.css.CSSStyledText;
12  import org.daisy.pipeline.braille.css.impl.BrailleCssStyle;
13  import org.daisy.pipeline.braille.css.TextStyleParser;
14  import org.daisy.pipeline.braille.css.xpath.impl.Stylesheet;
15  
16  public class StyledText {
17  
18  	private final TextStyleParser textStyleParser;
19  
20  	public StyledText(TextStyleParser textStyleParser) {
21  		this.textStyleParser = textStyleParser;
22  	}
23  
24  	public static String getText(CSSStyledText t) {
25  		return t.getText();
26  	}
27  
28  	/**
29  	 * @param t assumed to not change
30  	 */
31  	public static Optional<Style> getStyle(CSSStyledText t) {
32  		SimpleInlineStyle s = t.getStyle();
33  		if (s == null || s.isEmpty())
34  			return Optional.empty();
35  		return Optional.of(Stylesheet.of(BrailleCssStyle.of(s)));
36  	}
37  
38  	public static Optional<String> getLanguage(CSSStyledText t) {
39  		Locale l = t.getLanguage();
40  		if (l == null)
41  			return Optional.empty();
42  		return Optional.of(l.toLanguageTag());
43  	}
44  
45  	public static Map<String,String> getTextAttributes(CSSStyledText t) {
46  		Map<String,String> map = t.getTextAttributes();
47  		return map != null ? map : Collections.emptyMap();
48  	}
49  
50  	public CSSStyledText of(String text) {
51  		return of(text, Optional.empty());
52  	}
53  
54  	public CSSStyledText of(String text, Optional<Object> style) {
55  		return of(text, style, Optional.empty());
56  	}
57  
58  	public CSSStyledText of(String text, Optional<Object> style, Optional<String> language) {
59  		return of(text, style, language, null);
60  	}
61  
62  	public CSSStyledText of(String text, Optional<Object> style, Optional<String> language, Map<String,String> textAttributes) {
63  		Locale l = null; {
64  			if (language != null && language.isPresent()) {
65  				try {
66  					l = (new Locale.Builder()).setLanguageTag(language.get().replace('_','-')).build();
67  				} catch (IllformedLocaleException e) {
68  					throw new IllegalArgumentException("Locale '" + language.get() + "' could not be parsed", e);
69  				}
70  			}
71  		}
72  		if (style != null && style.isPresent()) {
73  			Object o = style.get();
74  			if (o instanceof Style) {
75  				SimpleInlineStyle s; {
76  					try {
77  						if (!(o instanceof Stylesheet))
78  							throw new IllegalArgumentException("Unexpected style argument");
79  						if (((Stylesheet)o).style == null)
80  							s = SimpleInlineStyle.EMPTY;
81  						else
82  							s = ((Stylesheet)o).style.asSimpleInlineStyle(true); // this will raise a UnsupportedOperationException
83  							                                                     // when the provided style is not a simple style
84  					} catch (UnsupportedOperationException e) {
85  						throw new IllegalArgumentException("Unexpected style argument");
86  					}
87  				}
88  				return new CSSStyledText(text, s, l, textAttributes);
89  			} else if (o instanceof String) {
90  				String s = (String)o;
91  				return new CSSStyledText(text, textStyleParser.parse(s), l, textAttributes);
92  			} else
93  				throw new IllegalArgumentException("Unexpected style argument: expected `Style` object or string");
94  		} else
95  			return new CSSStyledText(text, null, l, textAttributes);
96  	}
97  }
