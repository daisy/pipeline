68  	public static String toString(Term<?> term) {
69  		if (term instanceof TextTransformList ||
70  		    term instanceof ContentList ||
71  		    term instanceof StringSetList ||
72  		    term instanceof CounterSetList)
73  			return serializeTermList((List<Term<?>>)term);
74  		else if (term instanceof AttrFunction) {
75  			AttrFunction f = (AttrFunction)term;
76  			String s = "attr(" + toString(f.name);
77  			if (f.asURL)
78  				s = s + " url";
79  			s += ")";
80  			return s; }
81  		else if (term instanceof ContentFunction) {
82  			ContentFunction f = (ContentFunction)term;
83  			if (f.target.isPresent())
84  				return "target-content(" + toString(f.target.get()) + ")";
85  			else
86  				return "content()"; }
87  		else if (term instanceof StringFunction) {
88  			StringFunction f = (StringFunction)term;
89  			if (f.target.isPresent())
90  				return "target-string(" + toString(f.target.get()) + ", " + toString(f.name) + ")";
91  			else {
92  				String s = "string(" + toString(f.name);
93  				if (f.scope.isPresent())
94  					s = s + ", " + f.scope.get().toString();
95  				s += ")";
96  				return s; }}
97  		else if (term instanceof CounterFunction) {
98  			CounterFunction f = (CounterFunction)term;
99  			String s = "";
100 			if (f.target.isPresent())
101 				s = "target-counter(" + toString(f.target.get()) + ", " + toString(f.name);
102 			else
103 				s = "counter(" + toString(f.name);
104 			if (f.style.isPresent())
105 				s = s + ", " + toString(f.style.get());
106 			s += ")";
107 			return s; }
108 		else if (term instanceof TextFunction) {
109 			TextFunction f = (TextFunction)term;
110 			return "target-text(" + toString(f.target) + ")"; }
111 		else if (term instanceof LeaderFunction) {
112 			LeaderFunction f = (LeaderFunction)term;
113 			String s = "leader(" + toString(f.pattern);
114 			if (f.position.isPresent())
115 				s = s + ", " + toString(f.position.get());
116 			if (f.alignment.isPresent())
117 				s = s + ", " + f.alignment.get().toString();
118 			s += ")";
119 			return s; }
120 		else if (term instanceof FlowFunction) {
121 			FlowFunction f = (FlowFunction)term;
122 			String s = "flow(" + toString(f.from);
123 			if (f.scope.isPresent())
124 				s = s + ", " + f.scope.get().toString();
125 			s += ")";
126 			return s; }
127 		else
128 			return CssSerializer.toString(term, t -> toString(t));
129 	}

138 	public static String toString(BrailleCssStyle style) {
139 		if (style.serialized == null) {
140 			style.serialized = toString(style, null, null);
141 			// cache
142 			if (style.parser != null)
143 				style.parser.cache.put(style.context, style.serialized, style);
144 		} else {
145 			// access cache to keep entry longer in it
146 			if (style.parser != null)
147 				style.parser.cache.get(style.context, style.serialized);
148 		}
149 		return style.serialized;
150 	}

153 	 * @param relativeTo If not {@code null}, include only those declarations that are needed
154 	 *                   to reconstruct {@code style} with {@code relativeTo} as the parent
155 	 *                   style. Relativizes even if the parent style is empty.

157 	public static String toString(BrailleCssStyle style, BrailleCssStyle relativeTo) {
158 		if (relativeTo == null)
159 			return toString(style);
160 		String s = toString(style.relativize(relativeTo));
161 		// cache
162 		if (style.parser != null)
163 			style.parser.cache.put(style.context,
164 			                       s,
165 			                       relativeTo.declarations != null
166 			                           ? (ParsedDeclarations)relativeTo.declarations
167 			                           : ParsedDeclarations.EMPTY,
168 			                       true,
169 			                       style);
170 		return s;
171 	}

180 	private static String toString(BrailleCssStyle style, String base, String indent) {
181 		StringBuilder b = new StringBuilder();
182 		StringBuilder rel = new StringBuilder();
183 		if ("".equals(indent)) indent = null;
184 		String newline = indent != null ? "\n" : " ";
185 		if (style.declarations != null) {
186 			b.append(serializeDeclarationList(style.declarations, ";" + newline));
187 			if (indent != null && b.length() > 0) b.append(";");
188 		}
189 		if (style.nestedStyles != null)
190 			for (Map.Entry<String,BrailleCssStyle> e : style.nestedStyles.entrySet()) {
191 				if (base != null && e.getKey().startsWith("&")) {
192 					if (rel.length() > 0) rel.append(newline);
193 					rel.append(toString(e.getValue(), base + e.getKey().substring(1), indent));
194 				} else {
195 					if (b.length() > 0) {
196 						if (indent == null && b.charAt(b.length() - 1) != '}') b.append(";");
197 						b.append(newline);
198 					}
199 					b.append(toString(e.getValue(), e.getKey(), indent));
200 				}
201 			}
202 		if (base != null && b.length() > 0) {
203 			if (indent != null) {
204 				String s = b.toString();
205 				s = indent + s.trim().replaceAll("\n", "\n" + indent);
206 				b.setLength(0);
207 				b.append(s);
208 			}
209 			b.insert(0, base + " {" + newline);
210 			b.append(newline + "}");
211 		}
212 		if (rel.length() > 0) {
213 			if (b.length() > 0) {
214 				if (indent == null && b.charAt(b.length() - 1) != '}') b.append(";");
215 				b.append(newline);
216 			}
217 			b.append(rel);
218 		}
219 		return b.toString();
220 	}

222 	public static String toString(NodeData style) {
223 		List<String> declarations = new ArrayList<>();
224 		for (String p : style.getPropertyNames()) {
225 			String v = serializePropertyValue(style, p);
226 			if (v != null)
227 				declarations.add(p + ": " + v);
228 		}
229 		Collections.sort(declarations);
230 		StringBuilder s = new StringBuilder();
231 		Iterator<String> it = declarations.iterator();
232 		while (it.hasNext()) {
233 			s.append(it.next());
234 			if (it.hasNext()) s.append("; ");
235 		}
236 		return s.toString();
237 	}

269 	public static String toString(RuleRelativeBlock rule) {
270 		StringBuilder b = new StringBuilder();
271 		boolean first = true;
272 		for (Selector s : rule.getSelector()) {
273 			Combinator c = s.getCombinator();
274 			if (first) {
275 				if (c == null)
276 					b.append("&");
277 				else if (c != Combinator.CHILD)
278 					b.append(c.value());
279 				first = false;
280 			} else if (c != null) // should always be true
281 				b.append(c.value());
282 			b = OutputUtil.appendList(b, s, OutputUtil.EMPTY_DELIM);
283 		}
284 		b.append(" { ");
285 		b.append(toString((RuleBlock<Rule<?>>)rule));
286 		b.append(" }");
287 		return b.toString();
288 	}

306 	public static String serializeRuleBlockList(Iterable<? extends RuleBlock<? extends Rule<?>>> ruleBlocks) {
307 		String b = null;
308 		for (RuleBlock<? extends Rule<?>> r : ruleBlocks) {
309 			String s;
310 			if (r instanceof RuleMainBlock)
311 				s = BrailleCssSerializer.toString((RuleMainBlock)r);
312 			else if (r instanceof RuleRelativeBlock)
313 				s = BrailleCssSerializer.toString((RuleRelativeBlock)r);
314 			else
315 				s = BrailleCssSerializer.toString(r);
316 			if (!s.isEmpty())
317 				if (b == null)
318 					b = s;
319 				else {
320 					if (!(b.endsWith("}") || b.endsWith(";")))
321 						b = b + ";";
322 					b += " ";
323 					b += s; }}
324 		if (b == null) b = "";
325 		return b;
326 	}

340 	private static String serializeDeclarationList(Iterable<? extends Declaration> declarations, String separator) {
341 		List<String> sortedDeclarations = new ArrayList<>();
342 		for (Declaration d : declarations)
343 			sortedDeclarations.add(BrailleCssSerializer.toString(d));
344 		Collections.sort(sortedDeclarations);
345 		StringBuilder s = new StringBuilder();
346 		Iterator<String> it = sortedDeclarations.iterator();
347 		while (it.hasNext()) {
348 			s.append(it.next());
349 			if (it.hasNext()) s.append(separator);
350 		}
351 		return s.toString();
352 	}

440 	private static void toXml(BrailleCssStyle style,
441 	                          XMLStreamWriter w,
442 	                          boolean recursive) throws XMLStreamException {
443 		if (style.declarations != null && !Iterables.isEmpty(style.declarations)) {
444 			if (!recursive || style.nestedStyles != null)
445 				writeStartElement(w, CSS_RULE);
446 			List<Declaration> declarations = new ArrayList<>();
447 			for (Declaration d : style.declarations) declarations.add(d);
448 			Collections.sort(declarations, Ordering.natural().onResultOf(Declaration::getProperty));
449 			for (Declaration d : declarations)
450 				toXml(d, w);
451 			if (!recursive || style.nestedStyles != null)
452 				w.writeEndElement();
453 		}
454 		if (style.nestedStyles != null)
455 			for (Map.Entry<String,BrailleCssStyle> e : style.nestedStyles.entrySet()) {
456 				writeStartElement(w, CSS_RULE);
457 				writeAttribute(w, SELECTOR, e.getKey());
458 				toXml(e.getValue(), w, true);
459 				w.writeEndElement();
460 			}
461 	}

463 	public static void toXml(Declaration declaration, XMLStreamWriter writer) throws XMLStreamException {
464 		if (declaration instanceof PropertyValue)
465 			toXml((PropertyValue)declaration, writer);
466 		else {
467 			writeStartElement(writer, CSS_PROPERTY);
468 			writeAttribute(writer, NAME, declaration.getProperty());
469 			writeAttribute(writer, VALUE, serializeTermList(declaration));
470 			writer.writeEndElement();
471 		}

474 	public static void toXml(PropertyValue value, XMLStreamWriter writer) throws XMLStreamException {
475 		writeStartElement(writer, CSS_PROPERTY);
476 		writeAttribute(writer, NAME, value.getProperty());
477 		CSSProperty p = value.getCSSProperty();
478 		if (p == Content.NONE || p == Content.INITIAL)
479 			;
480 		else if (p == Content.INHERIT)
481 			writeAttribute(writer, VALUE, Content.INHERIT.toString());
482 		else if (p == Content.content_list) {
483 			if (value.getValue() instanceof ContentList)
484 				toXml((ContentList)value.getValue(), writer);
485 			else
486 				throw new IllegalArgumentException();
487 		} else if (p == StringSet.NONE || p == StringSet.INITIAL)
488 			;
489 		else if (p == StringSet.INHERIT)
490 			writeAttribute(writer, VALUE, StringSet.INHERIT.toString());
491 		else if (p == StringSet.list_values)
492 			if (value.getValue() instanceof StringSetList)
493 				toXml((StringSetList)value.getValue(), writer);
494 			else
495 				throw new IllegalArgumentException();
496 		else if (p == CounterSet.NONE || p == CounterSet.INITIAL ||
497 		         p == CounterReset.NONE || p == CounterReset.INITIAL ||
498 		         p == CounterIncrement.NONE || p == CounterIncrement.INITIAL)
499 			;
500 		else if (p == CounterSet.INHERIT ||
501 		         p == CounterReset.INHERIT ||
502 		         p == CounterIncrement.INHERIT)
503 			writeAttribute(writer, VALUE, "inherit");
504 		else if (p == CounterSet.list_values ||
505 		         p == CounterReset.list_values ||
506 		         p == CounterIncrement.list_values)
507 			if (value.getValue() instanceof CounterSetList)
508 				toXml((CounterSetList)value.getValue(), writer);
509 			else
510 				throw new IllegalArgumentException();
511 		else
512 			writeAttribute(writer, VALUE, serializePropertyValue(value));
513 		writer.writeEndElement();
514 	}

516 	public static void toXml(ContentList list, XMLStreamWriter w) throws XMLStreamException {
517 		for (Term<?> i : list)
518 			if (i instanceof TermString || i instanceof TermURI) {
519 				Term<String> s = (Term<String>)i;
520 				writeStartElement(w, CSS_STRING);
521 				writeAttribute(w, VALUE, s.getValue());
522 				w.writeEndElement(); }
523 			else if (i instanceof AttrFunction) {
524 				AttrFunction f = (AttrFunction)i;
525 				writeStartElement(w, CSS_ATTR);
526 				writeAttribute(w, NAME, f.name.getValue());
527 				w.writeEndElement(); }
528 			else if (i instanceof ContentFunction) {
529 				ContentFunction f = (ContentFunction)i;
530 				writeStartElement(w, CSS_CONTENT);
531 				if (f.target.isPresent())
532 					if (f.target.get().url != null) {
533 						TermURI t = f.target.get().url;
534 						URI url = URLs.asURI(t.getValue());
535 						if (t.getBase() != null)
536 							url = URLs.resolve(URLs.asURI(t.getBase()), url);
537 						writeAttribute(w, TARGET, url.toString());
538 					} else
539 						writeAttribute(w, TARGET_ATTRIBUTE, f.target.get().urlAttr.name.getValue());
540 				w.writeEndElement(); }
541 			else if (i instanceof StringFunction) {
542 				StringFunction f = (StringFunction)i;
543 				writeStartElement(w, CSS_STRING);
544 				writeAttribute(w, NAME, f.name.getValue());
545 				if (f.target.isPresent())
546 					if (f.target.get().url != null) {
547 						TermURI t = f.target.get().url;
548 						URI url = URLs.asURI(t.getValue());
549 						if (t.getBase() != null)
550 							url = URLs.resolve(URLs.asURI(t.getBase()), url);
551 						writeAttribute(w, TARGET, url.toString());
552 					} else
553 						writeAttribute(w, TARGET_ATTRIBUTE, f.target.get().urlAttr.name.getValue());
554 				else if (f.scope.isPresent())
555 					writeAttribute(w, SCOPE, f.scope.get().toString());
556 				w.writeEndElement(); }
557 			else if (i instanceof CounterFunction) {
558 				CounterFunction f = (CounterFunction)i;
559 				writeStartElement(w, CSS_COUNTER);
560 				writeAttribute(w, NAME, f.name.getValue());
561 				if (f.target.isPresent())
562 					if (f.target.get().url != null) {
563 						TermURI t = f.target.get().url;
564 						URI url = URLs.asURI(t.getValue());
565 						if (t.getBase() != null)
566 							url = URLs.resolve(URLs.asURI(t.getBase()), url);
567 						writeAttribute(w, TARGET, url.toString());
568 					} else
569 						writeAttribute(w, TARGET_ATTRIBUTE, f.target.get().urlAttr.name.getValue());
570 				if (f.style.isPresent())
571 					writeAttribute(w, STYLE, toString(f.style.get()));
572 				w.writeEndElement(); }
573 			else if (i instanceof TextFunction) {
574 				TextFunction f = (TextFunction)i;
575 				writeStartElement(w, CSS_TEXT);
576 				if (f.target.url != null) {
577 					TermURI t = f.target.url;
578 					URI url = URLs.asURI(t.getValue());
579 					if (t.getBase() != null)
580 						url = URLs.resolve(URLs.asURI(t.getBase()), url);
581 					writeAttribute(w, TARGET, url.toString());
582 				} else
583 					writeAttribute(w, TARGET_ATTRIBUTE, f.target.urlAttr.name.getValue());
584 				w.writeEndElement(); }
585 			else if (i instanceof LeaderFunction) {
586 				LeaderFunction f = (LeaderFunction)i;
587 				writeStartElement(w, CSS_LEADER);
588 				writeAttribute(w, PATTERN, f.pattern.getValue());
589 				if (f.position.isPresent())
590 					writeAttribute(w, POSITION, toString(f.position.get()));
591 				if (f.alignment.isPresent())
592 					writeAttribute(w, ALIGNMENT, f.alignment.get().toString());
593 				w.writeEndElement(); }
594 			else if (i instanceof FlowFunction) {
595 				FlowFunction f = (FlowFunction)i;
596 				writeStartElement(w, CSS_FLOW);
597 				writeAttribute(w, FROM, f.from.getValue());
598 				if (f.scope.isPresent())
599 					writeAttribute(w, SCOPE, f.scope.get().toString());
600 				w.writeEndElement(); }
601 			else if (i instanceof TermFunction) {
602 				TermFunction f = (TermFunction)i;
603 				writeStartElement(w, CSS_CUSTOM_FUNC);
604 				writeAttribute(w, NAME, f.getFunctionName());
605 				int k = 0;
606 				for (Term<?> arg : f) {
607 					k++;
608 					writeAttribute(w, new QName("arg" + k), toString(arg)); }
609 				w.writeEndElement(); }
610 			else
611 				throw new RuntimeException("coding error");
612 	}
