114 	public BrailleCssStyle parseInlineStyle(String style, Context context) {
115 		if (context == null)
116 			throw new IllegalArgumentException();
117 		BrailleCssStyle s = cache.get(context, style);
118 		if (s == null) {
119 			// try if a declaration was cached
120 			if (context == Context.ELEMENT) {
121 				Declaration d = declCache.get(style);
122 				if (d != null)
123 					s = BrailleCssStyle.of(d);
124 			}
125 			if (s == null)
126 				s = BrailleCssStyle.of(this, context, style);
127 			cache.put(context, style, s);
128 		}
129 		return s;
130 	}

133 	 * Concretizes "inherit" even if the parent style is null or empty.
134 	 *
135 	 * The returned object is immutable.
136 	 */
137 	public BrailleCssStyle parseInlineStyle(String style, Context context, BrailleCssStyle parent) {
138 		if (context != Context.ELEMENT)
139 			throw new IllegalArgumentException();
140 		ParsedDeclarations parentDecls; {
141 			if (parent == null || parent.declarations == null)
142 				parentDecls = ParsedDeclarations.EMPTY;
143 			else if (!(parent.declarations instanceof ParsedDeclarations))
144 				throw new IllegalArgumentException();
145 			else
146 				parentDecls = (ParsedDeclarations)parent.declarations;
147 		}
148 		BrailleCssStyle s = cache.get(context, style, parentDecls, true);
149 		if (s == null) {
150 			s = parseInlineStyle(style, context);
151 			s = s.inheritFrom(parentDecls);
152 			cache.put(context, style, parentDecls, true, s);
153 		}
154 		return s;
155 	}

158 	 * Style assumed to be specified in the context of a (pseudo-)element.
159 	 *
160 	 * @param context element for evaluating attr() and content() values against.
161 	 * @param mutable Whether the caller wishes to mutate the returned declaration.

163 	public Optional<Declaration> parseDeclaration(String property, String value, Element context, boolean mutable) {
164 		String style = String.format("%s: %s", property, value);
165 		Declaration declaration = declCache.get(style);
166 		if (declaration == null) {
167 			declaration = parseInlineStyle(style, Context.ELEMENT).getDeclaration(property);
168 			if (declaration == null)
169 				return Optional.empty();
170 			declCache.put(style, declaration);
171 		}
172 		if (context != null) {
173 			Declaration evaluated = BrailleCssStyle.evaluateDeclaration(declaration, context);
174 			if (evaluated != declaration)
175 				return Optional.of(evaluated);
176 		}
177 		if (mutable) {
178 			Declaration unlocked = BrailleCssStyle.unlockDeclaration(declaration);
179 			if (unlocked != declaration)
180 				return Optional.of(unlocked);
181 			return Optional.of((Declaration)declaration.clone());
182 		}
183 		return Optional.of(declaration);
184 	}

343 		@Override
344 		public ParsedDeclarations inheritFrom(NodeData parent) {
345 			if (!(parent instanceof ParsedDeclarations))
346 				throw new IllegalArgumentException();
347 			ParsedDeclarations r = (ParsedDeclarations)super.inheritFrom(parent);
348 			BrailleCssParser p = ((ParsedDeclarations)parent).parser;
349 			if (r.parser == null)
350 				r.parser = p;
351 			else if (p != null && r.parser != p)
352 				throw new IllegalArgumentException();
353 			Context c = ((ParsedDeclarations)parent).context;
354 			if (r.context == null)
355 				r.context = c;
356 			else if (c != null && r.context != c)
357 				throw new IllegalArgumentException();
358 			r.checkParserAndContext();
359 			if (!r.isEmpty()) {
360 				// Make sure that the resulting SimpleInlineStyle is not based on a parent and that it
361 				// is not concretized because that would result in an exception when inheritFrom() is
362 				// called on it. Note that concretize() does not have the same effect.
363 				// Also perform special inheritance of text-transform.
364 				List<PropertyValue> list = new ArrayList<>();
365 				r.forEach(list::add);
366 				ListIterator<PropertyValue> i = list.listIterator();
367 				while (i.hasNext()) {
368 					ParsedDeclaration v = (ParsedDeclaration)i.next();
369 					PropertyValue flat = null;
370 					if (!((ParsedDeclarations)parent).isEmpty() && v.getCSSProperty() instanceof TextTransform)
371 						for (PropertyValue vv : (ParsedDeclarations)parent)
372 							if (vv.getCSSProperty() instanceof TextTransform) {
373 								TextTransformList t = TextTransformList.of(v);
374 								t.inheritFrom(TextTransformList.of(vv)); // this mutates the value
375 								if (t != v.getValue())
376 									flat = new ParsedDeclaration(
377 										v.getParser(),
378 										v.getContext(),
379 										v.getProperty(),
380 										t.equalsAuto() ? TextTransform.AUTO
381 										               : t.equalsInitial() ? TextTransform.INITIAL
382 										                                   : t.equalsNone() ? TextTransform.NONE
383 										                                                    : TextTransform.list_values,
384 										t.equalsAuto() || t.equalsInitial() || t.equalsNone() ? null : t,
385 										v.getSourceDeclaration());
386 								break;
387 							}
388 					if (flat == null)
389 						flat = new ParsedDeclaration(
390 							v.getParser(),
391 							v.getContext(),
392 							v.getProperty(),
393 							v.getCSSProperty(),
394 							v.getValue(),
395 							v.getSourceDeclaration());
396 					i.set(flat);
397 				}
398 				r = new ParsedDeclarations(r.parser, r.context, list);
399 			}
400 			return r;
401 		}

437 				this.parser = ((Quadruple)this.propertyValue).getParser();

447 				this.context = context;
448 				this.propertyValue = new Quadruple(this.getProperty(),
449 				                                   this.getCSSProperty(),
450 				                                   this.getValue(),
451 				                                   this.getSourceDeclaration(),
452 				                                   this.getSupportedBrailleCSS());
453 			}
454 			this.quadruple = (Quadruple)this.propertyValue;
455 			this.enableCaching = (this.context == Context.ELEMENT);
456 		}

502 			if (valueSerialized == null) {
503 				valueSerialized = serializer.serializePropertyValue(new PropertyValue(this));
504 				serialized = getProperty() + ": " + valueSerialized;
505 				if (enableCaching)
506 					parser.declCache.put(serialized, this);
507 			} else {
508 				// access cache to keep entry longer in it
509 				if (enableCaching)
510 					parser.declCache.get(serialized);
511 			}
512 			return valueSerialized;
513 		}

515 		@Override
516 		public ParsedDeclaration clone() {
517 			ParsedDeclaration clone = (ParsedDeclaration)super.clone();
518 			// Clones are never cached because we assume that clones are made to expose a
519 			// declaration to the outside, where they may possibly be mutated. If a clone is made
520 			// for another purpose, such as for running evaluate() on it, caching might be
521 			// desirable. Luckily, evalutation is normally already done before a declaration is
522 			// serialized. An alternative would be to cache copies of the object every time
523 			// toString() is called, and to also make copies when objects are retreived from the
524 			// cache. This would allow objects to be mutated.
525 			clone.enableCaching = false;
526 			// forget serialization because object may be mutated after it has been cloned
527 			clone.valueSerialized = null;
528 			clone.serialized = null;
529 			return clone;
530 		}

585 	private static class CacheKey implements Comparable<CacheKey> {
586 		private final Context context;
587 		private final String style;
588 		private final String parent;
589 		private final boolean concretizeInherit;
590 		public CacheKey(Context context, String style, ParsedDeclarations parent, boolean concretizeInherit) {
591 			if (parent != null && context != Context.ELEMENT)
592 				throw new IllegalArgumentException();
593 			this.context = context;
594 			this.style = style;
595 			this.parent = parent != null ? serializer.toString(parent) : null;
596 			this.concretizeInherit = concretizeInherit;
597 		}
598 		@Override
599 		public int compareTo(CacheKey that) {
600 			int i = this.context.compareTo(that.context);
601 			if (i != 0)
602 				return i;
603 			i = this.style.compareTo(that.style);
604 			if (i != 0)
605 				return i;
606 			if (this.parent == null)
607 				return that.parent == null ? 0 : -1;
608 			else if (that.parent == null)
609 				return 1;
610 			i = this.parent.compareTo(that.parent);
611 			if (i != 0)
612 				return i;
613 			if (this.concretizeInherit == that.concretizeInherit)
614 				return 0;
615 			else
616 				return this.concretizeInherit ? -1 : 1;
617 		}
618 		@Override
619 		public int hashCode() {
620 			final int prime = 31;
621 			int hash = 1;
622 			hash = prime * hash + context.hashCode();
623 			hash = prime * hash + style.hashCode();
624 			hash = prime * hash + (parent == null ? 0 : parent.hashCode());
625 			hash = prime * hash + Boolean.hashCode(concretizeInherit);
626 			return hash;
627 		}
628 		@Override
629 		public boolean equals(Object o) {
630 			if (o == null)
631 				return false;
632 			if (!(o instanceof CacheKey))
633 				return false;
634 			CacheKey that = (CacheKey)o;
635 			if (this.context != that.context)
636 				return false;
637 			if (!this.style.equals(that.style))
638 				return false;
639 			if (this.parent == null)
640 				return that.parent == null;
641 			else if (that.parent == null)
642 				return false;
643 			else if (!this.parent.equals(that.parent))
644 				return false;
645 			else
646 				return this.concretizeInherit == that.concretizeInherit;
647 		}

663 		@Override
664 		public boolean parseDeclaration(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
665 			if ("text-transform".equalsIgnoreCase(d.getProperty())) {
666 				if (super.processTextTransform(d, properties, values)) {
667 					if (properties.get("text-transform") == TextTransform.list_values) {
668 						Term<?> value = values.get("text-transform");
669 						if (value instanceof TermList)
670 							values.put("text-transform", TextTransformList.of((TermList)value));
671 						else
672 							throw new IllegalStateException(); // should not happen
673 					}
674 					return true;
675 				} else
676 					return false;
677 			} else if ("content".equalsIgnoreCase(d.getProperty())) {
678 				if (super.processContent(d, properties, values)) {
679 					if (properties.get("content") == Content.content_list) {
680 						Term<?> value = values.get("content");
681 						if (value instanceof TermList) {
682 							ContentList l = ContentList.of(BrailleCssParser.this, extensions, context, (TermList)value);
683 							for (Term<?> t : l)
684 								if (t instanceof ContentFunction && !((ContentFunction)t).target.isPresent())
685 									throw new IllegalArgumentException("unexpected term in content list: " + t);
686 							values.put("content", l);
687 						} else
688 							throw new IllegalStateException(); // should not happen
689 					}
690 					return true;
691 				} else
692 					return false;
693 			} else if ("string-set".equalsIgnoreCase(d.getProperty())) {
694 				if (super.processStringSet(d, properties, values)) {

699 						else
700 							throw new IllegalStateException(); // should not happen
701 					}
702 					return true;
703 				} else
704 					return false;
705 			} else if ("counter-set".equalsIgnoreCase(d.getProperty())) {
706 				if (super.processCounterSet(d, properties, values)) {
707 					if (properties.get("counter-set") == CounterSet.list_values) {

711 						else
712 							throw new IllegalStateException(); // should not happen
713 					}
714 					return true;
715 				} else
716 					return false;
717 			} else if ("counter-reset".equalsIgnoreCase(d.getProperty())) {
718 				if (super.processCounterReset(d, properties, values)) {
719 					if (properties.get("counter-reset") == CounterReset.list_values) {

731 					if (properties.get("counter-increment") == CounterIncrement.list_values) {

757 			try {
758 				return FrameworkUtil.getBundle(OSGiHelper.class) != null;
759 			} catch (NoClassDefFoundError e) {
760 				return false;
761 			}
762 		}

764 		static Iterator<BrailleCSSExtension> getCSSExtensions() {
765 			return new Iterator<BrailleCSSExtension>() {
766 				private BundleContext bc = FrameworkUtil.getBundle(BrailleCssParser.class).getBundleContext();
767 				private ServiceReference[] refs;
768 				private int index = 0;  {
769 					try {
770 						refs = bc.getServiceReferences(BrailleCSSExtension.class.getName(), null);
771 					} catch (InvalidSyntaxException e) {
772 						throw new IllegalStateException(e); // should not happen
773 					}
774 				}
775 				public boolean hasNext() {
776 					return refs != null && index < refs.length;
777 				}
778 				public BrailleCSSExtension next() throws NoSuchElementException {
779 					if (!hasNext())
780 						throw new NoSuchElementException();
781 					return (BrailleCSSExtension)bc.getService(refs[index++]);
782 				}
783 				public void remove() {
784 					throw new UnsupportedOperationException();
785 				}
786 			};
787 		}
788 	}
