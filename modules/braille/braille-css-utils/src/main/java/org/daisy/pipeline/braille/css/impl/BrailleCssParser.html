113 	public BrailleCssStyle parseInlineStyle(String style, Context context) {
114 		if (context == null)
115 			throw new IllegalArgumentException();
116 		BrailleCssStyle s = cache.get(context, style);
117 		if (s == null) {
118 			// try if a declaration was cached
119 			if (context == Context.ELEMENT) {
120 				Declaration d = declCache.get(style);
121 				if (d != null)
122 					s = BrailleCssStyle.of(d);
123 			}
124 			if (s == null)
125 				s = BrailleCssStyle.of(this, context, style);
126 			cache.put(context, style, s);
127 		}
128 		return s;
129 	}

132 	 * Concretizes "inherit" even if the parent style is null or empty.
133 	 *
134 	 * The returned object is immutable.
135 	 */
136 	public BrailleCssStyle parseInlineStyle(String style, Context context, BrailleCssStyle parent) {
137 		if (context != Context.ELEMENT)
138 			throw new IllegalArgumentException();
139 		ParsedDeclarations parentDecls; {
140 			if (parent == null || parent.declarations == null)
141 				parentDecls = ParsedDeclarations.EMPTY;
142 			else if (!(parent.declarations instanceof ParsedDeclarations))
143 				throw new IllegalArgumentException();
144 			else
145 				parentDecls = (ParsedDeclarations)parent.declarations;
146 		}
147 		BrailleCssStyle s = cache.get(context, style, parentDecls, true);
148 		if (s == null) {
149 			s = parseInlineStyle(style, context);
150 			s = s.inheritFrom(parentDecls);
151 			cache.put(context, style, parentDecls, true, s);
152 		}
153 		return s;
154 	}

157 	 * Style assumed to be specified in the context of a (pseudo-)element.
158 	 *
159 	 * @param context element for evaluating attr() and content() values against.
160 	 * @param mutable Whether the caller wishes to mutate the returned declaration.
161 	 */
162 	public Optional<Declaration> parseDeclaration(String property, String value, Element context, boolean mutable) {
163 		String style = String.format("%s: %s", property, value);
164 		Declaration declaration = declCache.get(style);
165 		if (declaration == null) {
166 			declaration = parseInlineStyle(style, Context.ELEMENT).getDeclaration(property);
167 			if (declaration == null)
168 				return Optional.empty();
169 			declCache.put(style, declaration);
170 		}
171 		if (context != null) {
172 			Declaration evaluated = BrailleCssStyle.evaluateDeclaration(declaration, context);
173 			if (evaluated != declaration)
174 				return Optional.of(evaluated);
175 		}
176 		if (mutable) {
177 			Declaration unlocked = BrailleCssStyle.unlockDeclaration(declaration);
178 			if (unlocked != declaration)
179 				return Optional.of(unlocked);
180 			return Optional.of((Declaration)declaration.clone());
181 		}
182 		return Optional.of(declaration);
183 	}

342 		@Override
343 		public ParsedDeclarations inheritFrom(NodeData parent) {
344 			if (!(parent instanceof ParsedDeclarations))
345 				throw new IllegalArgumentException();
346 			ParsedDeclarations r = (ParsedDeclarations)super.inheritFrom(parent);
347 			BrailleCssParser p = ((ParsedDeclarations)parent).parser;
348 			if (r.parser == null)
349 				r.parser = p;
350 			else if (p != null && r.parser != p)
351 				throw new IllegalArgumentException();
352 			Context c = ((ParsedDeclarations)parent).context;
353 			if (r.context == null)
354 				r.context = c;
355 			else if (c != null && r.context != c)
356 				throw new IllegalArgumentException();
357 			r.checkParserAndContext();
358 			if (!r.isEmpty()) {
359 				// Make sure that the resulting SimpleInlineStyle is not based on a parent and that it
360 				// is not concretized because that would result in an exception when inheritFrom() is
361 				// called on it. Note that concretize() does not have the same effect.
362 				// Also perform special inheritance of text-transform.
363 				List<PropertyValue> list = new ArrayList<>();
364 				r.forEach(list::add);
365 				ListIterator<PropertyValue> i = list.listIterator();
366 				while (i.hasNext()) {
367 					ParsedDeclaration v = (ParsedDeclaration)i.next();
368 					PropertyValue flat = null;
369 					if (!((ParsedDeclarations)parent).isEmpty() && v.getCSSProperty() instanceof TextTransform)
370 						for (PropertyValue vv : (ParsedDeclarations)parent)
371 							if (vv.getCSSProperty() instanceof TextTransform) {
372 								TextTransformList t = TextTransformList.of(v);
373 								t.inheritFrom(TextTransformList.of(vv)); // this mutates the value
374 								if (t != v.getValue())
375 									flat = new ParsedDeclaration(
376 										v.getParser(),
377 										v.getContext(),
378 										v.getProperty(),
379 										t.equalsAuto() ? TextTransform.AUTO
380 										               : t.equalsInitial() ? TextTransform.INITIAL
381 										                                   : t.equalsNone() ? TextTransform.NONE
382 										                                                    : TextTransform.list_values,
383 										t.equalsAuto() || t.equalsInitial() || t.equalsNone() ? null : t,
384 										v.getSourceDeclaration());
385 								break;
386 							}
387 					if (flat == null)
388 						flat = new ParsedDeclaration(
389 							v.getParser(),
390 							v.getContext(),
391 							v.getProperty(),
392 							v.getCSSProperty(),
393 							v.getValue(),
394 							v.getSourceDeclaration());
395 					i.set(flat);
396 				}
397 				r = new ParsedDeclarations(r.parser, r.context, list);
398 			}
399 			return r;
400 		}

429 		/**

431 		 * @param context If {@code null}, context is derived from declaration
432 		 */
433 		ParsedDeclaration(BrailleCssParser parser, Context context, PropertyValue declaration) {
434 			super(declaration);
435 			if (this.propertyValue instanceof Quadruple) {
436 				this.parser = ((Quadruple)this.propertyValue).getParser();
437 				this.context = ((Quadruple)this.propertyValue).getContext();
438 				if (parser != null && this.parser != parser)
439 					throw new IllegalArgumentException();
440 				if (context != null && this.context != context)
441 					throw new IllegalArgumentException();
442 			} else {
443 				if (parser == null || context == null)
444 					throw new IllegalArgumentException();
445 				this.parser = parser;
446 				this.context = context;
447 				this.propertyValue = new Quadruple(this.getProperty(),
448 				                                   this.getCSSProperty(),
449 				                                   this.getValue(),
450 				                                   this.getSourceDeclaration(),
451 				                                   this.getSupportedBrailleCSS());
452 			}
453 			this.quadruple = (Quadruple)this.propertyValue;
454 			this.enableCaching = (this.context == Context.ELEMENT);
455 		}

503 				serialized = getProperty() + ": " + valueSerialized;
504 				if (enableCaching)

506 			} else {
507 				// access cache to keep entry longer in it
508 				if (enableCaching)
509 					parser.declCache.get(serialized);
510 			}
511 			return valueSerialized;

514 		@Override
515 		public ParsedDeclaration clone() {
516 			ParsedDeclaration clone = (ParsedDeclaration)super.clone();
517 			// Clones are never cached because we assume that clones are made to expose a
518 			// declaration to the outside, where they may possibly be mutated. If a clone is made
519 			// for another purpose, such as for running evaluate() on it, caching might be
520 			// desirable. Luckily, evalutation is normally already done before a declaration is
521 			// serialized. An alternative would be to cache copies of the object every time
522 			// toString() is called, and to also make copies when objects are retreived from the
523 			// cache. This would allow objects to be mutated.
524 			clone.enableCaching = false;
525 			// forget serialization because object may be mutated after it has been cloned
526 			clone.valueSerialized = null;
527 			clone.serialized = null;
528 			return clone;
529 		}

584 	private static class CacheKey implements Comparable<CacheKey> {
585 		private final Context context;
586 		private final String style;
587 		private final String parent;
588 		private final boolean concretizeInherit;
589 		public CacheKey(Context context, String style, ParsedDeclarations parent, boolean concretizeInherit) {
590 			if (parent != null && context != Context.ELEMENT)
591 				throw new IllegalArgumentException();
592 			this.context = context;
593 			this.style = style;
594 			this.parent = parent != null ? BrailleCssSerializer.toString(parent) : null;
595 			this.concretizeInherit = concretizeInherit;
596 		}
597 		@Override
598 		public int compareTo(CacheKey that) {
599 			int i = this.context.compareTo(that.context);
600 			if (i != 0)
601 				return i;
602 			i = this.style.compareTo(that.style);
603 			if (i != 0)
604 				return i;
605 			if (this.parent == null)
606 				return that.parent == null ? 0 : -1;
607 			else if (that.parent == null)
608 				return 1;
609 			i = this.parent.compareTo(that.parent);
610 			if (i != 0)
611 				return i;
612 			if (this.concretizeInherit == that.concretizeInherit)
613 				return 0;
614 			else
615 				return this.concretizeInherit ? -1 : 1;
616 		}
617 		@Override
618 		public int hashCode() {
619 			final int prime = 31;
620 			int hash = 1;
621 			hash = prime * hash + context.hashCode();
622 			hash = prime * hash + style.hashCode();
623 			hash = prime * hash + (parent == null ? 0 : parent.hashCode());
624 			hash = prime * hash + Boolean.hashCode(concretizeInherit);
625 			return hash;
626 		}
627 		@Override
628 		public boolean equals(Object o) {
629 			if (o == null)
630 				return false;
631 			if (!(o instanceof CacheKey))
632 				return false;
633 			CacheKey that = (CacheKey)o;
634 			if (this.context != that.context)
635 				return false;
636 			if (!this.style.equals(that.style))
637 				return false;
638 			if (this.parent == null)
639 				return that.parent == null;
640 			else if (that.parent == null)
641 				return false;
642 			else if (!this.parent.equals(that.parent))
643 				return false;
644 			else
645 				return this.concretizeInherit == that.concretizeInherit;
646 		}

662 		@Override
663 		public boolean parseDeclaration(Declaration d, Map<String,CSSProperty> properties, Map<String,Term<?>> values) {
664 			if ("text-transform".equalsIgnoreCase(d.getProperty())) {
665 				if (super.processTextTransform(d, properties, values)) {
666 					if (properties.get("text-transform") == TextTransform.list_values) {
667 						Term<?> value = values.get("text-transform");
668 						if (value instanceof TermList)
669 							values.put("text-transform", TextTransformList.of((TermList)value));
670 						else
671 							throw new IllegalStateException(); // should not happen
672 					}
673 					return true;
674 				} else
675 					return false;
676 			} else if ("content".equalsIgnoreCase(d.getProperty())) {
677 				if (super.processContent(d, properties, values)) {
678 					if (properties.get("content") == Content.content_list) {
679 						Term<?> value = values.get("content");
680 						if (value instanceof TermList) {
681 							ContentList l = ContentList.of(BrailleCssParser.this, extensions, context, (TermList)value);
682 							for (Term<?> t : l)
683 								if (t instanceof ContentFunction && !((ContentFunction)t).target.isPresent())
684 									throw new IllegalArgumentException("unexpected term in content list: " + t);
685 							values.put("content", l);
686 						} else
687 							throw new IllegalStateException(); // should not happen
688 					}
689 					return true;
690 				} else
691 					return false;
692 			} else if ("string-set".equalsIgnoreCase(d.getProperty())) {
693 				if (super.processStringSet(d, properties, values)) {
694 					if (properties.get("string-set") == StringSet.list_values) {
695 						Term<?> value = values.get("string-set");

698 						else
699 							throw new IllegalStateException(); // should not happen
700 					}
701 					return true;
702 				} else
703 					return false;
704 			} else if ("counter-set".equalsIgnoreCase(d.getProperty())) {
705 				if (super.processCounterSet(d, properties, values)) {
706 					if (properties.get("counter-set") == CounterSet.list_values) {

710 						else
711 							throw new IllegalStateException(); // should not happen
712 					}
713 					return true;
714 				} else
715 					return false;
716 			} else if ("counter-reset".equalsIgnoreCase(d.getProperty())) {
717 				if (super.processCounterReset(d, properties, values)) {
718 					if (properties.get("counter-reset") == CounterReset.list_values) {
719 						Term<?> value = values.get("counter-reset");

722 						else
723 							throw new IllegalStateException(); // should not happen

729 				if (super.processCounterIncrement(d, properties, values)) {
730 					if (properties.get("counter-increment") == CounterIncrement.list_values) {
731 						Term<?> value = values.get("counter-increment");

756 			try {
757 				return FrameworkUtil.getBundle(OSGiHelper.class) != null;
758 			} catch (NoClassDefFoundError e) {
759 				return false;
760 			}
761 		}

763 		static Iterator<BrailleCSSExtension> getCSSExtensions() {
764 			return new Iterator<BrailleCSSExtension>() {
765 				private BundleContext bc = FrameworkUtil.getBundle(BrailleCssParser.class).getBundleContext();
766 				private ServiceReference[] refs;
767 				private int index = 0;  {
768 					try {
769 						refs = bc.getServiceReferences(BrailleCSSExtension.class.getName(), null);
770 					} catch (InvalidSyntaxException e) {
771 						throw new IllegalStateException(e); // should not happen
772 					}
773 				}
774 				public boolean hasNext() {
775 					return refs != null && index < refs.length;
776 				}
777 				public BrailleCSSExtension next() throws NoSuchElementException {
778 					if (!hasNext())
779 						throw new NoSuchElementException();
780 					return (BrailleCSSExtension)bc.getService(refs[index++]);
781 				}
782 				public void remove() {
783 					throw new UnsupportedOperationException();
784 				}
785 			};
786 		}
