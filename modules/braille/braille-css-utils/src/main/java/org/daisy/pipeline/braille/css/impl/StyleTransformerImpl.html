1   package org.daisy.pipeline.braille.css.impl;
2   
3   import java.util.ArrayList;
4   import java.util.HashMap;
5   import java.util.HashSet;
6   import java.util.List;
7   import java.util.Map;
8   import java.util.Objects;
9   import java.util.Optional;
10  import java.util.Set;
11  
12  import cz.vutbr.web.css.CSSProperty;
13  import cz.vutbr.web.css.Declaration;
14  import cz.vutbr.web.css.Term;
15  
16  import org.daisy.braille.css.BrailleCSSParserFactory;
17  import org.daisy.braille.css.BrailleCSSParserFactory.Context;
18  import org.daisy.braille.css.PropertyValue;
19  import org.daisy.braille.css.SimpleInlineStyle;
20  import org.daisy.braille.css.SupportedBrailleCSS;
21  import org.daisy.pipeline.braille.css.impl.BrailleCssParser.ParsedDeclaration;
22  import org.daisy.pipeline.braille.css.impl.BrailleCssParser.ParsedDeclarations;
23  import org.daisy.pipeline.braille.css.StyleTransformer;
24  import org.daisy.pipeline.braille.css.xpath.Style;
25  import org.daisy.pipeline.braille.css.xpath.impl.Stylesheet;
26  
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  public class StyleTransformerImpl implements StyleTransformer {
31  
32  	private final static Logger logger = LoggerFactory.getLogger(StyleTransformer.class);
33  
34  	private final SupportedBrailleCSS fromSupportedBrailleCSS;
35  	private final SupportedBrailleCSS toSupportedBrailleCSS;
36  	private final BrailleCssParser parser;
37  	private final Set<String> propertiesWithDifferentDefault;
38  
39  	public StyleTransformerImpl(SupportedBrailleCSS fromSupportedBrailleCSS,
40  	                            SupportedBrailleCSS toSupportedBrailleCSS) {
41  		this.fromSupportedBrailleCSS = fromSupportedBrailleCSS;
42  		this.toSupportedBrailleCSS = toSupportedBrailleCSS;
43  		this.parser = new BrailleCssParser() {
44  				@Override
45  				public BrailleCSSParserFactory getBrailleCSSParserFactory() {
46  					return null; }; // should be fine to return null (assuming parseInlineStyle() will not
47  					                // be called on this parser object)
48  				@Override
49  				public Optional<SupportedBrailleCSS> getSupportedBrailleCSS(Context context) {
50  					return Optional.of(toSupportedBrailleCSS); }};
51  		this.propertiesWithDifferentDefault = new HashSet<>(); {
52  			for (String p : toSupportedBrailleCSS.getDefinedPropertyNames())
53  				if (fromSupportedBrailleCSS.isSupportedCSSProperty(p))
54  					if ("".equals(fromSupportedBrailleCSS.getDefaultProperty(p).toString())
55  					    ? (Objects.equals(fromSupportedBrailleCSS.getDefaultProperty(p),
56  					                      toSupportedBrailleCSS.getDefaultProperty(p)) &&
57  					       Objects.equals(fromSupportedBrailleCSS.getDefaultValue(p),
58  					                      toSupportedBrailleCSS.getDefaultValue(p)))
59  					    : Objects.equals(fromSupportedBrailleCSS.getDefaultProperty(p).toString(),
60  					                     toSupportedBrailleCSS.getDefaultProperty(p).toString()))
61  						; // same default
62  					else
63  						propertiesWithDifferentDefault.add(p);
64  		}
65  	}
66  
67  	@Override
68  	public final SimpleInlineStyle transform(SimpleInlineStyle style) {
69  		Map<String,CSSProperty> properties = new HashMap<>();
70  		Map<String,Term<?>> values = new HashMap<>();
71  		Map<String,Declaration> sourceDeclarations = new HashMap<>();
72  		List<ParsedDeclaration> declarations = new ArrayList<>();
73  		for (PropertyValue d : style) {
74  			if (d.getSupportedBrailleCSS() != fromSupportedBrailleCSS)
75  				throw new IllegalArgumentException();
76  			String p = d.getProperty();
77  			if (!toSupportedBrailleCSS.isSupportedCSSProperty(p)) {
78  				// ignore "inherit" (if style is based on parent, inherit has already been concretized;
79  				//                 if not based on parent, this is equivalent to "initial")
80  				if (d.getCSSProperty().equalsInherit())
81  					;
82  				// ignore "initial" (should already have been concretized; properties with different
83  				//                   default in the output model are handled below)
84  				else if (d.getCSSProperty().equalsInitial())
85  					;
86  				// ignore values that are the default (equivalent to "initial")
87  				else if (Objects.equals(d.getCSSProperty(), fromSupportedBrailleCSS.getDefaultProperty(p)) &&
88  				         Objects.equals(d.getValue(), fromSupportedBrailleCSS.getDefaultValue(p)))
89  					;
90  				else
91  					logger.warn("Property '{}' not supported", p);
92  			} else if (!toSupportedBrailleCSS.parseDeclaration(d, properties, values))
93  				logger.warn("Property '{}' not supported", d);
94  			else
95  				sourceDeclarations.put(p, d.getSourceDeclaration());
96  		}
97  		for (String p : toSupportedBrailleCSS.getDefinedPropertyNames()) {
98  			CSSProperty property = null;
99  			Term<?> value = null;
100 			if (properties.containsKey(p))
101 				declarations.add(new ParsedDeclaration(parser,
102 				                                       Context.ELEMENT,
103 				                                       p,
104 				                                       properties.get(p),
105 				                                       values.get(p),
106 				                                       sourceDeclarations.get(p)));
107 			else if (propertiesWithDifferentDefault.contains(p))
108 				declarations.add(new ParsedDeclaration(parser,
109 				                                       Context.ELEMENT,
110 				                                       p,
111 				                                       toSupportedBrailleCSS.getDefaultProperty(p),
112 				                                       toSupportedBrailleCSS.getDefaultValue(p),
113 				                                       null));
114 		}
115 		return new ParsedDeclarations(parser, Context.ELEMENT, declarations);
116 	}
117 
118 	@Override
119 	public final Style transform(Style style) {
120 		if (!(style instanceof Stylesheet))
121 			throw new IllegalArgumentException();
122 		Stylesheet fs = (Stylesheet)style;
123 		SimpleInlineStyle s = fs.style != null
124 			? fs.style.asSimpleInlineStyle(false)
125 			: SimpleInlineStyle.EMPTY;
126 		s = transform(s);
127 		return Stylesheet.of(BrailleCssStyle.of(s));
128 	}
129 }
