4   import java.util.Map;

6   import cz.vutbr.web.css.TermInteger;

8   import cz.vutbr.web.css.TermNumeric;

10  import org.daisy.pipeline.css.Dimension;
11  import org.daisy.pipeline.css.Dimension.Unit;
12  import org.daisy.pipeline.css.Medium;
13  
14  public class EmbossedMedium extends Medium {
15  
16  	private final double cellWidth;
17  	private final double cellHeight;
18  
19  	/**
20  	 * @param width Width of the printable area of the page (which forms the viewport)
21  	 * @param height Height of the printable area of the page (which forms the viewport)
22  	 * @param pageWidth Width of the page ({@code device-width})
23  	 * @param pageHeight Height of the page ({@code device-height})
24  	 * @param cellWidth Braille cell width in mm
25  	 * @param cellHeight Braille cell height in mm
26  	 */
27  	public EmbossedMedium(Dimension width, Dimension height, Dimension pageWidth, Dimension pageHeight,
28  	                      Map<String,Object> otherFeatures,
29  	                      double cellWidth, double cellHeight, MediumBuilder parser) {
30  		super(Type.EMBOSSED,
31  		      width,
32  		      height,
33  		      pageWidth,
34  		      pageHeight,
35  		      otherFeatures,
36  		      parser);
37  		if (width != null)
38  			switch (width.getUnit()) {
39  			case VW:
40  				throw new IllegalArgumentException("width must not be expressed in vw");
41  			case VH:
42  				if (height == null || height.getUnit() == Unit.VW)
43  					throw new IllegalArgumentException();
44  				break;
45  			}
46  		if (height != null)
47  			switch (height.getUnit()) {
48  			case VH:
49  				throw new IllegalArgumentException("height must not be expressed in vh");
50  			case VW:
51  				if (width == null || width.getUnit() == Unit.VH)
52  					throw new IllegalArgumentException();
53  				break;
54  			}
55  		if (width == null)
56  			if ((pageWidth != null && pageWidth.getUnit() == Unit.VW)
57  			    || (pageHeight != null && pageHeight.getUnit() == Unit.VW))
58  				throw new IllegalArgumentException("viewport width is not known");
59  		if (height == null)
60  			if ((pageWidth != null && pageWidth.getUnit() == Unit.VH)
61  			    || (pageHeight != null && pageHeight.getUnit() == Unit.VH))
62  				throw new IllegalArgumentException("viewport height is not known");
63  		this.cellWidth = cellWidth;
64  		this.cellHeight = cellHeight;
65  	}
66  
67  	/* ///////////////////// */
68  	/* RelativeDimensionBase */
69  	/* ///////////////////// */
70  
71  	/**
72  	 * Cell width in millimeters
73  	 */
74  	@Override
75  	public double getCh() {
76  		return cellWidth;
77  	}
78  
79  	/**
80  	 * Cell height in millimeters
81  	 */
82  	@Override
83  	public double getEm() {
84  		return cellHeight;
85  	}
86  
87  	/**
88  	 * The width of the viewport, in millimeters
89  	 *
90  	 * For resolving {@code vw} dimensions in media queries. Note that the meaning of {@code vw} in
91  	 * CSS properties may vary depending on the current page style.
92  	 */
93  	@Override
94  	public double getViewportWidth() {
95  		Dimension w = getWidth();
96  		if (w == null)
97  			super.getViewportWidth();
98  		return w.toUnit(Unit.MM, this).getValue().doubleValue();
99  	}
100 
101 	/**
102 	 * The height of the viewport, in millimeters
103 	 *
104 	 * For resolving {@code vh} dimensions in media queries. Note that the meaning of {@code vh} in
105 	 * CSS properties may vary depending on the current page style.
106 	 */
107 	@Override
108 	public double getViewportHeight() {
109 		Dimension h = getHeight();
110 		if (h == null)
111 			super.getViewportHeight();
112 		return h.toUnit(Unit.MM, this).getValue().doubleValue();
113 	}
114 
115 	public static class EmbossedMediumBuilder extends MediumBuilder {
116 
117 		public EmbossedMediumBuilder() {
118 			super();
119 			type = Medium.Type.EMBOSSED;
120 		}
121 
122 		@Override
123 		protected Dimension parseLength(Object value, String feature) throws IllegalArgumentException {
124 			if (value instanceof TermInteger)
125 				return parseLength(((TermInteger)value).getIntValue(), feature);
126 			if ((value instanceof Integer
127 			     || (value instanceof TermLength && ((TermLength)value).getUnit() == TermNumeric.Unit.none)))
128 				if ("width".equals(feature) || "device-width".equals(feature))
129 					return value instanceof Integer
130 						? new Dimension((Integer)value, Unit.CH)
131 						: new Dimension(((TermLength)value).getValue(), Unit.CH);
132 				else if ("height".equals(feature) || "device-height".equals(feature))
133 					return value instanceof Integer
134 						? new Dimension((Integer)value, Unit.EM)
135 						: new Dimension(((TermLength)value).getValue(), Unit.EM);
136 			return super.parseLength(value, feature);
137 		}
138 
139 		@Override
140 		public EmbossedMedium build() {
141 			return new EmbossedMedium((Dimension)features.remove("width"),
142 			                          (Dimension)features.remove("height"),
143 			                          (Dimension)features.remove("device-width"),
144 			                          (Dimension)features.remove("device-height"),
145 			                          features,
146 			                          6,
147 			                          10,
148 			                          this);
149 		}
150 	}
151 }
