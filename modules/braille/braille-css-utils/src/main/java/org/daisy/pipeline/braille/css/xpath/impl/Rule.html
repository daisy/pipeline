1   package org.daisy.pipeline.braille.css.xpath.impl;
2   
3   import java.util.Collections;
4   import java.util.Iterator;
5   import java.util.Optional;
6   
7   import javax.xml.stream.XMLStreamException;
8   import javax.xml.stream.XMLStreamWriter;
9   
10  import com.google.common.collect.Iterables;
11  
12  import org.daisy.pipeline.braille.css.impl.BrailleCssSerializer;
13  import org.daisy.pipeline.braille.css.impl.BrailleCssStyle;
14  import org.daisy.pipeline.braille.css.xpath.Style;
15  
16  public class Rule extends Stylesheet {
17  
18  	private final String selector;
19  
20  	/**
21  	 * @param rule style with a single nested style
22  	 */
23  	public Rule(BrailleCssStyle rule) {
24  		super(rule);
25  		if (rule == null || rule.getPropertyNames().iterator().hasNext())
26  			throw new IllegalArgumentException();
27  		this.selector = Iterables.getOnlyElement(rule.getSelectors());
28  	}
29  
30  	@Override
31  	protected Iterable<String> keys() {
32  		return Collections.singleton(selector);
33  	}
34  
35  	@Override
36  	protected Optional<String> property() {
37  		return Optional.empty();
38  	}
39  
40  	@Override
41  	protected Optional<String> selector() {
42  		return Optional.of(selector);
43  	}
44  
45  	@Override
46  	protected Optional<Style> get(String key) {
47  		if (selector.equals(key))
48  			return Optional.of(Stylesheet.of(style.getNestedStyle(key)));
49  		else
50  			return Optional.empty();
51  	}
52  
53  	@Override
54  	protected Optional<Style> remove(Iterator<String> keys) {
55  		while (keys.hasNext())
56  			if (selector.equals(keys.next()))
57  				return Optional.empty();
58  		return Optional.of(this);
59  	}
60  
61  	@Override
62  	protected Iterator<Object> iterate() {
63  		return Collections.<Object>singleton(this).iterator();
64  	}
65  
66  	@Override
67  	public String toString() {
68  		return BrailleCssSerializer.getInstance().toString(style);
69  	}
70  
71  	@Override
72  	protected String toString(Style parent) {
73  		return toString();
74  	}
75  }
