1   package org.daisy.pipeline.braille.css.xpath.impl;
2   
3   import java.util.Collections;
4   import java.util.Iterator;
5   import java.util.Optional;
6   
7   import javax.xml.stream.XMLStreamException;
8   import javax.xml.stream.XMLStreamWriter;
9   
10  import org.daisy.pipeline.braille.css.impl.BrailleCssSerializer;
11  import org.daisy.pipeline.braille.css.xpath.Style;
12  
13  public class Declaration extends Style {
14  
15  	public final cz.vutbr.web.css.Declaration declaration;
16  
17  	public Declaration(cz.vutbr.web.css.Declaration declaration) {
18  		if (declaration == null)
19  			throw new IllegalArgumentException();
20  		this.declaration = declaration;
21  	}
22  
23  	@Override
24  	protected Iterable<String> keys() {
25  		return Collections.singleton(declaration.getProperty());
26  	}
27  
28  	@Override
29  	protected Optional<String> property() {
30  		return Optional.of(declaration.getProperty());
31  	}
32  
33  	@Override
34  	protected Optional<String> selector() {
35  		return Optional.empty();
36  	}
37  
38  	@Override
39  	protected Optional<Style> get(String key) {
40  		if (declaration.getProperty().equals(key))
41  			return Optional.of(new Value(declaration));
42  		else
43  			return Optional.empty();
44  	}
45  
46  	@Override
47  	protected Optional<Style> remove(Iterator<String> keys) {
48  		while (keys.hasNext())
49  			if (declaration.getProperty().equals(keys.next()))
50  				return Optional.empty();
51  		return Optional.of(this);
52  	}
53  
54  	@Override
55  	protected Iterator<Object> iterate() {
56  		return Collections.<Object>singleton(this).iterator();
57  	}
58  
59  	@Override
60  	protected String toString(Style parent) {
61  		if (parent != null)
62  			throw new UnsupportedOperationException();
63  		return BrailleCssSerializer.toString(declaration);
64  	}
65  
66  	@Override
67  	protected String toPrettyString(String indentation) {
68  		return toString();
69  	}
70  
71  	@Override
72  	protected void toXml(XMLStreamWriter writer) throws XMLStreamException {
73  		// <css:property>
74  		BrailleCssSerializer.toXml(declaration, writer);
75  	}
76  }
