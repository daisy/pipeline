1   package org.daisy.pipeline.braille.common;
2   
3   import java.net.URI;
4   import java.net.URL;
5   import java.util.HashMap;
6   import java.util.Map;
7   
8   import static org.daisy.pipeline.braille.common.util.OS;
9   
10  public class BundledNativePath extends StandardNativePath {
11  	
12  	private static final String OS_FAMILY = "os.family";
13  	
14  	private BundledResourcePath resourcePath;
15  	
16  	protected ResourcePath delegate() {
17  		return resourcePath;
18  	}
19  	
20  	/**
21  	 * @throws RuntimeException if the binaries don't work on the current OS
22  	 */
23  	protected void activate(Map<?,?> properties, Class<?> context) throws RuntimeException {
24  		if (properties.get(OS_FAMILY) == null
25  				|| properties.get(OS_FAMILY).toString().isEmpty()) {
26  			throw new IllegalArgumentException(OS_FAMILY + " property must not be empty"); }
27  		if (OS.Family.valueOf(properties.get(OS_FAMILY).toString().toUpperCase()) != OS.getFamily())
28  			throw new RuntimeException(toString() + " does not work on " + OS.getFamily());
29  		if (properties.get(BundledResourcePath.UNPACK) != null)
30  			throw new IllegalArgumentException(BundledResourcePath.UNPACK + " property not supported");
31  		Map<Object,Object> props = new HashMap<Object,Object>(properties);
32  		props.remove(OS_FAMILY);
33  		props.put(BundledResourcePath.UNPACK, true);
34  		props.put(BundledResourcePath.EXECUTABLES, true);
35  		resourcePath = new BundledResourcePath();
36  		resourcePath.activate(props, context);
37  	}
38  	
39  	private boolean unpacked = false;
40  	
41  	@Override
42  	public URL resolve(URI resource) {
43  		URL resolved = super.resolve(resource);
44  		if (!unpacked && resolved != null) {
45  			// unpack everything (because of possible dependencies between binaries)
46  			resourcePath.resolve(getIdentifier());
47  			unpacked = true; }
48  		return resolved;
49  	}
50  	
51  	@Override
52  	public boolean equals(Object object) {
53  		if (this == object)
54  			return true;
55  		if (object == null)
56  			return false;
57  		if (getClass() != object.getClass())
58  			return false;
59  		return super.equals((BundledNativePath)object);
60  	}
61  }
