1   package org.daisy.pipeline.braille.common;
2   
3   /**
4    * Hyphenating means breaking within words.
5    */
6   public interface Hyphenator extends Transform {
7   	
8   	/**
9   	 * Hyphenate by inserting soft hyphens and zero width spaces in order to
10  	 * indicate all the break opportunities within words. Apart from the
11  	 * insertion of these characters no other transformations are
12  	 * allowed. This means that non-standard hyphenation can not be supported
13  	 * through this interface.
14  	 */
15  	public FullHyphenator asFullHyphenator() throws UnsupportedOperationException;
16  	
17  	/**
18  	 * Like `asFullHyphenator`, this interface can not be used for
19  	 * non-standard hyphenation.
20  	 */
21  	public XProc asXProc() throws UnsupportedOperationException;
22  	
23  	/**
24  	 * Hyphenate by breaking an input into lines of a preferred and maximal
25  	 * length. Transformations such as non-standard hyphenation are
26  	 * allowed.
27  	 */
28  	public LineBreaker asLineBreaker() throws UnsupportedOperationException;
29  	
30  	/* ------------------ */
31  	/* asFullHyphenator() */
32  	/* ------------------ */
33  	
34  	public interface FullHyphenator {
35  		
36  		public String transform(String text);
37  		public String[] transform(String[] text);
38  		
39  	}
40  	
41  	/* --------------- */
42  	/* asLineBreaker() */
43  	/* --------------- */
44  	
45  	public interface LineBreaker {
46  		
47  		public LineIterator transform(String input);
48  		
49  	}
50  	
51  	/* ------------ */
52  	/* LineIterator */
53  	/* ------------ */
54  	
55  	public interface LineIterator {
56  		
57  		/**
58  		 * Returns true if there is remaining text to make a new line of.
59  		 */
60  		public boolean hasNext();
61  		
62  		/**
63  		 * Get the next line of text, with a length as close as possible to, but not
64  		 * exceding `limit` characters, breaking a word if needed. `limit` must be greater
65  		 * than 0. If `force` is true, may not return an empty string. If `allowHyphens`
66  		 * is false, breaking words is not permitted unless `force` is true and an empty
67  		 * string would otherwise be returned. Must throw an exception if there is no
68  		 * remaining text to make a new line of.
69  		 */
70  		public String nextLine(int limit, boolean force, boolean allowHyphens);
71  		public String nextLine(int limit, boolean force);
72  		
73  		/**
74  		 * Whether or not a hyphen character should be inserted at the end of
75  		 * the current line to indicate a word break. Must throw an exception
76  		 * if `nextLine` hasn't been called yet.
77  		 */
78  		public boolean lineHasHyphen();
79  		
80  		public String remainder();
81  		
82  		/**
83  		 * Set a checkpoint to which the iterator can be `reset` later.
84  		 */
85  		public void mark();
86  		
87  		/**
88  		 * Roll back to a checkpoint previously set with `mark`, or to the
89  		 * beginning if `mark` hasn't been called yet.
90  		 */
91  		public void reset();
92  		
93  	}
94  }
