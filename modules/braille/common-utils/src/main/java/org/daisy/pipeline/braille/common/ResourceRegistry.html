1   package org.daisy.pipeline.braille.common;
2   
3   import java.net.URI;
4   import java.net.URL;
5   import java.util.HashMap;
6   import java.util.Map;
7   
8   import org.daisy.pipeline.braille.common.ResourceResolver.util.MemoizingResolver;
9   
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  public abstract class ResourceRegistry<T extends ResourcePath> implements ResourceResolver {
14  	
15  	protected void register(T path) {
16  		if (paths.containsKey(path.getIdentifier()))
17  			throw new RuntimeException("Resource registry already contains resource path with identifier " + path.getIdentifier());
18  		paths.put(path.getIdentifier(), path);
19  		resolver.invalidateCache();
20  		logger.debug("Adding resource path to registry: {}", path.getIdentifier());
21  	}
22  	
23  	protected void unregister(T path) {
24  		paths.remove(path.getIdentifier());
25  		resolver.invalidateCache();
26  		logger.debug("Removing resource path from registry: {}", path.getIdentifier());
27  	}
28  	
29  	protected final Map<URI,T> paths = new HashMap<URI,T>();
30  	
31  	/*
32  	 * Iterate over all registered resource paths and return as soon as one
33  	 * path can resolve the resource.
34  	 */
35  	protected final MemoizingResolver resolver = new MemoizingResolver() {
36  		public URL _apply(URI resource) {
37  			for (T path : paths.values()) {
38  				URL resolved = path.resolve(resource);
39  				if (resolved != null)
40  					return resolved; }
41  			return null;
42  		}
43  	};
44  	
45  	public URL resolve(URI resource) {
46  		return resolver.resolve(resource);
47  	}
48  	
49  	private static final Logger logger = LoggerFactory.getLogger(ResourceRegistry.class);
50  }
