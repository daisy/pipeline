1   package org.daisy.saxon.functions.file;
2   
3   import java.io.File;
4   import java.net.URI;
5   
6   import net.sf.saxon.expr.XPathContext;
7   import net.sf.saxon.lib.ExtensionFunctionCall;
8   import net.sf.saxon.lib.ExtensionFunctionDefinition;
9   import net.sf.saxon.om.AtomicSequence;
10  import net.sf.saxon.om.Sequence;
11  import net.sf.saxon.om.SequenceIterator;
12  import net.sf.saxon.om.StructuredQName;
13  import net.sf.saxon.trans.XPathException;
14  import net.sf.saxon.tree.iter.SingletonIterator;
15  import net.sf.saxon.type.BuiltInAtomicType;
16  import net.sf.saxon.value.SequenceType;
17  import net.sf.saxon.value.StringValue;
18  
19  @SuppressWarnings("serial")
20  public class Expand83 extends ExtensionFunctionDefinition {
21  
22  	private static final StructuredQName funcname = new StructuredQName("pf",
23  			"http://www.daisy.org/ns/pipeline/functions", "file-expand83");
24  
25  	@Override
26  	public SequenceType[] getArgumentTypes() {
27  		return new SequenceType[] { SequenceType.SINGLE_STRING };
28  	}
29  
30  	@Override
31  	public StructuredQName getFunctionQName() {
32  		return funcname;
33  	}
34  
35  	@Override
36  	public SequenceType getResultType(SequenceType[] arg0) {
37  		return SequenceType.SINGLE_STRING;
38  	}
39  
40  	@Override
41  	public ExtensionFunctionCall makeCallExpression() {
42  		return new ExtensionFunctionCall() {
43  
44  			@Override
45  			public Sequence call(XPathContext context, Sequence[] arguments)
46  					throws XPathException {
47  				String uri = ((AtomicSequence) arguments[0]).getStringValue();
48  				uri = Expand83.expand83(uri);
49  				return new StringValue(uri, BuiltInAtomicType.STRING);
50  			}
51  		};
52  	}
53  
54  	/**
55  	 * Expands 8.3 encoded path segments.
56  	 *
57  	 * For instance `C:\DOCUME~1\file.xml` will become `C:\Documents and
58  	 * Settings\file.xml`
59  	 */
60  	public static String expand83(String uri) throws XPathException {
61  		if (uri == null || !uri.startsWith("file:/")) {
62  			return uri;
63  		}
64  
65  		try {
66  			File file = new File(new URI(uri));
67  			String expandedUri = expand83(file);
68  			if (expandedUri == null) {
69  				return uri;
70  			} else {
71  				return expandedUri;
72  			}
73  
74  		} catch (Exception e) {
75  			throw new XPathException("pf:file-expand83 failed", e);
76  		}
77  	}
78  
79  	/**
80  	 * this is extracted out of `expand83(String)` because it can be unit tested
81  	 * with a custom File implementation.
82  	 */
83  	public static String expand83(File file) throws XPathException {
84  		try {
85  			if (file.exists()) {
86  				return file.getCanonicalFile().toURI().toString();
87  			} else {
88  				return file.toURI().toString();
89  			}
90  		} catch (Exception e) {
91  			throw new XPathException("pf:file-expand83 failed", e);
92  		}
93  	}
94  
95  }
