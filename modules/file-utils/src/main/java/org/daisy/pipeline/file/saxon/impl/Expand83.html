24  @Component(
25  	name = "pf:file-expand83",
26  	service = { ExtensionFunctionDefinition.class }
27  )
28  @SuppressWarnings("serial")
29  public class Expand83 extends ExtensionFunctionDefinition {
30  
31  	private static final StructuredQName funcname = new StructuredQName("pf",
32  			"http://www.daisy.org/ns/pipeline/functions", "file-expand83");
33  
34  	@Override
35  	public SequenceType[] getArgumentTypes() {
36  		return new SequenceType[] { SequenceType.SINGLE_STRING };
37  	}
38  
39  	@Override
40  	public StructuredQName getFunctionQName() {
41  		return funcname;
42  	}
43  
44  	@Override
45  	public SequenceType getResultType(SequenceType[] arg0) {
46  		return SequenceType.SINGLE_STRING;
47  	}
48  
49  	@Override
50  	public ExtensionFunctionCall makeCallExpression() {
51  		return new ExtensionFunctionCall() {
52  
53  			@Override
54  			public Sequence call(XPathContext context, Sequence[] arguments)
55  					throws XPathException {
56  				String uri = ((AtomicSequence) arguments[0]).getStringValue();
57  				uri = Expand83.expand83(uri);
58  				return new StringValue(uri, BuiltInAtomicType.STRING);
59  			}
60  		};
61  	}
62  
63  	/**
64  	 * Expands 8.3 encoded path segments.
65  	 *
66  	 * For instance `C:\DOCUME~1\file.xml` will become `C:\Documents and
67  	 * Settings\file.xml`
68  	 */
69  	public static String expand83(String uri) throws XPathException {
70  		if (uri == null || !uri.startsWith("file:/")) {
71  			return uri;
72  		}
73  		return expand83(URLs.asURI(uri)).toASCIIString();
74  	}
75  	
76  	private static URI expand83(URI uri) throws XPathException {
77  		try {
78  			String protocol = "file";
79  			String path = uri.getPath();
80  			String zipPath = null;
81  			if (path.contains("!/")) {
82  				// it is a path to a ZIP entry
83  				zipPath = path.substring(path.indexOf("!/")+1);
84  				path = path.substring(0, path.indexOf("!/"));
85  			}
86  			String query = uri.getQuery();
87  			String fragment = uri.getFragment();
88  			File file = new File(new URI(protocol, null, path, null, null));
89  			URI expandedUri = expand83(file, path.endsWith("/"));
90  			if (expandedUri == null) {
91  				return uri;
92  			} else {
93  				path = expandedUri.getPath();
94  				if (zipPath != null)
95  					path = path + "!" + new URI(null, null, zipPath, null, null).getPath();
96  				return new URI(protocol, null, path, query, fragment);
97  			}
98  		} catch (Throwable e) {
99  			throw new XPathException("Unexpected error in pf:file-expand83("+uri+")", e);
100 		}
101 	}
102 
103 	/**
104 	 * this is extracted out of `expand83(String)` because it can be unit tested
105 	 * with a custom File implementation.
106 	 */
107 	public static URI expand83(File file) throws XPathException {
108 		return expand83(file, false);
109 	}
110 	
111 	public static URI expand83(File file, boolean isDir) throws XPathException {
112 		try {
113 			if (file.exists()) {
114 				return URLs.asURI(file.getCanonicalFile());
115 			} else {
116 				// if the file does not exist a parent directory may exist which can be canonicalized
117 				String relPath = file.getName();
118 				if (isDir)
119 					relPath += "/";
120 				File dir = file.getParentFile();
121 				while (dir != null) {
122 					if (dir.exists())
123 						return URLs.resolve(URLs.asURI(dir.getCanonicalFile()), new URI(null, null, relPath, null, null));
124 					relPath = dir.getName() + "/" + relPath;
125 					dir = dir.getParentFile();
126 				}
127 				return URLs.asURI(file);
128 			}
129 		} catch (Throwable e) {
130 			throw new XPathException("Unexpected error in pf:file-expand83("+file+")", e);
131 		}
132 	}
