1   package org.daisy.pipeline.file;
2   
3   import java.io.Reader;
4   import java.io.StringReader;
5   
6   /**
7    * Various file/document related utility functions for use in XProc steps
8    */
9   public final class FileUtils {
10  
11  	private FileUtils() {}
12  
13  	/**
14  	 * Create a <a href="https://www.w3.org/TR/xproc/#cv.result"><code>c:result</code></a> document
15  	 *
16  	 * @param text the value of the text node that the c:result element should contain
17  	 * @return the document as a Reader object
18  	 */
19  	public static Reader cResultDocument(String text) {
20  		StringBuilder sb = new StringBuilder();
21  		sb.append("<c:result xmlns:c=\"http://www.w3.org/ns/xproc-step\">")
22  			.append(text)
23  			.append("</c:result>");
24  		return new StringReader(sb.toString());
25  	}
26  }
