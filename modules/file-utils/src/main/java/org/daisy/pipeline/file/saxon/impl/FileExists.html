1   package org.daisy.pipeline.file.saxon.impl;
2   
3   import java.io.File;
4   
5   import net.sf.saxon.expr.XPathContext;
6   import net.sf.saxon.lib.ExtensionFunctionCall;
7   import net.sf.saxon.lib.ExtensionFunctionDefinition;
8   import net.sf.saxon.om.AtomicSequence;
9   import net.sf.saxon.om.Sequence;
10  import net.sf.saxon.om.StructuredQName;
11  import net.sf.saxon.trans.XPathException;
12  import net.sf.saxon.type.BuiltInAtomicType;
13  import net.sf.saxon.value.BooleanValue;
14  import net.sf.saxon.value.SequenceType;
15  
16  import org.osgi.service.component.annotations.Component;
17  
18  @Component(
19  	name = "pf:file-exists",
20  	service = { ExtensionFunctionDefinition.class }
21  )
22  @SuppressWarnings("serial")
23  public class FileExists extends ExtensionFunctionDefinition {
24  
25  	private static final StructuredQName funcname = new StructuredQName("pf",
26  			"http://www.daisy.org/ns/pipeline/functions", "file-exists");
27  
28  	@Override
29  	public SequenceType[] getArgumentTypes() {
30  		return new SequenceType[] { SequenceType.SINGLE_STRING };
31  	}
32  
33  	@Override
34  	public StructuredQName getFunctionQName() {
35  		return funcname;
36  	}
37  
38  	@Override
39  	public SequenceType getResultType(SequenceType[] arg0) {
40  		return SequenceType.SINGLE_BOOLEAN;
41  	}
42  
43  	@Override
44  	public ExtensionFunctionCall makeCallExpression() {
45  		return new ExtensionFunctionCall() {
46  
47  			@SuppressWarnings({ "unchecked", "rawtypes" })
48  			public Sequence call(XPathContext context, Sequence[] arguments)
49  					throws XPathException {
50  
51  				try {
52  					String path = ((AtomicSequence) arguments[0])
53  							.getStringValue();
54  					boolean result = (path.isEmpty()) ? true : new File(path)
55  							.exists();
56  					return new BooleanValue(result, BuiltInAtomicType.BOOLEAN);
57  				} catch (Throwable e) {
58  					throw new XPathException("Unexpected error in pf:file-exists", e);
59  				}
60  			}
61  		};
62  	}
63  
64  }
