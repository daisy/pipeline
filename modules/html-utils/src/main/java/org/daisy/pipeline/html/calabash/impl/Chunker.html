59  // The transformation consists of two passes. In the first pass, the split points are computed based
60  // on the provided stylesheets. In the second pass, the document is split into chunks. Because the
61  // XMLStreamToXMLStreamTransformer interface does not yet support reading the input document more
62  // than once, we use the Saxon specific NodeToXMLStreamTransformer interface.
63  class Chunker implements NodeToXMLStreamTransformer {

101 	public void transform(Iterator<XdmNode> input, Supplier<BaseURIAwareXMLStreamWriter> output) throws TransformerException {
102 		XdmNode doc = Iterators.getOnlyElement(input);
103 		BaseURIAwareXMLStreamReader reader;
104 		try {
105 			reader = SaxonHelper.nodeReader(doc, config);
106 			SortedSet<BreakPosition> collectSplitPoints = new TreeSet<>();
107 			Map<String,Path> collectIds = new HashMap<>();
108 			getSplitPoints(doc, collectSplitPoints, collectIds);
109 			idToChunk = new HashMap<String,Integer>();
110 			int n = 1;
111 			for (BreakPosition sp : collectSplitPoints) {
112 				Iterator<Map.Entry<String,Path>> i = collectIds.entrySet().iterator();
113 				while (i.hasNext()) {
114 					Map.Entry<String,Path> e = i.next();
115 					if (sp.compareTo(e.getValue()) > 0) {
116 						idToChunk.put(e.getKey(), n);
117 						i.remove();
118 					}
119 				}
120 				n++;
121 			}
122 			for (String id : collectIds.keySet())
123 				idToChunk.put(id, n);
124 			splitPoints = collectSplitPoints.iterator();
125 		} catch (XPathException | SaxonApiException e) {
126 			throw new TransformerException(e);
127 		}
128 		if (splitPoints.hasNext()) {
129 			if (doc.getBaseURI() != null)
130 				output = setBaseURI(output, doc.getBaseURI());
131 			transform(reader, output);
132 		} else
133 			try {
134 				BaseURIAwareXMLStreamWriter writer = output.get();
135 				if (doc.getBaseURI() != null)
136 					writer.setBaseURI(doc.getBaseURI());
137 				writeDocument(writer, reader);
138 			} catch (XMLStreamException e) {
139 				throw new TransformerException(e);
140 			}
141 	}

143 	void transform(BaseURIAwareXMLStreamReader reader, Supplier<BaseURIAwareXMLStreamWriter> writers) throws TransformerException {
144 		nextSplitPoint = null;
145 		if (splitPoints.hasNext())
146 			nextSplitPoint = splitPoints.next();
147 		parents = new Stack<QName>();
148 		parentAttrs = new Stack<Map<QName,String>>();
149 		boolean containsSplitPoint = true;
150 		Stack<Boolean> containsSplitPointStack = new Stack<Boolean>();
151 		currentPath = new Path.Builder();
152 		BaseURIAwareXMLStreamWriter writer = writers.get();
153 		try {
154 			URI sourceBaseURI = reader.getBaseURI();
155 			writer.writeStartDocument();
156 		  loop: while (true)
157 				try {
158 					int event = reader.next();
159 					switch (event) {
160 					case START_ELEMENT:
161 						containsSplitPointStack.push(containsSplitPoint);
162 						if (!currentPath.isRoot()) {
163 							currentPath.nextElement();
164 							if (containsSplitPoint && isSplitPoint(currentPath, BreakPosition.Side.BEFORE, nextSplitPoint))
165 								split(writer, writer = writers.get());
166 							containsSplitPoint = containsSplitPoint(currentPath, nextSplitPoint);
167 						}
168 						writeStartElement(writer, reader);
169 						for (int i = 0; i < reader.getAttributeCount(); i++) {
170 							QName attr = reader.getAttributeName(i);
171 							String val = reader.getAttributeValue(i);
172 							if (sourceBaseURI != null)
173 								if (attr.equals(linkAttributeName) && val.startsWith("#")) {
174 									String id = val.substring(1);
175 									if (idToChunk.containsKey(id)) {
176 										URI currentBaseURI = writer.getBaseURI();
177 										URI targetBaseURI = getChunkBaseURI(sourceBaseURI, idToChunk.get(id));
178 										if (!currentBaseURI.equals(targetBaseURI))
179 											val = relativize(currentBaseURI, targetBaseURI).toASCIIString() + val; }}
180 							writeAttribute(writer, attr, val);
181 						}
182 						currentPath.down();
183 						if (containsSplitPoint) {
184 							parents.push(reader.getName());
185 							parentAttrs.push(getAttributes(reader));
186 						}
187 						break;
188 					case END_ELEMENT:
189 						if (containsSplitPoint) {
190 							parents.pop();
191 							parentAttrs.pop();
192 						}
193 						writer.writeEndElement();
194 						containsSplitPoint = containsSplitPointStack.pop();
195 						if (containsSplitPoint && isSplitPoint(currentPath, BreakPosition.Side.AFTER, nextSplitPoint))
196 							split(writer, writer = writers.get());
197 						currentPath.up();
198 						break;
199 					case CHARACTERS:
200 						if (currentPath.isElement())
201 							currentPath.inc();
202 						if (containsSplitPoint && isSplitPoint(currentPath, BreakPosition.Side.BEFORE, nextSplitPoint))
203 							split(writer, writer = writers.get());
204 						writeCharacters(writer, reader);
205 						if (containsSplitPoint && isSplitPoint(currentPath, BreakPosition.Side.AFTER, nextSplitPoint))
206 							split(writer, writer = writers.get());
207 						break;
208 					case START_DOCUMENT:
209 						break;
210 					case END_DOCUMENT:
211 						break loop;
212 					default:
213 						writeEvent(writer, event, reader);
214 					}
215 				} catch (NoSuchElementException e) {
216 					break;
217 				}
218 			writer.writeEndDocument();
219 		} catch (XMLStreamException e) {
220 			throw new TransformerException(e);
221 		}
222 	}

224 	void split(XMLStreamWriter writer, XMLStreamWriter newWriter) throws XMLStreamException {
225 		for (int i = parents.size(); i > 0; i--)
226 			writer.writeEndElement();
227 		writer.writeEndDocument();
228 		newWriter.writeStartDocument();
229 		for (int i = 0; i < parents.size(); i++) {
230 			writeStartElement(newWriter, parents.get(i));
231 			for (Map.Entry<QName,String> attr : parentAttrs.get(i).entrySet())
232 				if (!(attr.getKey().equals(_ID) || attr.getKey().equals(XML_ID)))
233 					writeAttribute(newWriter, attr);
234 		}
235 		nextSplitPoint = splitPoints.hasNext() ? splitPoints.next() : null;
236 	}

267 	void getSplitPoints(XdmNode source, SortedSet<BreakPosition> collectSplitPoints, Map<String,Path> collectIds)
268 			throws XPathException, SaxonApiException {
269 		SortedSet<BreakOpportunity> opportunities = new TreeSet<>();
270 		int totalBytes = getBreakOpportunities(source, opportunities, collectIds);
271 		collectSplitPoints.addAll(computeSplitPoints(opportunities, totalBytes, maxChunkSize * 1000));

274 	static SortedSet<BreakPosition> computeSplitPoints(SortedSet<BreakOpportunity> opportunities, int totalBytes, int maxSize) {
275 		SortedSet<BreakPosition> splitPoints = new TreeSet<>();
276 		if (maxSize <= 0) {
277 			for (BreakOpportunity o : opportunities)
278 				if (o.weight == BreakOpportunity.Weight.ALWAYS)
279 					splitPoints.add(o.position);
280 		} else {
281 			BreakOpportunity BEGIN = new BreakOpportunity(
282 				new BreakPosition(Path.ROOT, BreakPosition.Side.BEFORE),
283 				BreakOpportunity.Weight.ALWAYS,
284 				0);
285 			BreakOpportunity END = new BreakOpportunity(
286 				new BreakPosition(Path.ROOT, BreakPosition.Side.AFTER),
287 				BreakOpportunity.Weight.ALWAYS,
288 				totalBytes);
289 			opportunities.add(BEGIN);
290 			opportunities.add(END);
291 			
292 			// chunk candidates starting from a certain point
293 			int preferredSize = maxSize * 4 / 5;
294 			Function<BreakOpportunity,Set<BreakOpportunity>> neighbors = (current) -> {
295 				Set<BreakOpportunity> result = new TreeSet<>();
296 				if (current.equals(END))
297 					return result;
298 				Iterator<BreakOpportunity> i = opportunities.iterator();
299 				for (;;)
300 					try {
301 						BreakOpportunity n = i.next();
302 						if (n.equals(current))
303 							break; }
304 					catch (NoSuchElementException e) {
305 						throw new IllegalArgumentException(current+" is not a break opportunity"); }
306 				while (i.hasNext()) {
307 					BreakOpportunity n = i.next();
308 					if (n.bytesBefore - current.bytesBefore <= maxSize) {
309 						result.add(n);
310 						if (n.weight == BreakOpportunity.Weight.ALWAYS)
311 							break; // can not be skipped
312 					} else {
313 						if (result.isEmpty())
314 							result.add(n); // chunk exceeds maximum size
315 						break;
316 					}
317 				}
318 				return result;
319 			};
320 			
321 			// cost of a single chunk
322 			// total cost function can be expressed as the sum of the costs of individual chunks
323 			BiFunction<BreakOpportunity,BreakOpportunity,Float> cost = (begin, end) -> {
324 				float result = 0;
325 				result += .2; // the less chunks the better
326 				switch (end.weight) {
327 				case ALWAYS:
328 				case PREFER:
329 					break;
330 				case ALLOW:
331 					result += .2;
332 				}
333 				int size = end.bytesBefore - begin.bytesBefore;
334 				int diff = Math.abs(preferredSize - size);
335 				if (size > maxSize)
336 					diff *= 4; // higher penalty
337 				result += (float)diff / (float)preferredSize;
338 				return result;
339 			};
340 			
341 			// compute optimal chunking
342 			List<BreakOpportunity> solution = shortestPath(BEGIN, END, neighbors, cost);
343 			for (BreakOpportunity o : solution)
344 				splitPoints.add(o.position);
345 		}
346 		return splitPoints;
347 	}

349 	// Dijkstra's algorithm
350 	static <T> List<T> shortestPath(T from, T to, Function<T,Set<T>> neighbors, BiFunction<T,T,Float> distance) {
351 		Map<T,Float> tentativeDistance = new HashMap<>(); // unvisited = tentativeDistance.keySet()
352 		Set<T> visited = new TreeSet<>();
353 		Map<T,T> bestPredecessor = new HashMap<>();
354 		tentativeDistance.put(from, 0f);
355 		T current = from;
356 		for (;;) {
357 			for (T n : neighbors.apply(current)) {
358 				if (!visited.contains(n)) {
359 					float distFromCurrent = distance.apply(current, n);
360 					float distFromBegin = tentativeDistance.get(current) + distFromCurrent;
361 					if (!tentativeDistance.containsKey(n) || tentativeDistance.get(n) > distFromBegin) {
362 						tentativeDistance.put(n, distFromBegin);
363 						bestPredecessor.put(n, current);
364 					}}}
365 			visited.add(current);
366 			tentativeDistance.remove(current);
367 			try {
368 				current = Collections.min(
369 					tentativeDistance.entrySet(),
370 					Comparator.comparing(Map.Entry::getValue)).getKey();
371 			} catch (NoSuchElementException e) {
372 				throw new IllegalArgumentException("neighbors function cannot connect "+from+" and "+to);
373 			}
374 			if (current.equals(to)) {
375 				List<T> path = new ArrayList<>();
376 				current = to;
377 				for (;;) {
378 					current = bestPredecessor.get(current);
379 					if (current.equals(from))
380 						break;
381 					path.add(0, current);
382 				}
383 				return path;
384 			}}
385 	}

387 	int getBreakOpportunities(XdmNode source, SortedSet<BreakOpportunity> collectBreakOpportunities, Map<String,Path> collectIds)
388 			throws XPathException, SaxonApiException {
389 		net.sf.saxon.s9api.QName _ID = new net.sf.saxon.s9api.QName(Chunker._ID);
390 		net.sf.saxon.s9api.QName XML_ID = new net.sf.saxon.s9api.QName(Chunker.XML_ID);
391 		Predicate<XdmNode> allowBreakBefore = parseOption(allowBreakBeforeOption);
392 		Predicate<XdmNode> allowBreakAfter = parseOption(allowBreakAfterOption);
393 		Predicate<XdmNode> preferBreakBefore = parseOption(preferBreakBeforeOption);
394 		Predicate<XdmNode> preferBreakAfter = parseOption(preferBreakAfterOption);
395 		Predicate<XdmNode> alwaysBreakBefore = parseOption(alwaysBreakBeforeOption);
396 		Predicate<XdmNode> alwaysBreakAfter = parseOption(alwaysBreakAfterOption);
397 		return new Function<XdmNode,Integer>() {
398 			Path.Builder currentPath = new Path.Builder();
399 			public Integer apply(XdmNode node) {
400 				int bytesSeen = 0;
401 				if (node.getNodeKind() == XdmNodeKind.DOCUMENT) {
402 					for (XdmItem i : SaxonHelper.axisIterable(node, Axis.CHILD))
403 						bytesSeen += apply((XdmNode)i);
404 				} else if (node.getNodeKind() == XdmNodeKind.ELEMENT) {
405 					if (!currentPath.isRoot())
406 						currentPath.nextElement();
407 					String id = node.getAttributeValue(XML_ID);
408 					if (id == null) id = node.getAttributeValue(_ID);
409 					if (id != null)
410 						collectIds.put(id, currentPath.build());
411 					if (!currentPath.isRoot())
412 						if (alwaysBreakBefore.test(node))
413 							addOptionally(collectBreakOpportunities,
414 							              propagate(node, currentPath, BreakPosition.Side.BEFORE, BreakOpportunity.Weight.ALWAYS, bytesSeen));
415 						else if (preferBreakBefore.test(node))
416 							addOptionally(collectBreakOpportunities,
417 							              propagate(node, currentPath, BreakPosition.Side.BEFORE, BreakOpportunity.Weight.PREFER, bytesSeen));
418 						else if (allowBreakBefore.test(node))
419 							addOptionally(collectBreakOpportunities,
420 							              propagate(node, currentPath, BreakPosition.Side.BEFORE, BreakOpportunity.Weight.ALLOW, bytesSeen));
421 					currentPath.down();
422 					for (XdmItem i : SaxonHelper.axisIterable(node, Axis.CHILD))
423 						bytesSeen += apply((XdmNode)i);
424 					currentPath.up();
425 					if (!currentPath.isRoot())
426 						if (alwaysBreakAfter.test(node))
427 							addOptionally(collectBreakOpportunities,
428 							              propagate(node, currentPath, BreakPosition.Side.AFTER, BreakOpportunity.Weight.ALWAYS, bytesSeen));
429 						else if (preferBreakAfter.test(node))
430 							addOptionally(collectBreakOpportunities,
431 							              propagate(node, currentPath, BreakPosition.Side.AFTER, BreakOpportunity.Weight.PREFER, bytesSeen));
432 						else if (allowBreakAfter.test(node))
433 							addOptionally(collectBreakOpportunities,
434 							              propagate(node, currentPath, BreakPosition.Side.AFTER, BreakOpportunity.Weight.ALLOW, bytesSeen));
435 				} else if (node.getNodeKind() == XdmNodeKind.TEXT) {
436 					// FIXME: currently only taking into account character data
437 					// FIXME: we need to know the encoding in order to correctly compute the size
438 					bytesSeen += node.getStringValue().getBytes().length;
439 				}
440 				return bytesSeen;
441 			}
442 		}.apply(source);
443 	}

445 	Predicate<XdmNode> parseOption(RuntimeValue option) throws XPathException {
446 		if (!option.getString().equals("/*")) {
447 			XPathExpression matcher = SaxonHelper.compileExpression(option.getString(),
448 			                                                        option.getNamespaceBindings(),
449 			                                                        config);
450 			return n -> n.getNodeKind() == XdmNodeKind.ELEMENT && SaxonHelper.evaluateBoolean(matcher, n);
451 		} else
452 			return n -> false;
453 	}

460 	static Optional<BreakOpportunity> propagate(XdmNode element, Path.Builder path, BreakPosition.Side side,
461 	                                            BreakOpportunity.Weight weight, int bytesSeen) {
462 		path = path.builder();
463 		while (!path.isRoot()) {
464 			if (shouldPropagateBreak(element, side)) {
465 				element = (XdmNode)element.getParent();
466 				path.up();
467 				continue;
468 			}
469 			if (side == BreakPosition.Side.AFTER)
470 				bytesSeen += skipWhiteSpaceNodes(element, side, path);
471 			return Optional.of(new BreakOpportunity(new BreakPosition(path, side).normalize(), weight, bytesSeen));
472 		}
473 		return Optional.empty();
474 	}

476 	static boolean shouldPropagateBreak(XdmNode element, BreakPosition.Side side) {
477 		for (XdmItem i : SaxonHelper.axisIterable(element, side == BreakPosition.Side.BEFORE ? Axis.PRECEDING_SIBLING
478 		                                                                                     : Axis.FOLLOWING_SIBLING)) {
479 			XdmNode n = (XdmNode)i;
480 			if (n.getNodeKind() == XdmNodeKind.ELEMENT)
481 				return false;
482 			else if (n.getNodeKind() == XdmNodeKind.TEXT && !isWhiteSpaceNode(n))
483 				return false;
484 		}
485 		return true;
486 	}
487 	
488 	static int skipWhiteSpaceNodes(XdmNode element, BreakPosition.Side side, Path.Builder path) {
489 		int bytesSkipped = 0;
490 		for (XdmItem i : SaxonHelper.axisIterable(element, side == BreakPosition.Side.BEFORE ? Axis.PRECEDING_SIBLING
491 		                                                                                     : Axis.FOLLOWING_SIBLING)) {
492 			XdmNode n = (XdmNode)i;
493 			if (n.getNodeKind() == XdmNodeKind.ELEMENT) {
494 				if (side == BreakPosition.Side.AFTER)
495 					path.inc();
496 				else
497 					path.dec();
498 				return bytesSkipped;
499 			} else if (n.getNodeKind() == XdmNodeKind.TEXT) {
500 				if (isWhiteSpaceNode(n))
501 					bytesSkipped += n.getStringValue().getBytes().length;
502 				else
503 					return 0;
504 			}
505 		}
506 		return 0;
507 	}

520 	static URI relativize(URI base, URI child) {
521 		try {
522 			if (base.isOpaque() || child.isOpaque()
523 			    || !Optional.ofNullable(base.getScheme()).orElse("").equalsIgnoreCase(Optional.ofNullable(child.getScheme()).orElse(""))
524 			    || !Optional.ofNullable(base.getAuthority()).equals(Optional.ofNullable(child.getAuthority())))
525 				return child;
526 			else {
527 				String bp = base.normalize().getPath();
528 				String cp = child.normalize().getPath();
529 				String relativizedPath;
530 				if (cp.startsWith("/")) {
531 					String[] bpSegments = bp.split("/", -1);
532 					String[] cpSegments = cp.split("/", -1);
533 					int i = bpSegments.length - 1;
534 					int j = 0;
535 					while (i > 0) {
536 						if (bpSegments[j].equals(cpSegments[j])) {
537 							i--;
538 							j++; }
539 						else
540 							break; }
541 					relativizedPath = "";
542 					while (i > 0) {
543 						relativizedPath += "../";
544 						i--; }
545 					while (j < cpSegments.length) {
546 						relativizedPath += cpSegments[j] + "/";
547 						j++; }
548 					relativizedPath = relativizedPath.substring(0, relativizedPath.length() - 1); }
549 				else
550 					relativizedPath = cp;
551 				if (relativizedPath.isEmpty())
552 					relativizedPath = "./";
553 				return new URI(null, null, relativizedPath, child.getQuery(), child.getFragment()); }
554 		} catch (URISyntaxException e) {
555 			throw new RuntimeException(e);
556 		}
557 	}

559 	/*
560 	 * The path of a node in the tree is encoded as a list of integers where each integer
561 	 * is an index of a child node, starting with the ancestor of the node that is a child
562 	 * of the root element and ending with the node itself. Indexes are even for elements
563 	 * (2, 4, ...) and uneven for text nodes (1, 3, ...). This definition matches the
564 	 * definition of EPUB canonical fragment identifiers (see
565 	 * http://www.idpf.org/epub/linking/cfi/#sec-epubcfi-def). [x, y, z] corresponds with
566 	 * /x/y/z in EPUB CFI.
567 	 */
568 	static class Path implements Comparable<Path> {
569 		
570 		protected final Stack<Integer> path;
571 		
572 		final static Path ROOT = new Builder().build();
573 		
574 		private Path(Stack<Integer> path) {
575 			this.path = path;
576 		}
577 		
578 		public boolean isRoot() {
579 			return path.isEmpty();
580 		}
581 		
582 		public boolean isElement() {
583 			return path.isEmpty() || path.peek() % 2 == 0;
584 		}
585 		
586 		public Builder builder() {
587 			return new Builder(this);
588 		}
589 		
590 		/**
591 		 * @return -1 when this node comes before the given node, 1 when this node comes after the
592 		 *         given node, 0 when the nodes are the same or one is a descendant of the other.
593 		 */
594 		@Override
595 		public int compareTo(Path o) {
596 			int minsize = Math.min(this.path.size(), o.path.size());
597 			for (int i = 0; i < minsize; i++) {
598 				int c = this.path.get(i).compareTo(o.path.get(i));
599 				if (c != 0)
600 					return c;
601 			}
602 			return 0;
603 		}
604 		
605 		public boolean contains(Path o) {
606 			if (compareTo(o) != 0)
607 				return false;
608 			return o.path.size() > this.path.size();
609 		}
610 		
611 		static class Builder extends Path {
612 			
613 			public Builder() {
614 				super(new Stack<>());
615 			}
616 			
617 			private Builder(Path from) {
618 				super(new Stack<Integer>());
619 				for (Integer step : from.path)
620 					path.add(step);
621 			}
622 			
623 			public Builder down() {
624 				path.add(0);
625 				return this;
626 			}
627 			
628 			public Builder up() {
629 				path.pop();
630 				return this;
631 			}
632 			
633 			public Builder inc() {
634 				path.add(path.pop() + 1);
635 				return this;
636 			}
637 			
638 			public Builder dec() {
639 				path.add(path.pop() - 1);
640 				return this;
641 			}
642 			
643 			public Builder nextElement() {
644 				int step = path.pop() + 1;
645 				if (step % 2 == 1)
646 					step++;
647 				path.add(step);
648 				return this;
649 			}
650 			
651 			public Path build() {
652 				Stack<Integer> copy = new Stack<Integer>();
653 				for (Integer step : path)
654 					copy.add(step);
655 				return new Path(copy);
656 			}
657 		}
658 		
659 		@Override
660 		public int hashCode() {
661 			final int prime = 31;
662 			int result = 1;
663 			result = prime * result + path.hashCode();
664 			return result;
665 		}
666 		
667 		@Override
668 		public boolean equals(Object obj) {
669 			if (this == obj)
670 				return true;
671 			if (obj == null)
672 				return false;
673 			if (getClass() != obj.getClass())
674 				return false;
675 			Path other = (Path)obj;
676 			return this.path.equals(other.path);
677 		}
678 		
679 		@Override
680 		public String toString() {
681 			if (isRoot())
682 				return "/";
683 			StringBuilder b = new StringBuilder();
684 			for (Integer i : path)
685 				b.append('/').append(i);
686 			return b.toString();
687 		}
688 		
689 		// for testing
690 		static Path parse(String string) {
691 			Stack<Integer> path = new Stack<>();
692 			if ("/".equals(string))
693 				return ROOT;
694 			if (!string.matches("(/(0|[1-9][0-9]*))+"))
695 				throw new IllegalArgumentException();
696 			String[] steps = string.split("/");
697 			for (int i = 1; i < steps.length; i++)
698 				path.add(Integer.parseInt(steps[i]));
699 			return new Path(path);
700 		}
701 	}

703 	static class BreakPosition implements Comparable<BreakPosition> {
704 		
705 		enum Side { BEFORE, AFTER }
706 		final Path path;
707 		final Side side;
708 		
709 		BreakPosition(Path path, Side side) {
710 			if (path == null || side == null)
711 				throw new NullPointerException();
712 			this.path = path;
713 			this.side = side;
714 		}
715 		
716 		// normalize to BEFORE
717 		BreakPosition normalize() {
718 			if (side == Side.BEFORE)
719 				return this;
720 			else
721 				return new BreakPosition(path.builder().inc(), Side.BEFORE);
722 		}
723 		
724 		@Override
725 		public int compareTo(BreakPosition o) {
726 			int c = this.path.compareTo(o.path);
727 			if (c != 0)
728 				return c;
729 			if (this.path.contains(o.path))
730 				return this.side == Side.BEFORE ? -1 : 1;
731 			else if (o.path.contains(this.path))
732 				return o.side == Side.BEFORE ? 1 : -1;
733 			else
734 				return side.compareTo(o.side);
735 		}
736 		
737 		/**
738 		 * @param path The position of the node
739 		 * @return -1 when the split point comes before the node, 0 when the split point lies inside
740 		 *         the node, 1 when the split point comes after the node.
741 		 */
742 		public int compareTo(Path path) {
743 			int c = this.path.compareTo(path);
744 			if (c != 0)
745 				return c;
746 			if (this.path.contains(path))
747 				return 0;
748 			else if (this.side == Side.BEFORE)
749 				return -1;
750 			else
751 				return 1;
752 		}
753 		
754 		@Override
755 		public int hashCode() {
756 			final int prime = 31;
757 			int result = 1;
758 			result = prime * result + path.hashCode();
759 			result = prime * result + side.hashCode();
760 			return result;
761 		}
762 		
763 		@Override
764 		public boolean equals(Object obj) {
765 			if (this == obj)
766 				return true;
767 			if (obj == null)
768 				return false;
769 			if (getClass() != obj.getClass())
770 				return false;
771 			BreakPosition other = (BreakPosition)obj;
772 			return equals(other.path, other.side);
773 		}
774 		
775 		public boolean equals(Path path, Side side) {
776 			if (path == null)
777 				return false;
778 			if (this.side != side)
779 				return false;
780 			if (!this.path.equals(path))
781 				return false;
782 			return true;
783 		}
784 		
785 		@Override
786 		public String toString() {
787 			return side + " " + path;
788 		}
789 	}

791 	static class BreakOpportunity implements Comparable<BreakOpportunity> {
792 		
793 		enum Weight { ALWAYS, PREFER, ALLOW }
794 		final BreakPosition position;
795 		final Weight weight;
796 		final int bytesBefore;
797 		
798 		BreakOpportunity(BreakPosition position, Weight weight, int bytesBefore) {
799 			this.position = position;
800 			this.weight = weight;
801 			this.bytesBefore = bytesBefore;
802 		}
803 		
804 		@Override
805 		public int compareTo(BreakOpportunity o) {
806 			int c = this.position.compareTo(o.position);
807 			if (c != 0)
808 				return c;
809 			return this.weight.compareTo(o.weight);
810 		}
811 		
812 		@Override
813 		public int hashCode() {
814 			final int prime = 31;
815 			int result = 1;
816 			result = prime * result + bytesBefore;
817 			result = prime * result + position.hashCode();
818 			result = prime * result + weight.hashCode();
819 			return result;
820 		}
821 
822 		@Override
823 		public boolean equals(Object obj) {
824 			if (this == obj)
825 				return true;
826 			if (obj == null)
827 				return false;
828 			if (getClass() != obj.getClass())
829 				return false;
830 			BreakOpportunity other = (BreakOpportunity)obj;
831 			if (bytesBefore != other.bytesBefore)
832 				return false;
833 			if (!position.equals(other.position))
834 				return false;
835 			if (weight != other.weight)
836 				return false;
837 			return true;
838 		}
