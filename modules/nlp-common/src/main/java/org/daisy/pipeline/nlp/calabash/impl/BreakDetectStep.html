74  	@Override
75  	public void setOption(QName name, RuntimeValue value) {
76  		super.setOption(name, value);
77  		if ("inline-tags".equalsIgnoreCase(name.getLocalName())) {
78  			inlineTagsOption = value;
79  		} else if ("ensure-word-before".equalsIgnoreCase(name.getLocalName())) {
80  			wordBeforeOption = value;
81  		} else if ("ensure-word-after".equalsIgnoreCase(name.getLocalName())) {
82  			wordAfterOption = value;
83  		} else if ("ensure-sentence-before".equalsIgnoreCase(name.getLocalName())) {
84  			sentenceBeforeOption = value;
85  		} else if ("ensure-sentence-after".equalsIgnoreCase(name.getLocalName())) {
86  			sentenceAfterOption = value;
87  		} else if ("output-word-tag".equalsIgnoreCase(name.getLocalName())) {
88  			wordTagOption = value.getQName();
89  		} else if ("output-sentence-tag".equalsIgnoreCase(name.getLocalName())) {
90  			sentenceTagOption = value.getQName();
91  		} else {
92  			runtime.error(new RuntimeException("unrecognized option " + name));
93  		}
94  	}

105 	private Predicate<XdmNode> processXSLTMatchPatternOption(RuntimeValue option) throws XPathException {
106 		if (!option.getString().equals("")) {
107 			XPathExpression matcher = SaxonHelper.compileExpression(
108 				option.getString(),
109 				option.getNamespaceBindings(),
110 				runtime.getProcessor().getUnderlyingConfiguration());
111 			return n -> n.getNodeKind() == XdmNodeKind.ELEMENT && SaxonHelper.evaluateBoolean(matcher, n);
112 		} else
113 			return n -> false;
114 	}

116 	public void run() throws SaxonApiException {
117 		super.run();
118 
119 		HashMap<Locale, LexerToken> langToToken = new HashMap<Locale, LexerToken>();
120 
121 		//Retrieve a generic lexer that can handle unexpected languages.
122 		LexerToken generic;
123 		try {
124 			generic = mLexerRegistry.getFallbackToken(Collections.EMPTY_LIST);
125 		} catch (LexerInitException e1) {
126 			mRuntime.error(e1);
127 			return;
128 		}
129 		langToToken.put(null, generic);
130 
131 		FormatSpecifications formatSpecs;
132 		try {
133 			formatSpecs = new FormatSpecifications(
134 				sentenceTagOption, wordTagOption, "http://www.w3.org/XML/1998/namespace", "lang",
135 				processXSLTMatchPatternOption(inlineTagsOption),
136 				processXSLTMatchPatternOption(wordBeforeOption),
137 				processXSLTMatchPatternOption(wordAfterOption),
138 				processXSLTMatchPatternOption(sentenceBeforeOption),
139 				processXSLTMatchPatternOption(sentenceAfterOption));
140 		} catch (XPathException e) {
141 			throw new XProcException(step, e);
142 		}
143 
144 		XmlBreakRebuilder xmlRebuilder = new XmlBreakRebuilder();
145 
146 		long before = System.currentTimeMillis();
147 
148 		if (mLangDetector == null) {
149 			mLangDetector = new DummyLangDetector();
150 			mLangDetector.train();
151 		}
152 
153 		while (mSource.moreDocuments()) {
154 			XdmNode doc = mSource.read();
155 
156 			//init the lexers with the languages
157 			mLangs = new HashSet<Locale>();
158 			try {
159 				new InlineSectionFinder().find(doc, 0, formatSpecs, this,
160 				        Collections.EMPTY_SET);
161 				for (Locale lang : mLangs) {
162 					if (!langToToken.containsKey(lang)) {
163 						LexerToken token = mLexerRegistry.getTokenForLang(lang, langToToken
164 						        .values());
165 						if (token == null) {
166 							throw new LexerInitException(
167 							        "cannot find a lexer for the language: " + lang);
168 						}
169 						langToToken.put(lang, token);
170 					}
171 				}
172 			} catch (LexerInitException e) {
173 				mRuntime.error(e);
174 				continue;
175 			}
176 
177 			mLogger.debug("Total number of language(s): " + (langToToken.size() - 1));
178 			for (Map.Entry<Locale, LexerToken> entry : langToToken.entrySet()) {
179 				mLogger.debug("LexService for language '"
180 				        + (entry.getKey() == null ? "<ANY>" : entry.getKey()) + "': "
181 				        + entry.getValue().getLexService().getName());
182 			}
183 
184 			//rebuild the XML tree and lex the content on-the-fly
185 			List<String> parsingErrors = new ArrayList<String>();
186 			XdmNode tree;
187 			try {
188 				tree = xmlRebuilder.rebuild(this, langToToken, doc, formatSpecs,
189 				        mLangDetector, false, parsingErrors);
190 				mResult.write(tree);
191 			} catch (LexerInitException e) {
192 				mRuntime.error(e);
193 			}
194 			for (String error : parsingErrors) {
195 				mRuntime.info(null, null, doc.getBaseURI() + ": " + error);
196 			}
197 		}
198 
199 		for (LexerToken token : langToToken.values()) {
200 			mLexerRegistry.releaseToken(token);
201 		}
202 
203 		long after = System.currentTimeMillis();
204 		mLogger.debug("lexing time = " + (after - before) / 1000.0 + " s.");
205 
206 		mLangs = null;
207 	}

214 	@Override
215 	public void onInlineSectionFound(List<Leaf> leaves, List<String> text, Locale lang)
216 	        throws LexerInitException {
217 
218 		//TODO: find a way to not doing this multiple times (it is done also in the rebuilder)
219 		//If this is really too CPU-intensive, one can skip this detection.
220 		//As a result, no lexer would be loaded for the true language if the language is not
221 		//also used somewhere else, and the XMLRebuilder will eventually use a generic lexer for
222 		//the detected language.
223 
224 		lang = mLangDetector.findLang(lang, text);
225 
226 		if (lang != null) {
227 			mLangs.add(lang);
228 		}
229 	}
