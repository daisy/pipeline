1   package org.daisy.pipeline.nlp.calabash.impl;
2   
3   import org.daisy.common.xproc.calabash.XProcStep;
4   import org.daisy.common.xproc.calabash.XProcStepProvider;
5   import org.daisy.pipeline.nlp.lexing.LexServiceRegistry;
6   
7   import com.xmlcalabash.core.XProcRuntime;
8   import com.xmlcalabash.runtime.XAtomicStep;
9   
10  import org.osgi.service.component.annotations.Component;
11  import org.osgi.service.component.annotations.Reference;
12  import org.osgi.service.component.annotations.ReferenceCardinality;
13  import org.osgi.service.component.annotations.ReferencePolicy;
14  
15  /**
16   * OSGi component that provides an XProcStep on the top of LexServices satisfied
17   * by OSGi services.
18   */
19  @Component(
20  	name = "break-detector",
21  	service = { XProcStepProvider.class },
22  	property = { "type:String={http://www.daisy.org/ns/pipeline/xproc}break-detect" }
23  )
24  public class BreakDetectProvider implements XProcStepProvider {
25  
26  	private LexServiceRegistry mRegistry = null;
27  
28  	@Reference(
29  		name = "LexServiceRegistry",
30  		unbind = "unsetLexServiceRegistry",
31  		service = LexServiceRegistry.class,
32  		cardinality = ReferenceCardinality.MANDATORY,
33  		policy = ReferencePolicy.STATIC
34  	)
35  	public void setLexServiceRegistry(LexServiceRegistry r) {
36  		mRegistry = r;
37  	}
38  
39  	public void unsetLexServiceRegistry(LexServiceRegistry r) {
40  		mRegistry = null;
41  	}
42  
43  	@Override
44  	public XProcStep newStep(XProcRuntime runtime, XAtomicStep step) {
45  		return new BreakDetectStep(runtime, step, mRegistry);
46  	}
47  }
