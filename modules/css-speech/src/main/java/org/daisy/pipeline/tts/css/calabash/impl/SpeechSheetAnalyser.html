8   import java.util.ArrayList;
9   import java.util.Collection;
10  import java.util.List;
11  import java.util.regex.Matcher;
12  import java.util.regex.Pattern;
13  
14  import org.apache.commons.io.IOUtils;
15  import org.w3c.dom.Document;
16  
17  import cz.vutbr.web.css.CSSException;
18  import cz.vutbr.web.css.CSSFactory;
19  import cz.vutbr.web.css.NetworkProcessor;
20  import cz.vutbr.web.css.StyleSheet;

22  import cz.vutbr.web.csskit.antlr.CSSParserFactory;
23  import cz.vutbr.web.csskit.antlr.CSSSource;
24  import cz.vutbr.web.csskit.antlr.DefaultCSSSourceReader;
25  import cz.vutbr.web.domassign.Analyzer;
26  import cz.vutbr.web.domassign.StyleMap;
27  import cz.vutbr.web.domassign.SupportedCSS21;
28  
29  public class SpeechSheetAnalyser {
30  	private static SupportedCSS SupportedCSS;
31  	private static final String Medium = "speech";
32  	private static Pattern UrlPattern = Pattern.compile("url\\('?\"?([^)'\"]+)'?\"?\\)");
33  
34  	private static Pattern AbsoluteUrlPattern = Pattern
35  	        .compile("([a-zA-Z][-.+a-zA-Z0-9]*://)|/");
36  
37  	private Analyzer mAnalyzer;
38  	private static final CSSParserFactory parserFactory = CSSParserFactory.getInstance();
39  
40  	static {
41  		SupportedCSS = SupportedCSS21.getInstance();
42  		CSSFactory.registerSupportedCSS(SupportedCSS);
43  		CSSFactory.registerDeclarationTransformer(new SpeechDeclarationTransformer());
44  	}
45  
46  	public void analyse(Collection<URI> sheetURIs, Collection<String> embeddedCSS,
47  	        URI embedContainerURI, NetworkProcessor network) throws IOException, URISyntaxException, CSSException {
48  		if (!SupportedCSS.isSupportedMedia(Medium)) {
49  			throw new IllegalStateException("medium '" + Medium + "' is not supported");
50  		}
51  
52  		List<URI> alluris = new ArrayList<URI>();
53  		List<String> csscode = new ArrayList<String>(embeddedCSS);
54  		for (int k = 0; k < embeddedCSS.size(); ++k)
55  			alluris.add(embedContainerURI);
56  
57  		for (URI uri : sheetURIs) {
58  			if (uri != null) {
59  				alluris.add(uri);
60  				csscode.add(IOUtils.toString(uri.toURL().openStream(), "UTF-8"));
61  			}
62  		}
63  
64  		List<StyleSheet> styleSheets = new ArrayList<StyleSheet>();
65  		for (int k = 0; k < csscode.size(); ++k) {
66  			String basePath = alluris.get(k).resolve(".").toString();
67  			String withAbsURL = makeURLabsolute(csscode.get(k), basePath);
68  			styleSheets.add(parserFactory.parse(new CSSSource(withAbsURL, (String)null, new URL("http://base"), 0, 0),
69  			                                    new DefaultCSSSourceReader(network)));
70  			/*
71  			 * we cannot use CSSFactory.parse(withAbsURL) because it tries to
72  			 * convert the null base URL into a String
73  			 */
74  		}
75  
76  		mAnalyzer = new Analyzer(styleSheets);
77  	}
78  
79  	public StyleMap evaluateDOM(Document doc) {
80  		return mAnalyzer.evaluateDOM(doc, Medium, false);
81  	}
82  
83  	static String makeURLabsolute(String csscode, String basePath) {
84  		//This ugly hack is necessary because jStyleParser fails to take into 
85  		//account the base URI when it parses relative URIs
86  		Matcher m = UrlPattern.matcher(csscode);
87  		while (m.find()) {
88  			String url = m.group(1);
89  			Matcher m2 = AbsoluteUrlPattern.matcher(url);
90  			if (!m2.lookingAt()) {
91  				csscode = csscode.replace(m.group(0), "url('" + Paths.get(basePath, url)
92  				        + "')");
93  			}
94  		}
95  		return csscode;
96  	}
97  }
