40  	private final static String mStyleNsPrefix = "tts";
41  	private final static String xmlns = "http://www.w3.org/2000/xmlns/";
42  	private boolean mFirst;
43  	private TreeWriter mTreeWriter;
44  	private StyleMap mStyleMap;
45  	private String mStyleNS;
46  	private Map<String, String> mPrefixes;
47  	private static Set<String> mCSS2Properties = new HashSet<String>(Arrays.asList(
48  	        "voice-family", "stress", "richness", "cue", "cue-before", "cue-after", "pause",
49  	        "pause-after", "pause-before", "azimuth", "volume", "speak", "play-during",
50  	        "elevation", "speech-rate", "pitch", "pitch-range", "stress", "speak-punctuation",
51  	        "speak-numeral", "speak-header"));

53  	public XdmNode inline(TreeWriterFactory twFactory, URI docURI, XdmNode root,
54  	        SpeechSheetAnalyser analyzer, String styleNamespace) {
55  		// match the document with the style sheet
56  		DocumentOverNodeInfo doc = (DocumentOverNodeInfo) DocumentOverNodeInfo.wrap(root
57  		        .getUnderlyingNode());
58  
59  		Document wrapped = (Document) WrapperOverWrapper.wrap(doc);
60  		mStyleMap = analyzer.evaluateDOM(wrapped);
61  
62  		mStyleNS = styleNamespace;
63  		mFirst = true;
64  		mTreeWriter = twFactory.newInstance();
65  		mTreeWriter.startDocument(docURI);
66  
67  		Set<String> namespaces = new HashSet<String>();
68  		String defaultNs = getAllNamespaces(wrapped.getDocumentElement(), namespaces);
69  		namespaces.remove(xmlns);
70  		mPrefixes = new HashMap<String, String>();
71  		for (String ns : namespaces) {
72  			String p = "ns" + mPrefixes.size();
73  			mPrefixes.put(ns, p);
74  		}
75  		mPrefixes.put(mStyleNS, "tts");
76  		mPrefixes.put("http://www.w3.org/XML/1998/namespace", "xml");
77  		if (defaultNs != null)
78  			mPrefixes.put(defaultNs, "");
79  
80  		rebuildRec(wrapped.getOwnerDocument());
81  		mTreeWriter.endDocument();
82  
83  		XdmNode result = mTreeWriter.getResult();
84  
85  		//help the GC does its job
86  		mStyleNS = null;
87  		mTreeWriter = null;
88  		mStyleMap = null;
89  
90  		return result;
91  	}
92  
93  	private static String getAllNamespaces(Node n, Set<String> namespaces) {
94  		if (n.getNodeType() == Node.ELEMENT_NODE) {
95  			String ns = n.getNamespaceURI();
96  			if (ns != null && !ns.isEmpty())
97  				namespaces.add(ns);
98  			NamedNodeMap attributes = n.getAttributes();
99  			for (int i = 0; i < attributes.getLength(); i++) {
100 				Node attr = attributes.item(i);
101 				ns = attr.getNamespaceURI();
102 				if (ns != null && !ns.isEmpty())
103 					namespaces.add(ns);
104 			}
105 			String defaultNs = null;
106 			for (Node child = n.getFirstChild(); child != null; child = child.getNextSibling()) {
107 				String ret = getAllNamespaces(child, namespaces);
108 				if (ret != null)
109 					defaultNs = ret;
110 			}
111 			if (n.getPrefix() == null || n.getPrefix().isEmpty())
112 				return n.getNamespaceURI();
113 			return defaultNs;
114 		}
115 		return null;
116 	}

118 	private void rebuildRec(Node node) {
119 		if (node.getNodeType() == Node.COMMENT_NODE) {
120 			mTreeWriter.addComment(node.getNodeValue());
121 		} else if (node.getNodeType() == Node.TEXT_NODE) {
122 			mTreeWriter.addText(node.getNodeValue());
123 		} else if (node.getNodeType() == Node.PROCESSING_INSTRUCTION_NODE) {
124 			mTreeWriter.addPI(node.getNodeName(), node.getNodeValue());
125 		} else if (node.getNodeType() == Node.DOCUMENT_NODE) {
126 			for (Node child = node.getFirstChild(); child != null; child = child
127 			        .getNextSibling())
128 				rebuildRec(child);
129 		} else if (node.getNodeType() == Node.ELEMENT_NODE) {
130 			if (node.getNamespaceURI() == null || node.getNamespaceURI().isEmpty())
131 				mTreeWriter.addStartElement(new QName(null, node.getLocalName()));
132 			else
133 				mTreeWriter.addStartElement(new QName(mPrefixes.get(node.getNamespaceURI()),
134 				        node.getNamespaceURI(), node.getLocalName()));
135 
136 			if (mFirst) {
137 				for (Map.Entry<String, String> ns : mPrefixes.entrySet()) {
138 					mTreeWriter.addNamespace(ns.getValue(), ns.getKey());
139 				}
140 				mFirst = false;
141 			}
142 
143 			NamedNodeMap attributes = node.getAttributes();
144 			for (int i = 0; i < attributes.getLength(); i++) {
145 				Node attr = attributes.item(i);
146 				if (attr.getNamespaceURI() != null && !attr.getNamespaceURI().isEmpty()) {
147 					if (!attr.getNamespaceURI().equals(xmlns))
148 						mTreeWriter.addAttribute(new QName(mPrefixes.get(attr
149 						        .getNamespaceURI()), attr.getNamespaceURI(), attr
150 						        .getLocalName()), attr.getNodeValue());
151 				} else {
152 					mTreeWriter.addAttribute(new QName(null, attr.getLocalName()), attr
153 					        .getNodeValue());
154 				}
155 			}
156 
157 			// ===== start inlining ===== //
158 			Element elem = (Element) node;
159 			{
160 			NodeData nd = mStyleMap.get(elem);
161 			if (nd != null) {
162 				for (String property : nd.getPropertyNames()) {
163 					if (mCSS2Properties.contains(property)) {
164 						Term<?> t = nd.getValue(property, false);
165 						String str = null;
166 						if (t == null || t.getValue() == null) {
167 							//jStyleParser replaces '-' with '_'. Best workaround so far is to do the opposite:
168 							//(voice-family and cue aside, there is no property values with '_' in Aural CSS)
169 							str = nd.getProperty(property, false).toString().replace("_", "-")
170 							        .toLowerCase();
171 						} else if (t.getValue() instanceof List<?>) {
172 							List<?> li = (List<?>) t.getValue();
173 							StringBuilder sb = new StringBuilder();
174 							Iterator it = li.iterator();
175 							sb.append(it.next());
176 							while (it.hasNext()) {
177 								Term<?> term = (Term<?>) it.next();
178 								sb.append("," + term.getValue().toString());
179 							}
180 							str = sb.toString();
181 
182 						} else if (property.startsWith("cue")) {
183 							str = t.getValue().toString();
184 						} else {
185 							str = t.toString().replace("_", "-").toLowerCase();
186 						}
187 						mTreeWriter.addAttribute(
188 						        new QName(mStyleNsPrefix, mStyleNS, property), str);
189 
190 					}
191 				}
192 			}
193 			}
194 			// process ::after and ::before pseudo elements
195 
196 			for (PseudoElement pseudo : mStyleMap.pseudoSet(elem)) {
197 				if ("::after".equals(pseudo.toString()) || "::before".equals(pseudo.toString())) {
198 					NodeData nd = mStyleMap.get(elem, pseudo);
199 					for (String property : nd.getPropertyNames()) {
200 						if ("content".equals(property)) {
201 							switch ((Content)nd.getProperty(property)) {
202 							case list_values:
203 								StringBuilder value = new StringBuilder();
204 								for (Term<?> t : (TermList)nd.getValue(property, false)) {
205 									if (t instanceof TermString) {
206 										value.append(t.getValue());
207 										continue;
208 									} else if (t instanceof TermFunction) {
209 										TermFunction f = (TermFunction)t;
210 										if ("attr".equals(f.getFunctionName().toLowerCase())
211 										    && f.size() == 1
212 										    && f.get(0) instanceof TermIdent) {
213 											value.append(elem.getAttribute(((TermIdent)f.get(0)).getValue()));
214 											continue;
215 										}
216 									}
217 									logger.warn("Don't know how to speak content value " + t
218 									            + " within ::" + pseudo.getName() + " pseudo-element");
219 									value = null;
220 									break;
221 								}
222 								if (value != null && value.length() > 0)
223 									mTreeWriter.addAttribute(
224 										new QName(mStyleNsPrefix, mStyleNS, pseudo.getName()), value.toString());
225 								break;
226 							case NORMAL:
227 							case NONE:
228 							case INHERIT:
229 							case INITIAL:
230 							default:
231 								break;
232 							}
233 							break;
234 						} else if (mCSS2Properties.contains(property)) {
235 							logger.warn("Ignoring property '"+ property
236 							            + "' within ::" + pseudo.getName() + " pseudo-element");
237 						}
238 					}
239 				}
240 			}
241 
242 			mTreeWriter.startContent();
243 
244 			// ===== end inlining ===== //
245 
246 			for (Node child = elem.getFirstChild(); child != null; child = child
247 			        .getNextSibling())
248 				rebuildRec(child);
249 
250 			mTreeWriter.addEndElement();
251 		}
252 	}
