1   package org.daisy.pipeline.tts.css.calabash.impl;
2   
3   import java.net.URI;
4   import java.net.URISyntaxException;
5   import java.net.URL;
6   import java.util.ArrayList;
7   import java.util.Collection;
8   
9   import net.sf.saxon.s9api.QName;
10  import net.sf.saxon.s9api.XdmNode;
11  
12  import org.daisy.pipeline.tts.config.ConfigReader;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  public class CSSConfigExtension implements ConfigReader.Extension {
17  
18  	private Logger Logger = LoggerFactory.getLogger(CSSConfigExtension.class);
19  
20  	@Override
21  	public boolean parseNode(XdmNode node, URI documentURI) {
22  		String name = node.getNodeName().getLocalName();
23  		if ("css".equalsIgnoreCase(name)) {
24  			String href = node.getAttributeValue(new QName(null, "href"));
25  			if (href != null) {
26  				URL url = ConfigReader.URIinsideConfig(href, documentURI);
27  				if (url != null)
28  					try {
29  						mCSSuris.add(url.toURI());
30  					} catch (URISyntaxException e) {
31  					}
32  			} else {
33  				String content = node.getStringValue();
34  				if (content != null && !content.isEmpty()) {
35  					mEmbeddedCSS.add(content);
36  				}
37  			}
38  			return true;
39  		}
40  		return false;
41  	}
42  
43  	public Collection<URI> getCSSstylesheetURIs() {
44  		return mCSSuris;
45  	}
46  
47  	public Collection<String> getEmbeddedCSS() {
48  		return mEmbeddedCSS;
49  	}
50  
51  	@Override
52  	public void setParentReader(ConfigReader cr) {
53  	}
54  
55  	private Collection<URI> mCSSuris = new ArrayList<URI>();
56  	private Collection<String> mEmbeddedCSS = new ArrayList<String>();
57  }
