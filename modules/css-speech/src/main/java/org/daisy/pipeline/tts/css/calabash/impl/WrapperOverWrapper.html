1   package org.daisy.pipeline.tts.css.calabash.impl;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import net.sf.saxon.dom.AttrOverNodeInfo;
7   import net.sf.saxon.dom.DOMNodeList;
8   import net.sf.saxon.dom.DocumentOverNodeInfo;
9   import net.sf.saxon.dom.ElementOverNodeInfo;
10  import net.sf.saxon.dom.PIOverNodeInfo;
11  import net.sf.saxon.dom.TextOverNodeInfo;
12  
13  import org.w3c.dom.Node;
14  import org.w3c.dom.NodeList;
15  
16  /**
17   * This wrapper and the W3C adapters allow us to override getAttributeNode() of
18   * ElementOverNodeInfo so it does not take into account the prefix when
19   * attributes are compared.
20   */
21  public class WrapperOverWrapper {
22  
23  	public static Node wrap(Node n) {
24  		if (n instanceof ElementOverNodeInfo) {
25  			W3CElementAdapter res = new W3CElementAdapter();
26  			res.setUnderlyingNode(((ElementOverNodeInfo) n)
27  			        .getUnderlyingNodeInfo());
28  			return res;
29  		}
30  		if (n instanceof DocumentOverNodeInfo) {
31  			W3CDocumentAdapter res = new W3CDocumentAdapter();
32  			res.setUnderlyingNode(((DocumentOverNodeInfo) n)
33  			        .getUnderlyingNodeInfo());
34  			return res;
35  		}
36  		if (n instanceof AttrOverNodeInfo) {
37  			W3CAttrAdapter res = new W3CAttrAdapter();
38  			res.setUnderlyingNode(((AttrOverNodeInfo) n)
39  			        .getUnderlyingNodeInfo());
40  			return res;
41  		}
42  		if (n instanceof TextOverNodeInfo) {
43  			W3CTextAdapter res = new W3CTextAdapter();
44  			res.setUnderlyingNode(((TextOverNodeInfo) n)
45  			        .getUnderlyingNodeInfo());
46  			return res;
47  		}
48  		if (n instanceof PIOverNodeInfo) {
49  			W3CPiAdapter res = new W3CPiAdapter();
50  			res.setUnderlyingNode(((PIOverNodeInfo) n).getUnderlyingNodeInfo());
51  			return res;
52  		}
53  		return n;
54  	}
55  
56  	public static NodeList wrap(NodeList l) {
57  		List<Node> res = new ArrayList<Node>(10);
58  		for (int i = 0; i < l.getLength(); ++i)
59  			res.add(wrap(l.item(i)));
60  		return new DOMNodeList(res);
61  	}
62  
63  }
