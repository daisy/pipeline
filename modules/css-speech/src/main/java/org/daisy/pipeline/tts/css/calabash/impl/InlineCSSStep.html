53  	public InlineCSSStep(XProcRuntime runtime, XAtomicStep step, final URIResolver resolver) {

56  		mNetwork = new DefaultNetworkProcessor() {
57  			@Override
58  			public InputStream fetch(URL url) throws IOException {
59  				try {
60  					if (url != null) {
61  						Source resolved = resolver.resolve(url.toString(), "");
62  						if (resolved != null) {
63  							if (resolved instanceof StreamSource)
64  								return ((StreamSource)resolved).getInputStream();
65  							else
66  								url = new URL(resolved.getSystemId());
67  						}
68  					}
69  				} catch (TransformerException e) {
70  				} catch (MalformedURLException e) {
71  				}
72  				return super.fetch(url);
73  			}
74  		};
75  	}

105 	static URI buildAbsoluteURI(String path, XdmNode doc) {
106 		URI u;
107 		try {
108 			u = new URI(path);
109 		} catch (URISyntaxException e) {
110 			try {
111 				u = new URI("file://" + path);
112 			} catch (URISyntaxException e1) {
113 				return null;
114 			}
115 		}
116 		if (!u.isAbsolute()) {
117 			u = doc.getBaseURI().resolve(u);
118 		}
119 
120 		return u;
121 	}

123 	static Collection<URI> getCSSurisInContent(XdmNode doc) {
124 		Collection<URI> result = new ArrayList<URI>();
125 
126 		//add the CSS stylesheet URI from the processing-instructions
127 		XdmSequenceIterator it = doc.axisIterator(Axis.CHILD);
128 		while (it.hasNext()) {
129 			XdmNode next = (XdmNode) it.next();
130 			if (next.getNodeKind() == XdmNodeKind.PROCESSING_INSTRUCTION) {
131 				String content = next.getStringValue();
132 				String href = ProcInstParser.getPseudoAttribute(content, "href");
133 				if ("xml-stylesheet".equals(next.getNodeName().getLocalName())
134 				        && href != null
135 				        && !href.isEmpty()
136 				        && "text/css".equals(ProcInstParser
137 				                .getPseudoAttribute(content, "type"))) {
138 
139 					URI uri = buildAbsoluteURI(href, doc);
140 					if (uri != null)
141 						result.add(uri);
142 				}
143 			}
144 		}
145 
146 		//add the CSS stylesheet URIs from the headers
147 		it = doc.axisIterator(Axis.DESCENDANT);
148 		while (it.hasNext()) {
149 			XdmNode next = (XdmNode) it.next();
150 			if (next.getNodeKind() == XdmNodeKind.ELEMENT
151 			        && "link".equals(next.getNodeName().getLocalName())
152 			        && "stylesheet".equals(next.getAttributeValue(new QName(null, "rel")))) {
153 				String href = next.getAttributeValue(new QName(null, "href"));
154 				URI uri = buildAbsoluteURI(href, doc);
155 				if (uri != null)
156 					result.add(uri);
157 			}
158 		}
159 
160 		return result;
161 	}

163 	public void run() throws SaxonApiException {
164 		super.run();
165 
166 		//read config
167 		CSSConfigExtension cssExt = new CSSConfigExtension();
168 		XdmNode config = null; //it is allowed to not provide any config file
169 		if (mConfig.moreDocuments()) {
170 			config = mConfig.read();
171 		}
172 		new ConfigReader(mRuntime.getProcessor(), config, cssExt);
173 
174 		//read first document
175 		XdmNode doc = null;
176 		if (mSource.moreDocuments()) {
177 			doc = mSource.read();
178 		}
179 
180 		Collection<URI> alluris = new ArrayList<URI>();
181 		alluris.addAll(getCSSurisInContent(doc));
182 		alluris.addAll(cssExt.getCSSstylesheetURIs());
183 
184 		CSSInliner inliner = new CSSInliner();
185 		SpeechSheetAnalyser analyzer = new SpeechSheetAnalyser();
186 		try {
187 			analyzer.analyse(alluris, cssExt.getEmbeddedCSS(), config.getBaseURI(), mNetwork);
188 		} catch (Throwable t) {
189 			Logger.debug("error while analyzing CSS speech: " + t.getMessage());
190 			mResult.write(doc);
191 			return;
192 		}
193 
194 		// rebuild the document with the additional style info
195 		XdmNode rebuilt = inliner.inline(this, doc.getBaseURI(), doc, analyzer,
196 		        mStyleNsOption);
197 
198 		for (URI uri : alluris) {
199 			mRuntime.info(null, null, uri.toString() + " inlined");
200 		}
201 
202 		mResult.write(rebuilt);
203 	}
