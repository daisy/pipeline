33  
34  @Component(
35  	name = "pf:asciimath-to-ssml",
36  	service = { ExtensionFunctionProvider.class }
37  )
38  public class ASCIIMathToSSMLFunctionProvider extends ReflexiveExtensionFunctionProvider {
39  
40  	public ASCIIMathToSSMLFunctionProvider() {
41  		super();
42  		addExtensionFunctionDefinitionsFromClass(ASCIIMathToSSML.class, new ASCIIMathToSSML());
43  	}
44  
45  	private final List<TTSInputProcessor> processors = new ArrayList<>();
46  
47  	@Reference(
48  		name = "TTSInputProcessor",
49  		unbind = "-",
50  		service = TTSInputProcessor.class,
51  		cardinality = ReferenceCardinality.MULTIPLE,
52  		policy = ReferencePolicy.STATIC
53  	)
54  	public void bindTTSInputProcessor(TTSInputProcessor processor) {
55  		processors.add(processor);
56  	}
57  
58  	public class ASCIIMathToSSML {
59  
60  		private final static String MIME_ASCIIMATH = "text/x-asciimath";
61  		private final static String MIME_MATHML = "application/mathml+xml";
62  
63  		public void transform(XMLInputValue<?> asciimath, XMLOutputValue<?> ssml, Locale language, Configuration saxonConfig) {
64  			asciimath = asciimath.ensureSingleItem();
65  			TTSInputProcessor processor = null; {
66  				for (TTSInputProcessor p : processors)
67  					if ((p.supportsInputMediaType(MIME_ASCIIMATH) || p.supportsInputMediaType(MIME_MATHML))
68  					    && (processor == null || p.getPriority() > processor.getPriority()))
69  						processor = p.forInputMediaType(
70  							p.supportsInputMediaType(MIME_ASCIIMATH)
71  								? MIME_ASCIIMATH
72  								: MIME_MATHML);
73  			}
74  			if (processor != null) {
75  				SingleInSingleOutXMLTransformer transformer = null; {
76  					if (processor instanceof XMLTransformer)
77  						transformer = SingleInSingleOutXMLTransformer.from((XMLTransformer)processor);
78  					else
79  						; // could be a XProcStepProvider, but not supported for now
80  				}
81  				if (transformer != null) {
82  					XMLInputValue<?> math = asciimath;
83  					if (!processor.supportsInputMediaType(MIME_ASCIIMATH)) {
84  						// first convert to MathML
85  						Node n = asciimath.asNodeIterator().next();
86  						if (!(n instanceof Text))
87  							throw new IllegalArgumentException("Expected text node, but got;: " + n.getClass());
88  						Element mathml = ASCIIMathML.convert(n.getNodeValue());
89  						math = new SaxonInputValue(SaxonHelper.nodeInfoFromNode(mathml, saxonConfig));
90  					}
91  					transformer.transform(
92  						math,
93  						ssml,
94  						new InputValue<>(ImmutableMap.of(new QName("language"), new InputValue<>(language)))
95  					).run();
96  					return;
97  				}
98  			}
99  			throw new UnsupportedOperationException("No ASCIIMath to SSML processor found");
100 		}
101 	}
102 }
