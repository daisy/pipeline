1   package org.daisy.pipeline.css;
2   
3   import java.net.URL;
4   import java.util.Collection;
5   
6   /**
7    * Style sheet that should be applied by default for a given content
8    * media type and output medium.
9    */
10  public interface UserAgentStylesheet {
11  
12      /**
13       * Get the resource
14       */
15      public URL getURL();
16  
17      /**
18       * Get the style sheet type, e.g. "text/css" or "text/x-scss"
19       */
20      public String getType();
21  
22      /**
23       * Whether this style sheet applies to the given content media type.
24       *
25       * @param contentType e.g. "application/x-dtbook+xml" or "application/xhtml+xml"
26       */
27      public boolean matchesContentType(String contentType);
28  
29      /**
30       * Whether this style sheet matches the given medium
31       */
32      public boolean matchesMedium(Medium medium);
33  
34  }
