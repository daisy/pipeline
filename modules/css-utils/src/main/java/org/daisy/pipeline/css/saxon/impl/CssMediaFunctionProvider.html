27  @Component(
28  	name = "pf:css-parse-medium",
29  	service = { ExtensionFunctionProvider.class }
30  )
31  public class CssMediaFunctionProvider extends ReflexiveExtensionFunctionProvider {
32  
33  	public CssMediaFunctionProvider() {
34  		super();
35  		addExtensionFunctionDefinitionsFromClass(CssMediaFunctions.class, new CssMediaFunctions());
36  	}
37  
38  	private final List<MediumProvider> mediumProviders = new ArrayList<>();
39  	private final MediumProvider mediumProvider = MediumProvider.dispatch(mediumProviders, true);
40  
41  	@Reference(
42  		name = "MediumProvider",
43  		unbind = "-",
44  		service = MediumProvider.class,
45  		cardinality = ReferenceCardinality.MULTIPLE,
46  		policy = ReferencePolicy.STATIC
47  	)
48  	protected void bindMediumProvider(MediumProvider provider) {
49  		mediumProviders.add(provider);
50  	}
51  
52  	public class CssMediaFunctions {
53  
54  		private CssMediaFunctions() {}
55  
56  		// pf:css-parse-medium
57  		public Medium parse(Iterator<Object> args) throws IllegalArgumentException {
58  			MediumParser parser = new MediumParser(mediumProvider);
59  			boolean first = true;
60  			try {
61  				while (args.hasNext()) {
62  					Object arg = args.next();
63  					if (arg instanceof Medium) {
64  						if (!first || args.hasNext())
65  							throw new IllegalArgumentException("Medium object specified but not the only item in the sequence");
66  						return (Medium)arg;
67  					} else if (arg instanceof String) {
68  						parser.add((String)arg);
69  					} else if (arg instanceof Map) {
70  						Map<String,Object> features = new HashMap<>();
71  						for (Object k : ((Map)arg).keySet()) {
72  							if (!(k instanceof String))
73  								throw new IllegalArgumentException("Unexpected argument type: map with non-string key");
74  							Object v = ((Map)arg).get(k);
75  							if (v instanceof Attr)
76  								v = ((Attr)v).getNodeValue();
77  							else if (!(v instanceof String || v instanceof Integer || v instanceof Long || v instanceof Boolean || v instanceof Locale))
78  								throw new IllegalArgumentException("Unexpected argument type: map with value of type " + v.getClass());
79  							features.put((String)k, v);
80  						}
81  						parser.add(features);
82  					} else
83  						throw new IllegalArgumentException("Unexpected argument type: " + arg.getClass());
84  					first = false;
85  				}
86  				if (first)
87  					throw new IllegalArgumentException("Argument must not be an empty sequence");
88  				return parser.parse();
89  			} catch (IllegalArgumentException|NoSuchElementException e) {
90  				throw new IllegalArgumentException("Argument can not be resolved to a medium", e);
91  			}
92  		}
93  
94  		// pf:media-query-matches
95  		public boolean matches(String mediaQuery, Iterator<Object> medium) throws IllegalArgumentException {
96  			return parse(medium).matches(mediaQuery);
97  		}
98  	}
99  
100 	private static final Logger logger = LoggerFactory.getLogger(CssMediaFunctions.class);
