77  	@Override
78  	public void doInit() {
79  		super.doInit();
80  		if (!isAuthenticated())
81  			return;
82  		uriResolver = (URIResolver)getContext().getAttributes().get(URI_RESOLVER_KEY);
83  		datatypeRegistry = (DatatypeRegistry)getContext().getAttributes().get(DATATYPE_REGISTRY_KEY);

86  	/**
87  	 * Gets the resource.
88  	 *
89  	 * @return the resource
90  	 */
91  	@Post
92  	public Representation getResource(Representation representation) {
93  		logRequest();
94  		maybeEnableCORS();
95  		if (!isAuthenticated()) {
96  			setStatus(Status.CLIENT_ERROR_UNAUTHORIZED);
97  			return null;
98  		}
99  		if (representation == null) {
100 			setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
101 			return getErrorRepresentation("POST request with no entity");
102 		}
103 		Document request = null;
104 		ZippedJobResources data; {
105 			if (MediaType.MULTIPART_FORM_DATA.equals(representation.getMediaType(), true)) {
106 				MultipartRequestData multi = null;
107 				try {
108 					multi = MultipartRequestData.processMultipart(getRequest(),
109 					                                              STYLESHEET_PARAMETERS_DATA_FIELD,
110 					                                              STYLESHEET_PARAMETERS_REQUEST_FIELD,
111 					                                              new File(getConfiguration().getTmpDir()));
112 				} catch (Exception e) {
113 					return badRequest(e);
114 				}
115 				if (multi == null) {
116 					setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
117 					return getErrorRepresentation("Multipart data is empty");
118 				}
119 				request = multi.getXml();
120 				data = new ZippedJobResources(multi.getZipFile());
121 			} else {
122 				data = null;
123 				try {
124 					DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
125 					factory.setNamespaceAware(true);
126 					DocumentBuilder builder = factory.newDocumentBuilder();
127 					request = builder.parse(new InputSource(new StringReader(representation.getText())));
128 				} catch (IOException|ParserConfigurationException|SAXException e) {
129 					return badRequest(e);
130 				}
131 			}
132 		}
133 		if (logger.isDebugEnabled())
134 			logger.debug(XmlUtils.nodeToString(request));
135 		if (!XmlValidator.validate(request, STYLESHEET_PARAMETERS_REQUEST_SCHEMA_URL)) {
136 			setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
137 			return getErrorRepresentation("The request is not valid");
138 		}
139 
140 		// using ScriptInput.Builder to validate URIs and convert them to Source objects
141 		ScriptInput inputs; {
142 			ScriptInput.Builder builder = new ScriptInput.Builder(data);
143 			NodeList userStylesheets = request.getElementsByTagNameNS(NS_DAISY, "userStylesheets");
144 			if (userStylesheets.getLength() > 0) {
145 				NodeList files = ((Element)userStylesheets.item(0)).getElementsByTagNameNS(NS_DAISY, "file");
146 				for (int i = 0; i < files.getLength(); i++) {
147 					Element file = (Element)files.item(i);
148 					URI href = URI.create(file.getAttribute("href"));
149 					try {
150 						if ("file".equals(href.getScheme()) && !getConfiguration().isLocalFS())
151 							throw new FileNotFoundException(
152 								"WS does not allow local inputs but a href starting with 'file:' was found: " + href);
153 						else if ("http".equals(href.getScheme()) || "https".equals(href.getScheme()))
154 							builder.withInput("stylesheet",
155 							                  new ByteArrayInputStream(
156 								                  ("@import url(\"" + href + "\");").getBytes(StandardCharsets.UTF_8)));
157 						else
158 							builder.withInput("stylesheet", href);
159 					} catch (IllegalArgumentException|FileNotFoundException e) {
160 						return badRequest(e);
161 					}
162 				}
163 			}
164 			NodeList sourceDocument = request.getElementsByTagNameNS(NS_DAISY, "sourceDocument");
165 			if (sourceDocument.getLength() > 0) {
166 				NodeList file = ((Element)sourceDocument.item(0)).getElementsByTagNameNS(NS_DAISY, "file");
167 				URI href = URI.create(((Element)file.item(0)).getAttribute("href"));
168 				try {
169 					if ("file".equals(href.getScheme()) && !getConfiguration().isLocalFS())
170 						throw new FileNotFoundException(
171 							"WS does not allow local inputs but a href starting with 'file:' was found: " + href);
172 					builder.withInput("source", href);
173 				} catch (IllegalArgumentException|FileNotFoundException e) {
174 					return badRequest(e);
175 				}
176 			}
177 			inputs = builder.build();
178 		}
179 		Medium medium; {
180 			medium = null;
181 			NodeList media = request.getElementsByTagNameNS(NS_DAISY, "media");
182 			if (media.getLength() > 0)
183 				try {
184 					medium = Medium.parse(((Element)media.item(0)).getAttribute("value"));
185 				} catch (IllegalArgumentException e) {
186 					return badRequest(e);
187 				}
188 			if (medium == null)
189 				medium = Medium.parse("screen");
190 		}
191 		URI contextBase = URI.create("context:/");
192 		uriResolver = fallback(uriResolver, simpleURIResolver);
193 		URIResolver resolver = data == null
194 			? uriResolver
195 			: new URIResolver() {
196 					@Override
197 					public Source resolve(String href, String base) throws TransformerException {
198 						URI hrefURI = URLs.asURI(href);
199 						if (href.startsWith(contextBase.toString())
200 						    || (base != null && base.startsWith(contextBase.toString())
201 						        && !(hrefURI.isAbsolute() || hrefURI.getSchemeSpecificPart().startsWith("/")))) {
202 							if (!href.startsWith(contextBase.toString()))
203 								hrefURI = URLs.resolve(URLs.asURI(base), hrefURI);
204 							try {
205 								URL u = URLs.asURL(hrefURI);
206 								return new StreamSource(u.openStream(), // implemented in custom URLStreamHandler below
207 								                        hrefURI.toString());
208 							} catch (IOException e) {
209 								throw new TransformerException(e);
210 							}
211 						}
212 						return uriResolver.resolve(href, base);
213 					}
214 				};
215 		URLStreamHandlerFactory resetURLStreamHandlerFactory = null;
216 		if (data != null)
217 			resetURLStreamHandlerFactory = setURLStreamHandlerFactory(
218 				new URLStreamHandlerFactory() {
219 					@Override
220 					public URLStreamHandler createURLStreamHandler(String protocol) {
221 						if ("context".equals(protocol))
222 							return new URLStreamHandler() {
223 								@Override
224 								protected URLConnection openConnection(URL u) throws IOException {
225 									return new URLConnection(u) {
226 										@Override
227 										public void connect() throws IOException {
228 										}
229 										@Override
230 										public InputStream getInputStream() throws IOException {
231 											Supplier<InputStream> s = data.getResource(URLs.relativize(contextBase, URLs.asURI(u)).getPath());
232 											if (s == null)
233 												throw new FileNotFoundException();
234 											return s.get();
235 										}
236 									};
237 								}
238 							};
239 						else
240 							return null;
241 					}
242 				}
243 			);
244 		try {
245 			Function<Source,Source> handleZippedInput = s -> {
246 				if (s.getSystemId() == null)
247 					// this means we used a InputStream
248 					return s;
249 				URI u = URLs.asURI(s.getSystemId());
250 				if (u.getScheme() == null)
251 					// we know that u is a relative URI and that data contains the file
252 					u = URLs.resolve(contextBase, u);
253 				try {
254 					return resolver.resolve(u.toString(), "");
255 				} catch (TransformerException e) {
256 					throw new RuntimeException(e);
257 				}
258 			};
259 			Iterable<Source> userStylesheets = Iterables.transform(inputs.getInput("stylesheet"), handleZippedInput);
260 			Source sourceDocument = Iterables.getFirst(Iterables.transform(inputs.getInput("source"), handleZippedInput), null);
261 			Document parametersDoc; {
262 				parametersDoc = XmlUtils.createDom("parameters");
263 				Element parametersElem = parametersDoc.getDocumentElement();
264 				try {
265 					for (SassVariable v : new SassAnalyzer(medium, resolver, datatypeRegistry)
266 						                      .analyze(userStylesheets, sourceDocument)
267 					                          .getVariables()) {
268 						if (v.isDefault()) {
269 							Element parameterElem = parametersDoc.createElementNS(XmlUtils.NS_PIPELINE_DATA, "parameter");
270 							parameterElem.setAttribute("name", v.getName());
271 							parameterElem.setAttribute("nicename", v.getNiceName());
272 							parameterElem.setAttribute("description", v.getDescription());
273 							parameterElem.setAttribute("default", v.getValue());
274 							parameterElem.setAttribute("type", v.getType().getId());
275 							parameterElem.setAttribute("required", "false");
276 							parameterElem.setAttribute("sequence", "false");
277 							parameterElem.setAttribute("ordered", "false");
278 							parametersElem.appendChild(parameterElem);
279 						}
280 					}
281 				} catch (IOException e) {
282 					return badRequest(e);
283 				} catch (DOMException e) { // should not happen
284 					setStatus(Status.SERVER_ERROR_INTERNAL);
285 					return getErrorRepresentation(e.getMessage());
286 				}
287 				DomRepresentation dom = new DomRepresentation(MediaType.APPLICATION_XML, parametersDoc);
288 				setStatus(Status.SUCCESS_OK);
289 				logResponse(dom);
290 				return dom;
291 			}
292 		} finally {
293 			if (data != null)
294 				setURLStreamHandlerFactory(resetURLStreamHandlerFactory);
295 		}
296 	}

298 	private Representation badRequest(Exception e) {
299 		logger.error("bad request:", e);
300 		setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
301 		return getErrorRepresentation(e.getMessage());
302 	}

304 	private static URIResolver simpleURIResolver = new URIResolver() {
305 			@Override
306 			public Source resolve(String href, String base) throws TransformerException {
307 				try {
308 					URI uri; {
309 						if (base != null)
310 							uri = new URI(base).resolve(new URI(href));
311 						else
312 							uri = new URI(href);
313 					}
314 					return new SAXSource(new InputSource(uri.toASCIIString()));
315 				} catch (URISyntaxException e) {
316 					throw new TransformerException(e);
317 				}
318 			}
319 		};
320 
321 	private static URIResolver fallback(URIResolver... resolvers) {
322 		return new URIResolver() {
323 			@Override
324 			public Source resolve(String href, String base) throws TransformerException {
325 				Source source = null;
326 				Iterator<URIResolver> iterator = Iterators.forArray(resolvers);
327 				while (iterator.hasNext()) {
328 					source = iterator.next().resolve(href, base);
329 					if (source != null)
330 						break; }
331 				return source;
332 			}
333 		};
334 	}

336 	// see https://stackoverflow.com/a/18018891
337 	private static URLStreamHandlerFactory setURLStreamHandlerFactory(URLStreamHandlerFactory factory) {
338 		try {
339 			Field f = URL.class.getDeclaredField("factory");
340 			f.setAccessible(true);
341 			URLStreamHandlerFactory curFactory = (URLStreamHandlerFactory)f.get(null);
342 			f.set(null, null);
343 			URL.setURLStreamHandlerFactory(factory);
344 			return curFactory;
345 		} catch (NoSuchFieldException|SecurityException|IllegalArgumentException|IllegalAccessException e) {
346 			throw new RuntimeException();
347 		}
348 	}
