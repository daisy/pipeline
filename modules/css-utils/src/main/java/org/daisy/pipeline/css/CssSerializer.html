1   package org.daisy.pipeline.css;
2   
3   import java.net.URI;
4   import java.util.List;
5   
6   import cz.vutbr.web.css.Declaration;
7   import cz.vutbr.web.css.Term;
8   import cz.vutbr.web.css.TermFunction;
9   import cz.vutbr.web.css.TermInteger;
10  import cz.vutbr.web.css.TermList;
11  import cz.vutbr.web.css.TermNumber;
12  import cz.vutbr.web.css.TermPair;
13  import cz.vutbr.web.css.TermPercent;
14  import cz.vutbr.web.css.TermString;
15  import cz.vutbr.web.css.TermURI;
16  
17  import org.daisy.common.file.URLs;
18  
19  public final class CssSerializer {
20  
21  	private CssSerializer() {}
22  
23  	public static String toString(Term<?> term) {
24  		if (term instanceof TermInteger) {
25  			TermInteger integer = (TermInteger)term;
26  			return "" + integer.getIntValue(); }
27  		else if (term instanceof TermNumber) {
28  			TermNumber number = (TermNumber)term;
29  			Double value = number.getValue().doubleValue();
30  			if (value == Math.floor(value))
31  				return "" + value.intValue();
32  			else
33  				return "" + value; }
34  		else if (term instanceof TermPercent) {
35  			TermPercent percent = (TermPercent)term;
36  			Double value = percent.getValue().doubleValue();
37  			if (value == Math.floor(value))
38  				return "" + value.intValue() + "%";
39  			else
40  				return "" + value + "%"; }
41  		else if (term instanceof TermList
42  		         || term instanceof Declaration) {
43  			String s = serializeTermList((List<Term<?>>)term);
44  			if (term instanceof TermFunction) {
45  				TermFunction function = (TermFunction)term;
46  				s = function.getFunctionName() + "(" + s + ")"; }
47  			return s; }
48  		else if (term instanceof TermPair) {
49  			TermPair<?,?> pair = (TermPair<?,?>)term;
50  			Object val = pair.getValue();
51  			return "" + pair.getKey() + " " + (val instanceof Term ? toString((Term<?>)val) : val.toString()); }
52  		else if (term instanceof TermURI) {
53  			TermURI termURI = (TermURI)term;
54  			URI uri = URLs.asURI(termURI.getValue());
55  			if (termURI.getBase() != null)
56  				uri = URLs.resolve(URLs.asURI(termURI.getBase()), uri);
57  			return "url(\"" + uri + "\")"; }
58  		else if (term instanceof TermString) {
59  			TermString string = (TermString)term;
60  			return "'" + string.getValue().replaceAll("\n", "\\\\A ").replaceAll("'", "\\\\27 ") + "'"; }
61  		else
62  			return term.toString().replaceAll("^[,/ ]+", "");
63  	}
64  
65  	public static String serializeTermList(List<Term<?>> termList) {
66  		String s = "";
67  		for (Term<?> t : termList) {
68  			if (!s.isEmpty()) {
69  				Term.Operator o = t.getOperator();
70  				if (o != null)
71  					switch (o) {
72  					case COMMA:
73  						s += ","; }
74  				s += " "; }
75  			s += toString(t); }
76  		return s;
77  	}
78  }
