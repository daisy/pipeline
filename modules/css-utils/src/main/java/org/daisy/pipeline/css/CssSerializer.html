24  public class CssSerializer {
25  
26  	private static CssSerializer INSTANCE = new CssSerializer();
27  
28  	public static CssSerializer getInstance() {
29  		return INSTANCE;
30  	}
31  
32  	public String toString(Term<?> term) {
33  		if (term instanceof TermInteger) {
34  			TermInteger integer = (TermInteger)term;
35  			return "" + integer.getIntValue(); }
36  		else if (term instanceof TermNumber) {
37  			TermNumber number = (TermNumber)term;
38  			Double value = number.getValue().doubleValue();
39  			if (value == Math.floor(value))
40  				return "" + value.intValue();
41  			else
42  				return "" + value; }
43  		else if (term instanceof TermLength) {
44  			TermLength length = (TermLength)term;
45  			Double value = length.getValue().doubleValue();
46  			Unit unit = length.getUnit();
47  			StringBuilder s = new StringBuilder();
48  			if (value == Math.floor(value))
49  				s.append("" + value.intValue());
50  			else
51  				s.append("" + value);
52  			if (unit != null)
53  				s.append(unit.value());
54  			return s.toString(); }
55  		else if (term instanceof TermPercent) {
56  			TermPercent percent = (TermPercent)term;
57  			Double value = percent.getValue().doubleValue();
58  			if (value == Math.floor(value))
59  				return "" + value.intValue() + "%";
60  			else
61  				return "" + value + "%"; }
62  		else if (term instanceof TermList
63  		         || term instanceof Declaration) {
64  			String s = serializeTermList((List<Term<?>>)term);
65  			if (term instanceof TermFunction) {
66  				TermFunction function = (TermFunction)term;
67  				s = function.getFunctionName() + "(" + s + ")"; }
68  			return s; }
69  		else if (term instanceof TermPair) {
70  			TermPair<?,?> pair = (TermPair<?,?>)term;
71  			Object val = pair.getValue();
72  			return "" + pair.getKey() + " " + (val instanceof Term ? toString((Term<?>)val) : val.toString()); }
73  		else if (term instanceof TermURI) {
74  			TermURI termURI = (TermURI)term;
75  			URI uri = URLs.asURI(termURI.getValue());
76  			if (termURI.getBase() != null)
77  				uri = URLs.resolve(URLs.asURI(termURI.getBase()), uri);
78  			return "url(\"" + uri + "\")"; }
79  		else if (term instanceof TermString) {
80  			TermString string = (TermString)term;
81  			return "'" + string.getValue().replaceAll("\n", "\\\\A ").replaceAll("'", "\\\\27 ") + "'"; }
82  		else if (term instanceof TermIdent) {
83  			TermIdent ident = (TermIdent)term;
84  			return ident.getValue(); }
85  		else
86  			return term.toString().replaceAll("^[,/ ]+", "");
87  	}
88  
89  	public String serializeTermList(Collection<? extends Term<?>> termList) {
90  		String s = "";
91  		for (Term<?> t : termList) {
92  			if (!s.isEmpty()) {
93  				Term.Operator o = t.getOperator();
94  				if (o != null)
95  					switch (o) {
96  					case COMMA:
97  						s += ","; }
98  				s += " "; }
99  			s += toString(t); }
100 		return s;
101 	}
102 }
