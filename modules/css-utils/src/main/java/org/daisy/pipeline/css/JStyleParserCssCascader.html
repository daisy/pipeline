76  	public JStyleParserCssCascader(URIResolver uriResolver,
77  	                               SassCompiler sassCompiler,
78  	                               String defaultStyleSheet,
79  	                               Medium medium,
80  	                               QName attributeName,
81  	                               CSSParserFactory parserFactory,
82  	                               RuleFactory ruleFactory,
83  	                               SupportedCSS supportedCSS,
84  	                               DeclarationTransformer declarationTransformer) {
85  		this.defaultStyleSheet = defaultStyleSheet;
86  		this.medium = medium.asMediaSpec();
87  		this.attributeName = attributeName;
88  		this.parserFactory = parserFactory;
89  		this.ruleFactory = ruleFactory;
90  		this.supportedCSS = supportedCSS;
91  		this.declarationTransformer = declarationTransformer;
92  		NetworkProcessor defaultNetwork = new DefaultNetworkProcessor();
93  		/*
94  		 * CSSSourceReader that handles media type "text/x-scss". Throws a IOException if something
95  		 * goes wrong when resolving the source or if the SASS compilation fails.
96  		 */
97  		this.cssReader = new CSSSourceReader() {
98  				public boolean supportsMediaType(String mediaType, URL url) {
99  					if ("text/css".equals(mediaType))
100 						return true;
101 					else if (mediaType == null && (url == null || url.toString().endsWith(".css")))
102 						return true;
103 					else if (sassCompiler == null)
104 						return false;
105 					else if ("text/x-scss".equals(mediaType))
106 						return true;
107 					else if (mediaType == null && url != null && url.toString().endsWith(".scss"))
108 						return true;
109 					else
110 						return false;
111 				}
112 				public CSSInputStream read(CSSSource source) throws IOException {
113 					URL url = null;
114 					InputStream is; {
115 						switch (source.type) {
116 						case INLINE:
117 						case EMBEDDED:
118 							is = new ByteArrayInputStream(((String)source.source).getBytes());
119 							break;
120 						case URL:
121 							url = (URL)source.source;
122 							logger.debug("Fetching style sheet: " + url);
123 							Source resolved; {
124 								try {
125 									resolved = uriResolver.resolve(URLs.asURI(url).toASCIIString(), ""); }
126 								catch (javax.xml.transform.TransformerException e) {
127 									throw new IOException(e); }}
128 							if (resolved != null) {
129 								url = new URL(resolved.getSystemId());
130 								logger.debug("Resolved to :" + url); }
131 							if (resolved != null && resolved instanceof StreamSource)
132 								is = ((StreamSource)resolved).getInputStream();
133 							else
134 								is = defaultNetwork.fetch(url);
135 							// skip BOM
136 							is = new BOMInputStream(is);
137 							break;
138 						default:
139 							throw new RuntimeException("coding error");
140 						}
141 					}
142 					if (!supportsMediaType(source.mediaType, url))
143 						throw new IllegalArgumentException();
144 					if ("text/x-scss".equals(source.mediaType)
145 					    || (source.mediaType == null && url != null && url.toString().endsWith(".scss"))) {
146 						// sassCompiler must be non-null
147 						try {
148 							is = sassCompiler.compile(is, url != null ? url : source.base, source.encoding);
149 						} catch (RuntimeException e) {
150 							throw new IOException("SASS compilation failed", e);
151 						}
152 					}
153 					// FIXME: there should be a way to pass the resolved URL to the CSSInputStream
154 					// so that relative imports can be handled correctly
155 					return new CSSInputStream(is, source.encoding);
156 				}
157 			};
158 	}

162 	public Runnable transform(XMLInputValue<?> source, XMLOutputValue<?> result, InputValue<?> params) {
163 		if (source == null || result == null)
164 			throw new IllegalArgumentException();
165 		return () -> transform(source.ensureSingleItem().asNodeIterator().next(), result.asXMLStreamWriter());
166 	}

168 	private void transform(Node node, BaseURIAwareXMLStreamWriter output) throws TransformerException {
169 		if (!(node instanceof Document))
170 			throw new TransformerException(new IllegalArgumentException());
171 		Document document = (Document)node;
172 		BaseURIAwareXMLStreamWriter writer = output;
173 		try {
174 			URI baseURI = new URI(document.getBaseURI());
175 			URL baseURL; {
176 				if (baseURI == null || baseURI.toString().equals(""))
177 					// handling the case where base URI is empty, although this should in theory never happen
178 					// FIXME: find out why it happens
179 					baseURL = null;
180 				else
181 					baseURL = URLs.asURL(baseURI);
182 			}
183 			StyleMap styleMap;
184 			synchronized(JStyleParserCssCascader.class) {
185 				// CSSParserFactory injected in CSSAssignTraversal.<init> in CSSFactory.getUsedStyles
186 				CSSFactory.registerCSSParserFactory(parserFactory);
187 				// RuleFactory injected in
188 				// - SimplePreparator.<init> in CSSParserFactory.append in CSSFactory.getUsedStyles
189 				// - CSSTreeParser.<init> in CSSParserFactory.append in CSSFactory.getUsedStyles
190 				CSSFactory.registerRuleFactory(ruleFactory);
191 				// DeclarationTransformer injected in SingleMapNodeData.<init> in CSSFactory.createNodeData in Analyzer.evaluateDOM
192 				CSSFactory.registerDeclarationTransformer(declarationTransformer);
193 				// SupportedCSS injected in
194 				// - SingleMapNodeData.<init> in CSSFactory.createNodeData in Analyzer.evaluateDOM
195 				// - Repeater.assignDefaults in DeclarationTransformer.parseDeclaration in SingleMapNodeData.push in Analyzer.evaluateDOM
196 				// - Variator.assignDefaults in DeclarationTransformer.parseDeclaration in SingleMapNodeData.push in Analyzer.evaluateDOM
197 				CSSFactory.registerSupportedCSS(supportedCSS);
198 				styleSheet = (StyleSheet)ruleFactory.createStyleSheet().unlock();
199 				if (defaultStyleSheet != null) {
200 					StringTokenizer t = new StringTokenizer(defaultStyleSheet);
201 					while (t.hasMoreTokens()) {
202 						URL u = URLs.asURL(URLs.resolve(baseURI, URLs.asURI(t.nextToken())));
203 						if (!cssReader.supportsMediaType(null, u))
204 							logger.warn("Style sheet type not supported: " + u);
205 						else
206 							styleSheet = parserFactory.append(new CSSSource(u, (Charset)null, (String)null), cssReader, styleSheet);
207 					}
208 				}
209 				styleSheet = CSSFactory.getUsedStyles(document, null, baseURL, medium, cssReader, styleSheet);
210 				styleMap = new Analyzer(styleSheet).evaluateDOM(document, medium, false);
211 			}
212 			writer.setBaseURI(baseURI);
213 			writer.writeStartDocument();
214 			traverse(document.getDocumentElement(), styleMap, writer);
215 			writer.writeEndDocument();
216 		} catch (Exception e) {
217 			throw new TransformerException(e);
218 		} finally {
219 			styleSheet = null;
220 		}
221 	}

231 	private void traverse(Node node, StyleMap styleMap, BaseURIAwareXMLStreamWriter writer) throws XMLStreamException {
232 		if (node.getNodeType() == Node.ELEMENT_NODE) {
233 			Element elem = (Element)node;
234 			writeStartElement(writer, elem);
235 			NamedNodeMap attributes = node.getAttributes();
236 			for (int i = 0; i < attributes.getLength(); i++) {
237 				Node attr = attributes.item(i);
238 				if (!(attr.getPrefix() == null && "style".equals(attr.getLocalName())))
239 					writeAttribute(writer, attr); }
240 			String style = serializeStyle(
241 				styleMap.get(elem),
242 				new AbstractMap<PseudoElement,NodeData>() {
243 					Set<Entry<PseudoElement,NodeData>> entrySet = new AbstractSet<Entry<PseudoElement,NodeData>>() {
244 							Set<PseudoElement> keySet = styleMap.pseudoSet(elem);
245 							public Iterator<Entry<PseudoElement,NodeData>> iterator() {
246 								return Iterators.transform(
247 									keySet.iterator(),
248 									pseudo -> new SimpleImmutableEntry<PseudoElement,NodeData>(
249 										pseudo, styleMap.get(elem, pseudo)));
250 							}
251 							public int size() {
252 								return keySet.size();
253 							}
254 						};
255 					public Set<Entry<PseudoElement,NodeData>> entrySet() {
256 						return entrySet;
257 					}
258 				},
259 				elem);
260 			if (style != null)
261 				writeAttribute(writer, attributeName, style);
262 			for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling())
263 				traverse(child, styleMap, writer);
264 			writer.writeEndElement(); }
265 		else if (node.getNodeType() == Node.COMMENT_NODE)
266 			writeComment(writer, node);
267 		else if (node.getNodeType() == Node.TEXT_NODE)
268 			writeCharacters(writer, node);
269 		else if (node.getNodeType() == Node.PROCESSING_INSTRUCTION_NODE)
270 			writeProcessingInstruction(writer, node);
271 		else
272 			throw new UnsupportedOperationException("Unexpected node type");
273 	}
