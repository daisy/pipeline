182 	/**
183 	 * Evaluate a media expression about a non-standard feature, in <a
184 	 * href="https://drafts.csswg.org/mediaqueries-5/#mq-boolean-context">boolean context</a>.
185 	 */
186 	protected boolean matchesFeature(String feature) {
187 		if ("-daisy-document-locale".equals(feature))
188 			return false; // intended for medium selection, not valid in regular media queries
189 		Object v = getNonStandardFeature(feature);
190 		if (v == null)
191 			return false;
192 		else if (v instanceof String)
193 			return !"none".equals(v);
194 		else if (v instanceof Integer)
195 			return (Integer)v != 0;
196 		else if (v instanceof Boolean)
197 			return (Boolean)v == true;
198 		else if (v instanceof Dimension)
199 			return ((Dimension)v).getValue().equals(BigDecimal.ZERO);
200 		else
201 			return true;
202 	}
203 
204 	/**
205 	 * Evaluate a media expression about a non-standard feature.
206 	 */
207 	protected boolean matchesFeature(String feature, Term<?> value) {
208 		if ("-daisy-document-locale".equals(feature))
209 			return false; // intended for medium selection, not valid in regular media queries
210 		Object v = getNonStandardFeature(feature);
211 		if (v != null)
212 			try {
213 				if (value instanceof TermIdent)
214 					if (v instanceof Locale)
215 						return v.equals(parser.parseLocale(value));
216 					else
217 						return v.equals(parser.parseIdent(value));
218 				else if (value instanceof TermInteger)
219 					if (v instanceof Boolean)
220 						return v.equals(parser.parseBoolean(value));
221 					else
222 						return v.equals(parser.parseInteger(value));
223 				else if (value instanceof TermLength)
224 					return v.equals(parser.parseLength(value, feature));
225 			} catch (IllegalArgumentException e) {
226 			}
227 		return false;
228 	}

230 	@Override
231 	public String toString() {
232 		StringBuilder s  = new StringBuilder();
233 		if (type != null)
234 			s.append(type.toString());
235 		if (width != null) {
236 			if (s.length() > 0) s.append(" AND ");
237 			s.append("(width: ").append(width).append(")");
238 		}
239 		if (height != null) {
240 			if (s.length() > 0) s.append(" AND ");
241 			s.append("(height: ").append(height).append(")");
242 		}
243 		if (deviceWidth != null) {
244 			if (s.length() > 0) s.append(" AND ");
245 			s.append("(width: ").append(deviceWidth).append(")");
246 		}
247 		if (deviceHeight != null) {
248 			if (s.length() > 0) s.append(" AND ");
249 			s.append("(height: ").append(deviceHeight).append(")");
250 		}
251 		for (String f : nonStandardFeatures.keySet()) {
252 			if ("-daisy-document-locale".equals(f))
253 				continue;
254 			if (s.length() > 0) s.append(" AND ");
255 			s.append("(").append(f);
256 			Object v = nonStandardFeatures.get(f);
257 			if (v instanceof Boolean)
258 				s.append((Boolean)v ? "" : ": 0");
259 			else if (v instanceof Locale)
260 				s.append(": ").append(((Locale)v).toLanguageTag());
261 			else if (v.toString().matches("^[+-]?([0-9]*\\.)?[0-9]+([eE][+-]?[0-9]+)?([a-zA-Z]+|%)?"))
262 				// integer, number, dimension or percentage
263 				s.append(": ").append(v.toString());
264 			else
265 				// identifier: escape if needed
266 				s.append(": ").append(CssEscape.escapeCssIdentifierMinimal(v.toString()));
267 			s.append(")");
268 		}
269 		return s.toString();
270 	}

325 	/**
326 	 * The medium as a jStyleParser {@link MediaSpec} object, used by {@link
327 	 * JStyleParserCssCascader} for passing to {@link CSSFactory#getUsedStyles} and {@link
328 	 * Analyzer#evaluateDOM}.
329 	 */
330 	public MediaSpec asMediaSpec() {
331 		if (mediaSpec == null) {
332 			Comparator<Dimension> lengthComparator = Dimension.comparator(Medium.this);
333 			mediaSpec = new MediaSpec(type.toString()) {
334 				@Override
335 				protected boolean matchesIgnoreNegation(MediaExpression e) {
336 					String f = e.getFeature();
337 					boolean isMin = false;
338 					boolean isMax = false;
339 					if (f.startsWith("min-")) { isMin = true; f = f.substring(4); }
340 					else if (f.startsWith("max-")) { isMax = true; f = f.substring(4); }
341 					if (knownFeatures.contains(f)) {
342 						MediaSpec.Feature ff = getFeatureByName(f);
343 						switch (ff) {
344 						case WIDTH:
345 						case HEIGHT:
346 						case DEVICE_WIDTH:
347 						case DEVICE_HEIGHT:
348 							if (e.size() != 1)
349 								return false;
350 							Dimension v = null; {
351 								switch (ff) {
352 								case WIDTH: v = Medium.this.width; break;
353 								case HEIGHT: v = Medium.this.height; break;
354 								case DEVICE_WIDTH: v = Medium.this.deviceWidth; break;
355 								case DEVICE_HEIGHT: v = Medium.this.deviceHeight; break;
356 								}
357 							}
358 							if (v == null)
359 								return false;
360 							Dimension length; {
361 								try {
362 									length = parser.parseLength(e.get(0), f);
363 								} catch (IllegalArgumentException ex) {
364 									return false;
365 								}
366 							}
367 							int comparison = lengthComparator.compare(v, length);
368 							return isMin
369 								? comparison >= 0
370 								: isMax
371 									? comparison <= 0
372 									: comparison == 0;
373 						default:
374 							return super.matchesIgnoreNegation(e);
375 						}
376 					} else {
377 						if (isMin || isMax)
378 							return false;
379 						f = parser.normalizeFeature(f);
380 						if (e.size() > 1)
381 							return false;
382 						else if (e.size() == 0)
383 							return Medium.this.matchesFeature(f);
384 						else
385 							return Medium.this.matchesFeature(f, e.get(0));
386 					}
387 				}
388 			};
389 		}
390 		if (Medium.this.type == Type.EMBOSSED)
391 			mediaSpec.setGrid(1);
392 		// These are commented out because the unit conversion may fail and it is not so
393 		// important that these are set anyway: the corresponding getter methods in MediaSpec
394 		// are not used anywhere within the jStyleParser code anyway.
395 		/*if (width != null)
396 			mediaSpec.setWidth(width.toUnit(Dimension.Unit.PX, this).getValue().floatValue());
397 		if (height != null)
398 			mediaSpec.setHeight(height.toUnit(Dimension.Unit.PX, this).getValue().floatValue());
399 		if (deviceWidth != null)
400 			mediaSpec.setWidth(deviceWidth.toUnit(Dimension.Unit.PX, this).getValue().floatValue());
401 		if (deviceHeight != null)
402 			mediaSpec.setHeight(deviceHeight.toUnit(Dimension.Unit.PX, this).getValue().floatValue());*/
403 		return mediaSpec;
404 	}

407 	 * Helper class to build {@link Medium} from {@link MediaQuery}, e.g. for use in {@link
408 	 * MediumProvider} implementations.
409 	 */
410 	public static class MediumBuilder {
411 
412 		protected Type type = null;
413 		protected Map<String,Object> features = new HashMap<>();
414 
415 		public final MediumBuilder parse(MediaQuery query) throws IllegalArgumentException {
416 			if (query.getType() != null && !"all".equals(query.getType()))
417 				try {
418 					Type t = Type.valueOf(query.getType().toUpperCase());
419 					if (type == null)
420 						type = t;
421 					else if (!type.equals(t))
422 						throw new IllegalArgumentException(
423 							String.format("Incompatible values for medium type: %s AND %s", type, t));
424 				} catch (IllegalArgumentException e) {
425 					throw new IllegalArgumentException("Unexpected medium type: " + query.getType());
426 				}
427 			if (query.isNegative())
428 				throw new IllegalArgumentException("Unexpected medium: contains NOT: " + query);
429 			Map<String,Object> features = new HashMap<>();
430 			for (MediaExpression e : query) {
431 				if (e.isNegative())
432 					throw new IllegalArgumentException("Unexpected medium: contains NOT: " + query);
433 				String f = e.getFeature();
434 				if (f.startsWith("min-") || f.startsWith("max-"))
435 					throw new IllegalArgumentException("Unexpected medium: contains " + f.substring(0, 4));
436 				if (e.size() == 0)
437 					features.put(f, true);
438 				else if (e.size() == 1)
439 					features.put(f, e.get(0));
440 				else
441 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
442 			}
443 			return parse(features);
444 		}
445 
446 		protected Object parse(String feature, Object value) throws IllegalArgumentException {
447 			if ("width".equals(feature) || "height".equals(feature)
448 			    || "device-width".equals(feature) || "device-height".equals(feature))
449 				value = parseLength(value, feature);
450 			else if ("-daisy-document-locale".equals(feature) || "-daisy-locale".equals(feature))
451 				value = parseLocale(value);
452 			else if ("-daisy-duplex".equals(feature)
453 			         || "-daisy-saddle-stitch".equals(feature)
454 			         || "-daisy-z-folding".equals(feature)
455 			         || "-daisy-sheets-multiple-of-two".equals(feature)
456 			         || "-daisy-blank-last-page".equals(feature))
457 				value = parseBoolean(value);
458 			else
459 				// don't parse/validate unknown features, just check the type
460 				// note that any string can be represented by an ident (possibly using
461 				// escape characters)
462 				if (!(value instanceof String
463 				      || value instanceof Integer
464 				      || value instanceof Boolean
465 				      || value instanceof Dimension))
466 					throw new IllegalArgumentException();
467 			Object prev = features.get(feature);
468 			if (prev == null)
469 				features.put(feature, value);
470 			else if (!prev.equals(value))
471 				throw new IllegalArgumentException(
472 					String.format(
473 						"Incompatible values for medium feature: (%s: %s) AND (%s: %s)", feature, prev, feature, value));
474 			return value;
475 		}
476 
477 		protected String parseIdent(Object value) throws IllegalArgumentException {
478 			if (value instanceof String)
479 				return (String)value;
480 			else if (value instanceof TermIdent)
481 				return CssEscape.unescapeCss(value.toString());
482 			else
483 				throw new IllegalArgumentException("Not an ident: " + value);
484 		}
485 
486 		protected Integer parseInteger(Object value) throws IllegalArgumentException {
487 			if (value instanceof Integer)
488 				return (Integer)value;
489 			else if (value instanceof TermInteger)
490 				return ((TermInteger)value).getIntValue();
491 			else if (value instanceof String)
492 				try {
493 					return Integer.parseInt((String)value);
494 				} catch (NumberFormatException e) {
495 					throw new IllegalArgumentException("Not an integer: " + value, e);
496 				}
497 			else
498 				throw new IllegalArgumentException("Not an integer: " + value);
499 		}
500 
501 		protected Boolean parseBoolean(Object value) throws IllegalArgumentException {
502 			if (value instanceof Boolean)
503 				return (Boolean)value;
504 			else if (value instanceof Integer)
505 				switch ((Integer)value) {
506 				case 0:
507 					return false;
508 				case 1:
509 					return true;
510 				default:
511 					throw new IllegalArgumentException("Not a boolean: " + value);
512 				}
513 			else if (value instanceof TermInteger)
514 				return parseBoolean(((TermInteger)value).getIntValue());
515 			else if (value instanceof String)
516 				if ("1".equals(value))
517 					return true;
518 				else if ("0".equals(value))
519 					return false;
520 				else
521 					return Boolean.parseBoolean((String)value);
522 			else
523 				throw new IllegalArgumentException("Not a boolean: " + value);
524 		}
525 
526 		protected Locale parseLocale(Object value) throws IllegalArgumentException {
527 			if (value instanceof Locale)
528 				return (Locale)value;
529 			else if (value instanceof TermIdent)
530 				return parseLocale(value.toString());
531 			else if (value instanceof String)
532 				try {
533 					return new Locale.Builder().setLanguageTag((String)value).build();
534 				} catch (IllformedLocaleException e) {
535 					throw new IllegalArgumentException("Not a locale: " + value, e);
536 				}
537 			else
538 				throw new IllegalArgumentException("Not a locale: " + value);
539 		}
540 
541 		protected Dimension parseLength(Object value, String feature) throws IllegalArgumentException {
542 			if (value instanceof Dimension)
543 				return (Dimension)value;
544 			else if (value instanceof TermLength)
545 				return new Dimension(((TermLength)value).getValue(),
546 				                     ((TermLength)value).getUnit() == TermLength.Unit.none
547 				                         ? null
548 				                         : Dimension.Unit.parse(((TermLength)value).getUnit().name()));
549 			else if (value instanceof String)
550 				try {
551 					return Dimension.parse((String)value);
552 				} catch (IllegalArgumentException e) {
553 					try {
554 						return parseLength(parseInteger(value), feature);
555 					} catch (IllegalArgumentException ee) {
556 						throw e;
557 					}
558 				}
559 			else if (value instanceof Integer)
560 				// Grid based media may support integer lengths.
561 				// Interpreting these lengths (e.g. as multiple of em or ch) is left up to
562 				// subclasses.  Here we assume we are parsing a media query (see MediaQueryParser),
563 				// so the exact interpretation does not matter.
564 				return new Dimension((Integer)value, null);
565 			else if (value instanceof TermInteger)
566 				return parseLength(((TermInteger)value).getIntValue(), feature);
567 			else
568 				throw new IllegalArgumentException("Not a dimension: " + value);
569 		}

571 		private static String normalizeFeature(String feature) {
572 			if (knownDaisyFeatures.contains(feature))
573 				feature = "-daisy-" + feature;
574 			if ("-daisy-file-format".equals(feature))
575 				feature = "-daisy-format";
576 			return feature;
577 		}

590 	/* for use in MediaQueryParser */

593 		MediumBuilder parse(Map<String,Object> features) throws IllegalArgumentException {
594 			for (String feature : features.keySet()) {
595 				if (knownFeatures.contains(feature) && !(
596 					    "width".equals(feature)
597 					    || "height".equals(feature)
598 					    || "device-width".equals(feature)
599 					    || "device-height".equals(feature)))
600 					throw new IllegalArgumentException("Unsupported medium feature: " + feature);
601 				else if (feature.startsWith("-daisy-")
602 				         && !knownDaisyFeatures.contains(feature.substring("-daisy-".length())))
603 					throw new IllegalArgumentException("Unknown medium feature: " + feature);
604 				else {
605 					Object v = features.get(feature);
606 					feature = normalizeFeature(feature);
607 					if (v instanceof TermInteger)
608 						v = parseInteger(v);
609 					else if (v instanceof TermIdent)
610 						v = parseIdent(v);
611 					else if (v instanceof TermLength)
612 						v = parseLength(v, feature);
613 					try {
614 						parse(feature, v);
615 					} catch (IllegalArgumentException e) {
616 						throw new IllegalArgumentException(
617 							String.format("Unexpected value for medium feature: (%s: %s)", feature, v), e);
618 					}
619 				}
620 			}
621 			return this;
622 		}
623 	}

644 	private Medium and(Medium medium) throws IllegalArgumentException {
645 		Type type = this.type; {
646 			if (medium.type != null)
647 				if (type == null)
648 					type = medium.type;
649 				else if (!medium.type.equals(type))
650 					throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
651 		}
652 		Dimension width = this.width; {
653 			if (medium.width != null)
654 				if (width == null)
655 					width = medium.width;
656 				else if (!medium.width.equals(width))
657 					throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
658 		}
659 		Dimension height = this.height; {
660 			if (medium.height != null)
661 				if (height == null)
662 					height = medium.height;
663 				else if (!medium.height.equals(height))
664 					throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
665 		}
666 		Dimension deviceWidth = this.deviceWidth; {
667 			if (medium.deviceWidth != null)
668 				if (deviceWidth == null)
669 					deviceWidth = medium.deviceWidth;
670 				else if (!medium.deviceWidth.equals(deviceWidth))
671 					throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
672 		}
673 		Dimension deviceHeight = this.deviceHeight; {
674 			if (medium.deviceHeight != null)
675 				if (deviceHeight == null)
676 					deviceHeight = medium.deviceHeight;
677 				else if (!medium.deviceHeight.equals(deviceHeight))
678 					throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
679 		}
680 		Map<String,Object> nonStandardFeatures = this.nonStandardFeatures; {
681 			if (!medium.nonStandardFeatures.isEmpty())
682 				if (nonStandardFeatures.isEmpty())
683 					nonStandardFeatures = medium.nonStandardFeatures;
684 				else {
685 					nonStandardFeatures = new HashMap<String,Object>(nonStandardFeatures);
686 					for (String f : medium.nonStandardFeatures.keySet()) {
687 						Object v = medium.nonStandardFeatures.get(f);
688 						Object vv = nonStandardFeatures.get(f);
689 						if (vv != null && !vv.equals(v))
690 							throw new IllegalArgumentException("" + this + " can not be combined with " + medium);
691 						nonStandardFeatures.put(f, v);
692 					}
693 				}
694 		}
695 		return new Medium(type, width, height, deviceWidth, deviceHeight, nonStandardFeatures, parser);
696 	}

780 	private class MediaExpressionImpl extends AbstractList<Term<?>> implements MediaExpression {
781 
782 		private final String feature;
783 		private Term<?> value = null; // value computed lazily
784 
785 		private MediaExpressionImpl(String feature) {
786 			this.feature = feature;
787 		}
788 
789 		@Override
790 		public String getFeature() {
791 			return feature;
792 		}
793 
794 		@Override
795 		public int size() {
796 			return 1;
797 		}
798 
799 		@Override
800 		public Term<?> get(int index) {
801 			if (index != 0)
802 				throw new IndexOutOfBoundsException("index must be 0");
803 			if (value == null)
804 				if ("width".equals(feature))
805 					value = new UnmodifiableTermLength(width);
806 				else if ("height".equals(feature))
807 					value = new UnmodifiableTermLength(height);
808 				else if ("device-width".equals(feature))
809 					value = new UnmodifiableTermLength(deviceWidth);
810 				else if ("device-height".equals(feature))
811 					value = new UnmodifiableTermLength(deviceHeight);
812 				else {
813 					Object v = nonStandardFeatures.get(feature);
814 					if (v instanceof Integer)
815 						value = new UnmodifiableTermInteger((Integer)v);
816 					else if (v instanceof Boolean)
817 						value = new UnmodifiableTermInteger(((Boolean)v) ? 1 : 0);
818 					else if (v instanceof Dimension)
819 						value = new UnmodifiableTermLength((Dimension)v);
820 					else if (v instanceof Locale)
821 						value = new UnmodifiableTermIdent(((Locale)v).toLanguageTag());
822 					else
823 						value = new UnmodifiableTermIdent(CssEscape.escapeCssIdentifierMinimal(v.toString()));
824 				}
825 			return value;
826 		}
827 
828 		@Override
829 		public boolean isNegative() {
830 			return false;
831 		}
832 
833 		@Override
834 		public List<Term<?>> asList() {
835 			return this;
836 		}
837 
838 		@Override
839 		public void setFeature(String feature) {
840 			throw new UnsupportedOperationException("immutable object");
841 		}
842 
843 		@Override
844 		public Rule<Term<?>> replaceAll(List<Term<?>> replacement) {
845 			throw new UnsupportedOperationException("immutable object");
846 		}
847 
848 		@Override
849 		public void setNegative(boolean negative) {
850 			throw new UnsupportedOperationException("immutable object");
851 		}
852 
853 		@Override
854 		public Rule<Term<?>> unlock() {
855 			throw new UnsupportedOperationException("immutable object");
856 		}
857 	}
