21  public class Medium {
22  
23  	public enum Type {
24  		EMBOSSED,
25  		SPEECH,
26  		SCREEN,
27  		PRINT;
28  		public String toString() {
29  			return super.toString().toLowerCase();
30  		}
31  	}
32  
33  	private final Type type;
34  	private final Integer width;
35  	private final Integer height;
36  	private final Map<String,String> customFeatures;
37  
38  	// see https://drafts.csswg.org/mediaqueries/#media-descriptor-table
39  	private final static Set<String> knownFeatures = new HashSet<>(
40  		Arrays.asList(
41  			"any-hover", "any-pointer", "aspect-ratio", "color", "color-gamut", "color-index",
42  			"device-aspect-ratio", "device-height", "device-width", "grid", "height", "hover",
43  			"monochrome", "orientation", "overflow-block", "overflow-inline", "pointer", "resolution",
44  			"scan", "update", "width"));
45  
46  	/**
47  	 * @param customFeatures can be used to pass additional properties of the target medium. Must
48  	 *                       not include any <a
49  	 *                       href="https://drafts.csswg.org/mediaqueries/#media-descriptor-table">known
50  	 *                       features</a>
51  	 */
52  	private Medium(Type type, Integer width, Integer height, Map<String,String> customFeatures) {
53  		switch (type) {
54  		case EMBOSSED:
55  			this.type = type;
56  			this.width = width;
57  			this.height = height;
58  			break;
59  		case SPEECH:
60  		case SCREEN:
61  		case PRINT:
62  			if (width != null) {
63  				throw new IllegalArgumentException("Unexpected 'width' argument for medium 'print'");
64  			}
65  			if (height != null) {
66  				throw new IllegalArgumentException("Unexpected 'height' argument for medium 'print'");
67  			}
68  			this.type = type;
69  			this.width = this.height = null;
70  			break;
71  		default:
72  			throw new IllegalArgumentException("Unexpected medium: " + type);
73  		}
74  		this.customFeatures = customFeatures != null
75  			? Collections.unmodifiableMap(customFeatures)
76  			: Collections.emptyMap();
77  	}

99  		return height;
100 	}

109 	public boolean matches(String mediaQuery) {
110 		return asMediaSpec().matches(CSSParserFactory.getInstance().parseMediaQuery(mediaQuery));
111 	}
112 
113 	public String toString() {
114 		StringBuilder s  = new StringBuilder();
115 		s.append(type.toString());
116 		if (width != null)
117 			s.append(" AND (width: ").append(width).append(")");
118 		if (height != null)
119 			s.append(" AND (height: ").append(height).append(")");
120 		for (String f : customFeatures.keySet())
121 			s.append(" AND (").append(f).append(": ").append(customFeatures.get(f)).append(")");
122 		return s.toString();
123 	}

125 	public boolean equals(Object o) {
126 		if (!(o instanceof Medium))
127 			return false;
128 		Medium that = (Medium)o;
129 		return this.type == that.type
130 			&& this.width == that.width
131 			&& this.height == that.height;
132 	}

134 	public int hashCode() {
135 		final int prime = 31;
136 		int result = 1;
137 		result = prime * result + type.hashCode();
138 		result = prime * result + width;
139 		result = prime * result + height;
140 		return result;
141 	}

143 	public static Medium parse(String medium) {
144 		List<MediaQuery> q = CSSParserFactory.getInstance().parseMediaQuery(medium);
145 		if (q.size() != 1)
146 			throw new IllegalArgumentException("Unexpected medium: " + medium);
147 		Type type = null;
148 		try {
149 			type = Type.valueOf(q.get(0).getType().toUpperCase());
150 		} catch (IllegalArgumentException e) {
151 			throw new IllegalArgumentException("Unexpected medium type: " + q.get(0).getType());
152 		}
153 		if (q.get(0).isNegative())
154 			throw new IllegalArgumentException("Unexpected medium: contains NOT: " + medium);
155 		Integer width = null;
156 		Integer height = null;
157 		Map<String,String> customFeatures = null;
158 		for (MediaExpression e : q.get(0)) {
159 			String feature = e.getFeature();
160 			if ("width".equals(feature) || "height".equals(feature)) {
161 				if (e.size() != 1)
162 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
163 				Term<?> v = e.get(0);
164 				if (!(v instanceof TermInteger))
165 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
166 				Integer i = ((TermInteger)v).getIntValue();
167 				if ("width".equals(feature))
168 					width = i;
169 				else
170 					height = i;
171 			} else if (knownFeatures.contains(feature)) {
172 				throw new IllegalArgumentException("Unsupported medium feature: " + feature);
173 			} else {
174 				if (e.size() != 1)
175 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
176 				Term<?> v = e.get(0);
177 				if (!(v instanceof TermIdent))
178 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
179 				if (customFeatures == null)
180 					customFeatures = new HashMap<>();
181 				customFeatures.put(feature, v.toString());
182 			}
183 		}
184 		return new Medium(type, width, height, customFeatures);
185 	}

187 	private MediaSpec mediaSpec;
188 
189 	/**
190 	 * The medium as a jStyleParser {@link MediaSpec} object, used by {@link
191 	 * JStyleParserCssCascader} for passing to {@link CSSFactory#getUsedStyles} and {@link
192 	 * Analyzer#evaluateDOM}.
193 	 */
194 	public MediaSpec asMediaSpec() {
195 		if (mediaSpec == null)
196 			switch (type) {
197 			case EMBOSSED:
198 				mediaSpec = new MediaSpec(type.toString()) {
199 						// overriding getExpressionLengthPx() because MediaSpec expects width and height
200 						// to be in pixels but width and height are specified in braille cells
201 						@Override
202 						protected Float getExpressionLengthPx(MediaExpression e) {
203 							return (float)getExpressionInteger(e);
204 						}
205 						@Override
206 						public boolean matches(MediaExpression e) {
207 							String f = e.getFeature();
208 							if (knownFeatures.contains(f) || knownFeatures.contains(f.replaceAll("^(min|max)-", "")))
209 								return super.matches(e);
210 							else {
211 								if (!customFeatures.isEmpty())
212 									for (String ff : customFeatures.keySet())
213 										if (ff.equals(f))
214 											return customFeatures.get(ff).equals(getExpressionIdentifier(e));
215 								return false;
216 							}
217 						}
218 					};
219 				if (width != null)
220 					mediaSpec.setWidth((float)width);
221 				if (height != null)
222 					mediaSpec.setWidth((float)height);
223 				break;
224 			default:
225 				mediaSpec = new MediaSpec(type.toString()) {
226 						@Override
227 						public boolean matches(MediaExpression e) {
228 							String f = e.getFeature();
229 							if (knownFeatures.contains(f) || knownFeatures.contains(f.replaceAll("^(min|max)-", "")))
230 								return super.matches(e);
231 							else {
232 								if (!customFeatures.isEmpty())
233 									for (String ff : customFeatures.keySet())
234 										if (ff.equals(f))
235 											return customFeatures.get(ff).equals(getExpressionIdentifier(e));
236 								return false;
237 							}
238 						}
239 					};
240 				break;
241 			}
242 		return mediaSpec;
243 	}
