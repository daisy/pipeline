1   package org.daisy.pipeline.css;
2   
3   import java.util.List;
4   
5   import cz.vutbr.web.css.MediaExpression;
6   import cz.vutbr.web.css.MediaSpec;
7   import cz.vutbr.web.css.MediaQuery;
8   import cz.vutbr.web.css.Term;
9   import cz.vutbr.web.css.TermInteger;
10  import cz.vutbr.web.csskit.antlr.CSSParserFactory;
11  
12  public class Medium {
13  
14  	public enum Type {
15  		EMBOSSED,
16  		PRINT;
17  		public String toString() {
18  			return super.toString().toLowerCase();
19  		}
20  	}
21  
22  	private final Type type;
23  	private final Integer width;
24  	private final Integer height;
25  
26  	private Medium(Type type, Integer width, Integer height) {
27  		switch (type) {
28  		case EMBOSSED:
29  			this.type = type;
30  			this.width = width;
31  			this.height = height;
32  			break;
33  		case PRINT:
34  			if (width != null) {
35  				throw new IllegalArgumentException("Unexpected 'width' argument for medium 'print'");
36  			}
37  			if (height != null) {
38  				throw new IllegalArgumentException("Unexpected 'height' argument for medium 'print'");
39  			}
40  			this.type = type;
41  			this.width = this.height = null;
42  			break;
43  		default:
44  			throw new IllegalArgumentException("Unexpected medium: " + type);
45  		}
46  	}
47  
48  	/**
49  	 * The media type: "embossed" or "print".
50  	 */
51  	public Type getType() {
52  		return type;
53  	}
54  
55  	/**
56  	 * The width of the page, in braille cells, or null if undetermined or not applicable. Only
57  	 * applicable on type 'embossed'.
58  	 */
59  	public Integer getWidth() {
60  		return width;
61  	}
62  
63  	/**
64  	 * The height of the page, in braille cells, or null if undetermined or not applicable. Only
65  	 * applicable on type 'embossed'.
66  	 */
67  	public Integer getHeight() {
68  		return height;
69  	}
70  
71  	public boolean matches(String mediaQuery) {
72  		return asMediaSpec().matchesOneOf(CSSParserFactory.getInstance().parseMediaQuery(mediaQuery));
73  	}
74  
75  	public String toString() {
76  		StringBuilder s  = new StringBuilder();
77  		s.append(type.toString());
78  		if (width != null)
79  			s.append(" AND (width:").append(width).append(")");
80  		if (height != null)
81  			s.append(" AND (height:").append(height).append(")");
82  		return s.toString();
83  	}
84  
85  	public boolean equals(Object o) {
86  		if (!(o instanceof Medium))
87  			return false;
88  		Medium that = (Medium)o;
89  		return this.type == that.type
90  			&& this.width == that.width
91  			&& this.height == that.height;
92  	}
93  
94  	public int hashCode() {
95  		final int prime = 31;
96  		int result = 1;
97  		result = prime * result + type.hashCode();
98  		result = prime * result + width;
99  		result = prime * result + height;
100 		return result;
101 	}
102 
103 	public static Medium parse(String medium) {
104 		List<MediaQuery> q = CSSParserFactory.getInstance().parseMediaQuery(medium);
105 		if (q.size() != 1)
106 			throw new IllegalArgumentException("Unexpected medium: " + medium);
107 		Type type = null;
108 		Integer width = null;
109 		Integer height = null;
110 		try {
111 			type = Type.valueOf(q.get(0).getType().toUpperCase());
112 		} catch (IllegalArgumentException e) {
113 			throw new IllegalArgumentException("Unexpected medium type: " + q.get(0).getType());
114 		}
115 		if (q.get(0).isNegative())
116 			throw new IllegalArgumentException("Unexpected medium: contains NOT: " + medium);
117 		for (MediaExpression e : q.get(0)) {
118 			String feature = e.getFeature();
119 			if ("width".equals(feature) || "height".equals(feature)) {
120 				if (e.size() != 1)
121 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
122 				Term<?> v = e.get(0);
123 				if (!(v instanceof TermInteger))
124 					throw new IllegalArgumentException("Unexpected value for medium feature: " + e);
125 				Integer i = ((TermInteger)v).getIntValue();
126 				if ("width".equals(feature))
127 					width = i;
128 				else
129 					height = i;
130 			} else
131 				throw new IllegalArgumentException("Unexpected medium feature: " + feature);
132 		}
133 		return new Medium(type, width, height);
134 	}
135 
136 	private MediaSpec mediaSpec;
137 
138 	/**
139 	 * The medium as a jStyleParser {@link MediaSpec} object.
140 	 */
141 	public MediaSpec asMediaSpec() {
142 		if (mediaSpec == null)
143 			switch (type) {
144 			case EMBOSSED:
145 				MediaSpec spec = new MediaSpec(type.toString()) {
146 						// overriding getExpressionLengthPx() because MediaSpec expects width and height
147 						// to be in pixels but width and height are specified in braille cells
148 						@Override
149 						protected Float getExpressionLengthPx(MediaExpression e) {
150 							return (float)getExpressionInteger(e);
151 						}
152 					};
153 				if (width != null)
154 					spec.setWidth((float)width);
155 				if (height != null)
156 					spec.setWidth((float)height);
157 				mediaSpec = spec;
158 				break;
159 			default:
160 				mediaSpec = new MediaSpec(type.toString());
161 				break;
162 			}
163 		return mediaSpec;
164 	}
165 }
