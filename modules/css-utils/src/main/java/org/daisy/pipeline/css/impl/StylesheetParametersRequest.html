55  	public static StylesheetParametersRequest fromXML(Document xml) throws IllegalArgumentException {
56  		if (!XmlValidator.validate(xml, STYLESHEET_PARAMETERS_REQUEST_SCHEMA_URL))
57  			throw new IllegalArgumentException("Supplied XML is not a valid stylesheet parameters request");
58  		try {
59  			StylesheetParametersRequest req = new StylesheetParametersRequest();
60  			NodeList nodes = xml.getElementsByTagNameNS(XmlUtils.NS_DAISY, "userStylesheets");
61  			if (nodes.getLength() > 0) {
62  				List<URI> userStylesheets = new ArrayList<>();
63  				NodeList fileNodes = ((Element)nodes.item(0)).getElementsByTagNameNS(XmlUtils.NS_DAISY, "file");
64  				for (int i = 0; i < fileNodes.getLength(); i++) {
65  					try {
66  						userStylesheets.add(new URI(((Element)fileNodes.item(i)).getAttribute("href")));
67  					} catch (URISyntaxException e) {
68  						throw new IllegalArgumentException("invalid href: not a URI");
69  					}
70  				}
71  				req.userStylesheets = new UserStylesheets(userStylesheets);
72  			}
73  			nodes = xml.getElementsByTagNameNS(XmlUtils.NS_DAISY, "sourceDocument");
74  			if (nodes.getLength() > 0) {
75  				NodeList fileNode = ((Element)nodes.item(0)).getElementsByTagNameNS(XmlUtils.NS_DAISY, "file");
76  				try {
77  					req.sourceDocument = new URI(((Element)fileNode.item(0)).getAttribute("href"));
78  				} catch (URISyntaxException e) {
79  					throw new IllegalArgumentException("invalid href: not a URI");
80  				}
81  			}
82  			nodes = xml.getElementsByTagNameNS(XmlUtils.NS_DAISY, "media");
83  			if (nodes.getLength() > 0)
84  				req.media = new Media(MediaQueryParser.parseList(((Element)nodes.item(0)).getAttribute("value")));
85  			nodes = xml.getElementsByTagNameNS(XmlUtils.NS_DAISY, "userAgentStylesheet");
86  			if (nodes.getLength() > 0)
87  				req.mediaTypes = new MediaTypes(
88  					Arrays.asList(((Element)nodes.item(0)).getAttribute("mediaType").trim().split("\\s+")));
89  			return req;
90  		} catch (IllegalArgumentException e) {
91  			throw new IllegalArgumentException(
92  				"Supplied XML is not a valid stylesheet parameters request: " + e.getMessage(), e);
93  		}
94  	}

160 	private static class MediaDeserializer implements JsonDeserializer<Media> {
161 		public Media deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context) throws JsonParseException {
162 			try {
163 				JsonPrimitive j = json.getAsJsonPrimitive();
164 				if (!j.isString())
165 					throw new JsonParseException("invalid media: not a string");
166 				return new Media(MediaQueryParser.parseList(j.getAsString()));
167 			} catch (IllegalStateException e) {
168 				throw new JsonParseException("invalid media: not a string", e);
169 			} catch (IllegalArgumentException e) {
170 				throw new JsonParseException("invalid media: " + e.getMessage(), e);
171 			}
172 		}
173 	}

175 	private static class MediaTypesDeserializer implements JsonDeserializer<MediaTypes> {
176 		public MediaTypes deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context) throws JsonParseException {
177 			if (!json.isJsonObject())
178 				throw new JsonParseException("invalid userAgentStylesheet: not an object");
179 			JsonObject o = json.getAsJsonObject();
180 			if (!o.has("mediaType"))
181 				throw new JsonParseException("no mediaType was provided");
182 			if (o.size() > 0)
183 				for (String k : o.keySet())
184 					if (!"mediaType".equals(k))
185 						throw new JsonParseException("unknown key: " + k);
186 			JsonElement j = o.get("mediaType");
187 			if (j.isJsonPrimitive()) {
188 				try {
189 					j = j.getAsJsonPrimitive();
190 					if (!((JsonPrimitive)j).isString())
191 						throw new JsonParseException("invalid mediaType: not a string");
192 					return new MediaTypes(Arrays.asList(j.getAsString().trim().split("\\s+")));
193 				} catch (IllegalStateException e) {
194 					throw new JsonParseException("invalid mediaType: not a string");
195 				}
196 			} else if (j.isJsonArray()) {
197 				List<String> mediaTypes = new ArrayList<>();
198 				for (JsonElement jj : j.getAsJsonArray().asList()) {
199 					try {
200 						jj = jj.getAsJsonPrimitive();
201 						if (!((JsonPrimitive)jj).isString())
202 							throw new JsonParseException("invalid mediaType: not a string");
203 						if (jj.getAsString().contains(" "))
204 							throw new JsonParseException("invalid mediaType: should contain no spaces");
205 						mediaTypes.add(jj.getAsString());
206 					} catch (IllegalStateException e) {
207 						throw new JsonParseException("invalid mediaType: not a string");
208 					}
209 				}
210 				return new MediaTypes(mediaTypes);
211 			} else
212 				throw new JsonParseException("invalid mediaType: not a string or an array");
213 		}
214 	}

216 	private static class UserStylesheetsDeserializer implements JsonDeserializer<UserStylesheets> {
217 		public UserStylesheets deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context)
218 				throws JsonParseException {
219 
220 			if (!json.isJsonObject())
221 				throw new JsonParseException("invalid user stylesheets: not an object");
222 			return new UserStylesheets(
223 				Lists.transform(
224 					json.getAsJsonArray().asList(),
225 					j -> GSON.fromJson(j, URI.class)));
226 		}
