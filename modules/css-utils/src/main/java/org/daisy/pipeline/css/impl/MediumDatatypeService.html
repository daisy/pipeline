1   package org.daisy.pipeline.css.impl;
2   
3   import java.util.regex.Pattern;
4   
5   import javax.xml.parsers.DocumentBuilderFactory;
6   import javax.xml.XMLConstants;
7   
8   import org.daisy.pipeline.datatypes.DatatypeService;
9   import org.daisy.pipeline.datatypes.ValidationResult;
10  
11  import org.osgi.service.component.annotations.Component;
12  
13  import org.w3c.dom.Attr;
14  import org.w3c.dom.Document;
15  import org.w3c.dom.Element;
16  
17  @Component(
18  	name = "org.daisy.pipeline.css.impl.MediumDatatypeService",
19  	service = {
20  		DatatypeService.class
21  	}
22  )
23  public class MediumDatatypeService extends DatatypeService {
24  
25  	private final Pattern pattern;
26  	private Document xmlDefinition = null;
27  
28  	public MediumDatatypeService() {
29  		super("medium");
30  		pattern = MediumParser.asRegex();
31  	}
32  
33  	public ValidationResult validate(String content) {
34  		if (pattern.matcher(content).matches())
35  			return ValidationResult.valid();
36  		else
37  			return ValidationResult.notValid(
38  				"'" + content + "' does not match the regular expression pattern /" + pattern + "/");
39  	}
40  
41  	public Document asDocument() throws Exception {
42  		if (xmlDefinition == null) {
43  			Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder()
44  			                                     .getDOMImplementation().createDocument(null, "data", null);
45  			Element data = doc.getDocumentElement();
46  			Attr type = doc.createAttribute("type");
47  			type.setValue("string");
48  			data.setAttributeNode(type);
49  			Element documentation = (Element)data.appendChild(
50  				doc.createElementNS("http://relaxng.org/ns/compatibility/annotations/1.0",
51  				                    "documentation"));
52  			Attr lang = doc.createAttributeNS(XMLConstants.XML_NS_URI, "lang");
53  			lang.setValue("en");
54  			documentation.setAttributeNode(lang);
55  			Attr space = doc.createAttributeNS(XMLConstants.XML_NS_URI, "space");
56  			space.setValue("preserve");
57  			documentation.setAttributeNode(space);
58  			documentation.appendChild(doc.createTextNode(
59  				"A medium specification"                                                             + "\n" +
60  				                                                                                       "\n" +
61  				"A medium is specified using a resticted version of the "                            + "\n" +
62  				"[media query syntax](https://www.w3.org/TR/mediaqueries-4/), that does not allow "  + "\n" +
63  				"the `not` keyword or `min-` and `max-` prefixes."));
64  			Element param = (Element)data.appendChild(doc.createElement("param"));
65  			Attr name = doc.createAttribute("name");
66  			name.setValue("pattern");
67  			param.setAttributeNode(name);
68  			param.appendChild(doc.createTextNode("^(" + pattern.pattern().replaceAll("\\(\\?<[^>]+>", "(") + ")$"));
69  			xmlDefinition = doc;
70  		}
71  		return xmlDefinition;
72  	}
73  }
