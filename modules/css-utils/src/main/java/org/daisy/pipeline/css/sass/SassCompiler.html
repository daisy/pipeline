53  /**
54   * {@link CssPreProcessor} that handles media type "text/x-scss".
55   */
56  public class SassCompiler implements CssPreProcessor {
57  
58  	public boolean supportsMediaType(String mediaType, URL url) {
59  		if ("text/x-scss".equals(mediaType))
60  			return true;
61  		else if (mediaType == null && url != null && url.toString().endsWith(".scss"))
62  			return true;
63  		else
64  			return false;
65  	}

67  	private final Importer importer;
68  	private final StreamSourceURIResolver resolver;
69  	private final Map<String,String> env;
70  
71  	/**
72  	 * @param env SASS variables. The map is allowed to be mutated by the caller after the
73  	 *            SassCompiler is created.
74  	 */
75  	public SassCompiler(final URIResolver resolver, Map<String,String> env) {
76  		this.resolver = new StreamSourceURIResolver(resolver);
77  		this.env = env;
78  		importer = new Importer() {
79  				public Collection<Import> apply(String url, Import previous) {
80  					URI uri = URLs.asURI(url);
81  					URI base = previous.getAbsoluteUri();
82  					logger.debug("Importing SASS style sheet: " + uri + " (base = " + base + ")");
83  					try {
84  						try {
85  							StreamSource resolved = SassCompiler.this.resolver.resolve(uri.toString(), base.toString());
86  							URI abs = URLs.asURI(resolved.getSystemId());
87  							logger.debug("Resolved to: " + abs);
88  							try {
89  								return ImmutableList.of(
90  									new Import(uri, abs,
91  									           preProcess(
92  										           byteSource(resolved.getInputStream())
93  										           .asCharSource(StandardCharsets.UTF_8).read()))); }
94  							catch (RuntimeException e) {
95  								throw new IOException(e); }}
96  						catch (TransformerException e) {
97  							throw new IOException(e); }}
98  					catch (IOException e) {
99  						if (!url.endsWith(".scss"))
100 							return apply(url + ".scss", previous);
101 						else
102 							throw new RuntimeException("Failed to import " + uri + " (base = " + base + ")", e); }
103 				}
104 			};
105 	}

107 	private static final String scssNumber = "\\d*\\.\\d+";
108 	private static final String scssColor = "(#[\\da-zA-Z]+|(rgb|hsl)a?\\([^)]*\\))";
109 	private static final String scssBadStringChars = "!\"#$'()*+,\\.\\/:<=>?@\\[\\\\\\]^`{|}~-";
110 	private static final String scssNumberColorString = "\\s*("+ scssNumber
111 	                                                       +"|"+ scssColor
112 	                                                       +"|[^\\s"+scssBadStringChars+"]+"
113 	                                                       +"|\"([^\"]|\\\")*\""
114 	                                                       +"|'([^']|\\')*'"
115 	                                                       +")\\s*";
116 	private static final Pattern charsetRule = Pattern.compile("(@charset +\"(.+)\";?).*");
117 	private static final Pattern sourceMappingComment
118 		= Pattern.compile("/\\*# sourceMappingURL=data:application/json;base64,(.+)=* \\*/");
119 	private static final Base64.Decoder base64Decoder = Base64.getDecoder();
120 	
121 	/**
122 	 * @param encoding the encoding of the input or null if unknown
123 	 * @throws IOException if something goes wrong reading the input
124 	 * @throws RuntimeException if the compilation fails.
125 	 */
126 	public PreProcessingResult compile(Source sass, Charset encoding) throws IOException {
127 		String base = sass.getSystemId();
128 		if (sass instanceof StreamSource)
129 			return compile(((StreamSource)sass).getInputStream(), URLs.asURL(base), encoding);
130 		else
131 			try {
132 				return compile(resolver.resolve(base, base), encoding);
133 			} catch (TransformerException e) {
134 				throw new IOException(e);
135 			}
136 	}

138 	/**
139 	 * @param encoding the encoding of the input or null if unknown
140 	 * @throws IOException if something goes wrong reading the input
141 	 * @throws RuntimeException if the compilation fails.
142 	 */
143 	@Override
144 	public PreProcessingResult compile(InputStream sass, URL base, Charset encoding) throws IOException {
145 		Compiler sassCompiler = new Compiler();
146 		Options options = new Options();
147 		options.setIsIndentedSyntaxSrc(false);
148 		options.setOutputStyle(OutputStyle.EXPANDED);
149 		options.setSourceMapContents(false);
150 		options.setSourceMapEmbed(true);
151 		options.setSourceComments(false);
152 		options.setPrecision(5);
153 		options.setOmitSourceMapUrl(false);
154 		options.getImporters().add(importer);
155 		// FIXME: Note that the addition of these variables breaks the original line info in
156 		// sourceMap. Luckily this is not a real problem because only the base info is used.
157 		StringBuilder scss = new StringBuilder();
158 		if (env != null) {
159 			for (String var : env.keySet()) {
160 				String value = env.get(var);
161 				if (!value.matches(scssNumberColorString)) {
162 					// if value contains spaces or special characters that can mess up parsing; wrap it in single quotes
163 					logger.debug("scss variable '"+var+"' contains special characters: "+value);
164 					value = "'"+value.replace("\n", "\\A").replace("'","\\27")+"'";
165 					logger.debug("scss variable '"+var+"' was quoted                 : "+value);
166 				} else {
167 					logger.debug("scss variable '"+var+"' contains no special characters: "+value);
168 				}
169 				scss.append("$").append(var).append(": ").append(value).append(";\n");
170 			}
171 		}
172 		// FIXME: if stream starts with BOM, encoding should be UTF-8
173 		BufferedInputStream bufferedStream = new BufferedInputStream(sass);
174 		bufferedStream.mark(1000);
175 		BufferedReader r = new BufferedReader(new InputStreamReader(bufferedStream,
176 		                                                            encoding != null ? encoding : StandardCharsets.UTF_8));
177 		String firstLine = r.readLine();
178 		Matcher m = charsetRule.matcher(firstLine);
179 		if (m.matches()) {
180 			String charset = m.group(2);
181 			firstLine = firstLine.substring(m.group(1).length());
182 			if (encoding == null)
183 				try {
184 					encoding = Charset.forName(charset);
185 				} catch (UnsupportedCharsetException e) {
186 					logger.warn("Ignoring @charset \"" + charset + "\";");
187 				}
188 			else
189 				logger.warn("Ignoring @charset \"" + charset + "\";");
190 		}
191 		if (encoding == null)
192 			encoding = StandardCharsets.UTF_8;
193 		if (encoding != StandardCharsets.UTF_8) {
194 			if (firstLine.getBytes(StandardCharsets.UTF_8).length < 1000) {
195 				bufferedStream.reset();
196 				r = new BufferedReader(new InputStreamReader(bufferedStream, encoding));
197 				firstLine = r.readLine();
198 				m = charsetRule.matcher(firstLine);
199 				if (m.matches()) { // must be true
200 					firstLine = firstLine.substring(m.group(1).length());
201 				}
202 			} else {
203 				logger.warn("Ignoring @charset \"" + encoding + "\";");
204 			}
205 		}
206 		scss.append(firstLine).append("\n");
207 		scss.append(CharStreams.toString(r));
208 		r.close();
209 		try {
210 			// FIXME: Note that preProcess() breaks the original column info in sourceMap. Luckily
211 			// this is not a real problem because only the base info is used.
212 			Output result = sassCompiler.compileString(preProcess(scss.toString()), URLs.asURI(base), null, options);
213 			String css = result.getCss();
214 			String sourceMap = null; {
215 				int lastNewlineIdx = css.lastIndexOf('\n');
216 				String lastLine = css.substring(lastNewlineIdx + 1);
217 				if ((m = sourceMappingComment.matcher(lastLine)).matches()) {
218 					sourceMap = new String(base64Decoder.decode(m.group(1)));
219 					css = css.substring(0, lastNewlineIdx);
220 				}
221 			}
222 			// FIXME: Note that postProcess() breaks the column info in sourceMap. Luckily this only
223 			// happens in selectors, not in url() values.
224 			css = postProcess(css);
225 			logger.debug(base + " compiled to:\n\n" + css);
226 			return new PreProcessingResult(
227 				new ByteArrayInputStream(css.getBytes(StandardCharsets.UTF_8)),
228 				sourceMap,
229 				// in source map files are relative to the current working directory
230 				sourceMap != null
231 					? URLs.asURI(new File("").getAbsoluteFile())
232 					: URLs.asURI(base)); }
233 		catch (CompilationException e) {
234 			throw new RuntimeException("Could not compile SASS style sheet", e); }
235 	}

245 	/**
246 	 * URIResolver that is guaranteed to return a StreamSource with a systemId.
247 	 * Falls back to URL.openStream() if the backing resolver can not resolve the URI.
248 	 * Throws a TransformerException if an error happens in the backing resolver or the resource can not be found.
249 	 */
250 	private static class StreamSourceURIResolver implements URIResolver {
251 
252 		final URIResolver resolver;
253 
254 		StreamSourceURIResolver(URIResolver resolver) {
255 			this.resolver = resolver;
256 		}
257 
258 		@Override
259 		public StreamSource resolve(String href, String base) throws TransformerException {
260 			Source source = resolver.resolve(href, base);
261 			String systemId = null;
262 			if (source != null)
263 				systemId = source.getSystemId();
264 			if (systemId == null || systemId.equals(""))
265 				systemId = null;
266 			if (source != null && source instanceof StreamSource && systemId != null)
267 				return (StreamSource)source;
268 			if (systemId == null)
269 				systemId = URLs.resolve(URLs.asURI(base), URLs.asURI(href)).toASCIIString();
270 			InputStream stream;
271 			if (source != null && source instanceof StreamSource)
272 				stream = ((StreamSource)source).getInputStream();
273 			else
274 				try {
275 					URLConnection conn = URLs.asURL(systemId).openConnection();
276 					stream = conn.getInputStream();
277 					if ("gzip".equalsIgnoreCase(conn.getContentEncoding()))
278 						stream = new GZIPInputStream(stream); }
279 				catch (IOException e) {
280 					throw new TransformerException(e); }
281 			return new StreamSource(stream, systemId);
282 		}
283 	}

285 	private static final Logger logger = LoggerFactory.getLogger(SassCompiler.class);

287 	/**
288 	 * In order to fully support stacked pseudo-elements and pseudo-classes on pseudo-elements (also
289 	 * in combination with @extend), we need to pre-processed the SASS before it is compiled to
290 	 * CSS. Pseudo-elements are replaced with a child selector followed by a pseudo-element. This is
291 	 * reverted in {@link #postProcess(String)}.
292 	 */
293 	private static String preProcess(String sass) {
294 		return sass.replaceAll("::", ">::");
295 	}
296 
297 	/**
298 	 * Replace child selector followed by a pseudo-element with the pseudo-element.
299 	 */
300 	private static String postProcess(String css) {
301 		if (!css.contains("::"))
302 			return css;
303 		try {
304 			ANTLRInputStream input;
305 			try {
306 				input = new ANTLRInputStream(
307 					new ByteArrayInputStream(css.getBytes(StandardCharsets.UTF_8)),
308 					StandardCharsets.UTF_8.name());
309 			} catch (IOException e) {
310 				throw new RuntimeException(e); // should not happen
311 			}
312 			SassPostProcessLexer lexer = new SassPostProcessLexer(input);
313 			CommonTokenStream tokens = new CommonTokenStream(lexer);
314 			SassPostProcessParser parser = new SassPostProcessParser(tokens);
315 			return parser.stylesheet();
316 		} catch (RecognitionException e) {
317 			throw new RuntimeException("Error happened while parsing the CSS", e);
318 		}
319 	}
