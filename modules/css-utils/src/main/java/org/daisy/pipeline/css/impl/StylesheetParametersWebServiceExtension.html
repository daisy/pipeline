1   package org.daisy.pipeline.css.impl;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import javax.xml.transform.URIResolver;
7   
8   import org.daisy.pipeline.css.MediumProvider;
9   import org.daisy.pipeline.css.UserAgentStylesheetRegistry;
10  import org.daisy.pipeline.datatypes.DatatypeRegistry;
11  import org.daisy.pipeline.webservice.restlet.WebServiceExtension;
12  
13  import org.restlet.routing.Router;
14  
15  import org.osgi.service.component.annotations.Component;
16  import org.osgi.service.component.annotations.Reference;
17  import org.osgi.service.component.annotations.ReferenceCardinality;
18  import org.osgi.service.component.annotations.ReferencePolicy;
19  
20  @Component(
21  	name = "stylesheet-parameters-service-extension",
22  	service = { WebServiceExtension.class }
23  )
24  public class StylesheetParametersWebServiceExtension implements WebServiceExtension {
25  
26  	static final String STYLESHEET_PARAMETERS_ROUTE = "/stylesheet-parameters";
27  
28  	private URIResolver uriResolver;
29  
30  	@Reference(
31  		name = "URIResolver",
32  		unbind = "-",
33  		service = URIResolver.class,
34  		cardinality = ReferenceCardinality.MANDATORY,
35  		policy = ReferencePolicy.STATIC
36  	)
37  	protected void setURIResolver(URIResolver resolver) {
38  		uriResolver = resolver;
39  	}
40  
41  	private DatatypeRegistry datatypeRegistry;
42  
43  	@Reference(
44  		name = "DatatypeRegistry",
45  		unbind = "-",
46  		service = DatatypeRegistry.class,
47  		cardinality = ReferenceCardinality.MANDATORY,
48  		policy = ReferencePolicy.STATIC
49  	)
50  	protected void setDatatypeRegistry(DatatypeRegistry registry) {
51  		datatypeRegistry = registry;
52  	}
53  
54  	private UserAgentStylesheetRegistry userAgentStylesheetRegistry;
55  
56  	@Reference(
57  		name = "UserAgentStylesheetRegistry",
58  		unbind = "-",
59  		service = UserAgentStylesheetRegistry.class,
60  		cardinality = ReferenceCardinality.MANDATORY,
61  		policy = ReferencePolicy.STATIC
62  	)
63  	protected void setUserAgentStylesheetRegistry(UserAgentStylesheetRegistry registry) {
64  		userAgentStylesheetRegistry = registry;
65  	}
66  
67  	private final List<MediumProvider> mediumProviders = new ArrayList<>();
68  
69  	@Reference(
70  		name = "MediumProvider",
71  		unbind = "-",
72  		service = MediumProvider.class,
73  		cardinality = ReferenceCardinality.MULTIPLE,
74  		policy = ReferencePolicy.STATIC
75  	)
76  	protected void bindMediumProvider(MediumProvider provider) {
77  		mediumProviders.add(provider);
78  	}
79  
80  	public void attachTo(Router router) {
81  		router.getContext().getAttributes().put(StylesheetParametersResource.URI_RESOLVER_KEY, uriResolver);
82  		router.getContext().getAttributes().put(StylesheetParametersResource.DATATYPE_REGISTRY_KEY, datatypeRegistry);
83  		router.getContext().getAttributes().put(StylesheetParametersResource.USER_AGENT_STYLESHEET_REGISTRY_KEY, userAgentStylesheetRegistry);
84  		router.getContext().getAttributes().put(StylesheetParametersResource.MEDIUM_PROVIDER_REGISTRY_KEY, MediumProvider.dispatch(mediumProviders, true));
85  		router.attach(STYLESHEET_PARAMETERS_ROUTE, StylesheetParametersResource.class);
86  	}
87  }
