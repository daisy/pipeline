1   package org.daisy.pipeline.css;
2   
3   import java.util.List;
4   import java.util.Map;
5   
6   import cz.vutbr.web.css.MediaQuery;
7   import cz.vutbr.web.csskit.antlr.CSSParserFactory;
8   
9   /**
10   * Media query parser for media queries that are specifically intended for selecting a medium
11   * (i.e. for passing to {@link MediumProvider}).
12   */
13  public class MediaQueryParser {
14  	private MediaQueryParser() {}
15  
16  	// these are all immutable
17  	public static final MediaQuery EMBOSSED = Medium.EMBOSSED.toMediaQuery();
18  	public static final MediaQuery BRAILLE = Medium.BRAILLE.toMediaQuery();
19  	public static final MediaQuery SPEECH = Medium.SPEECH.toMediaQuery();
20  	public static final MediaQuery SCREEN = Medium.SCREEN.toMediaQuery();
21  	public static final MediaQuery PRINT = Medium.PRINT.toMediaQuery();
22  
23  	/**
24  	 * Parse a media query
25  	 *
26  	 * @return the parsed media query, validated and normalized
27  	 * @throws IllegalArgumentException if the media query can not be parsed due to a wrong syntax,
28  	 *                                  contains NOT operators, "min-" or "max-" prefixes, or
29  	 *                                  contains unsupported, invalid or conflicting media features
30  	 *                                  or types
31  	 */
32  	public static MediaQuery parse(String query) throws IllegalArgumentException {
33  		List<MediaQuery> q = parseList(query);
34  		if (q.size() != 1)
35  			throw new IllegalArgumentException("Media query is a comma-separated list: " + query.trim());
36  		return validate(q.get(0));
37  	}
38  
39  	public static List<MediaQuery> parseList(String query) throws IllegalArgumentException {
40  		List<MediaQuery> q = CSSParserFactory.getInstance().parseMediaQuery(query.trim());
41  		if (q == null)
42  			throw new IllegalArgumentException("Media query could not be parsed: " + query.trim());
43  		return q;
44  	}
45  
46  	/**
47  	 * Parse a list of media features
48  	 *
49  	 * @throws IllegalArgumentException if the input contains unsupported, invalid, or conflicting
50  	 *                                  media features.
51  	 */
52  	public static MediaQuery parse(Map<String,Object> query) throws IllegalArgumentException {
53  		return new Medium.MediumBuilder().parse(query).build().toMediaQuery();
54  	}
55  
56  	// validate and normalize
57  	private static MediaQuery validate(MediaQuery q) {
58  		if ("all".equals(q.getType()) && q.isNegative() && q.size() == 0)
59  			throw new IllegalArgumentException("Media query matches nothing: " + q);
60  		return Medium.fromMediaQuery(q).toMediaQuery();
61  	}
62  }
