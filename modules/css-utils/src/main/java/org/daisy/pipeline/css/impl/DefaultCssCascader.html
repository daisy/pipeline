60  /**
61   * For now this cascader handles only 'print' and 'screen' media without support for counters
62   * ({@code (-counter-support: none)}). It will generate marker contents by <a
63   * href="https://www.w3.org/TR/css-lists-3/#content-property">evaluating marker contents</a>, based
64   * on {@code ::marker} pseudo-element rules, 'list-style-type', 'counter-reset', 'counter-set' and
65   * 'counter-increment' properties and {@code @counter-style} rules, according to <a
66   * href="https://www.w3.org/TR/css-lists-3/#content-property">CSS Lists and Counters Module Level
67   * 3</a>. The marker contents are inserted as text nodes.
68   *
69   * For now all other styles are ignored and no attributes are added.
70   */
71  @Component(
72  	name = "DefaultCssCascader",
73  	service = { CssCascader.class }
74  )
75  public class DefaultCssCascader implements CssCascader {
76  
77  	public boolean supportsMedium(Medium medium) {
78  		switch (medium.getType()) {
79  		case PRINT:
80  		case SCREEN:
81  			return "none".equals(medium.getCustomFeatures().get("counter-support"));
82  		default:
83  			return false;
84  		}
85  	}
86  
87  	public XMLTransformer newInstance(Medium medium,
88  	                                  String userStylesheet,
89  	                                  URIResolver uriResolver,
90  	                                  CssPreProcessor preProcessor,
91  	                                  XsltProcessor xsltProcessor,
92  	                                  QName attributeName,
93  	                                  boolean multipleAttrs) {
94  		if (!supportsMedium(medium))
95  			throw new IllegalArgumentException("medium not supported: " + medium);
96  		return new Transformer(uriResolver, preProcessor, xsltProcessor, userStylesheet, medium,
97  		                       attributeName, multipleAttrs);
98  	}

108 	 * We can make use of JStyleParserCssCascader for the evaluation of marker contents because to

114 		private Map<String,CounterStyle> namedCounterStyles = null;
115 		private LinkedList<CounterStyle> listCounterStyle = new LinkedList<>();
116 		private LinkedList<Map<String,Integer>> counterValues = new LinkedList<>();
117 		private final QName markerAttributeName;
118 		private final QName markerContentAttributeName;

120 		private Transformer(URIResolver resolver, CssPreProcessor preProcessor, XsltProcessor xsltProcessor,
121 		                    String userStyleSheet, Medium medium, QName attributeName, boolean multipleAttrs) {
122 			super(resolver, preProcessor, xsltProcessor, userStyleSheet, medium, attributeName,
123 			      parserFactory, ruleFactory, supportedCSS, declarationTransformer);
124 			listCounterStyle.push(CounterStyle.DISC);
125 			counterValues.push(null);
126 			if (attributeName == null) {
127 				markerAttributeName= null;
128 				markerContentAttributeName = null;
129 			} else if (multipleAttrs) {
130 				markerAttributeName= null;
131 				markerContentAttributeName
132 					= new QName(attributeName.getNamespaceURI(), "marker-content", attributeName.getPrefix());
133 			} else {
134 				markerAttributeName= attributeName;
135 				markerContentAttributeName = null;
136 			}
137 		}

139 		@Override
140 		protected void processElement(Element element, StyleMap styleMap, BaseURIAwareXMLStreamWriter writer)
141 				throws XMLStreamException, IllegalArgumentException {
142 			XMLStreamWriterHelper.writeStartElement(writer, element);
143 			copyAttributes(element, writer);
144 			Map<PseudoElement,NodeData> pseudoStyles = new HashMap<>(); {
145 				for (PseudoElement pseudo : styleMap.pseudoSet(element))
146 					pseudoStyles.put(pseudo, styleMap.get(element, pseudo)); }
147 			NodeData mainStyle = styleMap.get(element);
148 			updateCounterValues(mainStyle);
149 			CounterStyle listCounterStyle = getCurrentListCounterStyle(mainStyle);
150 			String marker = generateMarkerContents(mainStyle, pseudoStyles, element, listCounterStyle);
151 			if (marker != null) {
152 				if (markerContentAttributeName != null)
153 					// if attribute namespace specified, insert as "marker-content" attribute
154 					XMLStreamWriterHelper.writeAttribute(writer, markerContentAttributeName, marker);
155 				else if (markerAttributeName != null)
156 					// if single attribute specified, insert as ::marker {} pseudo-element rule
157 					XMLStreamWriterHelper.writeAttribute(
158 						writer, markerAttributeName,
159 						String.format("&::marker { content: '%s' }", marker.replaceAll("'", "\\\\'")));
160 				else
161 					// if no attribute specified, insert as text
162 					writer.writeCharacters(marker);
163 			}
164 			this.listCounterStyle.push(listCounterStyle);
165 			this.counterValues.push(null);
166 			for (Node child = element.getFirstChild(); child != null; child = child.getNextSibling())
167 				traverse(child, styleMap, writer);
168 			this.listCounterStyle.pop();
169 			this.counterValues.pop();
170 			writer.writeEndElement();
171 		}

173 		/**
174 		 * Determine counter values (see https://www.w3.org/TR/css-lists-3/#creating-counters)
175 		 */
176 		private void updateCounterValues(NodeData style) {
177 			if (style.getProperty("counter-reset", false) == CounterReset.list_values) {
178 				List<Term<?>> list = style.getValue(TermList.class, "counter-reset", false);
179 				if (list != null) {
180 					for (Term<?> t : list) {
181 						if (!(t instanceof TermPair))
182 							throw new IllegalStateException("coding error");
183 						TermPair<String,Integer> p = (TermPair<String,Integer>)t;
184 						String name = p.getKey();
185 						Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
186 						if (createdBySiblingOrSelf == null) {
187 							createdBySiblingOrSelf = new HashMap<>();
188 							counterValues.set(0, createdBySiblingOrSelf);
189 						}
190 						createdBySiblingOrSelf.put(name, p.getValue());
191 					}
192 				}
193 			}
194 			// counter-set or counter-increment work in addition to counter-reset
195 			Set<String> done = null;
196 			if (style.getProperty("counter-set", false) == CounterSet.list_values) {
197 				List<Term<?>> list = style.getValue(TermList.class, "counter-set", false);
198 				if (list != null) {
199 					for (Term<?> t : list) {
200 						if (!(t instanceof TermPair))
201 							throw new IllegalStateException("coding error");
202 						TermPair<String,Integer> p = (TermPair<String,Integer>)t;
203 						String name = p.getKey();
204 						boolean exists = false;
205 						for (Map<String,Integer> c : counterValues) {
206 							if (c != null) {
207 								c.put(name, p.getValue());
208 								exists = true;
209 								break;
210 							}
211 						}
212 						if (!exists) {
213 							Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
214 							if (createdBySiblingOrSelf == null) {
215 								createdBySiblingOrSelf = new HashMap<>();
216 								counterValues.set(0, createdBySiblingOrSelf);
217 							}
218 							createdBySiblingOrSelf.put(name, p.getValue());
219 						}
220 						if (done == null)
221 							done = new HashSet<>();
222 						done.add(name);
223 					}
224 				}
225 			}
226 			if (style.getProperty("counter-increment", false) == CounterIncrement.list_values) {
227 				List<Term<?>> list = style.getValue(TermList.class, "counter-increment", false);
228 				if (list != null) {
229 					for (Term<?> t : list) {
230 						if (!(t instanceof TermPair))
231 							throw new IllegalStateException("coding error");
232 						TermPair<String,Integer> p = (TermPair<String,Integer>)t;
233 						String name = p.getKey();
234 						if (done == null || !done.contains(name)) {
235 							boolean exists = false;
236 							for (Map<String,Integer> c : counterValues) {
237 								if (c != null) {
238 									c.put(name, c.getOrDefault(name, 0) + p.getValue());
239 									exists = true;
240 									break;
241 								}
242 							}
243 							if (!exists) {
244 								Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
245 								if (createdBySiblingOrSelf == null) {
246 									createdBySiblingOrSelf = new HashMap<>();
247 									counterValues.set(0, createdBySiblingOrSelf);
248 								}
249 								createdBySiblingOrSelf.put(name, p.getValue());
250 							}
251 							if (done == null)
252 								done = new HashSet<>();
253 							done.add(name);
254 						}
255 					}
256 				}
257 			}
258 			// by default list items increment the 'list-item' counter
259 			if ((done == null || !done.contains("list-item")) && style.getProperty("display", false) == Display.LIST_ITEM) {
260 				boolean exists = false;
261 				for (Map<String,Integer> c : counterValues) {
262 					if (c != null) {
263 						c.put("list-item", c.getOrDefault("list-item", 0) + 1);
264 						exists = true;
265 						break;
266 					}
267 				}
268 				if (!exists) {
269 					Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
270 					if (createdBySiblingOrSelf == null) {
271 						createdBySiblingOrSelf = new HashMap<>();
272 						counterValues.set(0, createdBySiblingOrSelf);
273 					}
274 					createdBySiblingOrSelf.put("list-item", 1);
275 				}
276 			}
277 		}

279 		private CounterStyle getCurrentListCounterStyle(NodeData style) {
280 			ListStyleType type = style.getProperty("list-style-type", false);
281 			if (type == null)
282 				return this.listCounterStyle.peek(); // value is inherited from parent
283 			else {
284 				switch (type) {
285 				case INHERIT:
286 					// should not happen
287 					return this.listCounterStyle.peek();
288 				case INITIAL:
289 					return CounterStyle.DISC; // see https://www.w3.org/TR/css-lists-3/#propdef-list-style-type
290 				case counter_style_name:
291 					return getNamedCounterStyle(style.getValue(TermIdent.class, "list-style-type").getValue());
292 				case symbols_fn:
293 					return CounterStyle.fromSymbolsFunction(style.getValue(TermFunction.class, "list-style-type"));
294 				case braille_string: // note that braille CSS does not support any string, as standard CSS does
295 					return null;
296 				case DECIMAL:
297 					return CounterStyle.DECIMAL;
298 				case LOWER_ALPHA:
299 					return CounterStyle.LOWER_ALPHA;
300 				case LOWER_ROMAN:
301 					return CounterStyle.LOWER_ROMAN;
302 				case UPPER_ALPHA:
303 					return CounterStyle.UPPER_ALPHA;
304 				case UPPER_ROMAN:
305 					return CounterStyle.UPPER_ROMAN;
306 				case NONE:
307 				default:
308 					return null;
309 				}
310 			}
311 		}

313 		private CounterStyle getNamedCounterStyle(String name) {
314 			if (namedCounterStyles == null)
315 				namedCounterStyles = CounterStyle.parseCounterStyleRules(
316 					Iterables.filter(getParsedStyleSheet(), RuleCounterStyle.class));
317 			return namedCounterStyles.getOrDefault(name, CounterStyle.DECIMAL);
318 		}

320 		private String evaluateCounter(String name, CounterStyle style, boolean withPrefixAndSuffix) {
321 			Integer value = null;
322 			boolean exists = false;
323 			for (Map<String,Integer> c : counterValues)
324 				if (c != null && c.containsKey(name)) {
325 					value = c.get(name);
326 					break;
327 				}
328 			if (value == null) {
329 				value = 0;
330 				Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
331 				if (createdBySiblingOrSelf == null) {
332 					createdBySiblingOrSelf = new HashMap<>();
333 					counterValues.set(0, createdBySiblingOrSelf);
334 				}
335 				createdBySiblingOrSelf.put(name, value);
336 			}
337 			return style.format(value, withPrefixAndSuffix);
338 		}

340 		private String evaluateCounters(String name, CounterStyle style, String separator) {
341 			StringBuilder s = new StringBuilder();
342 			for (Map<String,Integer> c : counterValues)
343 				if (c != null && c.containsKey(name)) {
344 					if (s.length() > 0)
345 						s.insert(0, separator);
346 					s.insert(0, style.format(c.get(name)));
347 				}
348 			if (s.length() == 0) {
349 				Map<String,Integer> createdBySiblingOrSelf = counterValues.peek();
350 				if (createdBySiblingOrSelf == null) {
351 					createdBySiblingOrSelf = new HashMap<>();
352 					counterValues.set(0, createdBySiblingOrSelf);
353 				}
354 				createdBySiblingOrSelf.put(name, 0);
355 				s.append(style.format(0));
356 			}
357 			return s.toString();
358 		}

360 		private String evaluateContent(List<Term<?>> content, Element element) throws IllegalArgumentException {
361 			StringBuilder s = new StringBuilder();
362 			for (Term<?> t : content) {
363 				if (t instanceof TermString) {
364 					s.append(((TermString)t).getValue());
365 				} else if (t instanceof TermFunction) {
366 					TermFunction f = (TermFunction)t;
367 					if ("counter".equalsIgnoreCase(f.getFunctionName())) {
368 						String name = null;
369 						CounterStyle style = null;
370 						for (Term<?> arg : f)
371 							if (name == null)
372 								if (arg instanceof TermIdent)
373 									name = ((TermIdent)arg).getValue();
374 								else
375 									throw new IllegalArgumentException(
376 										"invalid first argument of counter() function: should be the counter name");
377 							else if (style == null) {
378 								if (arg instanceof TermIdent)
379 									style = getNamedCounterStyle(((TermIdent)arg).getValue());
380 								else if (arg instanceof TermFunction)
381 									try {
382 										style = CounterStyle.fromSymbolsFunction((TermFunction)arg);
383 									} catch (IllegalArgumentException e) {
384 										throw new IllegalArgumentException(
385 											"invalid second argument of counter() function: should be a counter style", e);
386 									}
387 								else
388 									throw new IllegalArgumentException(
389 										"invalid second argument of counter() function: should be a counter style"); }
390 							else
391 								throw new IllegalArgumentException(
392 									"unexpected argument of counter() function: function takes at most two arguments");
393 						if (name == null)
394 							throw new IllegalArgumentException("counter() function requires at least one argument");
395 						if (style == null)
396 							style = CounterStyle.DECIMAL;
397 						s.append(evaluateCounter(name, style, false));
398 					} else if ("counters".equalsIgnoreCase(f.getFunctionName())) {
399 						String name = null;
400 						String separator = null;
401 						CounterStyle style = null;
402 						for (Term<?> arg : f)
403 							if (name == null)
404 								if (arg instanceof TermIdent)
405 									name = ((TermIdent)arg).getValue();
406 								else
407 									throw new IllegalArgumentException(
408 										"invalid first argument of counters() function: should be the counter name");
409 							else if (separator == null) {
410 								if (arg instanceof TermString)
411 									separator = ((TermString)arg).getValue();
412 								else
413 									throw new IllegalArgumentException(
414 										"invalid second argument of counters() function: should be a string"); }
415 							else if (style == null) {
416 								if (arg instanceof TermIdent)
417 									style = getNamedCounterStyle(((TermIdent)arg).getValue());
418 								else if (arg instanceof TermFunction)
419 									try {
420 										style = CounterStyle.fromSymbolsFunction((TermFunction)arg);
421 									} catch (IllegalArgumentException e) {
422 										throw new IllegalArgumentException(
423 											"invalid third argument of counters() function: should be a counter style", e);
424 									}
425 								else
426 									throw new IllegalArgumentException(
427 										"invalid third argument of counters() function: should be a counter style"); }
428 							else
429 								throw new IllegalArgumentException(
430 									"unexpected argument of counters() function: function takes at most three arguments");
431 						if (name == null || separator == null)
432 							throw new IllegalArgumentException("counters() function requires at least two arguments");
433 						if (style == null)
434 							style = CounterStyle.DECIMAL;
435 						s.append(evaluateCounters(name, style, separator));
436 					} else
437 						throw new RuntimeException(f.getFunctionName() + "() function not supported in content list"); // FIXME
438 				} else
439 					throw new IllegalStateException(); // cannot happen
440 			}
441 			return s.toString();
442 		}

444 		/**
445 		 * Evaluate marker contents (see https://www.w3.org/TR/css-lists-3/#content-property)
446 		 */
447 		private String generateMarkerContents(NodeData mainStyle, Map<PseudoElement,NodeData> pseudoStyles, Element element,
448 		                                      CounterStyle listCounterStyle) throws IllegalArgumentException {
449 			if (mainStyle.getProperty("display", false) == Display.LIST_ITEM) {
450 				for (PseudoElement pseudo : pseudoStyles.keySet()) {
451 					if ("marker".equals(pseudo.getName())) {
452 						NodeData pseudoStyle = pseudoStyles.get(pseudo);
453 						if (pseudoStyle.getProperty("content", false) == Content.content_list) {
454 							List<Term<?>> content = pseudoStyle.getValue(TermList.class, "content", false);
455 							if (content != null)
456 								return evaluateContent(content, element);
457 						}
458 						return null;
459 					}
460 				}
461 				if (listCounterStyle != null)
462 					return evaluateCounter("list-item", listCounterStyle, true);
463 			}
464 			return null;
465 		}
