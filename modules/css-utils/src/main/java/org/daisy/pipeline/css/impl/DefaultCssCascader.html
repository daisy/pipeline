55  /**
56   * For now this cascader handles only 'print' and 'screen' media without support for counters
57   * ({@code (counter-support: none)}). It will generate marker contents by <a
58   * href="https://www.w3.org/TR/css-lists-3/#content-property">evaluating marker contents</a>, based
59   * on {@code ::marker} pseudo-element rules, 'list-style-type', 'counter-reset', 'counter-set' and
60   * 'counter-increment' properties and {@code @counter-style} rules, according to <a
61   * href="https://www.w3.org/TR/css-lists-3/#content-property">CSS Lists and Counters Module Level
62   * 3</a>. The marker contents are inserted as text nodes.
63   *
64   * For now all other styles are ignored and no attributes are added.
65   */
66  @Component(
67  	name = "DefaultCssCascader",
68  	service = { CssCascader.class }
69  )
70  public class DefaultCssCascader implements CssCascader {
71  
72  	public boolean supportsMedium(Medium medium) {
73  		if (medium.getType() == null)
74  			return false;
75  		switch (medium.getType()) {
76  		case PRINT:
77  		case SCREEN:
78  			return medium.matches("(counter-support: none)");
79  		default:
80  			return false;
81  		}
82  	}
83  
84  	public XMLTransformer newInstance(Medium medium,
85  	                                  String userStylesheet,
86  	                                  URIResolver uriResolver,
87  	                                  CssPreProcessor preProcessor,
88  	                                  XsltProcessor xsltProcessor,
89  	                                  QName attributeName,
90  	                                  boolean multipleAttrs) {
91  		if (!supportsMedium(medium))
92  			throw new IllegalArgumentException("medium not supported: " + medium);
93  		return new Transformer(uriResolver, preProcessor, xsltProcessor, userStylesheet, medium,
94  		                       attributeName, multipleAttrs);
95  	}

97  	// using braille-css because @counter-style is not supported by jStyleParser
98  	private static final CSSParserFactory parserFactory = new BrailleCSSParserFactory();
99  	private static final RuleFactory ruleFactory = RuleFactoryImpl.getInstance();
100 	private static final SupportedBrailleCSS supportedCSS = new SupportedBrailleCSS(false, true); // FIXME: support "list-style" shorthand
101 
102 	/*
103 	 * We can make use of JStyleParserCssCascader for the evaluation of marker contents because to
104 	 * compute the value we only need to know about preceding and ancestor nodes, which is a
105 	 * condition that is fulfilled because of how traverse() is called.
106 	 */
107 	private static class Transformer extends JStyleParserCssCascader {
108 
109 		private Map<String,CounterStyle> namedCounterStyles = null;
110 		private CounterEvaluator<Element> evaluator;
111 		private final QName markerAttributeName;
112 		private final QName markerContentAttributeName;
113 
114 		private Transformer(URIResolver resolver, CssPreProcessor preProcessor, XsltProcessor xsltProcessor,
115 		                    String userStyleSheet, Medium medium, QName attributeName, boolean multipleAttrs) {
116 			super(resolver, preProcessor, xsltProcessor, userStyleSheet, medium, attributeName,
117 			      parserFactory, ruleFactory, supportedCSS, supportedCSS);
118 			if (attributeName == null) {
119 				markerAttributeName= null;
120 				markerContentAttributeName = null;
121 			} else if (multipleAttrs) {
122 				markerAttributeName= null;
123 				markerContentAttributeName
124 					= new QName(attributeName.getNamespaceURI(), "marker-content", attributeName.getPrefix());
125 			} else {
126 				markerAttributeName= attributeName;
127 				markerContentAttributeName = null;
128 			}
129 		}
130 
131 		@Override
132 		protected void processElement(Element element, StyleMap styleMap, BaseURIAwareXMLStreamWriter writer)
133 				throws XMLStreamException, IllegalArgumentException {
134 			if (evaluator == null)
135 				// styleMap is constant so we can define evaluator based on styleMap argument of first call
136 				evaluator = new CounterEvaluator<Element>() {
137 						@Override
138 						protected Collection<TermPair<String,Integer>> getCounterReset(Element element) {
139 							NodeData style = styleMap.get(element);
140 							if (style.getProperty("counter-reset", false) == CounterReset.list_values) {
141 								List<Term<?>> list = style.getValue(TermList.class, "counter-reset", false);
142 								if (list != null) {
143 									for (Term<?> t : list) {
144 										if (!(t instanceof TermPair))
145 											throw new IllegalStateException("coding error");
146 										TermPair p = (TermPair)t;
147 										if (!(p.getKey() instanceof String))
148 											throw new IllegalStateException("coding error");
149 										if (!(p.getValue() instanceof Integer))
150 											throw new IllegalStateException("coding error");
151 									}
152 									return (List<TermPair<String,Integer>>)(List<?>)list;
153 								}
154 							}
155 							return null;
156 						}
157 						@Override
158 						protected Collection<TermPair<String,Integer>> getCounterSet(Element element) {
159 							NodeData style = styleMap.get(element);
160 							if (style.getProperty("counter-set", false) == CounterSet.list_values) {
161 								List<Term<?>> list = style.getValue(TermList.class, "counter-set", false);
162 								if (list != null) {
163 									for (Term<?> t : list) {
164 										if (!(t instanceof TermPair))
165 											throw new IllegalStateException("coding error");
166 										TermPair p = (TermPair)t;
167 										if (!(p.getKey() instanceof String))
168 											throw new IllegalStateException("coding error");
169 										if (!(p.getValue() instanceof Integer))
170 											throw new IllegalStateException("coding error");
171 									}
172 									return (List<TermPair<String,Integer>>)(List<?>)list;
173 								}
174 							}
175 							return null;
176 						}
177 						@Override
178 						protected Collection<TermPair<String,Integer>> getCounterIncrement(Element element) {
179 							NodeData style = styleMap.get(element);
180 							if (style.getProperty("counter-increment", false) == CounterIncrement.list_values) {
181 								List<Term<?>> list = style.getValue(TermList.class, "counter-increment", false);
182 								if (list != null) {
183 									for (Term<?> t : list) {
184 										if (!(t instanceof TermPair))
185 											throw new IllegalStateException("coding error");
186 										TermPair p = (TermPair)t;
187 										if (!(p.getKey() instanceof String))
188 											throw new IllegalStateException("coding error");
189 										if (!(p.getValue() instanceof Integer))
190 											throw new IllegalStateException("coding error");
191 									}
192 									return (List<TermPair<String,Integer>>)(List<?>)list;
193 								}
194 							}
195 							return null;
196 						}
197 						@Override
198 						protected Display getDisplay(Element element) {
199 							NodeData style = styleMap.get(element);
200 							return style.getProperty("display", false);
201 						}
202 						@Override
203 						protected Collection<Term<?>> getMarkerContent(Element element) {
204 							NodeData style = styleMap.get(element);
205 							for (PseudoElement pseudo : styleMap.pseudoSet(element)) {
206 								if ("marker".equals(pseudo.getName())) {
207 									NodeData pseudoStyle = styleMap.get(element, pseudo);
208 									if (pseudoStyle.getProperty("content", false) == Content.content_list) {
209 										List<Term<?>> content = pseudoStyle.getValue(TermList.class, "content", false);
210 										if (content != null)
211 											return content;
212 									}
213 									return Collections.emptyList();
214 								}
215 							}
216 							return null;
217 						}
218 						@Override
219 						protected CounterStyle getListStyleType(Element element, CounterStyle parentListStyleType) {
220 							NodeData style = styleMap.get(element);
221 							ListStyleType type = style.getProperty("list-style-type", false);
222 							if (type == null)
223 								return parentListStyleType; // value is inherited from parent
224 							else {
225 								switch (type) {
226 								case INHERIT:
227 									// should not happen
228 									return parentListStyleType;
229 								case INITIAL:
230 									return CounterStyle.DISC; // see https://www.w3.org/TR/css-lists-3/#propdef-list-style-type
231 								case counter_style_name:
232 									return getNamedCounterStyle(style.getValue(TermIdent.class, "list-style-type").getValue());
233 								case symbols_fn:
234 									return CounterStyle.fromSymbolsFunction(style.getValue(TermFunction.class, "list-style-type"));
235 								case braille_string: // note that braille CSS does not support any string, as standard CSS does
236 									return null;
237 								case DECIMAL:
238 									return CounterStyle.DECIMAL;
239 								case LOWER_ALPHA:
240 									return CounterStyle.LOWER_ALPHA;
241 								case LOWER_ROMAN:
242 									return CounterStyle.LOWER_ROMAN;
243 								case UPPER_ALPHA:
244 									return CounterStyle.UPPER_ALPHA;
245 								case UPPER_ROMAN:
246 									return CounterStyle.UPPER_ROMAN;
247 								case NONE:
248 								default:
249 									return null;
250 								}
251 							}
252 						}
253 						@Override
254 						protected CounterStyle getNamedCounterStyle(String name) {
255 							if (namedCounterStyles == null)
256 								namedCounterStyles = CounterStyle.parseCounterStyleRules(
257 									Iterables.filter(getParsedStyleSheet(), RuleCounterStyle.class));
258 							return namedCounterStyles.getOrDefault(name, CounterStyle.DECIMAL);
259 						}
260 					};
261 			XMLStreamWriterHelper.writeStartElement(writer, element);
262 			copyAttributes(element, writer);
263 			evaluator.startElement(element);
264 			String marker = evaluator.generateMarkerContents(element);
265 			if (marker != null) {
266 				if (markerContentAttributeName != null)
267 					// if attribute namespace specified, insert as "marker-content" attribute
268 					XMLStreamWriterHelper.writeAttribute(writer, markerContentAttributeName, marker);
269 				else if (markerAttributeName != null)
270 					// if single attribute specified, insert as ::marker {} pseudo-element rule
271 					XMLStreamWriterHelper.writeAttribute(
272 						writer, markerAttributeName,
273 						String.format("&::marker { content: '%s' }", marker.replaceAll("'", "\\\\'")));
274 				else
275 					// if no attribute specified, insert as text
276 					writer.writeCharacters(marker);
277 			}
278 			for (Node child = element.getFirstChild(); child != null; child = child.getNextSibling())
279 				traverse(child, styleMap, writer);
280 			evaluator.endElement();
281 			writer.writeEndElement();
282 		}
283 
284 		protected Map<QName,String> serializeStyle(NodeData mainStyle, Map<PseudoElement,NodeData> pseudoStyles, Element context) {
285 			throw new UnsupportedOperationException(); // not needed
286 		}
287 
288 		protected String serializeValue(Term<?> value, String property) {
289 			return CssSerializer.getInstance().toString(value);
290 		}
291 	}
