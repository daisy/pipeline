107 	public static class EpubCheckStep extends DefaultStep implements XProcStep {
108 
109 		private static final QName _epubFile = new QName("epub");
110 		private static final QName _epubVersion = new QName("version");
111 		private static final QName _mode = new QName("mode"); // epub/opf/xhtml/svg/mo/nav
112 		private static final QName _tempDir = new QName("temp-dir");
113 
114 		private WritablePipe report = null;
115 
116 		private EpubCheckStep(XProcRuntime runtime, XAtomicStep step) {
117 			super(runtime, step);
118 		}
119 
120 		@Override
121 		public void setOutput(String port, WritablePipe pipe) {
122 			report = pipe;
123 		}
124 
125 		@Override
126 		public void reset() {
127 			report.resetWriter();
128 		}
129 
130 		@Override
131 		public void run() throws SaxonApiException {
132 			super.run();
133 
134 			Archive epub = null;
135 			try {
136 
137 				URI epubURI = new URI(getOption(_epubFile).getString());
138 				File epubFile = new File(epubURI);
139 				String path = epubFile.getCanonicalPath();
140 
141 				String epubVersionString = getOption(_epubVersion).getString();
142 
143 				if (!("3".equals(epubVersionString) || "2".equals(epubVersionString))) {
144 					throw new InvalidVersionException("'"+_epubVersion.getLocalName()+"' should be either '2' or '3'");
145 				}
146 
147 				EPUBVersion epubVersion = EPUBVersion.VERSION_2.equals(epubVersionString) ? EPUBVersion.VERSION_2 : EPUBVersion.VERSION_3;
148 
149 				String mode = getOption(_mode).getString();
150 				if ("epub".equals(mode))
151 					mode = null;
152 
153 				File fileOut = File.createTempFile("epubcheck", null);
154 				Report xmlReport = new XmlReportImpl(new PrintWriter(fileOut, "UTF-8"), epubFile.getName(), EpubCheck.version());
155 
156 				String toolDate = EpubCheck.buildDate();
157 				if (toolDate != null && !toolDate.startsWith("$"))
158 					xmlReport.info(null, FeatureEnum.TOOL_DATE, toolDate);
159 
160 				if (mode != null) {
161 					xmlReport.info(null, FeatureEnum.EXEC_MODE, String.format(messages.get("single_file"), mode, epubVersion.toString(), EPUBProfile.DEFAULT));
162 
163 					if ("expanded".equals(mode) || "exp".equals(mode)) {
164 						epub = new Archive(path, true);
165 						String epubName = new File(path).getName() + ".epub";
166 						File tempDir; {
167 							RuntimeValue o = getOption(_tempDir);
168 							String tempDirOpt = o == null ? null : o.getString();
169 							if (tempDirOpt != null && !tempDirOpt.equals("")) {
170 								try {
171 									tempDir = new File(new URI(tempDirOpt));
172 								} catch (URISyntaxException e) {
173 									throw new IllegalArgumentException("temp-dir option invalid: " + tempDirOpt);
174 								}
175 								if (tempDir.exists())
176 									throw new IllegalArgumentException("temp-dir option must be a non-existing directory: "+tempDirOpt);
177 							} else {
178 								tempDir = Files.createTempDirectory("epubcheck-").toFile();
179 								tempDir.deleteOnExit();
180 							}
181 						}
182 						File zippedEpubFile = new File(tempDir, epubName);
183 						logger.debug("Creating zipped EPUB from " + path + " at " + zippedEpubFile);
184 						tempDir.mkdirs();
185 						epub.createArchive(zippedEpubFile);
186 						EpubCheck check = new EpubCheck(zippedEpubFile, xmlReport);
187 						check.validate();
188 					}
189 
190 				}
191 
192 				if (epub == null) {
193 					GenericResourceProvider resourceProvider;
194 
195 					if (path.startsWith("http://") || path.startsWith("https://"))
196 						resourceProvider = new URLResourceProvider(path);
197 					else
198 						resourceProvider = new FileResourceProvider(path);
199 
200 					OPSType opsType = new OPSType(mode, epubVersion);
201 
202 					DocumentValidatorFactory factory = (DocumentValidatorFactory) documentValidatorFactoryMap.get(opsType);
203 
204 					if (factory == null) {
205 						xmlReport.message(new Message(null, Severity.FATAL, messages.get("mode_version_not_supported", mode, epubVersion), null), EPUBLocation.create(PathUtil.removeWorkingDirectory(path), 0, 0), mode, epubVersion);
206 
207 						throw new RuntimeException(messages.get("mode_version_not_supported", mode, epubVersion));
208 					}
209 					
210 					ValidationContext validationContext = new ValidationContext.ValidationContextBuilder()
211 															.report(xmlReport)
212 															.path(path)
213 															.resourceProvider(resourceProvider)
214 															.mimetype((String) modeMimeTypeMap.get(opsType))
215 															.version(epubVersion)
216 															.build();
217 					DocumentValidator check = factory.newInstance(validationContext);
218 					check.validate();
219 				}
220 				
221 				if (xmlReport.generate() == 0) {
222 					XdmNode reportXml = runtime.getProcessor().newDocumentBuilder().build(fileOut);
223 					fileOut.delete();
224 					report.write(reportXml);
225 				} else {
226 					throw new Exception("An error occured while trying to write epubcheck report to: "+fileOut.getAbsolutePath());
227 				}
228 			}
229 
230 			catch (Throwable e) {
231 				throw XProcStep.raiseError(e, step);
232 			}
233 
234 			finally {
235 				// delete temporary epub file (expanded mode)
236 				// but not if temp-dir option was passed
237 				if (epub != null && getOption(_tempDir, "").isEmpty())
238 					epub.deleteEpubFile();
239 			}
240 		}
241 	}
