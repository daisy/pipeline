50      /**
51       * Export a openoffice graphic object within a document to
52       * @param xGraphicProvider
53       * @param xGraphic
54       * @param fileName
55       * @throws Exception
56       */
57      public static void exportGraphicObject(XGraphicProvider xGraphicProvider, XGraphic xGraphic, String fileName) throws Exception {
58          if (xGraphicProvider == null){
59              throw new Exception("XGraphicProvider is null.");
60          }
61  
62          if (xGraphic == null){
63              throw new Exception("XGraphic is null.");
64          }
65  
66          PropertyValue[] properties = new PropertyValue[2];
67          properties[0] = new PropertyValue();
68          properties[0].Name = "URL";
69          properties[0].Value = fileName;
70          properties[1] = new PropertyValue();
71          properties[1].Name = "MimeType";
72          properties[1].Value = "image/" + fileName.trim().substring(fileName.length() - 3);
73  
74          xGraphicProvider.storeGraphic(xGraphic, properties);
75      }

77      public static void exportAllEmbeddedGraphics(
78              XGraphicProvider xGraphicProvider,
79              XGraphicExportFilter xGraphicExportFilter,
80              XTextDocument xTextDocument,
81              String outDir,
82              String outputPrefix
83      ) throws Exception {
84          if (xGraphicProvider == null){
85              throw new Exception("XGraphicProvider is null.");
86          }
87  
88          if (xGraphicExportFilter == null){
89              throw new Exception("XGraphicExportFilter is null.");
90          }
91  
92          if (xTextDocument == null){
93              throw new Exception("XTextDocument is null.");
94          }
95  
96          XServiceInfo xServiceInfoDocument = UnoRuntime.queryInterface(XServiceInfo.class, xTextDocument);
97          if (xServiceInfoDocument != null) {
98              String[] documentServices = xServiceInfoDocument.getSupportedServiceNames();
99              xServiceInfoDocument = null;
100         }
101 
102         XDrawPageSupplier xDrawPageSupplier = UnoRuntime.queryInterface(XDrawPageSupplier.class, xTextDocument);
103         if(xDrawPageSupplier != null){
104             XDrawPage xDrawPage = xDrawPageSupplier.getDrawPage();
105 
106             if (xDrawPage != null) {
107                 int count = xDrawPage.getCount();
108                 for(int i = 0; i < count; ++i){
109                     Object elem = xDrawPage.getByIndex(i);
110                     XPropertySet xPropertySet = UnoRuntime.queryInterface(XPropertySet.class, elem);
111                     Property[] temp1 = xPropertySet.getPropertySetInfo().getProperties();
112 
113                     String desc = "";
114                     List<String> propertiesNames = new ArrayList<>();
115                     for (Property t: temp1 ) {
116                         propertiesNames.add(t.Name);
117                         if(t.Name == "Description"){
118                             desc = xPropertySet.getPropertyValue(t.Name).toString();
119                         }
120                     }
121                     XServiceInfo xServiceInfo = UnoRuntime.queryInterface(XServiceInfo.class, elem);
122                     XNamed xNamed = UnoRuntime.queryInterface(XNamed.class, elem);
123                     if (xServiceInfo != null){
124                         String[] services = xServiceInfo.getSupportedServiceNames();
125                         if(xServiceInfo.supportsService("com.sun.star.text.TextContent")
126                                 && xServiceInfo.supportsService("com.sun.star.text.TextGraphicObject")) {
127 
128                             try{
129                                 Property[] temp = xPropertySet.getPropertySetInfo().getProperties();
130                                 XGraphic xGraphic = UnoRuntime.queryInterface(
131                                         XGraphic.class, xPropertySet.getPropertyValue("Graphic")
132                                 );
133                                 if (xGraphic != null){
134                                     //String fileName = UUID.randomUUID() + ".png";
135                                     String fileName = (xNamed != null ? xNamed.getName() : UUID.randomUUID()) + ".png";
136                                     System.out.println("Filename: " + outDir + fileName);
137                                     exportGraphicObject(xGraphicProvider, xGraphic, outDir + outputPrefix + fileName);

139                                     xGraphic = null;

142 
143                             }
144 
145                         } else if (xServiceInfo.supportsService("com.sun.star.drawing.Shape")){
146                             XShape test = UnoRuntime.queryInterface(
147                                     XShape.class, elem
148                             );
149                             String type = test.getShapeType();
150 
151                             XComponent xCompo = UnoRuntime.queryInterface(
152                                     XComponent.class, elem
153                             );
154 
155                             String name = xNamed != null ? xNamed.getName() : "";

171                                 System.out.println("no component ?");

175                             System.out.println("Undefined drawing ?");
176                         }
177 
178                         xServiceInfo = null;
179                     }
180                 }

182             }

190     public static void ProcessShapes(String inputPath, String outputPath) throws IOException, Exception {
191         Optional<String> sofficePath = BinaryFinder.find("soffice");
192         if(!sofficePath.isPresent()){
193             throw new IOException("No openoffice compatible runtime found in system");
194         }
195         File oooExecFolder = new File(sofficePath.get()).getParentFile();
196         XComponentContext xContext = null;
197         XMultiComponentFactory xMultiComponentFactory = null;
198         Object oDesktop = null;
199         XComponentLoader xComponentLoader = null;
200         // Images export
201         XGraphicProvider xGraphicProvider = null;
202         // Shapes export
203         XGraphicExportFilter xGraphicExportFilter = null;
204 
205         try {
206             Method method = URLClassLoader.class.getDeclaredMethod("addURL", new Class[]{URL.class});
207             method.setAccessible(true);
208             method.invoke(Bootstrap.class.getClassLoader(), new Object[] {oooExecFolder.toURI().toURL()});
209             List<String> temp = new ArrayList<>(Arrays.asList(Bootstrap.getDefaultOptions()));
210             temp.add("--invisible");
211             temp.add("--headless");
212             xContext = Bootstrap.bootstrap(temp.toArray(new String[temp.size()]));
213         } catch (BootstrapException e) {
214             throw new IOException("Boostrap failed! Failed to start OpenOffice.", e);
215         } catch (MalformedURLException | NoSuchMethodException | InvocationTargetException | IllegalAccessException e) {
216             throw new RuntimeException(e);
217         }
218         try{
219             xMultiComponentFactory  = (XMultiComponentFactory) xContext.getServiceManager();
220             oDesktop = xMultiComponentFactory.createInstanceWithContext(
221                     "com.sun.star.frame.Desktop", xContext);
222             xComponentLoader = (XComponentLoader)UnoRuntime.queryInterface(
223                     com.sun.star.frame.XComponentLoader.class, oDesktop);
224             if (xComponentLoader == null) {
225                 throw new IOException("Failed to create openoffice XComponentLoader");
226             }
227 
228             try {
229                 Object oGraphicProvider = xMultiComponentFactory.createInstanceWithContext(
230                         "com.sun.star.graphic.GraphicProvider", xContext);
231                 xGraphicProvider = (XGraphicProvider) UnoRuntime.queryInterface(
232                         XGraphicProvider.class, oGraphicProvider);
233 
234                 Object oGraphicExportFilter = xMultiComponentFactory.createInstanceWithContext(
235                         "com.sun.star.drawing.GraphicExportFilter", xContext);
236                 xGraphicExportFilter = UnoRuntime.queryInterface(
237                         XGraphicExportFilter.class, oGraphicExportFilter);
238             } catch (Exception e) {
239                 throw new IOException("Failed to create openoffice XGraphicProvider or xGraphicExportFilter", e);
240             }
241 			/*try {
242 				Object oTextShapesSupplier = xMultiComponentFactory.createInstanceWithContext(
243 						"com.sun.star.text.XTextShapesSupplier", xContext);
244 				xTextShapesSupplier = (XTextShapesSupplier) UnoRuntime.queryInterface(
245 						XTextShapesSupplier.class, oTextShapesSupplier);
246 			} catch (Exception e) {
247 				System.out.println("Failed to create XGraphicProvider!");
248 				return;
249 			}*/
250 
251             PropertyValue[] properties = new PropertyValue[2];
252             properties[0] = new PropertyValue();
253             properties[0].Name = "Hidden";
254             properties[0].Value = false; // put true to hide OpenOffice window
255             properties[1] = new PropertyValue();
256             properties[1].Name = "CharacterSet";
257             properties[1].Value = "Unicode (UTF-8)";
258 
259             XComponent document = null;
260             try {
261                 document = xComponentLoader.loadComponentFromURL(inputPath, "_blank", 0, properties);
262             } catch (com.sun.star.io.IOException e) {
263                 e.printStackTrace();
264                 throw e;
265             } catch (IllegalArgumentException e) {
266                 e.printStackTrace();
267                 throw e;
268             }
269 
270             XTextDocument xTextDocument = (XTextDocument) UnoRuntime.queryInterface(
271                     XTextDocument.class, document);
272             // Exporting shapes
273             String inputNameWithoutExtension = FilenameUtils.getName(inputPath);
274             inputNameWithoutExtension = inputNameWithoutExtension.substring(0, inputNameWithoutExtension.length() - FilenameUtils.getExtension(inputNameWithoutExtension).length() - 1 );
275             exportAllEmbeddedGraphics(xGraphicProvider, xGraphicExportFilter, xTextDocument, outputPath, inputNameWithoutExtension+"-Shape");
276 
277             XCloseable xCloseable = (XCloseable)UnoRuntime.queryInterface(
278                     XCloseable.class, xContext);
279             if(xCloseable != null)
280             {
281                 xCloseable.close(true);
282                 xCloseable = null;
283             }
284 
285             xGraphicProvider = null;
286             xComponentLoader = null;
287             oDesktop = null;
288             xMultiComponentFactory = null;
289             xContext = null;
290 
291         } catch (java.lang.Exception e){
292             throw e;
293         } finally {
294             try {
295                 Runtime.getRuntime().exec("tskill soffice");
296             } catch (java.io.IOException e) {
297                 e.printStackTrace();
298             }
299         }
300     }
