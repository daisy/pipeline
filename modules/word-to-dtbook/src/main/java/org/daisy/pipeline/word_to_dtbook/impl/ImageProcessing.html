1   package org.daisy.pipeline.word_to_dtbook.impl;
2   
3   import java.awt.Graphics2D;
4   import java.awt.image.BufferedImage;
5   import java.awt.RenderingHints;
6   import java.io.File;
7   import java.io.UnsupportedEncodingException;
8   import java.net.URLEncoder;
9   
10  import javax.imageio.ImageIO;
11  
12  public class ImageProcessing {
13  
14  	static BufferedImage ResampleImage(BufferedImage SrcImage, float resampleValue) {
15  		float scaleFactorX = resampleValue / 96; // FIXME: assuming 96 dpi
16  		float scaleFactorY = resampleValue / 96; // FIXME: assuming 96 dpi
17  		int Width = (int)(SrcImage.getWidth() * scaleFactorX);
18  		int Height = (int)(SrcImage.getHeight() * scaleFactorY);
19  		BufferedImage bmPhoto = new BufferedImage(Width, Height, BufferedImage.TYPE_INT_RGB);
20  		try {
21  			Graphics2D grPhoto = bmPhoto.createGraphics();
22  			grPhoto.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
23  			grPhoto.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
24  			grPhoto.drawImage(SrcImage, 0, 0, Width, Height, null);
25  			grPhoto.dispose();
26  		} catch (Exception e) {
27  			throw new RuntimeException("Unable to resample the image", e);
28  		}
29  		return bmPhoto;
30  	}
31  
32  	static String SaveProcessedImage(BufferedImage ProcessedImage,
33  	                                 File OutPutFolder, String ImageName, String ImageFrmt) {
34  		ImageFormat Format = ImageFormat.from(ImageFrmt);
35  		try {
36  			File tempSave = new File(OutPutFolder, ImageName + Format.getExtension());
37  			ImageIO.write(ProcessedImage, Format.getFormatName(), tempSave);
38  			return UriEscape(tempSave.getName());
39  		} catch (Exception e) {
40  			throw new RuntimeException("Unable to save the image", e);
41  		}
42  	}
43  
44  	static String UriEscape(String imgName) {
45  		try {
46  			return URLEncoder.encode(imgName, "UTF-8").replace("+", "%20");
47  		} catch (UnsupportedEncodingException e) {
48  			throw new RuntimeException(e);
49  		}
50  	}
51  }
