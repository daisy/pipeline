366 		String tempString = "";
367 		if (numFormat.equals("decimal") || numFormat.equals("decimalZero")) {
368 			tempString = lvlText;
369 		} else if (numFormat.equals("lowerLetter")) {
370 			tempString = PageNumLowerAlphabet(Integer.parseInt(lvlText));
371 		} else if (numFormat.equals("upperLetter")) {
372 			tempString = PageNumUpperAlphabet(Integer.parseInt(lvlText));
373 		} else if (numFormat.equals("upperRoman")) {
374 			tempString = PageNumUpperRoman(Integer.parseInt(lvlText));
375 		} else if (numFormat.equals("lowerRoman")) {
376 			tempString = PageNumLowerRoman(Integer.parseInt(lvlText));
377 		} else if (numFormat.equals("bullet")) {
378 			tempString = bulletChar.get(iLvl);
379 		} else if (numFormat.equals("none")) {
380 			tempString = "";
381 		} else {
382 			tempString = lvlText;
383 		}
384 		return tempString;
385 	}

411 			throws XPathExpressionException {
412 		if (namespaces != null)
413 			xpath.setNamespaceContext(
414 				new NamespaceContext() {
415 					public String getNamespaceURI(String prefix) {
416 						return namespaces.get(prefix); }
417 					public String getPrefix(String namespaceURI) {
418 						for (String prefix : namespaces.keySet())
419 							if (namespaces.get(prefix).equals(namespaceURI))
420 								return prefix;
421 						return null; }
422 					public Iterator<String> getPrefixes(String namespaceURI) {
423 						List<String> prefixes = new ArrayList<String>();
424 						for (String prefix : namespaces.keySet())
425 							if (namespaces.get(prefix).equals(namespaceURI))
426 								prefixes.add(prefix);
427 						return prefixes.iterator(); }});
428 		else
429 			xpath.setNamespaceContext(null);
430 		return xpath.compile(expression);

540 					ImageFormat.Png.getFormatName(),
541 					new File(f.getAbsolutePath().replace(GetExtension(f), ".png")));
542 				strImgName = strImgName.replace(GetExtension(f), ".png");

554 			return "translation.oox2Daisy.ImageContent";
