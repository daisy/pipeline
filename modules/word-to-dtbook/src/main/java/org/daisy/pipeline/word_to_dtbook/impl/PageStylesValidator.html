1   package org.daisy.pipeline.word_to_dtbook.impl;
2   
3   import java.util.List;
4   
5   public class PageStylesValidator {
6   
7   	protected PageStyle _lastStyle = PageStyle.Empty;
8   
9   	public ValidationResult ValidateParagraph(List<PageStyle> paragraphStyles) {
10  		ValidationResult result = new ValidationResult();
11  		if (paragraphStyles.size() != 0) {
12  			if (paragraphStyles.size() > 1) {
13  				result = new ValidationResult(
14  					"More than one page style applied to paragraph, please applay only"
15  					+ "one style from Frontmatter (DAISY), Bodymatter (DAISY) and Rearmatter (DAISY)");
16  				_lastStyle = paragraphStyles.get(paragraphStyles.size() - 1);
17  			} else {
18  				PageStyle currentStyle = paragraphStyles.get(0);
19  				switch (_lastStyle) {
20  				case Empty:
21  					if (currentStyle == PageStyle.Rearmatter) {
22  						result = new ValidationResult(
23  							"Rearmatter (Daisy) can be applied only after Bodymetter (DAISY) style");
24  					}
25  					break;
26  				case Frontmatter:
27  					if (currentStyle == PageStyle.Frontmatter) {
28  							result = new ValidationResult(
29  								"Frontmatter (DAISY) style should be applied only one time");
30  					} else if (currentStyle == PageStyle.Rearmatter) {
31  						result = new ValidationResult(
32  							"Rearmatter (DAISY) style can be applied only after Bodymatter (DAISY) style");
33  					}
34  					break;
35  				case Bodymatter:
36  					if (currentStyle == PageStyle.Frontmatter) {
37  						result = new ValidationResult(
38  							"Frontmatter (DAISY) style can not be applied after Bodymatter (DAISY) style");
39  					} else if (currentStyle == PageStyle.Bodymatter) {
40  						result = new ValidationResult(
41  							"Bodymatter (DAISY) style should be applied only one time");
42  					}
43  					break;
44  				case Rearmatter:
45  					result = new ValidationResult(
46  						"Frontmatter (DAISY), Bodymatter (DAISY) and Rearmatter (DAISY) styles "
47  						+ "can not be applied after Rearmatter (DAISY) style");
48  					break;
49  				default:
50  					throw new RuntimeException();
51  				}
52  				_lastStyle = currentStyle;
53  			}
54  		}
55  		return result;
56  	}
57  
58  	public ValidationResult ValidateLastStyle() {
59  		if (_lastStyle == PageStyle.Frontmatter)
60  			return new ValidationResult(
61  				"Frontmatter (DAISY) can not be applied without Bodymatter (DAISY) style ");
62  		return new ValidationResult();
63  	}
64  
65  	public static PageStyle GetPageStyle(String style) {
66  		if (style.startsWith("Frontmatter"))
67  			return PageStyle.Frontmatter;
68  		else if (style.startsWith("Bodymatter"))
69  			return PageStyle.Bodymatter;
70  		else if (style.startsWith("Rearmatter"))
71  			return PageStyle.Rearmatter;
72  		return PageStyle.Empty;
73  	}
74  }
