33  public abstract class WordBasedScript implements ScriptService<Script>, ScriptServiceProvider {
34  
35  	private final String id;
36  	private final String version;
37  	private final String formatId;
38  	private final String[] formatChain;
39  	private ScriptRegistry scriptRegistry = null;
40  	private ScriptService<?> wordToDTBookService = null;
41  	private ScriptService<?>[] scriptServiceChain = null;
42  
43  	private static final Logger logger = LoggerFactory.getLogger(WordBasedScript.class);
44  
45  	public WordBasedScript(String... formatChain) {
46  		if (formatChain.length == 0)
47  			throw new IllegalArgumentException();
48  		for (String format : formatChain)
49  			if (!("daisy3".equals(format)
50  			      || "daisy202".equals(format)
51  			      || "ebraille".equals(format)
52  			      || "epub3".equals(format)
53  			      || "html".equals(format)
54  			      || "mp3".equals(format)))
55  				throw new IllegalArgumentException();
56  		this.formatChain = formatChain;
57  		this.formatId = formatChain[formatChain.length - 1];
58  		this.id = "word-to-" + this.formatId;
59  		Properties mavenProps = new Properties();
60  		try (InputStream is = URLs.getResourceFromJAR("/maven.properties", WordBasedScript.class)
61  		                          .openStream()) {
62  			mavenProps.load(is);
63  		} catch (IOException e) {
64  			throw new RuntimeException(e);
65  		}
66  		this.version = mavenProps.getProperty("version");
67  	}

77  	private Iterable<ScriptService<?>> scripts = null;
78  
79  	@Override
80  	public Iterable<ScriptService<?>> getScripts() {
81  		if (scripts == null) {
82  			if (scriptRegistry == null)
83  				throw new IllegalStateException("setScriptRegistry() must be called first");
84  			String wordToDTBookScriptId = "word-to-dtbook";
85  			wordToDTBookService = scriptRegistry.getScript(wordToDTBookScriptId);
86  			if (wordToDTBookService == null) {
87  				logger.debug("failed to load script " + id + ": no script found with ID " + wordToDTBookScriptId);
88  				return Collections.emptyList();
89  			}
90  			scriptServiceChain = new ScriptService<?>[formatChain.length];
91  			int i = 0;
92  			String from = "dtbook";
93  			for (String to : formatChain) {
94  				String scriptId = from + "-to-" + to;
95  				ScriptService<?> scriptService = scriptRegistry.getScript(scriptId);
96  				if (scriptService == null) {
97  					logger.debug("failed to load script " + id + ": no script found with ID " + scriptId);
98  					return Collections.emptyList();
99  				}
100 				scriptServiceChain[i++] = scriptService;
101 				from = to;
102 			}
103 			scripts = Collections.singleton(this);
104 		}
105 		return scripts;
106 	}

122 	private Script script = null;
123 	private Script wordToDTBook = null;
124 	private Script[] scriptChain = null;
125 
126 	@Override
127 	public Script load() {
128 		if (wordToDTBookService == null || scriptServiceChain == null)
129 			throw new IllegalStateException(); // the ScriptService interface is supposed to be used only
130 			                                   // after ScriptServiceProvider/ getScripts() has been called
131 		boolean rebuildScript = false;
132 		if (script == null)
133 			rebuildScript = true;
134 		Script wordToDTBook = wordToDTBookService.load();
135 		if (wordToDTBook != this.wordToDTBook) {
136 			rebuildScript = true;
137 			this.wordToDTBook = wordToDTBook;
138 		}
139 		if (scriptChain == null)
140 			scriptChain = new Script[scriptServiceChain.length];
141 		int i = 0;
142 		for (ScriptService<?> ss : scriptServiceChain) {
143 			if (ss == null)
144 				throw new IllegalStateException();
145 			Script s = ss.load();
146 			if (s != scriptChain[i]) {
147 				rebuildScript = true;
148 				scriptChain[i] = s;
149 			}
150 			i++;
151 		}
152 		if (rebuildScript) {
153 			Script.Builder builder = new Builder();
154 			String formatName = getFormatName(formatId);
155 			builder = builder
156 				.withInputFileset("docx")
157 				.withOutputFileset(formatId)
158 				.withShortName("Word to " + formatName + " (experimental)")
159 				.withDescription(
160 					"Transforms a Microsoft Office Word (.docx) document into "
161 					+ (formatId.equals("mp3")
162 					      ? "a folder structure with MP3 files suitable for playback on MegaVoice Envoy devices"
163 					      : (formatName + " format")) + ".")
164 				// FIXME: home page has yet to be created
165 				.withHomepage(
166 					"http://daisy.github.io/pipeline/Get-Help/User-Guide/Scripts/word-to-"
167 					+ formatId + "/")
168 				;
169 			for (ScriptPort p : wordToDTBook.getInputPorts())
170 				builder = builder.withInputPort(p.getName(), p);
171 			for (ScriptOption o : wordToDTBook.getOptions()) {
172 				String name = o.getName();
173 				if (   "title".equals(name)
174 				    || "creator".equals(name)
175 				    || "publisher".equals(name)
176 				    || "uid".equals(name)
177 				    || "subject".equals(name)
178 				    || "accept-revisions".equals(name)
179 				    || "repair".equals(name)
180 				    || "tidy".equals(name)
181 				)
182 					builder = builder.withOption(name, o);
183 			}
184 			Script lastScript = null;
185 			for (Script s : scriptChain) {
186 				for (ScriptOption o : s.getOptions()) {
187 					String name = o.getName();                     //  daisy3  |  daisy202  |  ebraille  |  epub3  |  html  |  mp3
188 					if (   "include-tts-log".equals(name)          //  x          x                         x                  x
189 					    || "tts-config".equals(name)               //  x          x                         x                  x
190 					    || "language".equals(name)                 //             x                         x         x
191 					    || "validation".equals(name)               //             x                         x         x
192 					    || "folder-depth".equals(name)             //                                                          x
193 					    || "epub-package".equals(name)             //                          x
194 					    || "attach-stylesheet".equals(name)        //                          x
195 					    || "braille-code".equals(name)             //                          x
196 					    || "include-original-text".equals(name)    //                          x
197 					    || (!"mp3".equals(formatId) &&
198 					         (   "audio".equals(name)              //  x          x                         x
199 					          || "with-text".equals(name)          //  x
200 					          || "word-detection".equals(name)))   //  x
201 					)
202 						// note that this will override any previously defined options with the same name
203 						builder = builder.withOption(name, o);
204 				}
205 				lastScript = s;
206 			}
207 			// take outputs of last script
208 			for (ScriptPort p : lastScript.getOutputPorts())
209 				builder = builder.withOutputPort(p.getName(), p);
210 			script = builder.build();
211 		}
212 		return script;
213 	}

228 	private class ScriptImpl extends Script {
229 
230 		private final Script wordToDTBook;
231 		private final Script[] scriptChain;
232 
233 		private ScriptImpl(String name, String description, String homepage,
234 		                   Map<String,ScriptPort> inputPorts, Map<String,ScriptPort> outputPorts,
235 		                   Map<String,ScriptOption> options, List<String> inputFilesets,
236 		                   List<String> outputFilesets) {
237 			super(id, version, name, description, homepage, inputPorts, outputPorts, options,
238 			      inputFilesets, outputFilesets);
239 			this.wordToDTBook = WordBasedScript.this.wordToDTBook;
240 			this.scriptChain = WordBasedScript.this.scriptChain.clone();
241 		}
242 
243 		@Override
244 		public Status run(ScriptInput input, Map<String,String> properties,
245 		                  MessageAppender messages, JobResultSet.Builder resultBuilder,
246 		                  File resultDir) throws IOException {
247 			ScriptInput.Builder stepInput = new ScriptInput.Builder(input.getResources());
248 			for (Source s : input.getInput("source"))
249 				stepInput = stepInput.withInput("source", s);
250 			stepInput = pickOptions(wordToDTBook, stepInput, input);
251 			JobResultSet.Builder stepResultBuilder = new JobResultSet.Builder(wordToDTBook);
252 			File stepResultDir = new File(resultDir, "word-to-dtbook");
253 			mkdirs(stepResultDir);
254 			Status status = null;
255 			BigDecimal portion = BigDecimal.ONE.divide(
256 				new BigDecimal(1 + scriptChain.length), MathContext.DECIMAL128);
257 			try (MessageAppender stepMessages = messages != null
258 			         ? messages.append(new MessageBuilder().withProgress(portion))
259 			         : null) {
260 				status = wordToDTBook.run(stepInput.build(),
261 			                              properties,
262 			                              stepMessages,
263 			                              stepResultBuilder,
264 			                              stepResultDir);
265 			}
266 			Script prevStep = wordToDTBook;
267 			String prevStepResultFormat = "dtbook";
268 			int i = 0;
269 			for (Script step : scriptChain) {
270 				switch (status) {
271 				case ERROR:
272 				case FAIL:
273 					return status;
274 				case SUCCESS:
275 				default:
276 				}
277 				JobResultSet stepResult = stepResultBuilder.build();
278 				stepInput = new ScriptInput.Builder();
279 				for (JobResult r : stepResult.getResults("result"))
280 					if (isPrimaryFile(r.getPath().getName(), prevStepResultFormat))
281 						stepInput = stepInput.withInput("source", r.getPath());
282 				stepInput = pickOptions(step, stepInput, input);
283 				if ("mp3".equals(formatId))
284 					// certain options must have a fixed value for mp3 output
285 					for (ScriptOption o : step.getOptions()) {
286 						String name = o.getName();
287 						if ("audio".equals(name))
288 							stepInput = stepInput.withOption(name, "true");
289 						else if ("with-text".equals(name))
290 							stepInput = stepInput.withOption(name, "false");
291 						else if ("word-detection".equals(name))
292 							stepInput = stepInput.withOption(name, "false");
293 					}
294 				String stepResultFormat = formatChain[i];
295 				stepResultBuilder = formatId.equals(stepResultFormat)
296 					? resultBuilder // this is the last step
297 					: new JobResultSet.Builder(step);
298 				stepResultDir = new File(resultDir, step.getId());
299 				mkdirs(stepResultDir);
300 				try (MessageAppender stepMessages = messages != null
301 				         ? messages.append(
302 				             new MessageBuilder().withProgress(portion)
303 				                                 .withText("Converting " + getFormatName(prevStepResultFormat)
304 				                                           + " to " + getFormatName(stepResultFormat))
305 				                                 .withLevel(Message.Level.INFO))
306 				         : null) {
307 					status = step.run(stepInput.build(),
308 					                  properties,
309 					                  stepMessages,
310 					                  stepResultBuilder,
311 					                  stepResultDir);
312 				}
313 				prevStep = step;
314 				prevStepResultFormat = stepResultFormat;
315 				i++;
316 			}
317 			return status;
318 		}
319 	}

321 	private static String getFormatName(String formatId) {
322 		if ("dtbook".equals(formatId))
323 			return "DTBook";
324 		else if ("daisy3".equals(formatId))
325 			return "DAISY 3";
326 		else if ("daisy202".equals(formatId))
327 			return "DAISY 2.02";
328 		else if ("ebraille".equals(formatId))
329 			return "eBraille";
330 		else if ("epub3".equals(formatId))
331 			return "EPUB 3";
332 		else if ("html".equals(formatId))
333 			return "HTML";
334 		else if ("mp3".equals(formatId))
335 			return "navigable MP3 file-set";
336 		else
337 			throw new IllegalArgumentException("unknown format " + formatId);
338 	}
339 
340 	private static boolean isPrimaryFile(String fileName, String formatId) {
341 		if ("dtbook".equals(formatId))
342 			return fileName.endsWith(".xml");
343 		if ("daisy3".equals(formatId))
344 			return fileName.endsWith(".opf");
345 		if ("daisy202".equals(formatId))
346 			return fileName.equals("ncc.html"); // assume output of epub3-to-daisy202
347 		else if ("epub3".equals(formatId))
348 			return fileName.endsWith(".epub");
349 		else
350 			throw new IllegalArgumentException();
351 	}

353 	private static void mkdirs(File dir) throws IOException {
354 		if (!dir.exists() && !dir.mkdirs())
355 			throw new IOException("Could not create directory:" + dir.getAbsolutePath());
356 	}
357 
358 	private static ScriptInput.Builder pickOptions(Script script, ScriptInput.Builder toInput, ScriptInput fromInput) {
359 		for (ScriptOption o : script.getOptions()) {
360 			String name = o.getName();
361 			for (String s : fromInput.getOption(name))
362 				toInput = toInput.withOption(name, s);
363 		}
364 		return toInput;
365 	}
