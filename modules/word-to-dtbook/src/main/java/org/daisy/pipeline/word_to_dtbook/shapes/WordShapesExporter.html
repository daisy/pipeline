1   package org.daisy.pipeline.word_to_dtbook.shapes;
2   import org.daisy.common.file.URLs;
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   import java.io.*;
7   import java.net.URI;
8   import java.net.URL;
9   import java.nio.file.Files;
10  
11  public class WordShapesExporter {
12      private static final Logger LOGGER = LoggerFactory.getLogger(WordShapesExporter.class);
13  
14  
15      synchronized public static void ProcessShapes(String inputPath, String outputPath) throws Exception {
16          String exeFileName = "ExportShapesWithWord.exe";
17          // Write ExportShapesWithWord in a temp folder
18          if (!System.getProperty("os.name").toLowerCase().startsWith("windows"))
19              throw new IOException("Not on Windows");
20          URL exePath = URLs.getResourceFromJAR("exe/" + exeFileName, WordShapesExporter.class);
21          File exeFile; {
22              try {
23                  exeFile = new File(URLs.asURI(exePath));
24              } catch (IllegalArgumentException iae) {
25                  try {
26                      File tmpDirectory = Files.createTempDirectory("pipeline-").toFile();
27                      tmpDirectory.deleteOnExit();
28                      exeFile = new File(tmpDirectory, exeFileName);
29                      exeFile.deleteOnExit();
30                      exeFile.getParentFile().mkdirs();
31                      exeFile.createNewFile();
32                      FileOutputStream writer = new FileOutputStream(exeFile);
33                      exePath.openConnection();
34                      InputStream reader = exePath.openStream();
35                      byte[] buffer = new byte[153600];
36                      int bytesRead;
37                      while ((bytesRead = reader.read(buffer)) > 0) {
38                          writer.write(buffer, 0, bytesRead);
39                          buffer = new byte[153600];
40                      }
41                      writer.close();
42                      reader.close();
43                  } catch (IOException e) {
44                      throw new IOException("Could not unpack " + exeFileName, e);
45                  }
46              }
47  
48          }
49          // Launched the program with inputPath and outputPath as argument
50          ProcessBuilder procbuilder;
51          try{
52              procbuilder = new ProcessBuilder(
53                      exeFile.getAbsolutePath(),
54                      new File(new URI(inputPath)).getAbsolutePath(),
55                      new File(new URI(outputPath)).getAbsolutePath()
56              );
57              procbuilder.redirectErrorStream();
58  
59          } catch (Exception e){
60              throw new Exception("Could not export shapes with word", e);
61          }
62  
63  
64          Process exporter = procbuilder.start();
65          BufferedReader stdOut = new BufferedReader(new
66                  InputStreamReader(exporter.getInputStream()));
67  
68          BufferedReader stdError = new BufferedReader(new
69                  InputStreamReader(exporter.getErrorStream()));
70  
71          String s = null;
72          while ((s = stdOut.readLine()) != null) {
73              LOGGER.info(s);
74          }
75  
76          while ((s = stdError.readLine()) != null) {
77              LOGGER.error(s);
78          }
79          int returnCode = exporter.waitFor();
80          if(returnCode != 0){
81              throw new IOException("Word shapes exporter returned error code " + returnCode);
82          }
83      }
84  }
