3   /**
4    * Unzip extension for non-XML files
5    *
6    * @author Lars Wittmar -- le-tex publishing services GmbH
7    * @date   2012-03-08
8    * 
9    * A Java unzip for XProc, particularly useful 
10   * if you want to access zip contents as extracted files
11   * (and don't want to retrieve single files recursively, and
12   * potentially base64 encoded).
13   * 
14   * Written by Lars Wittmar, le-tex publishing services GmbH
15   * (small portions also by Gerrit Imsieke)
16   * 
17   * Configure letex:unzip step by passing options to it:
18   * zip (required):      file name (not URI) of a zip file
19   * dest-dir (required): directory (relative or absolute; not an URI)
20   * overwrite ('yes'/'no', optional, default 'no'): overwrite individual files even if they already exist
21   * file (optional):     file name relative to zip root. If omitted, extract complete zip file
22   * 
23   * The directory (and missing subdirectories) will be created if
24   * missing. 
25   * 
26   * CAVEAT: If overwrite='yes' and, e.g., dest-dir='/', it will attempt to remove the root directory.
27   * 
28   * DO NOT USE THIS AS A USER WITH ADMIN PRIVILEGES.
29   * 
30   *
31   * 2015-02-27, Jostein Austvik Jacobsen (NLB - Norwegian library of talking books and braille):
32   *     - Wrapped into a org.daisy.common.xproc.calabash.XProcStepProvider
33   *     - removed Apache Commons dependency
34   *     - Use file URIs instead of OS-specific paths for options
35   * 2016-02-10, Jostein Austvik Jacobsen (NLB - Norwegian library of talking books and braille):
36   *     - slightly changed the meaning of the 'overwrite' option
37   */

92      public static class UnZip
93              extends DefaultStep implements XProcStep
94      {
95          public UnZip(XProcRuntime runtime, XAtomicStep step)
96          {
97              super(runtime,step);
98          }
99  
100         @Override
101         public void setOutput(String port, WritablePipe pipe)
102         {
103             myResult = pipe;
104         }
105 
106         @Override
107         public void reset()
108         {
109             myResult.resetWriter();
110         }
111 
112         @Override
113         public void run()
114             throws SaxonApiException
115         {
116             super.run();
117             
118             String result = null;
119             File folder = null, source = null;
120             try {
121                 RuntimeValue href = getOption(new QName("zip"));
122                 URI sourceUri = href.getBaseURI().resolve(href.getString());
123                 href = getOption(new QName("dest-dir"));
124                 URI destinationUri = href.getBaseURI().resolve(href.getString());
125     			
126     			if (!"file".equals(sourceUri.getScheme())) {
127     				throw new XProcException(step.getNode(), "Only file: scheme URIs are supported by zip URI.");
128     			} else {
129     				source = new File(sourceUri.getPath());
130     				if (!source.isFile()) {
131     					throw new XProcException(step.getNode(), "The zip URI must refer to a file.");
132     				}
133     			}
134     			
135     			if (!"file".equals(destinationUri.getScheme())) {
136     				throw new XProcException(step.getNode(), "Only file: scheme URIs are supported by dest-dir URI.");
137     			} else {
138     				folder = new File(destinationUri.getPath());
139     				if (!folder.exists()) {
140     					folder.mkdirs();
141     				}
142     				if (!folder.isDirectory()) {
143     					throw new XProcException(step.getNode(), "The dest-dir URI must refer to a directory.");
144     				}
145     				if (folder.getParentFile() == null || folder.getParentFile().getParentFile() == null) {
146     					// Basic safety valve; should catch stuff like "/", "/home/", "C:\", "C:\Windows\" etc.
147     					// Mainly aimed at catching typos or bugs in scripts during developent;
148     					// running without admin-privileges is the only fairly safe option here 
149     					throw new XProcException(step.getNode(), "The dest-dir URI must not be the file system root directory (or any of its children) (was: \""+folder.getCanonicalPath()+"\").");
150     				}
151     			}
152     			
153                 String overwriteS = getOption(new QName("overwrite")).getString();
154                 boolean overwrite = false; 
155                 if (overwriteS != null) {
156                     if (overwriteS.matches("yes")) {
157                         overwrite = true;
158                     }
159                 }
160 
161     			String filename = "";
162     			if (getOption(new QName("file"))!=null) {
163                     filename = getOption(new QName("file")).getString();
164     			}
165     			
166                 if(! folder.exists()) { 
167                     folder.mkdirs(); 
168                 }
169     			
170     			final int BUFFER = 2048;
171     			BufferedOutputStream dest = null;
172     			FileInputStream fis = new FileInputStream(source);
173     			ZipInputStream zis = new ZipInputStream(new BufferedInputStream(fis));
174     			ZipEntry entry;
175 
176                 int numFiles = 0;
177     			result = "<c:files xmlns:c=\"http://www.w3.org/ns/xproc-step\" xml:base=\"" + folder.toURI() + "\">";
178     			//without given fileNode extract everything
179     			if(filename.equals("")) {
180                     while((entry = zis.getNextEntry()) != null) {
181                         int count;
182                         byte data[] = new byte[BUFFER];
183                         
184                         if(entry.isDirectory()) {
185                         	File dir = new File(folder, entry.getName());
186                             if (!dir.exists()) { dir.mkdirs(); }
187                         } else {
188                             numFiles++;
189                             result += "<c:file name=\"" + entry.getName() + "\"/>";
190                             File destFile = new File(folder, entry.getName());
191                             if (destFile.exists()) {
192                                 if (overwrite) {
193                                     deleteDirectory(destFile);
194                                 } else {
195                                     continue;
196                                 }
197                             }
198                             File destdir = destFile.getParentFile();
199                             if (!destdir.exists()) { destdir.mkdirs(); }
200                             FileOutputStream fos = new FileOutputStream(destFile);
201                             dest = new BufferedOutputStream(fos, BUFFER);
202                             while ((count = zis.read(data, 0, BUFFER)) != -1) {
203                                 dest.write(data, 0, count);
204                             }
205                             dest.flush();
206                             dest.close();
207                         }
208                     }
209     			} else {
210                     while((entry = zis.getNextEntry()) != null) {
211                         if(filename.equals(entry.getName())) {
212                             int count;
213                             byte data[] = new byte[BUFFER];
214                             
215                             numFiles++;
216                             result += "<c:file name=\"" + entry.getName() + "\"/>";
217                             File destFile = new File(folder, entry.getName());
218                             if (destFile.exists()) {
219                                 if (overwrite) {
220                                     deleteDirectory(destFile);
221                                 } else {
222                                     continue;
223                                 }
224                             }
225                             File destdir = destFile.getParentFile();
226                             if (!destdir.exists()) { destdir.mkdirs(); }
227                             FileOutputStream fos = new FileOutputStream(destFile);
228                             dest = new BufferedOutputStream(fos, BUFFER);
229                             while ((count = zis.read(data, 0, BUFFER)) != -1) {
230                                 dest.write(data, 0, count);
231                             }
232                             dest.flush();
233                             dest.close();
234                         }							 
235                     }
236     			}
237     			
238     			zis.close();
239     			if (numFiles == 0) {
240     			  result = "<c:error xmlns:c=\"http://www.w3.org/ns/xproc-step\" xmlns:letex=\"http://www.le-tex.de/namespace\" code=\"zip-error\" href=\""+folder.toURI()+source.toURI()+"\">No content processed. Zip file may empty or corrupted.</c:error>";
241     			} else {
242     			  result += "</c:files>";
243     			}
244                 
245     		  } catch(Exception e) {
246     			  StringWriter stackTraceWriter = new StringWriter();
247     			  e.printStackTrace(new PrintWriter(stackTraceWriter));
248     			  String stackTrace = stackTraceWriter.toString().replaceAll("-", " - "); // just to be sure that the comment is valid
249     			  result = "<c:error xmlns:c=\"http://www.w3.org/ns/xproc-step\" xmlns:letex=\"http://www.le-tex.de/namespace\" code=\"zip-error\" href=\""+folder.toURI()+source.toURI()+"\">Zip file seems to be corrupted: "+e.getMessage()+"<!-- "+stackTrace+" --></c:error>";
250     		  }
251             DocumentBuilder builder = runtime.getProcessor().newDocumentBuilder();
252             Source src = new StreamSource(new StringReader(result));
253             XdmNode doc = builder.build(src);
254 
255             myResult.write(doc);
256         }

258         /** http://stackoverflow.com/a/3775718/281065 */
259         public static boolean deleteDirectory(File directory) {
260             if(directory.exists()){
261                 File[] files = directory.listFiles();
262                 if(null!=files){
263                     for(int i=0; i<files.length; i++) {
264                         if(files[i].isDirectory()) {
265                             deleteDirectory(files[i]);
266                         }
267                         else {
268                             files[i].delete();
269                         }
270                     }
271                 }
272             }
273             return(directory.delete());
274         }
275 
276         private WritablePipe myResult = null;
277     }
